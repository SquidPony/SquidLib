020    public enum Measurement {
021
022        /**
023         * The distance it takes when only the four primary directions can be
024         * moved in. The default.
025         */
026        MANHATTAN,
027        /**
028         * The distance it takes when diagonal movement costs the same as
029         * cardinal movement.
030         */
031        CHEBYSHEV,
032        /**
033         * The distance it takes as the crow flies. This will NOT affect movement cost when calculating a path,
034         * only the preferred squares to travel to (resulting in drastically more reasonable-looking paths).
035         */
036        EUCLIDEAN
037    }

039    /**
040     * This affects how distance is measured on diagonal directions vs. orthogonal directions. MANHATTAN should form a
041     * diamond shape on a featureless map, while CHEBYSHEV and EUCLIDEAN will form a square. EUCLIDEAN does not affect
042     * the length of paths, though it will change the DijkstraMap's gradientMap to have many non-integer values, and
043     * that in turn will make paths this finds much more realistic and smooth (favoring orthogonal directions unless a
044     * diagonal one is a better option).
045     */
046    public Measurement measurement = Measurement.MANHATTAN;

297     * Used to initialize or re-initialize a DijkstraMap that needs a new PhysicalMap because it either wasn't given
298     * one when it was constructed, or because the contents of the terrain have changed permanently (not if a
299     * creature moved; for that you pass the positions of creatures that block paths to scan() or findPath() ).
300     *
301     * @param level
302     * @return
303     */
304    public DijkstraMap initialize(final double[][] level) {
305        width = level.length;
306        height = level[0].length;
307        gradientMap = new double[width][height];
308        safetyMap = new double[width][height];
309        physicalMap = new double[width][height];
310        costMap = new double[width][height];
311        targetMap = new Coord[width][height];
312        for (int x = 0; x < width; x++) {
313            System.arraycopy(level[x], 0, gradientMap[x], 0, height);
314            System.arraycopy(level[x], 0, physicalMap[x], 0, height);
315            Arrays.fill(costMap[x], 1.0);
316        }
317        initialized = true;
318        return this;
319    }

444    /**
445     * Gets the appropriate DijkstraMap.Measurement to pass to a constructor if you already have a Radius.
446     * Matches SQUARE or CUBE to CHEBYSHEV, DIAMOND or OCTAHEDRON to MANHATTAN, and CIRCLE or SPHERE to EUCLIDEAN.
447     *
448     * @param radius the Radius to find the corresponding Measurement for
449     * @return a DijkstraMap.Measurement that matches radius; SQUARE to CHEBYSHEV, DIAMOND to MANHATTAN, etc.
450     */
451    public static Measurement findMeasurement(Radius radius) {
452        if (radius.equals2D(Radius.SQUARE))
453            return DijkstraMap.Measurement.CHEBYSHEV;
454        else if (radius.equals2D(Radius.DIAMOND))
455            return DijkstraMap.Measurement.MANHATTAN;
456        else
457            return DijkstraMap.Measurement.EUCLIDEAN;
458    }

465     * @return a DijkstraMap.Measurement that matches radius; CHEBYSHEV to SQUARE, MANHATTAN to DIAMOND, etc.

468        switch (measurement) {
469            case CHEBYSHEV:
470                return Radius.SQUARE;
471            case EUCLIDEAN:
472                return Radius.CIRCLE;
473            default:
474                return Radius.DIAMOND;

1234     * Scans the dungeon using DijkstraMap.scan with the listed goals and start point, and returns a list
1235     * of Coord positions (using the current measurement) needed to get closer to the closest reachable
1236     * goal. The maximum length of the returned list is given by length; if moving the full length of
1237     * the list would place the mover in a position shared by one of the positions in onlyPassable
1238     * (which is typically filled with friendly units that can be passed through in multi-tile-
1239     * movement scenarios), it will recalculate a move so that it does not pass into that cell.
1240     * The keys in impassable should be the positions of enemies and obstacles that cannot be moved
1241     * through, and will be ignored if there is a goal overlapping one.
1242     * <br>
1243     * This caches its result in a member field, path, which can be fetched after finding a path and will change with
1244     * each call to a pathfinding method.
1245     * @param length       the length of the path to calculate
1246     * @param impassable   a Set of impassable Coord positions that may change (not constant like walls); can be null
1247     * @param onlyPassable a Set of Coord positions that this pathfinder cannot end a path occupying (typically allies); can be null
1248     * @param start        the start of the path, should correspond to the minimum-x, minimum-y position of the pathfinder
1249     * @param targets      a vararg or array of Coord that this will try to pathfind toward
1250     * @return an ArrayList of Coord that will contain the locations of this creature as it goes toward a target. Copy of path.
1251     */
1252    public ArrayList<Coord> findPath(int length, Set<Coord> impassable,
1253                                     Set<Coord> onlyPassable, Coord start, Coord... targets) {
1254        if (!initialized) return null;
1255        path.clear();
1256        LinkedHashSet<Coord> impassable2;
1257        if (impassable == null)
1258            impassable2 = new LinkedHashSet<>();
1259        else
1260            impassable2 = new LinkedHashSet<>(impassable);
1261        if (onlyPassable == null)
1262            onlyPassable = new LinkedHashSet<>();
1263
1264        resetMap();
1265        for (Coord goal : targets) {
1266            setGoal(goal.x, goal.y);
1267        }
1268        if (goals.isEmpty())
1269            return new ArrayList<>(path);
1270        scan(impassable2);
1271        Coord currentPos = start;
1272        double paidLength = 0.0;
1273        while (true) {
1274            if (frustration > 500) {
1275                path.clear();
1276                break;
1277            }
1278            double best = gradientMap[currentPos.x][currentPos.y];
1279            final Direction[] dirs = appendDir(shuffleDirs(rng), Direction.NONE);
1280            int choice = rng.nextInt(dirs.length);
1281
1282            for (int d = 0; d < dirs.length; d++) {
1283                Coord pt = Coord.get(currentPos.x + dirs[d].deltaX, currentPos.y + dirs[d].deltaY);
1284                if (gradientMap[pt.x][pt.y] < best) {
1285                    if (dirs[choice] == Direction.NONE || !path.contains(pt)) {
1286                        best = gradientMap[pt.x][pt.y];
1287                        choice = d;
1288                    }
1289                }
1290            }
1291
1292            if (best >= gradientMap[currentPos.x][currentPos.y] || physicalMap[currentPos.x + dirs[choice].deltaX][currentPos.y + dirs[choice].deltaY] > FLOOR) {
1293                path.clear();
1294                break;
1295            }
1296            currentPos = currentPos.translate(dirs[choice].deltaX, dirs[choice].deltaY);
1297            path.add(currentPos);
1298            paidLength += costMap[currentPos.x][currentPos.y];
1299            frustration++;
1300            if (paidLength > length - 1.0) {
1301                if (onlyPassable.contains(currentPos)) {
1302
1303                    closed.put(currentPos, WALL);
1304                    impassable2.add(currentPos);
1305                    return findPath(length, impassable2, onlyPassable, start, targets);
1306                }
1307                break;
1308            }
1309            if (gradientMap[currentPos.x][currentPos.y] == 0)
1310                break;
1311        }
1312        frustration = 0;
1313        goals.clear();
1314        return new ArrayList<>(path);
1315    }
