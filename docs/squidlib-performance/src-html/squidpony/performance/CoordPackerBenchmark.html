145    public static final int DIMENSION = 64;
146    public static DungeonGenerator dungeonGen =
147            new DungeonGenerator(DIMENSION, DIMENSION, new StatefulRNG(0x1337BEEFDEAL));
148    public static SerpentMapGenerator serpent = new SerpentMapGenerator(DIMENSION, DIMENSION,
149            new StatefulRNG(0x1337BEEFDEAL));
150
151    public static final char[][][] maps;
152    public static final short[][] floors, walls, visibleWalls;
153    public static GreasedRegion[] floorsG, wallsG, visibleWallsG;
154    static {
155        serpent.putWalledBoxRoomCarvers(1);
156        maps = new char[0x1000][][];
157        floors = new short[0x1000][];
158        walls = new short[0x1000][];
159        visibleWalls = new short[0x1000][];
160        floorsG = new GreasedRegion[0x1000];
161        wallsG = new GreasedRegion[0x1000];
162        visibleWallsG = new GreasedRegion[0x1000];
163        for (int i = 0; i < 0x1000; i++) {
164            maps[i] = dungeonGen.generate(serpent.generate());
165            floors[i] = CoordPacker.pack(maps[i], '.');
166            walls[i] = CoordPacker.pack(maps[i], '#');
167            visibleWalls[i] = CoordPacker.fringe(floors[i], 1, DIMENSION, DIMENSION, true);
168            floorsG[i] = new GreasedRegion(maps[i], '.');
169            wallsG[i] = new GreasedRegion(maps[i], '#');
170            visibleWallsG[i] = new GreasedRegion(floorsG[i]).fringe8way();
171        }
172    }

174    public long doPack()
175    {
176        long l = 0;
177        for (int i = 0; i < 0x1000; i++) {
178            l += CrossHash.hash(CoordPacker.pack(maps[i], '.'));
179        }
180        return l;
181    }

524     *

530    public static void main(String[] args) throws RunnerException {

532                .include(CoordPackerBenchmark.class.getSimpleName())
533                .timeout(TimeValue.seconds(30))
534                .warmupIterations(3)
535                .measurementIterations(3)
536                .forks(1)
537                .build();
538
539
540        new Runner(opt).run();
541    }
