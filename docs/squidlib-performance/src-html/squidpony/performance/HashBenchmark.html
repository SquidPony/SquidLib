142    public long doFNV()
143    {
144        final LongPeriodRNG rng = new LongPeriodRNG(seed);
145
146        for (int i = 0; i < 1000000; i++) {
147            rng.nextLong();
148            seed += CrossHash.hash64(rng.state);
149        }
150        return seed;
151    }

577    /*
578     * ============================== HOW TO RUN THIS TEST: ====================================
579     *
580     * You are expected to see the different run modes for the same benchmark.
581     * Note the units are different, scores are consistent with each other.
582     *
583     * You can run this test:
584     *
585     * a) Via the command line from the squidlib-performance module's root folder:
586     *    $ mvn clean install
587     *    $ java -jar target/benchmarks.jar HashBenchmark -wi 8 -i 8 -f 1 -gc true
588     *
589     *    (we requested 8 warmup/measurement iterations, single fork)
590     *
591     * b) Via the Java API:
592     *    (see the JMH homepage for possible caveats when running from IDE:
593     *      http://openjdk.java.net/projects/code-tools/jmh/)
594     */
595
596    public static void main(String[] args) throws RunnerException {
597        Options opt = new OptionsBuilder()
598                .include(HashBenchmark.class.getSimpleName())
599                .timeout(TimeValue.seconds(60))
600                .warmupIterations(8)
601                .measurementIterations(8)
602                .forks(1)
603                .build();
604
605        new Runner(opt).run();
606    }
