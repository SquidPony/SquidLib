026/**
027 * A panel to display some text using libgdx directly (i.e. without using
028 * {@link SquidPanel}) as in these examples (no scrolling first, then with a
029 * scroll bar):
030 * 
031 * <p>
032 * <ul>
033 * <li><img src="http://i.imgur.com/EqEXqlu.png"/></li>
034 * <li><img src="http://i.imgur.com/LYbxQZE.png"/></li>
035 * </ul>
036 * </p>
037 * 
038 * <p>
039 * It supports vertical scrolling, i.e. it'll put a vertical scrollbar if
040 * there's too much text to display. This class does a lot of stuff, you
041 * typically only have to provide the textures for the scrollbars and the scroll
042 * knobs (see example below).
043 * </p>
044 * 
045 * <p>
046 * A typical usage of this class is as follows:
047 * 
048 * <pre>
049 * final TextPanel<Color> tp = new TextPanel<>(new GDXMarkup(), font);
050 * tp.init(screenWidth, screenHeight, text); <- first 2 params: for fullscreen
051 * final ScrollPane sp = tp.getScrollPane();
052 * sp.setScrollPaneStyle(new ScrollPaneStyle(...)); <- set textures
053 * stage.addActor(sp);
054 * stage.setKeyboardFocus(sp);
055 * stage.setScrollFocus(sp);
056 * stage.draw();
057 * </pre>
058 * </p>
059 * 
060 * <p>
061 * In addition to what {@link ScrollPane} does (knobs, handling of the wheel);
062 * this class plugs scrolling with arrow keys (up, down, page up, page down) and
063 * vim shortcuts (j/k).
064 * </p>
065 * 
066 * @author smelC
067 * 
068 * @see ScrollPane A libGDX widget for general scrolling through only the visible part of a large widget
069 * @see LinesPanel An alternative for displaying lines of text in a variable-width font
070 */
071public class TextPanel<T extends Color> {

117    /**
118     * The text to display MUST be set later on with
119     * {@link #init(float, float, Collection)}.
120     *
121     * @param markup
122     *            An optional way to compute markup.
123     * @param font
124     *            The font to use. It can be set later using
125     *            {@link #setFont(BitmapFont)}, but it MUST be set before
126     *            drawing this panel.
127     */
128    public TextPanel(/* @Nullable */IMarkup<T> markup, /* @Nullable */ BitmapFont font) {
129        if (markup != null)
130            setMarkup(markup);
131        if (font != null)
132            setFont(font);
133        builder = new StringBuilder(512);
134        textActor = new TextActor();
135
136        this.scrollPane = new ScrollPane(textActor);
137
138        this.scrollPane.addListener(new InputListener() {
139            @Override
140            public boolean keyDown(InputEvent event, int keycode) {
141                                /* To receive key up */
142                return true;
143            }
144
145            @Override
146            public boolean keyUp(InputEvent event, int keycode) {
147                final YMoveKind d = UIUtil.YMoveKind.of(keycode, vimShortcuts);
148                if (d == null)
149                    return false;
150                else {
151                    switch (d) {
152                        case DOWN:
153                        case UP: {
154                            handleArrow(!d.isDown());
155                            return true;
156                        }
157                        case PAGE_DOWN:
158                        case PAGE_UP:
159                            final float scrollY = scrollPane.getScrollY();
160                            final int mult = d.isDown() ? 1 : -1;
161                            scrollPane.setScrollY(scrollY + mult * textActor.getHeight());
162                            return true;
163                    }
164                    throw new IllegalStateException(
165                            "Unmatched " + YMoveKind.class.getSimpleName() + ": " + d);
166                }
167            }
168
169            @Override
170            public boolean keyTyped(InputEvent event, char character) {
171                if (vimShortcuts && (character == 'j' || character == 'k'))
172                    return true;
173                else
174                    return super.keyTyped(event, character);
175            }
176
177            private void handleArrow(boolean up) {
178                final float scrollY = scrollPane.getScrollY();
179                final int mult = up ? -1 : 1;
180                scrollPane.setScrollY(scrollY + (scrollPane.getHeight() * 0.8f * mult));
181            }
182        });
183    }

186     * The text to display MUST be set later on with
187     * {@link #init(float, float, Collection)}.

189     * @param markup
190     *            An optional way to compute markup.
191     * @param distanceFieldFont
192     *            A distance field font as a TextCellFactory to use.
193     *            Won't be used for drawing in cells, just the distance field code it has matters.
194     */
195    public TextPanel(/* @Nullable */IMarkup<T> markup, /* @Nullable */ TextCellFactory distanceFieldFont) {
196        if (markup != null)
197            setMarkup(markup);
198        if (distanceFieldFont != null)
199        {
200            tcf = distanceFieldFont;
201            distanceField = distanceFieldFont.distanceField;
202            tcf.initBySize();
203            font = tcf.font();
204            if (markup != null)
205                font.getData().markupEnabled = true;
206        }
207        builder = new StringBuilder(512);
208        textActor = new TextActor();
209        scrollPane = new ScrollPane(textActor);
210
211        this.scrollPane.addListener(new InputListener() {
212            @Override
213            public boolean keyDown(InputEvent event, int keycode) {
214                                /* To receive key up */
215                return true;
216            }
217
218            @Override
219            public boolean keyUp(InputEvent event, int keycode) {
220                final YMoveKind d = UIUtil.YMoveKind.of(keycode, vimShortcuts);
221                if (d == null)
222                    return false;
223                else {
224                    switch (d) {
225                        case DOWN:
226                        case UP: {
227                            handleArrow(!d.isDown());
228                            return true;
229                        }
230                        case PAGE_DOWN:
231                        case PAGE_UP:
232                            final float scrollY = scrollPane.getScrollY();
233                            final int mult = d.isDown() ? 1 : -1;
234                            scrollPane.setScrollY(scrollY + mult * textActor.getHeight());
235                            return true;
236                    }
237                    throw new IllegalStateException(
238                            "Unmatched " + YMoveKind.class.getSimpleName() + ": " + d);
239                }
240            }
241
242            @Override
243            public boolean keyTyped(InputEvent event, char character) {
244                if (vimShortcuts && (character == 'j' || character == 'k'))
245                    return true;
246                else
247                    return super.keyTyped(event, character);
248            }
249
250            private void handleArrow(boolean up) {
251                final float scrollY = scrollPane.getScrollY();
252                final int mult = up ? -1 : 1;
253                scrollPane.setScrollY(scrollY + (scrollPane.getHeight() * 0.8f * mult));
254            }
255        });
256    }

258    /**
259         * @param m
260         *            The markup to use.

263                if (font != null)
264                        font.getData().markupEnabled |= true;

268        /**
269         * Sets the font to use. This method should be called once before
270         * {@link #init(float, float, Collection)} if the font wasn't given at
271         * creation-time.
272         * 
273         * @param font
274         *            The font to use.
275         */
276        public void setFont(BitmapFont font) {
277                this.font = font;
278        tcf = new TextCellFactory().font(font).height(MathUtils.ceil(font.getLineHeight()))
279                .width(MathUtils.round(font.getSpaceWidth()));
280                if (markup != null)
281                        font.getData().markupEnabled |= true;
282        }

284        /**
285         * This method sets the sizes of {@link #scrollPane} and {@link #textActor}.
286         * This method MUST be called before rendering.
287         * 
288         * @param maxHeight
289         *            The maximum height that the scrollpane can take (equal or
290         *            smaller than the height of the text actor).
291         * @param width
292         *            The width of the scrollpane and the text actor.
293         * @param text
294         */
295        public void init(float width, float maxHeight, Collection<? extends IColoredString<T>> text) {
296        this.text = new ArrayList<>(text);
297
298        scrollPane.setWidth(width);
299        textActor.setWidth(width);
300
301        if (tcf == null)
302            throw new NullPointerException(
303                    "The font should be set before calling " + TextPanel.class.getSimpleName() + "::init");
304
305        final BitmapFontCache cache = font.getCache();
306        final List<String> toDisplay = getTypesetText();
307        float totalTextHeight = tcf.height();
308        GlyphLayout layout = cache.addText(builder, 0, 0, 0, builder.length(), width, Align.left, true);
309        totalTextHeight += layout.height;
310        if(totalTextHeight < 0)
311            totalTextHeight = 0;
312                textActor.setHeight(/* Entire height */ totalTextHeight);
313                final boolean yscroll = maxHeight < totalTextHeight;
314                scrollPane.setHeight(/* Maybe not the entire height */ Math.min(totalTextHeight, maxHeight));
315        scrollPane.setWidget(new TextActor());
316                yScrollingCallback(yscroll);
317        }
318
319    public void init(float width, float maxHeight, T color, String... text)
320    {
321        ArrayList<IColoredString.Impl<T>> coll = new ArrayList<>(text.length);
322        for(String t : text)
323        {
324            coll.add(new IColoredString.Impl<T>(t, color));
325        }
326        init(width, maxHeight, coll);
327    }

329        /**
330         * Draws the border. You have to call this method manually, because the
331         * border is outside the actor and hence should be drawn at the very end,
332         * otherwise it can get overwritten by UI element.
333         * 
334         * @param batch
335         */
336        public void drawBorder(Batch batch) {
337                if (borderColor != null && 0 < borderSize) {
338                        final boolean reset = batch.isDrawing();
339                        if (reset)
340                                batch.end();
341
342                        final ShapeRenderer sr = getRenderer();
343                        final Matrix4 m = batch.getTransformMatrix();
344                        sr.setTransformMatrix(m);
345                        sr.begin(ShapeType.Filled);
346                        sr.setColor(borderColor);
347                        UIUtil.drawMarginsAround(sr, scrollPane.getX(), scrollPane.getY(), scrollPane.getWidth(),
348                                        scrollPane.getHeight() - 1, borderSize, borderColor, borderStyle, 1f, 1f);
349                        sr.end();
350
351                        if (reset)
352                                batch.begin();
353                }
354        }

356        /**
357         * @return The text to draw, after applying {@link #present(IColoredString)}
358         *         and {@link #applyMarkup(IColoredString)}.
359         */
360        public /* @Nullable */ List<String> getTypesetText() {
361                if (text == null)
362                        return null;
363        builder.delete(0, builder.length());
364                final List<String> result = new ArrayList<>();
365                for (IColoredString<T> line : text) {
366                        /* This code must be consistent with #draw in the custom Actor */
367                        final IColoredString<T> tmp = present(line);
368            final String marked = applyMarkup(tmp);
369                        result.add(marked);
370            builder.append(marked);
371            builder.append('\n');
372                }
373        if(builder.length() > 0)
374            builder.deleteCharAt(builder.length() - 1);
375                return result;
376        }

405        /**
406         * Callback done to do stuff according to whether y-scrolling is required
407         * 
408         * @param required
409         *            Whether y scrolling is required.
410         */
411        protected void yScrollingCallback(boolean required) {
412                if (required) {
413                        /* Disable borders, they don't mix well with scrollbars */
414                        borderSize = 0;
415                        scrollPane.setFadeScrollBars(false);
416                        scrollPane.setForceScroll(false, true);
417                }
418        }

421         * @param ics
422         *            Text set when building {@code this}
423         * @return The text to display to screen. If you wanna
424         *         {@link squidpony.IColorCenter#filter(IColoredString) filter} your
425         *         text , do it here.
426         */
427        protected IColoredString<T> present(IColoredString<T> ics) {
428                return ics;
429        }

431        /**
432         * @param ics
433         * @return The text obtained after applying {@link #markup}.

435        protected String applyMarkup(IColoredString<T> ics) {
436                return markup == null ? ics.toString() : ics.presentWithMarkup(markup);
437        }

489            }
490
491            if (font == null)
492                throw new NullPointerException(
493                        "The font should be set when drawing a " + getClass().getSimpleName());
494            if (text == null)
495                throw new NullPointerException(
496                        "The text should be set when drawing a " + getClass().getSimpleName());
497            if (tcf != null) {
498                tcf.configureShader(batch);
499            }
500            float yscroll = scrollPane.getScrollY();
501
502            final float destx = tx, offY = (tcf != null) ? tcf.height * 0.5f : 0;
503            getTypesetText();
504            font.draw(batch, builder, destx, theight + yscroll - offY,
505                    0, builder.length(), twidth, Align.left, true);

507        }
