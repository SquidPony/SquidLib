026        public IdentityFilter()

037    /**
038     * A Filter that converts all colors passed to it to grayscale, like a black and white film.
039     */
040    public static class GrayscaleFilter implements IFilter<Color>
041    {
042        public GrayscaleFilter()
043        {
044                /* Nothing to do */
045        }
046
047        @Override
048        public Color alter(float r, float g, float b, float a) {
049            float v = (r + g + b) / 3f;
050            return new Color(v, v, v, a);
051        }

055    /**
056     * A Filter that tracks the highest brightness for any component it was assigned and stores it in state as the first
057     * and only element.
058     */
059    public static class MaxValueFilter extends Filter<Color>
060    {
061        public MaxValueFilter()
062        {
063            state = new float[]{0};
064        }
065
066        @Override
067        public Color alter(float r, float g, float b, float a) {
068            state[0] = Math.max(state[0], Math.max(r, Math.max(g, b)));
069            return new Color(r, g, b, a);
070        }
071

074    /**
075     * A Filter that performs a brightness adjustment to make dark areas lighter and light areas not much less bright.
076     */
077    public static class GammaCorrectFilter extends Filter<Color> {
078        /**
079         * Sets up a GammaCorrectFilter with the desired gamma adjustment.
080         *
081         * @param gamma    should be 1.0 or less, and must be greater than 0. Typical values are between 0.4 to 0.8.
082         * @param maxValue the maximum brightness in the colors this will be passed; use MaxValueFilter for this
083         */
084        public GammaCorrectFilter(float gamma, float maxValue) {
085            state = new float[]{gamma, 1f / (float) Math.pow(maxValue, gamma)};
086        }
087
088        @Override
089        public Color alter(float r, float g, float b, float a) {
090            return new Color(state[1] * (float) Math.pow(r, state[0]),
091                    state[1] * (float) Math.pow(g, state[0]),
092                    state[1] * (float) Math.pow(b, state[0]),
093                    a);
094        }
095    }
096    /**
097     * A Filter that is constructed with a color and linear-interpolates any color it is told to alter toward the color
098     * it was constructed with.
099     */
100    public static class LerpFilter extends Filter<Color> {
101        /**
102         * Sets up a LerpFilter with the desired color to linearly interpolate towards.
103         *
104         * @param r the red component to lerp towards
105         * @param g the green component to lerp towards
106         * @param b the blue component to lerp towards
107         * @param a the opacity component to lerp towards
108         * @param amount the amount to lerp by, should be between 0.0 and 1.0
109         */
110        public LerpFilter(float r, float g, float b, float a, float amount) {
111            state = new float[]{r, g, b, a, amount};
112        }
113        /**
114         * Sets up a LerpFilter with the desired color to linearly interpolate towards.
115         *
116         * @param color the Color to lerp towards
117         * @param amount the amount to lerp by, should be between 0.0 and 1.0
118         */
119        public LerpFilter(Color color, float amount) {
120            state = new float[]{color.r, color.g, color.b, color.a, amount};
121        }
122
123        @Override
124        public Color alter(float r, float g, float b, float a) {
125            return new Color(r, g, b, a).lerp(state[0], state[1], state[2], state[3], state[4]);
126        }

129    /**
130     * A Filter that is constructed with a group of colors and linear-interpolates any color it is told to alter toward
131     * the color it was constructed with that has the closest hue.
132     */
133    public static class MultiLerpFilter extends Filter<Color> {
134        private SquidColorCenter globalSCC;
135        /**
136         * Sets up a MultiLerpFilter with the desired colors to linearly interpolate towards; the lengths of each given
137         * array should be identical.
138         *
139         * @param r the red components to lerp towards
140         * @param g the green components to lerp towards
141         * @param b the blue components to lerp towards
142         * @param a the opacity components to lerp towards
143         * @param amount the amounts to lerp by, should each be between 0.0 and 1.0
144         */
145        public MultiLerpFilter(float[] r, float[] g, float[] b, float[] a, float[] amount) {
146            state = new float[Math.min(r.length, Math.min(g.length, Math.min(b.length,
147                    Math.min(a.length, amount.length)))) * 6];
148            globalSCC = DefaultResources.getSCC();
149            for (int i = 0; i < state.length / 6; i++) {
150                state[i * 6] = r[i];
151                state[i * 6 + 1] = g[i];
152                state[i * 6 + 2] = b[i];
153                state[i * 6 + 3] = a[i];
154                state[i * 6 + 4] = amount[i];
155                state[i * 6 + 5] = globalSCC.getHue(r[i], g[i], b[i]);
156            }
157        }/**
158         * Sets up a MultiLerpFilter with the desired colors to linearly interpolate towards and their amounts.
159         *
160         * @param colors the Colors to lerp towards
161         * @param amount the amounts to lerp by, should each be between 0.0 and 1.0
162         */
163        public MultiLerpFilter(Color[] colors, float[] amount) {
164            state = new float[Math.min(colors.length, amount.length) * 6];
165            globalSCC = DefaultResources.getSCC();
166            for (int i = 0; i < state.length / 6; i++) {
167                state[i * 6] = colors[i].r;
168                state[i * 6 + 1] = colors[i].g;
169                state[i * 6 + 2] = colors[i].b;
170                state[i * 6 + 3] = colors[i].a;
171                state[i * 6 + 4] = amount[i];
172                state[i * 6 + 5] = globalSCC.getHue(colors[i]);
173            }
174        }
175
176        @Override
177        public Color alter(float r, float g, float b, float a) {
178            float givenH = globalSCC.getHue(r, g, b), givenS = globalSCC.getSaturation(r, g, b),
179                    minDiff = 999.0f, temp;
180            if(givenS < 0.05)
181                return new Color(r, g, b, a);
182            int choice = 0;
183            for (int i = 5; i < state.length; i += 6) {
184                temp = state[i] - givenH;
185                temp = (temp >= 0.5f) ? Math.abs(temp - 1f) % 1f : Math.abs(temp);
186                if(temp < minDiff) {
187                    minDiff = temp;
188                    choice = i / 6; // rounds down
189                }
190            }
191            return new Color(r, g, b, a).lerp(state[choice * 6], state[choice * 6 + 1], state[choice * 6 + 2],
192                    state[choice * 6 + 3], state[choice * 6 + 4]);
193        }
194    }

196    /**
197     * A Filter that is constructed with a color and makes any color it is told to alter have the same hue as the given
198     * color, have saturation that is somewhere between the given color's and the altered colors, and chiefly is
199     * distinguishable from other colors by value. Useful for sepia effects, which can be created satisfactorily with
200     * {@code new Filters.ColorizeFilter(SColor.CLOVE_BROWN, 0.6f, 0.0f)}.
201     */
202    public static class ColorizeFilter extends Filter<Color> {
203        private SquidColorCenter globalSCC;
204        /**
205         * Sets up a ColorizeFilter with the desired color to colorize towards.
206         *
207         * @param r the red component to colorize towards
208         * @param g the green component to colorize towards
209         * @param b the blue component to colorize towards
210         */
211        public ColorizeFilter(float r, float g, float b) {
212            globalSCC = DefaultResources.getSCC();
213
214            state = new float[]{globalSCC.getHue(r, g, b), globalSCC.getSaturation(r, g, b), 1f, 0f};
215        }
216        /**
217         * Sets up a ColorizeFilter with the desired color to colorize towards.
218         *
219         * @param color the Color to colorize towards
220         */
221        public ColorizeFilter(Color color) {
222            globalSCC = DefaultResources.getSCC();
223            state = new float[]{globalSCC.getHue(color), globalSCC.getSaturation(color), 1f, 0f};
224        }
225        /**
226         * Sets up a ColorizeFilter with the desired color to colorize towards.
227         *
228         * @param r the red component to colorize towards
229         * @param g the green component to colorize towards
230         * @param b the blue component to colorize towards
231         * @param saturationMultiplier a multiplier to apply to the final color's saturation; may be greater than 1
232         * @param valueModifier a modifier that affects the final brightness value of any color this alters;
233         *                      typically very small, such as in the -0.2f to 0.2f range
234         */
235        public ColorizeFilter(float r, float g, float b, float saturationMultiplier, float valueModifier) {
236            globalSCC = DefaultResources.getSCC();
237
238            state = new float[]{
239                    globalSCC.getHue(r, g, b),
240                    globalSCC.getSaturation(r, g, b),
241                    saturationMultiplier,
242                    valueModifier};
243        }
244        /**
245         * Sets up a ColorizeFilter with the desired color to colorize towards.
246         *
247         * @param color the Color to colorize towards
248         * @param saturationMultiplier a multiplier to apply to the final color's saturation; may be greater than 1
249         * @param valueModifier a modifier that affects the final brightness value of any color this alters;
250         *                      typically very small, such as in the -0.2f to 0.2f range
251         */
252        public ColorizeFilter(Color color, float saturationMultiplier, float valueModifier) {
253            globalSCC = DefaultResources.getSCC();
254            state = new float[]{
255                    globalSCC.getHue(color),
256                    globalSCC.getSaturation(color),
257                    saturationMultiplier,
258                    valueModifier};
259        }
260
261        @Override
262        public Color alter(float r, float g, float b, float a) {
263            return globalSCC.getHSV(
264                    state[0],
265                    Math.max(0f, Math.min((globalSCC.getSaturation(r, g, b) + state[1]) * 0.5f * state[2], 1f)),
266                    globalSCC.getValue(r, g, b) * (1f - state[3]) + state[3],
267                    a);
268        }
269    }
270    /**
271     * A Filter that makes the colors requested from it highly saturated, with the original hue, value and a timer that
272     * increments very slowly altering hue, with hue, value and the timer altering saturation, and the original hue,
273     * saturation, and value all altering value. It should look like a hallucination.
274     * <br>
275     * A short (poorly recorded) video can be seen here http://i.imgur.com/SEw2LXe.gifv ; performance should be smoother
276     * during actual gameplay.
277     */
278    public static class HallucinateFilter extends Filter<Color> {
279        private SquidColorCenter globalSCC;
280        /**
281         * Sets up a HallucinateFilter with the timer at 0..
282         */
283        public HallucinateFilter() {
284            globalSCC = DefaultResources.getSCC();
285
286            state = new float[]{0f};
287        }
288        @Override
289        public Color alter(float r, float g, float b, float a) {
290            state[0] += 0.00003f;
291            if(state[0] >= 1.0f)
292                state[0] = 0f;
293            float h = globalSCC.getHue(r, g, b),
294                    s = globalSCC.getSaturation(r, g, b),
295                    v = globalSCC.getValue(r, g, b);
296            return globalSCC.getHSV(
297                    (v * 4f + h + state[0]) % 1.0f,
298                    Math.max(0f, Math.min((h + v) * 0.65f + state[0] * 0.4f, 1f)),
299                    (h + v + s) * 0.35f + 0.7f,
300                    a);
301        }
302
303    }

306    /**
307     * A Filter that multiplies the saturation of any color requested from it by a number given during construction.
308     */
309    public static class SaturationFilter extends Filter<Color> {
310        private SquidColorCenter globalSCC;
311        /**
312         * Sets up a SaturationFilter with the desired saturation multiplier. Using a multiplier of 0f, as you would
313         * expect, makes the image grayscale. Using a multiplier of 0.5 make the image "muted", with no truly bright
314         * colors, while 1.0f makes no change, and and any numbers higher than 1.0f will make the image more saturated,
315         * with the exception of colors that were already grayscale or close to it. This clamps the result, so there's
316         * no need to worry about using too high of a saturation multiplier.
317         *
318         * @param multiplier the amount to multiply each requested color's saturation by; 1.0f means "no change"
319         */
320        public SaturationFilter(float multiplier) {
321            globalSCC = DefaultResources.getSCC();
322
323            state = new float[]{multiplier};
324        }
325
326        @Override
327        public Color alter(float r, float g, float b, float a) {
328            return globalSCC.getHSV(
329                    globalSCC.getHue(r, g, b),
330                    Math.max(0f, Math.min((globalSCC.getSaturation(r, g, b) * state[0]), 1f)),
331                    globalSCC.getValue(r, g, b),
332                    a);
333        }
334
335    }

356    /**
357     * A Filter that is constructed with a group of colors and forces any color it is told to alter to exactly
358     * the color it was constructed with that has the closest red, green, and blue components. A convenient way to
359     * use this is to pass in one of the color series from SColor, such as RED_SERIES or ACHROMATIC_SERIES.
360     *
361     * Preview using BLUE_GREEN_SERIES foreground, ACHROMATIC_SERIES background: http://i.imgur.com/2HdZpC9.png
362     */
363    public static class PaletteFilter extends Filter<Color> {
364        /**
365         * Sets up a PaletteFilter with the exact colors to use as individual components; the lengths of each given
366         * array should be identical.
367         *
368         * @param r the red components to use
369         * @param g the green components to use
370         * @param b the blue components to use
371         * @param a the opacity components to use
372         */
373        public PaletteFilter(float[] r, float[] g, float[] b, float[] a) {
374            state = new float[Math.min(r.length, Math.min(g.length, Math.min(b.length,
375                    a.length))) * 4];
376            for (int i = 0; i < state.length / 4; i++) {
377                state[i * 4] = MathUtils.clamp(r[i], 0f, 1f);
378                state[i * 4 + 1] = MathUtils.clamp(g[i], 0f, 1f);
379                state[i * 4 + 2] = MathUtils.clamp(b[i], 0f, 1f);
380                state[i * 4 + 3] = MathUtils.clamp(a[i], 0f, 1f);
381            }
382        }/**
383         * Sets up a PaletteFilter with the exact colors to use as Colors. A convenient way to
384         * use this is to pass in one of the color series from SColor, such as RED_SERIES or ACHROMATIC_SERIES.
385         *
386         * @param colors the Colors to use
387         */
388        public PaletteFilter(Color[] colors) {
389            state = new float[colors.length * 4];
390            for (int i = 0; i < colors.length; i++) {
391                state[i * 4] = colors[i].r;
392                state[i * 4 + 1] = colors[i].g;
393                state[i * 4 + 2] = colors[i].b;
394                state[i * 4 + 3] = colors[i].a;
395            }
396        }
397
398        @Override
399        public Color alter(float r, float g, float b, float a) {
400            float diff = 9999.0f, temp;
401            int choice = 0;
402            for (int i = 0; i < state.length; i += 4) {
403                temp = Math.abs(state[i] - r) + Math.abs(state[i + 1] - g) + Math.abs(state[i + 2] - b);
404                if(temp < diff) {
405                    diff = temp;
406                    choice = i;
407                }
408            }
409            return new Color(state[choice], state[choice + 1], state[choice + 2],
410                    a);
411        }
412    }
413    /**
414     * A Filter that alters primarily-red and primarily-green colors so they can be more easily be distinguished by
415     * people with at least some forms of red-green color-blindness (deuteranopia should be handled well, protanopia
416     * very well, and tritanopia may not benefit at all). Causes reds to be darkened and greens to be lightened if the
417     * other of the pair is not present in similar quantities (which is the case for yellows and blues).
418     */
419    public static class DistinctRedGreenFilter implements IFilter<Color> {
420        /**
421         * Constructs a DistinctRedGreenFilter. This class is a simple wrapper around a function that doesn't need
422         * member variables, so there should be little overhead with this filter.
423         */
424        public DistinctRedGreenFilter() {
425        }
426
427        @Override
428        public Color alter(float r, float g, float b, float a) {
429            float diff = g - r;
430            if(diff > 0.4f)
431                return new Color(Math.min(1f, r * (0.8f + diff * 0.5f)), Math.min(1f, g * (0.9f + diff * 0.5f)),
432                        Math.min(1f, b * (0.8f + diff * 0.5f)), a);
433            else if(diff < -0.3f)
434                return new Color(r * (0.6f - diff), g * (0.7f - diff),
435                        b * (0.7f - diff), a);
436            else
437                return new Color(r, g, b, a);
438        }
