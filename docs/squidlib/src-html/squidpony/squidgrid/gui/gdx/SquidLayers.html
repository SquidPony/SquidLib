450    /**
451     * Add an extra layer on top of the foreground layer. Use putInto methods to specify the layer when adding a char (0
452     * is background, 1 is unused, 2 is foreground, and the first call to this method creates layer 3).
453     *
454     * @return this for chaining
455     */

457        SquidPanel sp;
458        if(width != foregroundPanel.getTotalWidth() || height != foregroundPanel.getTotalHeight())
459            sp = new SquidPanel(width, height, textFactory, foregroundPanel.getColorCenter(), 0, 0,
460                    ArrayTools.fill(' ', foregroundPanel.getTotalWidth(), foregroundPanel.getTotalHeight()));
461        else
462            sp = new SquidPanel(width, height, textFactory);
463        addActor(sp);
464        extraPanels.add(sp);
465        return this;
466    }

499    /**
500     * Sets the size of the text in the given layer  (but not the size of the cells) to the given width and height in
501     * pixels (which may be stretched by viewports later on, if your program uses them).
502     * @param layer the layer to affect; 0 is background, 1 is unused, 2 is foreground, 3 and higher are extra panels
503     * @param wide the width of a glyph in pixels
504     * @param high the height of a glyph in pixels
505     * @return this for chaining
506     */
507    public SquidLayers setTextSize(int layer, float wide, float high)
508    {
509        SquidPanel p = backgroundPanel;
510        switch (layer) {
511            case 0:
512                break;
513            case 1:
514                break;
515            case 2:
516                p = foregroundPanel;
517                break;
518            default:
519                p = extraPanels.get(layer - 3);
520        }
521        p.setTextSize(wide, high);
522        return this;
523    }

525    /**
526     * Sets the size of the text in all layers (but not the size of the cells) to the given width and height in pixels
527     * (which may be stretched by viewports later on, if your program uses them).
528     * @param wide the width of a glyph in pixels
529     * @param high the height of a glyph in pixels
530     * @return this for chaining
531     */
532    public SquidLayers setTextSize(float wide, float high)
533    {
534        textFactory.tweakHeight(high).tweakWidth(wide).initBySize();
535        setTextSize(0, wide, high);
536        setTextSize(2, wide, high);
537        for (int i = 0; i < extraPanels.size(); i++) {
538            setTextSize(i + 3, wide, high);
539        }
540        return this;
541    }
