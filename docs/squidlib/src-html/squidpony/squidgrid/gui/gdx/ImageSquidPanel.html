016/**
017 * Displays text and images in a grid pattern, like SquidPanel, but will automatically render certain chars as images.
018 * Supports basic animations, such as sliding, wiggling, or fading the contents of a cell.
019 * <br>
020 * Grid width and height settings are in terms of number of cells. Cell width and height are in terms of number of
021 * pixels (when there is no stretching taking place due to viewport or window size).
022 * <br>
023 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
024 * @author Tommy Ettinger
025 */
026public class ImageSquidPanel extends SquidPanel {

123    @Override
124    public void draw (Batch batch, float parentAlpha) {
125        textFactory.configureShader(batch);
126        int inc = onlyRenderEven ? 2 : 1, widthInc = inc * cellWidth, heightInc = inc * cellHeight;
127        TextureRegion tr;
128        float screenX = xOffset - (gridOffsetX <= 0 ? 0 : cellWidth)/* - getX()*/,
129                screenY_base = 1f + yOffset + (gridOffsetY <= 0 ? 0 : cellHeight) + gridHeight * cellHeight, screenY;
130        char c;
131        for (int x = Math.max(0, gridOffsetX-1), xx = (gridOffsetX <= 0) ? 0 : -1; xx <= gridWidth && x < contents.length; x += inc, xx += inc, screenX += widthInc) {
132            screenY = screenY_base;
133            for (int y = Math.max(0, gridOffsetY-1), yy = (gridOffsetY <= 0) ? 0 : -1; yy <= gridHeight && y < contents[x].length; y += inc, yy += inc, screenY -= heightInc) {
134                c = contents[x][y];
135                tr = imageMap.get(c);
136                if(tr == null)
137                    textFactory.draw(batch, c, colors[x][y], screenX, screenY);
138                else
139                    textFactory.draw(batch, tr, colors[x][y], screenX, screenY);
140            }
141        }
142        super.draw(batch, parentAlpha);
143        int len = animatedEntities.size();
144        for (int i = 0; i < len; i++) {
145
146            animatedEntities.getAt(i).actor.act(Gdx.graphics.getDeltaTime());
147        }
148        len = autoActors.size();
149        Actor a;
150        for (int i = 0; i < len; i++) {
151            a = autoActors.getAt(i);
152            if(a == null) continue;
153            drawActor(batch, parentAlpha, a);
154            a.act(Gdx.graphics.getDeltaTime());
155        }
156    }

169    /**
170     * Draws one AnimatedEntity, specifically the Actor it contains. Batch must be between start() and end()
171     * @param batch Must have start() called already but not stop() yet during this frame.
172     * @param parentAlpha This can be assumed to be 1.0f if you don't know it
173     * @param ac The Actor to draw; the position to draw ac is modified and reset based on some fields of this object
174     */
175    public void drawActor(Batch batch, float parentAlpha, Actor ac)
176    {
177        float prevX = ac.getX(), prevY = ac.getY();
178        ac.setPosition(prevX - gridOffsetX * cellWidth, prevY + gridOffsetY * cellHeight);
179        ac.draw(batch, parentAlpha);
180        ac.setPosition(prevX, prevY);
181    }

766    /**
767     * Gets a "snapshot" of the data represented by this ImageSquidPanel; stores the dimensions, the char data, and the
768     * color data in a way that can be set back to a SquidPanel or ImageSquidPanel using
769     * {@link #setFromSnapshot(String, int, int, int, int)} or its overload that takes a StringBuilder. The actual
770     * contents of the returned StringBuilder are unlikely to be legible in any way if read as text, and are meant to be
771     * concise and stable across versions. This overload allows the first x and y position used to be specified, as well
772     * as the width and height to use (the actual width and height stored may be different if this SquidPanel's
773     * gridWidth and/or gridHeight are smaller than the width and/or height given).
774     * <br>
775     * NOTE: For this version, the mapping of chars to images is not stored in the snapshot, allowing alternate mappings
776     * to be used, such as while graphics are being updated frequently. This also allows the snapshot to be read in from
777     * both normal SquidPanels and ImageSquidPanels.
778     * @param startX the first x position to use in the snapshot, inclusive
779     * @param startY the first y position to use in the snapshot, inclusive
780     * @param width how wide the snapshot area should be; x positions from startX to startX + width - 1 will be used
781     * @param height how tall the snapshot area should be; y positions from startY to startY + height - 1 will be used
782     * @return a StringBuilder representation of this SquidPanel's data that can be passed later to {@link #setFromSnapshot(StringBuilder, int, int, int, int)} or converted to String and passed to its overload
783     */
784    public StringBuilder getSnapshot(int startX, int startY, int width, int height) {
785        width = Math.min(gridWidth - startX, width);
786        height = Math.min(gridHeight - startY, height);
787        StringBuilder sb = new StringBuilder(width * height * 9 + 12);
788        sb.append(width).append('x').append(height).append(':');
789        for (int x = startX, i = 0; i < width; x++, i++) {
790            sb.append(contents[x], startY, height);
791        }
792        char[] reuse = new char[8];
793        for (int x = startX, i = 0; i < width; x++, i++) {
794            for (int y = startY, j = 0; j < height; y++, j++) {
795                sb.append(SColor.floatToChars(reuse, colors[x][y]));
796            }
797        }
798        return sb;
799    }

825     * Given a "snapshot" from {@link #getSnapshot(int, int, int, int)}, this assigns the chars and colors in this
826     * ImageSquidPanel from the position given by putX,putY (inclusive) to putX+limitWidth,putY+limitHeight (exclusive)
827     * so they have the values stored in the snapshot. If limitWidth or limitHeight is negative, this uses the full
828     * width and height of the snapshot (stopping early if it would extend past the gridWidth or gridHeight of this
829     * ImageSquidPanel).
830     * @param snapshot a StringBuilder in a special format as produced by {@link #getSnapshot(int, int, int, int)}
831     * @param putX where to start placing the data from the snapshot, x position
832     * @param putY where to start placing the data from the snapshot, y position
833     * @param limitWidth if negative, uses all of snapshot's width as possible, otherwise restricts the width allowed
834     * @param limitHeight if negative, uses all of snapshot's height as possible, otherwise restricts the height allowed
835     * @return this after setting, for chaining
836     */
837    public ImageSquidPanel setFromSnapshot(StringBuilder snapshot, int putX, int putY, int limitWidth, int limitHeight)
838    {
839        if(putX >= gridWidth || putY >= gridHeight || snapshot == null || snapshot.length() < 4) return this;
840        if(putX < 0) putX = 0;
841        if(putY < 0) putY = 0;
842        int start = snapshot.indexOf(":")+1, width = StringKit.intFromDec(snapshot),
843                height = StringKit.intFromDec(snapshot, snapshot.indexOf("x") + 1, start),
844                run = start;
845        if(limitWidth < 0)
846            limitWidth = Math.min(width, gridWidth - putX);
847        else
848            limitWidth = Math.min(limitWidth, Math.min(width, gridWidth - putX));
849
850        if(limitHeight < 0)
851            limitHeight = Math.min(height, gridHeight - putY);
852        else
853            limitHeight = Math.min(limitHeight, Math.min(height, gridHeight - putY));
854        for (int x = putX, i = 0; i < limitWidth; x++, i++, run += height) {
855            snapshot.getChars(run, run + limitHeight, contents[x], putY);
856        }
857        run = start + width * height;
858        for (int x = putX, i = 0; i < limitWidth; x++, i++) {
859            for (int y = putY, j = 0; j < limitHeight; y++, j++) {
860                colors[x][y] = SColor.charsToFloat(snapshot, run);
861                run += 8;
862            }
863        }
864        return this;

975    /**
976     * Meant for taking easy-to-write chars and generating chars that can map to images, while still allowing the
977     * original easy-to-write char to be used as its own non-image char. Given a char that should be from the
978     * "relatively common" parts of Unicode (essentially, anything with a codepoint before 0x1000, or with a small risk
979     * of collision, anything before 0x8000), this returns a character from the "Private Use Area" of Unicode that will
980     * probably be unique for most kinds of input. 4096 possible chars can be returned by this method.
981     *
982     * As an example, you may want to map the char 'g' to an image of a goblin, but still sometimes show the actual 'g'
983     * for some other monster or a variant on goblins you don't have an image for. You could call
984     * {@code char goblinChar = ImageSquidPanel.getUnusedChar('g');} and then register goblinChar in the image mapping
985     * with {@code myImageSquidPanel.setImageSwap(goblinChar, goblinImage);}, which would allow you to draw 'g' directly
986     * by placing a 'g' with put(), or draw goblinImage by placing either goblinChar or
987     * {@code ImageSquidPanel.getUnusedChar('g')} with put().
988     *
989     * @param initial a char that ideally should be from the earlier parts of Unicode (before 0x1000 is ideal)
990     * @return a char from Unicode's private use area that is unlikely to be found otherwise or accidentally
991     */
992    public static char getUnusedChar(final char initial)
993    {
994        return (char)(((initial & 0x7000) >>> 3 ^ initial) & 0x0FFF | 0xE000);
995    }
