012/**

041    private final Color tmpColor = new Color();
042    @Override
043    protected Color create(int red, int green, int blue, int opacity) {
044        if (filter == null)
045                        /* No filtering */
046            return new Color(red / 255f, green / 255f, blue / 255f, opacity / 255f);
047        else
048                        /* Some filtering */
049            return filter.alter(red / 255f, green / 255f, blue / 255f, opacity / 255f);
050    }
051    @Override
052    public Color filter(Color c)
053    {
054        if(c == null)
055            return Color.CLEAR;
056        else
057            return super.filter(c);
058    }
059
060    public Color get(long c)
061    {
062        return get((int)((c >> 24) & 0xff), (int)((c >> 16) & 0xff), (int)((c >> 8) & 0xff), (int)(c & 0xff));
063    }
064    public Color get(float r, float g, float b, float a)
065    {
066        return get(Math.round(255 * r), Math.round(255 * g), Math.round(255 * b), Math.round(255 * a));
067    }

069    /**
070     * Gets the linear interpolation from Color start to Color end, changing by the fraction given by change.
071     * @param start the initial Color
072     * @param end the "target" color
073     * @param change the degree to change closer to end; a change of 0.0f produces start, 1.0f produces end
074     * @return a new Color
075     */
076    @Override
077    public Color lerp(Color start, Color end, float change)
078    {
079        if(start == null || end == null)
080            return Color.CLEAR;
081        return get(
082                start.r + change * (end.r - start.r),
083                start.g + change * (end.g - start.g),
084                start.b + change * (end.b - start.b),
085                start.a + change * (end.a - start.a)
086        );
087    }
088        @Override
089        public int getRed(Color c) {
090                return Math.round(c.r * 255f);
091        }
092
093        @Override
094        public int getGreen(Color c) {
095                return Math.round(c.g * 255f);
096        }
097
098        @Override
099        public int getBlue(Color c) {
100                return Math.round(c.b * 255f);
101        }
102
103        @Override
104        public int getAlpha(Color c) {
105                return Math.round(c.a * 255f);
106        }

108    public static int encode (Color color) {
109        if (color == null)
110            return 0;
111        return (Math.round(color.r * 255.0f) << 24)
112                | (Math.round(color.g * 255.0f) << 16)
113                | (Math.round(color.b * 255.0f) << 8)
114                | Math.round(color.a * 255.0f);
115    }

118     * Gets a modified copy of color as if it is lit with a colored light source.
119     * @param color the color to shine the light on
120     * @param light the color of the light source
121     * @return a copy of the Color color that factors in the lighting of the Color light.

123    public Color lightWith(Color color, Color light)
124    {
125        return filter(color.cpy().mul(light));

206     * Gets a fully-desaturated version of the given color (keeping its brightness, but making it grayscale).

345    /**
346     * Finds a gradient with the specified number of steps going from fromColor to toColor, both included in the
347     * gradient. The interpolation argument can be used to make the color stay close to fromColor and/or toColor longer
348     * than it would normally, or shorter if the middle colors are desirable.
349     * @param fromColor the color to start with, included in the gradient
350     * @param toColor the color to end on, included in the gradient
351     * @param steps the number of elements to use in the gradient
352     * @param interpolation a libGDX Interpolation that defines how quickly the color changes during the transition
353     * @return an ArrayList composed of the blending steps from fromColor to toColor, with length equal to steps
354     */
355    public ArrayList<Color> gradient(Color fromColor, Color toColor, int steps, Interpolation interpolation)
356    {
357        ArrayList<Color> colors = new ArrayList<>((steps > 1) ? steps : 1);
358        colors.add(filter(fromColor));
359        if(steps < 2)
360            return colors;
361        for (float i = 1; i < steps; i++) {
362            colors.add(lerp(fromColor, toColor, interpolation.apply(i / (steps - 1))));
363        }
364        return colors;
365    }

367    /**
368     * Finds a gradient with the specified number of steps going from fromColor to midColor, then midColor to (possibly)
369     * fromColor, with both included in the gradient but fromColor only repeated at the end if the number of steps is
370     * odd. The interpolation argument can be used to make the color linger for a while with colors close to fromColor
371     * or midColor, or to do the opposite and quickly change from one and spend more time in the middle.
372     * @param fromColor the color to start with (and end with, if steps is an odd number), included in the gradient
373     * @param midColor the color to use in the middle of the loop, included in the gradient
374     * @param steps the number of elements to use in the gradient, will be at least 3
375     * @param interpolation a libGDX Interpolation that defines how quickly the color changes at the start and end of
376     *                      each transition, both from fromColor to midColor as well as back to fromColor
377     * @return an ArrayList composed of the blending steps from fromColor to midColor to fromColor again, with length equal to steps
378     */
379    public ArrayList<Color> loopingGradient(Color fromColor, Color midColor, int steps, Interpolation interpolation)
380    {
381        ArrayList<Color> colors = new ArrayList<>((steps > 3) ? steps : 3);
382        colors.add(filter(fromColor));
383        for (float i = 1; i < steps / 2; i++) {
384            colors.add(lerp(fromColor, midColor, interpolation.apply(i / (steps / 2))));
385        }
386        for (float i = 0, c = steps / 2; c < steps; i++, c++) {
387            colors.add(lerp(midColor, fromColor, interpolation.apply(i / (steps / 2))));
388        }
389        return colors;
390    }

393     * Generates a hue-shifted rainbow of colors, starting at red and going through orange, yellow, green, blue, and
394     * purple before getting close to red at the end again. If the given number of steps is less than 6 or so, you should
395     * expect to see only some of those colors; if steps is larger (36 may be reasonable for gradients), you are more
396     * likely to see colors that appear for shorter spans on the color wheel, like orange.
397     * Produces fully saturated and max-brightness colors on the rainbow, which is what many people expect for a rainbow.
398     * @param steps the number of different Color elements to generate in the returned ArrayList
399     * @return an ArrayList of Color where each element goes around the color wheel, starting at red, then orange, etc.
400     */
401    public ArrayList<Color> rainbow(int steps)
402    {
403        return rainbow(1.0f, 1.0f, steps);
404    }

406    /**
407     * Generates a hue-shifted rainbow of colors, starting at red and going through orange, yellow, green, blue, and
408     * purple before getting close to red at the end again. If the given number of steps is less than 6 or so, you should
409     * expect to see only some of those colors; if steps is larger (36 may be reasonable for gradients), you are more
410     * likely to see colors that appear for shorter spans on the color wheel, like orange.
411     * Uses the given saturation and value for all colors in the rainbow, and only changes hue.
412     * @param saturation the saturation of the rainbow's colors; 1.0 is boldest and 0.0 is grayscale
413     * @param value the brightness of the rainbow's colors; 1.0 is brightest
414     * @param steps the number of different Color elements to generate in the returned ArrayList
415     * @return an ArrayList of Color where each element goes around the color wheel, starting at red, then orange, etc.
416     */
417    public ArrayList<Color> rainbow(float saturation, float value, int steps)
418    {
419        steps = (steps > 1) ? steps : 1;
420        ArrayList<Color> colors = new ArrayList<>(steps);
421        for (float i = 0; i < 1f - 0.5f / steps; i+= 1.0f / steps) {
422            colors.add(filter(getHSV(i, saturation, value)));
423        }
424        return colors;
425    }

427    /**
428     * Finds a gradient with the specified number of steps going from fromColor to toColor, both included in the
429     * gradient. This does not typically take a direct path on its way between fromColor and toColor, and is useful to
430     * generate a wide variety of colors that can be confined to a rough amount of maximum difference by choosing values
431     * for fromColor and toColor that are more similar.
432     * <br>
433     * Try using colors for fromColor and toColor that have different r, g, and b values, such as gray and white, then
434     * compare to colors that don't differ on, for example, r, such as bright red and pink. In the first case, red,
435     * green, blue, and many other colors will be generated if there are enough steps; in the second case, red will be
436     * at the same level in all generated colors (very high, so no pure blue or pure green, but purple and yellow are
437     * possible). This should help illustrate how this chooses how far to "zig-zag" off the straight-line path.
438     * @param fromColor the color to start with, included in the gradient
439     * @param toColor the color to end on, included in the gradient
440     * @param steps the number of elements to use in the gradient; ideally no greater than 345 to avoid duplicates
441     * @return an ArrayList composed of the zig-zag steps from fromColor to toColor, with length equal to steps
442     */
443    public ArrayList<Color> zigzagGradient(Color fromColor, Color toColor, int steps)
444    {
445        ArrayList<Color> colors = new ArrayList<>((steps > 1) ? steps : 1);
446        colors.add(filter(fromColor));
447        if(steps < 2)
448            return colors;
449        float dr = toColor.r - fromColor.r, dg = toColor.g - fromColor.g, db = toColor.b - fromColor.b,
450                a = fromColor.a, cr, cg, cb;
451        int decoded;
452        for (float i = 1; i < steps; i++) {
453            // 345 happens to be the distance on our 3D Hilbert curve that corresponds to (7,7,7).
454            decoded = Math.round(345 * (i / (steps - 1)));
455            cr = (CoordPacker.hilbert3X[decoded] / 7f) * dr + fromColor.r;
456            cg = (CoordPacker.hilbert3Y[decoded] / 7f) * dg + fromColor.g;
457            cb = (CoordPacker.hilbert3Z[decoded] / 7f) * db + fromColor.b;
458            colors.add(get(cr, cg, cb, a));
459        }
460        return colors;
461    }
