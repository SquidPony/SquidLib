001package squidpony.squidgrid.gui.gdx;
002
003import com.badlogic.gdx.Gdx;
004import com.badlogic.gdx.graphics.Camera;
005import com.badlogic.gdx.graphics.OrthographicCamera;
006import com.badlogic.gdx.math.MathUtils;
007import com.badlogic.gdx.math.Vector2;
008import com.badlogic.gdx.scenes.scene2d.Actor;
009import com.badlogic.gdx.scenes.scene2d.ui.WidgetGroup;
010import com.badlogic.gdx.utils.Scaling;
011import com.badlogic.gdx.utils.viewport.ScalingViewport;
012import squidpony.annotation.Beta;
013
014/**
015 * A single-Actor container that should stretch the container to fill the dimensions of the container.
016 * It's unclear if this actually does anything yet; there are issues testing it that may or may not be related to this.
017 * Created by Tommy Ettinger on 12/29/2016.
018 */
019@Beta
020public class StretchContainer extends WidgetGroup {
021    public WrapperViewport viewport;
022    protected float prefWidth, prefHeight;
023    public class WrapperViewport extends ScalingViewport
024    {
025        public WrapperViewport (float worldWidth, float worldHeight) {
026            this(worldWidth, worldHeight, new OrthographicCamera());
027        }
028        public WrapperViewport (float worldWidth, float worldHeight, Camera camera) {
029            super(Scaling.stretch, worldWidth, worldHeight, camera);
030        }
031
032        @Override
033        public void update (int screenWidth, int screenHeight, boolean centerCamera) {
034            Vector2 scaled = Scaling.stretch.apply(getWorldWidth(), getWorldHeight(),
035                    getWidth(), getHeight());
036            int viewportWidth = MathUtils.round(scaled.x);
037            int viewportHeight = MathUtils.round(scaled.y);
038            // Center.
039            setScreenBounds(MathUtils.round(getX()), MathUtils.round(getY()), viewportWidth, viewportHeight);
040            apply(true);
041        }
042
043        public Scaling getScaling () {
044            return Scaling.stretch;
045        }
046    }
047    public StretchContainer(Actor... actors)
048    {
049        if(actors == null || actors.length == 0)
050        {
051            viewport = new WrapperViewport(1, 1);
052        }
053        else {
054            viewport = new WrapperViewport(prefWidth = actors[0].getWidth(), prefHeight = actors[0].getHeight());
055            setSize(prefWidth, prefHeight);
056            addActor(actors[0]);
057
058        }
059        setFillParent(true);
060    }
061
062
063    @Override
064    public float getMinWidth() {
065        return 0;
066    }
067
068    @Override
069    public float getMinHeight() {
070        return 0;
071    }
072
073    @Override
074    public float getPrefWidth() {
075        return prefWidth;
076    }
077
078    @Override
079    public float getPrefHeight() {
080        return prefHeight;
081    }
082
083    @Override
084    public void invalidate() {
085        viewport.update(Gdx.graphics.getWidth(), Gdx.graphics.getHeight(), true);
086        super.invalidate();
087    }
088}
