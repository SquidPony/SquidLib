143     *
144     * If the TextCellFactory has not yet been initialized, then it will be sized at 12x12 px per cell. If it is null

146     *
147     * @param gridWidth the number of cells horizontally
148     * @param gridHeight the number of cells vertically
149     * @param factory the factory to use for cell rendering
150     * @param center
151     *                  The color center to use. Can be {@code null}, but then must be set later on with
152     *          {@link #setColorCenter(IColorCenter)}.
153     */
154    public SquidPanel(int gridWidth, int gridHeight, TextCellFactory factory, IColorCenter<Color> center,
155                      float xOffset, float yOffset, char[][] actualMap) {
156        this.gridWidth = gridWidth;
157        this.gridHeight = gridHeight;
158        if(center == null)
159            scc = DefaultResources.getSCC();
160        else
161            scc = center;
162
163        if (factory == null) {
164            textFactory = new TextCellFactory();
165        }
166        else
167            textFactory = factory;
168        if (!textFactory.initialized()) {
169            textFactory.initByFont();
170        }
171
172        cellWidth = MathUtils.round(textFactory.actualCellWidth);
173        cellHeight = MathUtils.round(textFactory.actualCellHeight);
174
175        if(actualMap == null || actualMap.length <= 0)
176            contents = ArrayTools.fill(' ', gridWidth, gridHeight);
177        else
178            contents = actualMap;
179        colors = ArrayTools.fill(scc.filter(Color.CLEAR).toFloatBits(), contents.length, contents[0].length);
180
181        int w = gridWidth * cellWidth;
182        int h = gridHeight * cellHeight;
183        this.xOffset = xOffset;
184        this.yOffset = yOffset;
185        setSize(w, h);
186        animatedEntities = new OrderedSet<>();
187        autoActors = new OrderedSet<>();
188    }

374    /**
375     * Puts the given string in the chosen direction, with the first character shown (not necessarily the first in the
376     * string) at the given offset. If you use {@link Direction#LEFT}, then this effectively reverses the String and
377     * prints it with the last character of the String at the minimum-x position, which is the same position that the
378     * first character would be if you printed normally or if you gave this RIGHT (x is xOffset, y is yOffset). Giving
379     * UP acts similarly to LEFT, but has the last character at the minimum-y position and has the first character below
380     * it. The diagonals act as you would expect, combining the behavior of one of UP or DOWN with one of LEFT or RIGHT.
381     * <br>
382     * Does not word wrap. Characters that are not renderable (due to being at negative offsets or offsets greater than
383     * the grid size) will not be shown but will not cause any malfunctions.
384     *
385     * @param xOffset the x coordinate of the first character
386     * @param yOffset the y coordinate of the first character
387     * @param string the characters to be displayed
388     * @param foreground the color to draw the characters
389     * @param direction the direction the text should be written in, such as {@link Direction#RIGHT} for normal layout
390     */
391    public void put(int xOffset, int yOffset, String string, Color foreground, Direction direction) {
392        float enc = scc.filter(foreground).toFloatBits();
393        switch (direction)
394        {
395            case DOWN:
396                for (int i = 0; i < string.length(); i++) {
397                    put(xOffset, yOffset + i, string.charAt(i), enc);
398                }
399                break;
400            case UP:
401                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
402                    put(xOffset, yOffset + p, string.charAt(i), enc);
403                }
404                break;
405            case LEFT:
406                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
407                    put(xOffset + p, yOffset, string.charAt(i), enc);
408                }
409                break;
410            case DOWN_RIGHT:
411                for (int i = 0; i < string.length(); i++) {
412                    put(xOffset + i, yOffset + i, string.charAt(i), enc);
413                }
414                break;
415            case UP_RIGHT:
416                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
417                    put(xOffset + i, yOffset + p, string.charAt(i), enc);
418                }
419                break;
420            case UP_LEFT:
421                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
422                    put(xOffset + p, yOffset + p, string.charAt(i), enc);
423                }
424                break;
425            case DOWN_LEFT:
426                for (int i = 0, p = string.length() - 1; i < string.length(); i++, p--) {
427                    put(xOffset + p, yOffset + i, string.charAt(i), enc);
428                }
429                break;
430            default:
431                for (int i = 0; i < string.length(); i++) {
432                    put(xOffset + i, yOffset, string.charAt(i), enc);
433                }
434        }
435    }
