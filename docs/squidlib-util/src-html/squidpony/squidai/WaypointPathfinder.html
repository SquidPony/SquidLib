028    /**
029     * Calculates and stores the doors and doors-like connections ("chokepoints") on the given map as waypoints.
030     * Will use the given Radius enum to determine how to handle DijkstraMap measurement in future pathfinding.
031     * Uses rng for all random choices, or a new unseeded RNG if the parameter is null.
032     * @param map a char[][] that stores a "complete" dungeon map, with any chars as features that pathfinding needs.
033     * @param measurement a Radius that should correspond to how you want path distance calculated.
034     * @param rng an RNG object or null (which will make this use a new RNG); will be used for all random choices
035     */
036    public WaypointPathfinder(char[][] map, Radius measurement, RNG rng)
037    {
038        if(rng == null)
039            this.rng = new StatefulRNG();
040        else
041            this.rng = rng;
042        this.map = map;
043        width = map.length;
044        height = map[0].length;
045        char[][] simplified = DungeonUtility.simplifyDungeon(map);
046        ArrayList<Coord> centers = PoissonDisk.sampleMap(simplified,
047                Math.min(width, height) * 0.4f, this.rng, '#');
048        int centerCount = centers.size();
049        expansionMap = new int[width][height];
050        waypoints = new OrderedMap<>(64);
051        dm = new DijkstraMap(simplified, DijkstraMap.Measurement.MANHATTAN);
052
053        for (Coord center : centers) {
054            dm.clearGoals();
055            dm.resetMap();
056            dm.setGoal(center);
057            dm.scan(null);
058            double current;
059            for (int i = 0; i < width; i++) {
060                for (int j = 0; j < height; j++) {
061                    current = dm.gradientMap[i][j];
062                    if (current >= DijkstraMap.FLOOR)
063                        continue;
064                    if (center.x == i && center.y == j)
065                        expansionMap[i][j]++;
066                    for (Direction dir : Direction.CARDINALS) {
067                        if (dm.gradientMap[i + dir.deltaX][j + dir.deltaY] == current + 1 ||
068                                dm.gradientMap[i + dir.deltaX][j + dir.deltaY] == current - 1)
069                            expansionMap[i][j]++;
070                    }
071                }
072            }
073        }
074
075        for (int i = 0; i < width; i++) {
076            for (int j = 0; j < height; j++) {
077                expansionMap[i][j] /= centerCount;
078            }
079        }
080
081        OrderedSet<Coord> chokes = new OrderedSet<>(128);
082        for (int i = 0; i < width; i++) {
083            ELEMENT_WISE:
084            for (int j = 0; j < height; j++) {
085                if(expansionMap[i][j] <= 0)
086                    continue;
087                int current = expansionMap[i][j];
088                boolean good = false;
089                for(Direction dir : Direction.CARDINALS) {
090                    if (chokes.contains(Coord.get(i + dir.deltaX, j + dir.deltaY)))
091                        continue ELEMENT_WISE;
092                    if (expansionMap[i + dir.deltaX][j + dir.deltaY] > 0 && expansionMap[i + dir.deltaX][j + dir.deltaY] > current + 1 ||
093                            (expansionMap[i + dir.deltaX][j + dir.deltaY] > current && expansionMap[i][j] <= 2)) {
094                        if (expansionMap[i - dir.deltaX][j - dir.deltaY] > 0 && expansionMap[i - dir.deltaX][j - dir.deltaY] >= current) {
095                            good = true;
096                        }
097                    }
098                }
099
100                if(good) {
101                    Coord chk = Coord.get(i, j);
102                    chokes.add(chk);
103                    waypoints.put(chk, new OrderedMap<Coord, Edge>());
104                }
105            }
106        }
107
108        /*
109        for (int y = 0; y < height; y++) {
110            for (int x = 0; x < width; x++) {
111                if(expansionMap[x][y] <= 0)
112                    System.out.print('#');
113                else
114                    System.out.print((char)(expansionMap[x][y] + 64));
115            }
116            System.out.println();
117        }
118
119        for (int y = 0; y < height; y++) {
120            for (int x = 0; x < width; x++) {
121                if(expansionMap[x][y] <= 0)
122                    System.out.print('#');
123                else if(chokes.contains(Coord.get(x, y)))
124                    System.out.print('@');
125                else if(centers.contains(Coord.get(x, y)))
126                    System.out.print('*');
127                else
128                    System.out.print('.');
129            }
130            System.out.println();
131        }
132*/
133
134        dm = new DijkstraMap(map, DijkstraMap.findMeasurement(measurement));
135
136        int e = 0;
137        for(Map.Entry<Coord, OrderedMap<Coord, Edge>> n : waypoints.entrySet())
138        {
139            chokes.remove(n.getKey());
140            if(chokes.isEmpty())
141                break;
142            dm.clearGoals();
143            dm.resetMap();
144            dm.setGoal(n.getKey());
145            dm.scan(null);
146            for(Coord c : chokes)
147            {
148                n.getValue().put(c, new Edge(n.getKey(), c, dm.findPathPreScanned(c), dm.gradientMap[c.x][c.y]));
149            }
150        }

422    /**
423     * Finds the appropriate one of the already-calculated, possibly-long paths this class stores to get from a waypoint
424     * to another waypoint, then quickly finds a path to get on the long path, and returns the total path. This does
425     * not need to perform any full-map scans with DijkstraMap.
426     * @param self the pathfinder's position
427     * @param approximateTarget the Coord that represents the approximate area to pathfind to; will be randomized if
428     *                          it is not walkable.
429     * @return an ArrayList of Coord that will go from a cell adjacent to self to a waypoint near approximateTarget
430     */
431    public ArrayList<Coord> getKnownPath(Coord self, Coord approximateTarget) {
432        ArrayList<Coord> near = dm.findNearestMultiple(approximateTarget, 5, waypoints.keySet());
433        Coord me = dm.findNearest(self, waypoints.keySet());
434        double bestCost = 999999.0;
435        ArrayList<Coord> path = new ArrayList<>();
436        /*if (waypoints.containsKey(me)) {
437            Edge[] ed = waypoints.get(me).values().toArray(new Edge[waypoints.get(me).size()]);
438            Arrays.sort(ed);
439            path = ed[0].path;
440        */
441        boolean reversed = false;
442        for (Coord test : near) {
443            if (waypoints.containsKey(test)) {
444                Edge ed;
445                if(waypoints.get(test).containsKey(me)) {
446                    ed = waypoints.get(test).get(me);
447                    reversed = true;
448                }
449                else if(waypoints.containsKey(me) && waypoints.get(me).containsKey(test))
450                    ed = waypoints.get(me).get(test);
451                else
452                    continue;
453                if (ed.cost < bestCost) {
454                    bestCost = ed.cost;
455                    path = new ArrayList<>(ed.path);
456                }
457            }
458        }
459        if(path.isEmpty())
460            return path;
461        if(reversed)
462            Collections.reverse(path);
463        ArrayList<Coord> getToPath = dm.findShortcutPath(self, path.toArray(new Coord[path.size()]));
464        if (getToPath.size() > 0)
465        {
466            getToPath.remove(getToPath.size() - 1);
467            getToPath.addAll(path);
468            path = getToPath;
469        }
470        return path;
471    }

473    /**
474     * If a creature is interrupted or obstructed on a "highway" path, it may need to travel off the path to its goal.
475     * This method gets a straight-line path back to the path to goal. It does not contain the "highway" path, only the
476     * "on-ramp" to enter the ideal path.
477     * @param currentPosition the current position of the pathfinder, which is probably not on the ideal path
478     * @param path the ideal path, probably returned by getKnownPath
479     * @return an ArrayList of Coord that go from a cell adjacent to currentPosition to a Coord on or adjacent to path.
480     */
481    public ArrayList<Coord> goBackToPath(Coord currentPosition, ArrayList<Coord> path)
482    {
483        return dm.findShortcutPath(currentPosition, path.toArray(new Coord[path.size()]));
484    }

491    private static class Edge implements Comparable<Edge>
492    {
493        public Coord from;
494        public Coord to;
495        public ArrayList<Coord> path;
496        public double cost;
497        public Edge(Coord from, Coord to, ArrayList<Coord> path, double cost)
498        {
499            this.from = from;
500            this.to = to;
501            this.path = path;
502            this.cost = cost;
503        }
504
505        @Override
506        public boolean equals(Object o) {
507            if (this == o) return true;
508            if (o == null || getClass() != o.getClass()) return false;
509
510            Edge edge = (Edge) o;
511
512            if (Double.compare(edge.cost, cost) != 0) return false;
513            if (!from.equals(edge.from)) return false;
514            return to.equals(edge.to);
515
516        }
517
518        @Override
519        public int hashCode() {
520            int result;
521            long temp;
522            result = from.hashCode();
523            result = 31 * result + to.hashCode();
524            temp = NumberTools.doubleToLongBits(cost);
525            result = 31 * result + (int) (temp ^ (temp >>> 32));
526            return result;
527        }
528
529        /**
530         * Compares this object with the specified object for order.  Returns a
531         * negative integer, zero, or a positive integer as this object is less
532         * than, equal to, or greater than the specified object.
533         *
534         * Note: this class has a natural ordering that is
535         * inconsistent with equals.
536         * @param o the object to be compared.
537         * @return a negative integer, zero, or a positive integer as this object
538         * is less than, equal to, or greater than the specified object.
539         * @throws NullPointerException if the specified object is null
540         * @throws ClassCastException   if the specified object's type prevents it
541         *                              from being compared to this object.
542         */
543        @Override
544        public int compareTo(Edge o) {
545            return (cost - o.cost > 0) ? 1 : (cost - o.cost < 0) ? -1 : 0;
546        }
