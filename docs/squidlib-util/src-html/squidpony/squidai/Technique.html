001package squidpony.squidai;
002
003import squidpony.squidmath.Coord;
004import squidpony.squidmath.OrderedMap;
005
006import java.util.ArrayList;
007import java.util.Collection;
008import java.util.Set;
009
010/**
011 * A simple struct-like class that stores various public fields which describe the targeting properties of a skill,
012 * spell, tech, or any other game-specific term for a targeted (typically offensive) ability we call a Technique.
013 *
014 * The typical usage of a Technique is:
015 * <ul>
016 * <li>Construct any AOE implementation the Technique would use (if the Technique affects multiple grid cells).</li>
017 * <li>Construct the Technique (passing the AOE as a parameter if needed).</li>
018 * <li>Call setMap() before considering the Technique if it has not been called yet, if the physical map (including
019 * doors and destructible objects) has changed since setMap() was last called, or simply on every Technique every time
020 * the map changes if there are few enemies with few Techniques. PERFORMING ANY SUBSEQUENT STEPS WITHOUT SETTING THE
021 * MAP TO THE CURRENT ACTUAL PHYSICAL MAP WILL HAVE BAD CONSEQUENCES FOR LOGIC AND MAY CAUSE CRASHING BUGS DUE TO
022 * ARRAY BOUNDS VIOLATIONS IF YOU HAVEN'T SET THE MAP ARRAY IN THE FIRST PLACE. The map should be bounded by wall chars
023 * ('#'), which is done automatically by squidpony.squidgrid.mapping.DungeonGenerator .</li>
024 * <li>When the Technique is being considered by an AI, call idealLocations() with the values of targets,
025 * lesserTargets, and/or priorityTargets set to beings that the AI can see (likely using FOV) and wants to affect with
026 * this Technique (enemies for offensive Techniques, allies for supporting ones), and requiredExclusions typically set
027 * to allies for offensive Techniques that can cause friendly-fire damage, or to null for supporting ones or Techniques
028 * that don't affect allies.</li>
029 * <li>When an ideal location has been determined from the previous step, and the AI decides (possibly randomly, by an
030 * aggression ("aggro") level tracked per-enemy, or by weights on Techniques for different situations) to use this
031 * Technique on a specific target point, call apply() with the user position as a Coord and the chosen Coord, and
032 * proceed to process the effects of the Technique as fitting for your game on the returned Map of Coord keys to Double
033 * values that describe the amount of effect (from 0.0 for none to 1.0 for full) that Coord receives.</li>
034 * </ul>
035 *
036 * A Technique always has an AOE implementation that it uses to determine which cells it actually affects, and
037 * Techniques that do not actually affect an area use the default single-cell "Area of Effect" implementation, PointAOE.
038 * You typically will need to construct the implementing class of the AOE interface in a different way for each
039 * implementation; BeamAOE, LineAOE, and ConeAOE depend on the user's position, BurstAOE and BlastAOE treat radii
040 * differently from BeamAOE and LineAOE, and CloudAOE has a random component that can be given a seed.
041 *
042 * A Technique has a String  name, which typically should be in a form that can be presented to a user, and a
043 * String id, which defaults to the same value as name but can be given some value not meant for users that records
044 * any additional identifying characteristics the game needs for things like comparisons.
045 *
046 * Created by Tommy Ettinger on 7/27/2015.
047 */
048public class Technique {
049    public String name;
050    public String id;
051    public AOE aoe;
052    protected char[][] dungeon;
053    protected static final Coord DEFAULT_POINT = Coord.get(0, 0);
054
055    /**
056     * Creates a Technique that can target any adjacent single Coord, using
057     * Chebyshev (8-way square) distance.
058     */
059    public Technique() {
060        name = "Default Technique";
061        id = name;
062        aoe = new PointAOE(DEFAULT_POINT);
063    }
064
065    /**
066     * Creates a Technique that can target any adjacent single Coord,
067     * using Chebyshev (8-way square) distance.
068     * @param name An identifier that may be displayed to the user. Also used for id.
069     */
070    public Technique(String name) {
071        this.name = name;
072        id = name;
073        aoe = new PointAOE(DEFAULT_POINT);
074    }
075
076    /**
077     * Creates a Technique that can target a Coord at a range specified by the given AOE's minRange and maxRange,
078     * using a distance metric from the AOE, and use that target Coord for the given AOE.
079     * @param name An identifier that may be displayed to the user. Also used for id.
080     * @param aoe An implementation of the AOE interface; typically needs construction beforehand.
081     */
082    public Technique(String name, AOE aoe) {
083        this.name = name;
084        id = name;
085        this.aoe = aoe;
086    }
087
088
089    /**
090     * Creates a Technique that can target a Coord at a range specified by the given AOE's minRange and maxRange,
091     * using a distance metric from the AOE, and use that target Coord for the given AOE. Takes an id parameter.
092     * @param name An identifier that may be displayed to the user.
093     * @param id An identifier that should always be internal, and will probably never be shown to the user.
094     * @param aoe An implementation of the AOE interface; typically needs construction beforehand.
095     */
096    public Technique(String name, String id, AOE aoe) {
097        this.name = name;
098        this.id = id;
099        this.aoe = aoe;
100    }
101
102
103    /**
104     * VITAL: Call this method before any calls to idealLocations() or apply(), and call it again if the map changes.
105     *
106     * This simple method sets the map that this Technique can find targets in to a given char 2D array with '#' for
107     * walls and any other character (including characters for open and closed doors) treated as a floor for most
108     * purposes (certain AOE implementations may treat open and closed doors differently, specifically any that use
109     * FOV internally and can yield values other than 1.0 from their findArea() method, like BlastAOE and ConeAOE).
110     * @param map A char 2D array like one generated by squidpony.squidgrid.mapping.DungeonGenerator, with '#' for walls and bounded edges.
111     */
112    public void setMap(char[][] map)
113    {
114        dungeon = map;
115        aoe.setMap(map);
116    }
117
118    /**
119     * Get a mapping of Coord keys representing locations to apply this Technique to, to ArrayList of Coord values
120     * representing which targets (by their location) are effected by choosing that Coord. All targets with this method
121     * are valued equally, and the ideal location affects as many as possible without hitting any requiredExclusions.
122     *
123     * YOU MUST CALL setMap() with the current map status at some point before using this method, and call it again if
124     * the map changes. Failure to do so can cause serious bugs, from logic errors where monsters consider a door
125     * closed when it is open or vice versa, to an ArrayIndexOutOfBoundsException being thrown if the player moved to a
126     * differently-sized map and the Technique tries to use the previous map with coordinates from the new one.
127     *
128     * @param user The location of the user of this Technique
129     * @param targets Set of Coord of desirable targets to include in the area of this Technique, as many as possible.
130     * @param requiredExclusions Set of Coord where each value is something this Technique will really try to avoid.
131     * @return OrderedMap of Coord keys representing target points to pass to apply, to ArrayList of Coord values representing what targets' locations will be affected.
132     */
133    public OrderedMap<Coord, ArrayList<Coord>> idealLocations(Coord user, Collection<Coord> targets, Collection<Coord> requiredExclusions) {
134        aoe.setOrigin(user);
135        return aoe.idealLocations(targets, requiredExclusions);
136
137    }
138
139    /**
140     * Get a mapping of Coord keys representing locations to apply this Technique to, to ArrayList of Coord values
141     * representing which targets (by their location) are effected by choosing that Coord. This method will strongly
142     * prefer including priorityTargets in its area, especially multiple one if possible, and primarily uses
143     * lesserTargets as a tiebreaker if two locations have the same number of included priorityTargets but one has more
144     * lesserTargets in its area.
145     *
146     * YOU MUST CALL setMap() with the current map status at some point before using this method, and call it again if
147     * the map changes. Failure to do so can cause serious bugs, from logic errors where monsters consider a door
148     * closed when it is open or vice versa, to an ArrayIndexOutOfBoundsException being thrown if the player moved to a
149     * differently-sized map and the Technique tries to use the previous map with coordinates from the new one.
150     *
151     * @param user The location of the user of this Technique
152     * @param priorityTargets Set of Coord of important targets to include in the area of this Technique, preferring to target a single priorityTarget over four lesserTargets.
153     * @param lesserTargets Set of Coord of desirable targets to include in the area of this Technique, as many as possible without excluding priorityTargets.
154     * @param requiredExclusions Set of Coord where each value is something this Technique will really try to avoid.
155     * @return OrderedMap of Coord keys representing target points to pass to apply, to ArrayList of Coord values representing what targets' locations will be affected.
156     */
157    public OrderedMap<Coord, ArrayList<Coord>> idealLocations(Coord user, Set<Coord> priorityTargets, Set<Coord> lesserTargets, Set<Coord> requiredExclusions) {
158        aoe.setOrigin(user);
159        return aoe.idealLocations(priorityTargets, lesserTargets, requiredExclusions);
160    }
161
162    /**
163     * This does one last validation of the location aimAt (checking that it is within the valid range for this
164     * Technique) before getting the area affected by the AOE targeting that cell. It considers the origin of the AOE
165     * to be the Coord parameter user, for purposes of directional limitations and for AOE implementations that need
166     * the user's location, such as ConeAOE and LineAOE.
167     *
168     * YOU MUST CALL setMap() with the current map status at some point before using this method, and call it again if
169     * the map changes. Failure to do so can cause serious bugs, from logic errors where monsters consider a door
170     * closed when it is open or vice versa, to an ArrayIndexOutOfBoundsException being thrown if the player moved to a
171     * differently-sized map and the Technique tries to use the previous map with coordinates from the new one.
172     *
173     * @param user The position of the Technique's user, x first, y second.
174     * @param aimAt A target Coord typically obtained from idealLocations that determines how to position the AOE.
175     * @return a HashMap of Coord keys to Double values from 1.0 (fully affected) to 0.0 (unaffected).
176     */
177    public OrderedMap<Coord, Double> apply(Coord user, Coord aimAt)
178    {
179        aoe.setOrigin(user);
180        aoe.shift(aimAt);
181        return aoe.findArea();
182    }
183}
