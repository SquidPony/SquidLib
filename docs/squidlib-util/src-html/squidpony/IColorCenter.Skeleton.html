010/**
011 * How to manage colors, making sure that a color is allocated at most once.
012 * 
013 * <p>
014 * If you aren't using squidlib's gdx part, you should use this interface (and
015 * the {@link Skeleton} implementation), because it caches instances.
016 * </p>
017 * 
018 * <p>
019 * If you are using squidlib's gdx part, you should use this interface (and the
020 * {@code SquidColorCenter} implementation) if:
021 * 
022 * <ul>
023 * <li>You don't want to use preallocated instances (if you do, check out
024 * {@code squidpony.squidgrid.gui.Colors})</li>
025 * <li>You don't want to use named colors (if you do, check out
026 * {@code com.badlogic.gdx.graphics.Colors})</li>
027 * <li>You don't like libgdx's Color representation (components as floats
028 * in-between 0 and 1) but prefer components within 0 (inclusive) and 256
029 * (exclusive); and don't mind the overhead of switching the representations. My
030 * personal opinion is that the overhead doesn't matter w.r.t other intensive
031 * operations that we have in roguelikes (path finding).</li>
032 * </ul>
033 * 
034 * @author smelC
035 * 
036 * @param <T>
037 *            The concrete type of colors
038 */
039public interface IColorCenter<T> {

307        /**
308         * You may want to copy colors between IColorCenter instances that have different create() methods -- and as
309         * such, will have different values for the same keys in the cache. This allows you to copy the cache from other
310         * into this Skeleton, but using this Skeleton's create() method.
311         * @param other another Skeleton of the same type that will have its cache copied into this Skeleton
312         */
313        public void copyCache(Skeleton<T> other)
314        {
315            for (Map.Entry<Long, T> k : other.cache.entrySet())
316            {
317                cache.put(k.getKey(), create(getRed(k.getValue()), getGreen(k.getValue()), getBlue(k.getValue()),
318                        getAlpha(k.getValue())));
319            }
320        }

337        @Override
338        public T get(int red, int green, int blue, int opacity) {
339            tempValue = getUniqueIdentifier(red, green, blue, opacity);
340            T t = cache.get(tempValue);
341            if (t == null) {
342                                /* Miss */
343                t = create(red, green, blue, opacity);
344                                /* Put in cache */
345                cache.put(tempValue, t);
346            }
347            return t;
348        }

355        @Override
356        public T getHSV(float hue, float saturation, float value, float opacity) {
357            if ( saturation < 0.0001 )                       //HSV from 0 to 1
358            {
359                return get(Math.round(value * 255), Math.round(value * 255), Math.round(value * 255),
360                        Math.round(opacity * 255));
361            }
362            else
363            {
364                float h = hue * 6f;
365                if ( h >= 6 ) h = 0;      //H must be < 1
366                int i = (int)h;             //Or ... var_i = floor( var_h )
367                float a = value * ( 1 - saturation );
368                float b = value * ( 1 - saturation * ( h - i ) );
369                float c = value * ( 1 - saturation * ( 1 - ( h - i ) ) );
370
371                switch (i)
372                {
373                    case 0: return get(Math.round(value * 255), Math.round(c * 255), Math.round(a * 255),
374                            Math.round(opacity * 255));
375                    case 1: return get(Math.round(b * 255), Math.round(value * 255), Math.round(a * 255),
376                            Math.round(opacity * 255));
377                    case 2: return get(Math.round(a * 255), Math.round(value * 255), Math.round(c * 255),
378                            Math.round(opacity * 255));
379                    case 3: return get(Math.round(a * 255), Math.round(b * 255), Math.round(value * 255),
380                            Math.round(opacity * 255));
381                    case 4: return get(Math.round(c * 255), Math.round(a * 255), Math.round(value * 255),
382                            Math.round(opacity * 255));
383                    default: return get(Math.round(value * 255), Math.round(a * 255), Math.round(b * 255),
384                            Math.round(opacity * 255));
385                }
386            }
387        }

415        /**
416         * @param r the red component in 0.0 to 1.0 range, typically
417         * @param g the green component in 0.0 to 1.0 range, typically
418         * @param b the blue component in 0.0 to 1.0 range, typically
419         * @return the saturation of the color from 0.0 (a grayscale color; inclusive) to 1.0 (a
420         * bright color, exclusive)
421         */
422        public float getSaturation(float r, float g, float b) {
423            float min = Math.min(Math.min(r, g ), b);    //Min. value of RGB
424            float max = Math.max(Math.max(r, g), b);    //Min. value of RGB
425            float delta = max - min;                     //Delta RGB value
426
427            float saturation;
428
429            if ( delta < 0.0001f )                     //This is a gray, no chroma...
430            {
431                saturation = 0;
432            }
433            else                                    //Chromatic data...
434            {
435                saturation = delta / max;
436            }
437            return saturation;
438        }

474        /**
475         * @param r the red component in 0.0 to 1.0 range, typically
476         * @param g the green component in 0.0 to 1.0 range, typically
477         * @param b the blue component in 0.0 to 1.0 range, typically
478         * @return The hue of the color from 0.0 (red, inclusive) towards orange, then yellow, and
479         * eventually to purple before looping back to almost the same red (1.0, exclusive)
480         */
481        public float getHue(float r, float g, float b) {
482            float min = Math.min(Math.min(r, g ), b);    //Min. value of RGB
483            float max = Math.max(Math.max(r, g), b);    //Min. value of RGB
484            float delta = max - min;                     //Delta RGB value
485
486            float hue;
487
488            if ( delta < 0.0001f )                     //This is a gray, no chroma...
489            {
490                hue = 0;                                //HSV results from 0 to 1
491            }
492            else                                    //Chromatic data...
493            {
494                float rDelta = ( ( ( max - r ) / 6f ) + ( delta / 2f ) ) / delta;
495                float gDelta = ( ( ( max - g ) / 6f ) + ( delta / 2f ) ) / delta;
496                float bDelta = ( ( ( max - b ) / 6f ) + ( delta / 2f ) ) / delta;
497
498                if       ( r == max ) hue = bDelta - gDelta;
499                else if ( g == max ) hue = ( 1f / 3f ) + rDelta - bDelta;
500                else                 hue = ( 2f / 3f ) + gDelta - rDelta;
501
502                if ( hue < 0 ) hue += 1f;
503                else if ( hue > 1 ) hue -= 1;
504            }
505            return hue;
506        }

525                @Override
526                public IColoredString<T> filter(IColoredString<T> ics) {
527                        /*
528                         * It is common not to have a filter or to have the identity one. To
529                         * avoid always copying strings in this case, we first roll over the
530                         * string to see if there'll be a change.
531                         * 
532                         * This is clearly a subjective design choice but my industry
533                         * experience is that minimizing allocations is the thing to do for
534                         * performances, hence I prefer iterating twice to do that.
535                         */
536                        boolean change = false;
537                        for (IColoredString.Bucket<T> bucket : ics) {
538                                final T in = bucket.getColor();
539                                if (in == null)
540                                        continue;
541                                final T out = filter(in);
542                                if (in != out) {
543                                        change = true;
544                                        break;
545                                }
546                        }
547
548                        if (change) {
549                                final IColoredString<T> result = IColoredString.Impl.create();
550                                for (IColoredString.Bucket<T> bucket : ics)
551                                        result.append(bucket.getText(), filter(bucket.getColor()));
552                                return result;
553                        } else
554                                /* Only one allocation: the iterator, yay \o/ */
555                                return ics;
556                }
557                /**
558                 * Gets a copy of t and modifies it to make a shade of gray with the same brightness.
559                 * The doAlpha parameter causes the alpha to be considered in the calculation of brightness and also changes the
560                 * returned alpha of the color.
561                 * Not related to reified types or any usage of "reify."
562                 * @param t a T to copy; only the copy will be modified
563                 * @param doAlpha
564                 *            Whether to include (and hereby change) the alpha component.
565                 * @return A monochromatic variation of {@code t}.
566                 */
567                @Override
568                public T greify(T t, boolean doAlpha) {
569                        if (t == null)
570                                /* Cannot do */
571                                return null;
572                        final int red = getRed(t);
573                        final int green = getGreen(t);
574                        final int blue = getBlue(t);
575                        final int alpha = getAlpha(t);
576                        final int rgb = red + green + blue;
577                        final int mean;
578                        final int newAlpha;
579                        if (doAlpha) {
580                                mean = (rgb + alpha) / 4;
581                                newAlpha = mean;
582                        } else {
583                                mean = rgb / 3;
584                                /* No change */
585                                newAlpha = alpha;
586                        }
587                        return get(mean, mean, mean, newAlpha);
588                }

590                /**
591                 * Gets the linear interpolation from Color start to Color end, changing by the fraction given by change.
592         * This implementation tries to work with colors in a way that is as general as possible, using getRed() instead
593         * of some specific detail that depends on how a color is implemented. Other implementations that specialize in
594         * a specific type of color may be able to be more efficient.
595         * @param start the initial color T
596         * @param end the "target" color T
597         * @param change the degree to change closer to end; a change of 0.0f produces start, 1.0f produces end
598         * @return a new T between start and end
599                 */
600                public T lerp(T start, T end, float change)
601                {
602                        if(start == null || end == null)
603                                return null;
604                        final int sr = getRed(start), sg = getGreen(start), sb = getBlue(start), sa = getAlpha(start),
605                                        er = getRed(end), eg = getGreen(end), eb = getBlue(end), ea = getAlpha(end);
606                        return get(
607                                        (int)(sr + change * (er - sr)),
608                                        (int)(sg + change
609                                                        * (eg - sg)),
610                                        (int)(sb + change * (eb - sb)),
611                                        (int)(sa + change * (ea - sa))
612                        );
613                }
614                /**
615                 * Gets a fully-desaturated version of the given color (keeping its brightness, but making it grayscale).
616                 * Keeps alpha the same; if you want alpha to be considered (and brightness to be calculated differently), then
617                 * you can use greify() in this class instead.
618                 * @param color the color T to desaturate (will not be modified)
619                 * @return the grayscale version of color
620                 */
621                public T desaturated(T color)
622                {
623                        int f = (int)Math.min(255, getRed(color) * 0.299f + getGreen(color) * 0.587f + getBlue(color) * 0.114f);
624                        return get(f, f, f, getAlpha(color));
625                }

629                 * Alpha is left unchanged.

631                 * @param degree a float between 0.0f and 1.0f; more makes it less colorful
632                 * @return the desaturated (and if a filter is used, also filtered) new color T
633                 */
634                public T desaturate(T color, float degree)
635                {
636                        return lerp(color, desaturated(color), degree);

640         * Fully saturates color (makes it a vivid color like red or green and less gray) and returns the modified copy.
641         * Leaves alpha unchanged.
642         * @param color the color T to saturate (will not be modified)
643         * @return the saturated version of color
644         */
645        public T saturated(T color)
646        {
647            return getHSV(getHue(color), 1f, getValue(color), getAlpha(color));
648        }
649                /**
650                 * Saturates color (makes it closer to a vivid color like red or green and less gray) by the specified degree and
651                 * returns the new color (saturated somewhat). If this is called on a color that is very close to gray, this is
652                 * likely to produce a red hue by default (if there's no hue to make vivid, it needs to choose something).
653                 * @param color the color T to saturate
654                 * @param degree a float between 0.0f and 1.0f; more makes it more colorful
655                 * @return the saturated (and if a filter is used, also filtered) new color
656                 */
657                public T saturate(T color, float degree)
658                {
659                        return lerp(color, saturated(color), degree);
660                }

662                @Override
663                public ArrayList<T> gradient(T fromColor, T toColor) {
664                        return gradient(fromColor, toColor, 16);
665                }
666
667                @Override
668                public ArrayList<T> gradient(T fromColor, T toColor, int steps) {
669                        ArrayList<T> colors = new ArrayList<>((steps > 1) ? steps : 1);
670                        colors.add(filter(fromColor));
671                        if(steps < 2)
672                                return colors;
673                        for (float i = 1; i < steps; i++) {
674                                colors.add(lerp(fromColor, toColor, i / (steps - 1f)));
675                        }
676                        return colors;
677
678                }
