029    /**
030     * Stupidly simple convenience method that produces a range from 0 to end, not including end, as an int array.
031     *
032     * @param end the exclusive upper bound on the range
033     * @return the range of ints as an int array
034     */
035    public static int[] range(int end) {
036        if (end <= 0)
037            return new int[0];
038        int[] r = new int[end];
039        for (int i = 0; i < end; i++) {
040            r[i] = i;
041        }
042        return r;
043    }

045    /**
046     * Stupidly simple convenience method that produces a range from start to end, not including end, as an int array.
047     *
048     * @param start the inclusive lower bound on the range
049     * @param end   the exclusive upper bound on the range
050     * @return the range of ints as an int array
051     */
052    public static int[] range(int start, int end) {
053        if (end - start <= 0)
054            return new int[0];
055        int[] r = new int[end - start];
056        for (int i = 0, n = start; n < end; i++, n++) {
057            r[i] = n;
058        }
059        return r;
060    }

062    /**
063     * Stupidly simple convenience method that produces a char range from start to end, including end, as a char array.
064     *
065     * @param start the inclusive lower bound on the range, such as 'a'
066     * @param end   the inclusive upper bound on the range, such as 'z'
067     * @return the range of chars as a char array
068     */
069    public static char[] charSpan(char start, char end) {
070        if (end - start <= 0)
071            return empty;
072        if (end == 0xffff) {
073
074            char[] r = new char[0x10000 - start];
075            for (char i = 0, n = start; n < end; i++, n++) {
076                r[i] = n;
077            }
078            r[0xffff - start] = 0xffff;
079            return r;
080        }
081        char[] r = new char[end - start + 1];
082        for (char i = 0, n = start; n <= end; i++, n++) {
083            r[i] = n;
084        }
085        return r;
086    }

088    /**
089     * Stupidly simple convenience method that produces a char array containing only letters that can be reasonably
090     * displayed (with SquidLib's default text display assets, at least). The letters are copied from a single source
091     * of 256 chars; if you need more chars or you don't need pure letters, you can use {@link #charSpan(char, char)}.
092     * This set does not contain "visual duplicate" letters, such as Latin alphabet capital letter 'A' and Greek
093     * alphabet capital letter alpha, 'Α'; it does contain many accented Latin letters and the visually-distinct Greek
094     * letters, up to a point.
095     * @param charCount the number of letters to return in an array; the maximum this will produce is 256
096     * @return the range of letters as a char array
097     */
098    public static char[] letterSpan(int charCount) {
099        if (charCount <= 0)
100            return empty;
101        char[] r = new char[Math.min(charCount, 256)];
102        System.arraycopy(letters, 0, r, 0, r.length);
103        return r;
104    }

107     * Gets the nth letter from the set that SquidLib is likely to support; from index 0 (returning 'A') to 255
108     * (returning the Greek lower-case letter gamma, 'γ') and wrapping around if given negative numbers or numbers
109     * larger than 255. This set does not contain "visual duplicate" letters, such as Latin alphabet capital letter 'A'
110     * and Greek alphabet capital letter alpha, 'Α'; it does contain many accented Latin letters and the
111     * visually-distinct Greek letters, up to a point.

113     * @return the letter at the given index in a 256-element portion of the letters SquidLib usually supports

115    public static char letterAt(int index)

117        return letters[index & 255];

121     * Gets a copy of the 2D char array, source, that has the same data but shares no references with source.
122     *
123     * @param source a 2D char array
124     * @return a copy of source, or null if source is null
125     */
126    public static char[][] copy(char[][] source) {
127        if (source == null)
128            return null;
129        if (source.length < 1)
130            return new char[0][0];
131        char[][] target = new char[source.length][];
132        for (int i = 0; i < source.length && i < target.length; i++) {
133            target[i] = new char[source[i].length];
134            System.arraycopy(source[i], 0, target[i], 0, source[i].length);
135        }
136        return target;
137    }

196    /**
197     * Inserts as much of source into target at the given x,y position as target can hold or source can supply.
198     * Modifies target in-place and also returns target for chaining.
199     * Used primarily to place a smaller array into a different position in a larger array, often freshly allocated.
200     *
201     * @param source a 2D char array that will be copied and inserted into target
202     * @param target a 2D char array that will be modified by receiving as much of source as it can hold
203     * @param x      the x position in target to receive the items from the first cell in source
204     * @param y      the y position in target to receive the items from the first cell in source
205     * @return a modified copy of target with source inserted into it at the given position
206     */
207    public static char[][] insert(char[][] source, char[][] target, int x, int y) {
208        if (source == null || target == null)
209            return target;
210        if (source.length < 1 || source[0].length < 1)
211            return copy(target);
212        for (int i = 0; i < source.length && x + i < target.length; i++) {
213            System.arraycopy(source[i], 0, target[x + i], y, Math.min(source[i].length, target[x + i].length - y));
214        }
215        return target;
216    }

383     * Not to be confused with {@link #fill(boolean, int, int)}, which makes a new 2D array.
384     * @param array2d a 2D array that will be modified in-place
385     * @param value the value to fill all of array2D with
386     */
387    public static void fill(boolean[][] array2d, boolean value) {
388        final int width = array2d.length;
389        final int height = width == 0 ? 0 : array2d[0].length;
390        if(width > 0) {
391            for (int i = 0; i < height; i++) {
392                array2d[0][i] = value;
393            }
394        }
395        for (int x = 1; x < width; x++) {
396            System.arraycopy(array2d[0], 0, array2d[x], 0, height);
397        }

460     * Not to be confused with {@link #fill(double[][], double)}, which fills a 2D array instead of a 3D one, or with
461     * {@link #fill(double, int, int)}, which makes a new 2D array.
462     * @param array3d a 3D array that will be modified in-place
463     * @param value the value to fill all of array3d with
464     */
465    public static void fill(double[][][] array3d, double value) {
466        final int depth = array3d.length;
467        final int width = depth == 0 ? 0 : array3d[0].length;
468        final int height = width == 0 ? 0 : array3d[0][0].length;
469        if(depth > 0 && width > 0) {
470            for (int i = 0; i < height; i++) {
471                array3d[0][0][i] = value;
472            }
473        }
474        for (int x = 1; x < width; x++) {
475            System.arraycopy(array3d[0][0], 0, array3d[0][x], 0, height);
476        }
477        for (int z = 1; z < depth; z++) {
478            for (int x = 0; x < width; x++) {
479                System.arraycopy(array3d[0][0], 0, array3d[z][x], 0, height);
480            }
481        }
482    }

523     * Rearranges an ArrayList to use the given ordering, returning a copy; random orderings can be produced with
524     * {@link squidpony.squidmath.RNG#randomOrdering(int)} or
525     * {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}. These orderings will never repeat an earlier element,
526     * and the returned ArrayList may be shorter than the original if {@code ordering} isn't as long as {@code list}.
527     * Using a random ordering is like shuffling, but allows you to repeat the shuffle exactly on other collections of
528     * the same size. A reordering can also be inverted with {@link #invertOrdering(int[])} or
529     * {@link #invertOrdering(int[], int[])}, getting the change that will undo another ordering.
530     *
531     * @param list     an ArrayList that you want a reordered version of; will not be modified.
532     * @param ordering an ordering, typically produced by one of RNG's randomOrdering methods.
533     * @param <T>      any generic type
534     * @return a modified copy of {@code list} with its ordering changed to match {@code ordering}.
535     */
536    public static <T> ArrayList<T> reorder(ArrayList<T> list, int... ordering) {
537        int ol;
538        if (list == null || ordering == null || (ol = Math.min(list.size(), ordering.length)) == 0)
539            return list;
540        ArrayList<T> alt = new ArrayList<T>(ol);
541        for (int i = 0; i < ol; i++) {
542            alt.add(list.get((ordering[i] % ol + ol) % ol));
543        }
544        return alt;
545    }

547    /**
548     * Given an ordering such as one produced by {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}, this finds
549     * its inverse, able to reverse the reordering and vice versa.
550     *
551     * @param ordering the ordering to find the inverse for
552     * @return the inverse of ordering
553     */
554    public static int[] invertOrdering(int[] ordering) {
555        int ol = 0;
556        if (ordering == null || (ol = ordering.length) == 0) return ordering;
557        int[] next = new int[ol];
558        for (int i = 0; i < ol; i++) {
559            if (ordering[i] < 0 || ordering[i] >= ol) return next;
560            next[ordering[i]] = i;
561        }
562        return next;
563    }

565    /**
566     * Given an ordering such as one produced by {@link squidpony.squidmath.RNG#randomOrdering(int, int[])}, this finds
567     * its inverse, able to reverse the reordering and vice versa. This overload doesn't allocate a new int
568     * array, and instead relies on having an int array of the same size as ordering passed to it as an
569     * additional argument.
570     *
571     * @param ordering the ordering to find the inverse for
572     * @param dest     the int array to put the inverse reordering into; should have the same length as ordering
573     * @return the inverse of ordering; will have the same value as dest
574     */
575    public static int[] invertOrdering(int[] ordering, int[] dest) {
576        int ol = 0;
577        if (ordering == null || dest == null || (ol = Math.min(ordering.length, dest.length)) == 0)
578            return ordering;
579        for (int i = 0; i < ol; i++) {
580            if (ordering[i] < 0 || ordering[i] >= ol) return dest;
581            dest[ordering[i]] = i;
582        }
583        return dest;
584    }

588     * @param data an array that will be reversed in-place
589     * @return the array passed in, after reversal
590     */
591    public static boolean[] reverse(boolean[] data)
592    {
593        int sz;
594        if(data == null || (sz = data.length) <= 0) return data;
595        boolean t;
596        for (int i = 0, j = sz - 1; i < j; i++, j--) {
597            t = data[j];
598            data[j] = data[i];
599            data[i] = t;
600        }
601        return data;
602    }
