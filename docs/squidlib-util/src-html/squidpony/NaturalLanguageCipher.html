010/**
011 * Class that builds up a dictionary of words in a source text to words generated by a FakeLanguageGen, and can
012 * "translate" a source text to a similarly-punctuated, similarly-capitalized fake text. Uses a hash of each word in the
013 * source text to determine the RNG seed that FakeLanguageGen will use, so the translation is not random. Can cipher a
014 * typically English text and generate a text with FakeLanguageGen, but also decipher such a generated text with a
015 * fully-complete, partially-complete, or partially-incorrect vocabulary.
016 * <br>
017 * This defaults to caching source-language words to their generated-language word translations in the field table, as
018 * well as the reverse translation in reverse. This can be changed to reduce memory usage for large vocabularies with
019 * {@code setCacheLevel()}, where it starts at 2 (writing to table and reverse), and can be lowered to 1 (writing to
020 * table only) if you don't need reverse to decipher a language easily, or to 0 (writing to neither) if you expect that
021 * memory will be at a premium and don't mind re-generating the same word each time it occurs in a source text. If
022 * cacheLevel is 1 or less, then this will not check for overlap between previously-generated words (it won't have an
023 * easy way to look up previously-generated ones) and so may be impossible to accurately decipher. As an example, one
024 * test of level 1 generated "he" as the translation for both "a" and "at", so every time "a" had been ciphered and then
025 * deciphered, the reproduced version said "at" instead. This won't happen by default, but the default instead relies on
026 * words being entered as inputs to cipher() or lookup() in the same order. If words are entered in two different orders
027 * to different runs of the program, they may have different generated results if cacheLevel is 2. One way to handle
028 * this is to use cacheLevel 2 and cipher the whole game script, or just the unique words in it (maybe just a large word
029 * list, such as http://wordlist.aspell.net/12dicts/ ), then serialize the LanguageCipher for later usage.
030 * @author Tommy Ettinger
031 * Created by Tommy Ettinger on 5/1/2016.
032 */
033public class NaturalLanguageCipher implements Serializable{

036        private static final long serialVersionUID = 1287835632461186341L;
037        private long state;
038        public SemiRandom()
039        {
040            state = (long) (Long.MAX_VALUE * (Math.random() * 2.0 - 1.0));
041        }
042        public SemiRandom(long state)
043        {
044            this.state = state;
045        }
046        /**
047         * Get the current internal state of the StatefulRandomness as a long.
048         *
049         * @return the current internal state of this object.
050         */
051        @Override
052        public long getState() {
053            return state;
054        }
055
056        /**
057         * Set the current internal state of this StatefulRandomness with a long.
058         *
059         * @param state a 64-bit long. You should avoid passing 0, even though some implementations can handle that.
060         */
061        @Override
062        public void setState(long state) {
063            this.state = state;
064        }
065
066        /**
067         * Using this method, any algorithm that might use the built-in Java Random
068         * can interface with this randomness source.
069         *
070         * @param bits the number of bits to be returned
071         * @return the integer containing the appropriate number of bits
072         */
073        @Override
074        public int next(int bits) {
075            return (int) ((state += 0x41041041041041L) & ~(-1 << bits));
076        }
077
078        /**
079         * Using this method, any algorithm that needs to efficiently generate more
080         * than 32 bits of random data can interface with this randomness source.
081         * <p>
082         * Get a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
083         *
084         * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
085         */
086        @Override
087        public long nextLong() {
088            return state += 0x41041041041041L;
089        }
090
091        public double nextDouble()
092        {
093            return NumberTools.longBitsToDouble(0x3FFL << 52 | (state += 0x41041041041041L) >>> 12) - 1.0;
094        }
095
096        /**
097         * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
098         * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
099         * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
100         *
101         * @return a copy of this RandomnessSource
102         */
103        @Override
104        public RandomnessSource copy() {
105            return new SemiRandom(state);
106        }
107    }

340    private String addPart(String original, int syllables)
341    {
342        String done;
343        Pattern[] checks = null;
344        if(original.endsWith("-"))
345        {
346            checks = additionalPrefixChecks;
347        }
348        else if(original.startsWith("-"))
349        {
350            checks = additionalSuffixChecks;
351        }
352        syllables <<= 1;
353        do {
354            done = language.word(rng, false, syllables >> 1, checks);
355            if(cacheLevel < 2 || ++syllables > 5)
356                break;
357        }while(reverse.containsKey(done));
358        switch (cacheLevel) {
359            case 2: reverse.put(done, original);
360            case 1: table.put(original, done);
361        }
362        return done;
363    }
364    /**
365     * Constructs a LanguageCipher that will use the given style of language generator to produce its text.
366     * @param language a FakeLanguageGen, typically one of the static constants in that class or a mix of them.
367     * @param shift any long; this will be used to alter the specific words generated unless it is 0
368     */
369    public NaturalLanguageCipher(FakeLanguageGen language, long shift)
370    {
371        this.shift = shift;
372        this.language = language.copy();
373        rs = new SemiRandom(0xDF58476D1CE4E5B9L + shift);
374        rng = new RNG(rs);
375        table = new HashMap<>(512);
376        reverse = new HashMap<>(512);
377        pluralSuffix = addPart("-s", 0);
378        nounySuffix = addPart("-y", 0);
379        nounicSuffix = addPart("-ic", 0);
380        nouniveSuffix = addPart("-ive", 0);
381        nounistSuffix = addPart("-ist", 0);
382        nounismSuffix = addPart("-ism", 1 + (rng.nextIntHasty(3) >> 1));
383        nounenSuffix = addPart("-en", 0);
384        verbedSuffix = addPart("-ed", 0);
385        verberSuffix = addPart("-er", 0);
386        verbingSuffix = addPart("-ing", 1);

438    public static long phoneticHash64(char[] data, int start, int end)
439    {
440        long h = 0, b = 0;
441        if(data == null || end <= start || start >= data.length)
442            return h;
443        int current, next, count = 0, used = 0;
444        long got, vc = 0;
445        boolean vowelStream = false;
446        for (int i = start; i < end && count < 10; i++, count++) { // && vc < 7
447            current = data[i] - 'a';
448            if(current > 26)
449                continue;
450            if(i + 1 < end) {
451                if((next = data[i + 1] - 'a') > 26)
452                    continue;
453            }
454            else
455                next = 26;
456            got = bigrams[27 * current + next];
457            if(got == 0)
458                continue;
459            h <<= 6;
460            //b <<= 3;
461            got = bigrams[27 * current + next];
462            i += got & 1L;
463            h |= got >>= 1;
464            used++;
465            //used += 5;
466            if(count == 0) {
467                vowelStream = got > 0 && got < 12;
468            }else if (vowelStream != (got > 0 && got < 12)) {
469                vc+= vowelStream ? 1 : 0;
470                vowelStream = !vowelStream;
471            }
472            //b |= (got >> 2);
473        }
474        vc += vowelStream ? 1 : 0;
475
476        if(used > 0 && count > 0) {
477            got = h;
478            for (; count < 11; count += used) {
479                h |= got << (6 * count);
480            }
481            h &= 0xFFFFFFFFFFFFFFFL; // 60 bits
482        }
483        /*b &= ~(-1 << (35-Math.min(used, 35)));
484        if(used <= 20)
485            b ^= b << 8;
486        h ^= ((vc & 7L) << 39) | (b << (used + 3));
487        */
488        vc = Math.max(1L, vc);
489        return h | ((vc & 15L) << 60);
490    }

575    /**
576     * Given a word in the source language (usually English), looks up an existing translation for that word, or if none
577     * exists, generates a new word based on the hash of the source word and this LanguageCipher's FakeLanguageGen.
578     * @param source a word in the source language
579     * @return a word in the fake language
580     */
581    public String lookup(String source)
582    {
583        if(source == null || source.isEmpty())
584            return "";
585        String s2 = source.toLowerCase(), ciphered;
586        if(table.containsKey(s2))
587            ciphered = table.get(s2);
588        else {
589            CharSequence altered = FakeLanguageGen.removeAccents(s2);
590            for (int i = 0; i < preproc.length; i++) {
591                altered = preproc[i].replace(altered);
592            }
593
594            char[] sc = ((String)altered).toCharArray(), scO = s2.toCharArray();
595            int start = 0, end = sc.length, endO = scO.length;
596            long mods = 0;
597            /*
598            boolean plural = false, verbing = false, verbed = false, verber = false, verbation = false,
599                    verbment = false, nouny = false, nounen = false, nounist = false, nounism = false,
600                    nounic = false, nounive = false, adjectively = false, adjectivest = false,
601                    //prefixes
602                    reverb = false, ennoun = false, preverb = false, postverb = false,
603                    proverb = false, antiverb = false, disnoun = false;
604            */
605            if(end >= 4 && endO >= 4 && sc[end-1]=='s')
606            {
607                mods |= PLURAL;
608                end--;
609                endO--;
610                if(scO[endO-1] == 'e')
611                {
612                    end--;
613                    endO--;
614                }
615            }
616            if(end >= 5 && endO >= 5 && sc[end - 2] == 'l' && sc[end-1] == 'y')
617            {
618                mods |= ADJECTIVELY;
619                end -= 2;
620                endO -= 2;
621            }
622            /*
623            else if(end >= 4 && endO >= 4 && scO[endO-1] == 'y')
624            {
625                mods |= NOUNY;
626                end--;
627                endO--;
628            }*/
629            if(end >= 5 && endO >= 5 && scO[endO-3] == 'i' && scO[endO-2] == 'n' && scO[endO-1]=='g')
630            {
631                mods |= VERBING;
632                end-=3;
633                endO -= 3;
634            }
635            if(end >= 4 && endO >= 4 && (scO[endO-3] == 'a' || scO[endO-3] == 'o') && scO[endO-2] == 'd' && scO[endO-1]=='e')
636            {
637                mods |= VERBED;
638                end-=3;
639                endO-=3;
640            }
641            else if(end >= 4 && endO >= 4 && scO[endO-2] == 'e' && scO[endO-1] == 'd')
642            {
643                mods |= VERBED;
644                end-=2;
645                endO-=2;
646            }
647            else if(end >= 5 && endO >= 5 && sc[end - 3] == 'e' && sc[end - 2] == 's' && sc[end-1] == 't')
648            {
649                mods |= ADJECTIVEST;
650                end -= 3;
651                endO -= 3;
652            }
653            if(end >= 5 && endO >= 5 && scO[endO-2] == 'i' && scO[endO-1] == 'c')
654            {
655                mods |=NOUNIC;
656                end -= 2;
657                endO-=2;
658            }
659            else if(end >= 6 && endO >= 6 && scO[endO-3] == 'i' && scO[endO-2] == 'v' && scO[endO-1] == 'e') {
660                mods |= NOUNIVE;
661                end -= 3;
662                endO -= 3;
663                if (end >= 4 && endO >= 4 && (scO[endO - 2] == 'a' || scO[endO - 2] == 'i') && scO[endO - 1] == 't') {
664                    end -= 2;
665                    endO -= 2;
666                }
667            }
668            if(end >= 5 && sc[end-3] == 'i' && sc[end-2] == 's' && sc[end-1] == 't')
669            {
670                mods |=NOUNIST;
671                end -= 3;
672                endO-=3;
673                if(endO >= 5 && scO[endO-2] == 'i' && scO[endO-1] == 'v')
674                {
675                    mods |= NOUNIVE;
676                    end-=2;
677                    endO-=2;
678                }
679            }
680            if(end >= 5 && sc[end-3] == 'i' && sc[end-2] == 's' && sc[end-1] == 'm')
681            {
682                mods |=NOUNISM;
683                end -= 3;
684                endO -= 3;
685                if(endO >= 5 && scO[endO-2] == 'i' && scO[endO-1] == 'v')
686                {
687                    mods |= NOUNIVE;
688                    end-=2;
689                    endO-=2;
690                }
691
692            }
693            if(end >= 8 && endO >= 8 && (scO[endO - 4] == 't' || scO[endO - 4] == 's' || scO[endO - 4] == 'c') && scO[endO-3] == 'i' && scO[endO-2] == 'o' && scO[endO-1]=='n')
694            {
695                mods |=VERBATION;
696                end-=4;
697                endO -= 4;
698            }
699            if(end >= 6 && sc[end-4] == 'm' && sc[end-3] == 'e' && sc[end-2] == 'n' && sc[end-1] == 't')
700            {
701                mods |=VERBMENT;
702                end-=4;
703                endO -= 4;
704            }
705            if(end >= 7 && endO >= 7 && scO[endO-3] == 'i' && scO[endO-2] == 'a' && scO[endO-1]=='n')
706            {
707                mods |=VERBER;
708                end-=3;
709                endO-=3;
710            }
711            else if(end >= 4 && endO >= 4 && (sc[end-2] == 'e' || sc[end-2] == 'o') && sc[end-1] == 'r')
712            {
713                mods |= VERBER;
714                end-=2;
715            }
716            if(end >= 4 && sc[end-2] == 'e' && sc[end-1]=='n')
717            {
718                mods |=NOUNEN;
719                end-=2;
720            }
721            if(end - start >= 5 && sc[start] == 'p' && sc[start+1] == 'r' && sc[start+2] == 'e')
722            {
723                mods |=PREVERB;
724                start += 3;
725            }
726            if(end - start >= 6 && sc[start] == 'p' && sc[start+1] == 'o' && sc[start+2] == 's' && sc[start+3] == 't')
727            {
728                mods |= POSTVERB;
729                start += 4;
730            }
731
732            if(end - start >= 5 && sc[start] == 'p' && sc[start+1] == 'r' && sc[start+2] == 'o')
733            {
734                mods |= PROVERB;
735                start += 3;
736            }
737            else {
738                if (end - start >= 6 && sc[start] == 'a' && sc[start + 1] == 'n' && sc[start + 2] == 't' && sc[start + 3] == 'i') {
739                    mods |= ANTIVERB;
740                    start += 4;
741                }
742                else if (end - start >= 8 && sc[start] == 'c' && sc[start + 1] == 'o' && sc[start + 2] == 'n' && sc[start + 3] == 't' && sc[start + 4] == 'r' && sc[start + 5] == 'a') {
743                    mods |= ANTIVERB;
744                    start += 6;
745                }
746            }
747            if(end - start >= 4 && sc[start] == 'r' && sc[start+1] == 'e')
748            {
749                mods |= REVERB;
750                start += 2;
751            }
752            if(end - start >= 5 && sc[start] == 'd' && sc[start+1] == 'i' && sc[start+2] == 's')
753            {
754                mods |= DISNOUN;
755                start += 3;
756            }
757            if(end - start >= 4 && sc[start] == 'u' && sc[start+1] == 'n')
758            {
759                mods |= ANTIVERB;
760                start += 2;
761            }
762            if(end - start >= 4 && (sc[start] == 'e' || sc[start] == 'i') && sc[start+1] == 'n')
763            {
764                mods |= ENNOUN;
765                start += 2;
766            }
767            long h = phoneticHash64(sc, start, end) ^ (shift & 0xFFFFFFFFFFFFFFFL) ^ (shift >>> 14), frustration = 0;
768            //System.out.print(source + ":" + ((h >>> 60) & 7) + ":" + StringKit.hex(h) + ", ");
769            rs.setState(h);
770            do {
771                ciphered = conjugate(language.word(rng, false, (int) Math.ceil((h >>> 60) / (0.9 + 0.5 * rng.nextDouble()))), mods);
772                if(cacheLevel < 2 || frustration++ > 9)
773                    break;
774            }while (reverse.containsKey(ciphered));
775            switch (cacheLevel) {
776                case 2: reverse.put(ciphered, s2);
777                case 1: table.put(s2, ciphered);
778            }
779        }
780        char[] chars = ciphered.toCharArray();
781        // Lu is the upper case letter category in Unicode; we're using regexodus for this because GWT won't
782        // respect unicode case data on its own (see
783        // https://github.com/gwtproject/gwt/blob/2.6.1/user/super/com/google/gwt/emul/java/lang/Character.java#L54-L61
784        // ). We are using GWT to capitalize, though, which appears to work in practice and the docs agree.
785        if(Category.Lu.contains(source.charAt(0)))
786            chars[0] = Character.toUpperCase(chars[0]);
787        if(source.length() > 1 && Category.Lu.contains(source.charAt(1))) {
788            for (int i = 1; i < chars.length; i++) {
789                chars[i] = Character.toUpperCase(chars[i]);
790            }
791        }
792        return new String(chars);
793    }

797     * translates each word to the fake language, using existing translations if previous calls to cipher() or lookup()

808    private class CipherSubstitution implements Substitution
809    {
810        @Override
811        public void appendSubstitution(MatchResult match, TextBuffer dest) {
812            dest.append(lookup(match.group(0)));
813        }
814    }
815    private class DecipherSubstition implements Substitution
816    {
817        private final Map<String, String> vocabulary;
818        DecipherSubstition(final Map<String, String> vocabulary)
819        {
820            this.vocabulary = vocabulary;
821        }
822        public void appendSubstitution(MatchResult match, TextBuffer dest) {
823            String translated = match.group(0);
824            if(translated == null) {
825                return;
826            }
827            translated = translated.toLowerCase();
828            translated = vocabulary.get(translated);
829            if(translated == null) {
830                dest.append(match.group(0));
831                return;
832            }
833            char[] chars = translated.toCharArray();
834            if(Category.Lu.contains(match.charAt(0)))
835                chars[0] = Character.toUpperCase(chars[0]);
836            if(match.length() > 1 && Category.Lu.contains(match.charAt(1))) {
837                for (int i = 1; i < chars.length; i++) {
838                    chars[i] = Character.toUpperCase(chars[i]);
839                }
840            }
841            dest.append(chars, 0, chars.length);
842        }
843    }

845    /**
846     * Deciphers words in an already-ciphered text with a given String-to-String Map for a vocabulary. This Map could be
847     * the reverse field of this LanguageCipher, which would give a complete translation, or it could be a
848     * partially-complete or partially-correct vocabulary of words the player has learned. The vocabulary should
849     * typically have entries added using the quick and accurate learnTranslations() method, unless you want to add
850     * translations one word at a time (then use learnTranslation() ) or you want incorrect or biased translations added
851     * (then use mismatchTranslation() ). You don't need to use one of these methods if you just pass the whole of the
852     * reverse field as a vocabulary, which will translate every word. If making your own vocabulary without the learn
853     * methods, the keys need to be lower-case because while regex Patterns can be case-insensitive, Map lookups cannot.
854     * @param text a text in the fake language
855     * @param vocabulary a Map of Strings in the fake language to Strings in the source language
856     * @return a deciphered version of text that has any words as keys in vocabulary translated to the source language
857     */
858    public String decipher(String text, final Map<String, String> vocabulary)
859    {
860        Pattern pat;
861        Replacer rep;
862        StringBuilder sb = new StringBuilder(128);
863        sb.append("(?:");
864        for(String k : vocabulary.keySet())
865        {
866            sb.append("(?:\\Q");
867            sb.append(k);
868            sb.append("\\E)|");
869        }
870        sb.deleteCharAt(sb.length() - 1);
871        sb.append(')');
872
873        pat = Pattern.compile("(?<![\\pL\\&-])(?=[\\pL\\&-])" + sb + "(?![\\pL\\&-])", "ui");
874
875        rep = pat.replacer(new DecipherSubstition(vocabulary));
876        return rep.replace(text);
877    }
