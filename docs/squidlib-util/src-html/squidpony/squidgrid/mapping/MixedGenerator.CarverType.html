016/**
017 * A dungeon generator that can use a mix of techniques to have part-cave, part-room dungeons. Not entirely intended for
018 * normal use outside of this library, though it can be very useful when you want to make a dungeon match a specific
019 * path and existing generators that use MixedGenerator aren't sufficient. You may want to use a simpler generator based
020 * on this, like SerpentMapGenerator, which generates a long, winding path that loops around on itself. This supports
021 * the getEnvironment() method, which can be used in conjunction with RoomFinder to find where separate room, corridor,
022 * and cave areas have been placed.
023 * <br>
024 * Based on Michael Patraw's excellent Drunkard's Walk dungeon generator.
025 * http://mpatraw.github.io/libdrunkard/
026 * @see squidpony.squidgrid.mapping.SerpentMapGenerator a normal use for MixedGenerator that makes winding dungeons
027 * @see squidpony.squidgrid.mapping.SerpentDeepMapGenerator uses MixedGenerator as it makes a multi-level dungeon
028 * Created by Tommy Ettinger on 10/22/2015.
029 */
030public class MixedGenerator {
031    public enum CarverType
032    {
033        CAVE,
034        BOX,
035        ROUND,
036        BOX_WALLED,
037        ROUND_WALLED
038    }

080    /**
081     * Internal use.
082     * @param width dungeon width in cells
083     * @param height dungeon height in cells
084     * @param rng rng to use
085     * @return evenly spaced Coord points in a list made by PoissonDisk, trimmed down so they aren't all used
086     * @see PoissonDisk used to make the list
087     */
088    protected static List<Coord> basicPoints(int width, int height, RNG rng)
089    {
090        return PoissonDisk.sampleRectangle(Coord.get(2, 2), Coord.get(width - 3, height - 3),
091                8.5f * (width + height) / 120f, width, height, 35, rng);
092    }

094    /**
095     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
096     * This version of the constructor uses Poisson Disk sampling to generate the points it will draw caves and
097     * corridors between, ensuring a minimum distance between points, but it does not ensure that paths between points
098     * will avoid overlapping with rooms or other paths. You call the different carver-adding methods to affect what the
099     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
100     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
101     * @param width the width of the final map in cells
102     * @param height the height of the final map in cells
103     * @param rng an RNG object to use for random choices; this make a lot of random choices.
104     * @see PoissonDisk used to ensure spacing for the points.
105     */
106    public MixedGenerator(int width, int height, RNG rng) {
107        this(width, height, rng, basicPoints(width, height, rng));
108    }

110    /**
111     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
112     * This version of the constructor uses a List of Coord points from some other source to determine the path to add
113     * rooms or caves to and then connect. You call the different carver-adding methods to affect what the
114     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
115     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
116     * @param width the width of the final map in cells
117     * @param height the height of the final map in cells
118     * @param rng an RNG object to use for random choices; this make a lot of random choices.
119     * @param sequence a List of Coord to connect in order; index 0 is the start, index size() - 1 is the end.
120     * @see SerpentMapGenerator a class that uses this technique
121     */
122    public MixedGenerator(int width, int height, RNG rng, List<Coord> sequence) {
123        this.width = width;
124        this.height = height;
125        this.roomWidth = width / 64.0f;
126        this.roomHeight = height / 64.0f;
127        if(width <= 2 || height <= 2)
128            throw new IllegalStateException("width and height must be greater than 2");
129        this.rng = rng;
130        dungeon = new char[width][height];
131        environment = new int[width][height];
132        marked = new boolean[width][height];
133        walled = new boolean[width][height];
134        fixedRooms = new boolean[width][height];
135        Arrays.fill(dungeon[0], '#');
136        Arrays.fill(environment[0], UNTOUCHED);
137        for (int i = 1; i < width; i++) {
138            System.arraycopy(dungeon[0], 0, dungeon[i], 0, height);
139            System.arraycopy(environment[0], 0, environment[i], 0, height);
140        }
141        totalPoints = sequence.size() - 1;
142        points = new IntVLA(totalPoints);
143        for (int i = 0; i < totalPoints; i++) {
144            Coord c1 = sequence.get(i), c2 = sequence.get(i + 1);
145            points.add(((c1.x & 0xff) << 24) | ((c1.y & 0xff) << 16) | ((c2.x & 0xff) << 8) | (c2.y & 0xff));
146        }
147        carvers = new EnumMap<>(CarverType.class);
148    }
149    /**
150     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
151     * This version of the constructor uses a Map with Coord keys and Coord array values to determine a
152     * branching path for the dungeon to take; each key will connect once to each of the Coords in its value, and you
153     * usually don't want to connect in both directions. You call the different carver-adding methods to affect what the
154     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
155     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
156     * @param width the width of the final map in cells
157     * @param height the height of the final map in cells
158     * @param rng an RNG object to use for random choices; this make a lot of random choices.
159     * @param connections a Map of Coord keys to arrays of Coord to connect to next; shouldn't connect both ways
160     * @see SerpentMapGenerator a class that uses this technique
161     */
162    public MixedGenerator(int width, int height, RNG rng, Map<Coord, List<Coord>> connections)
163    {
164        this(width, height, rng, connections, 0.8f);
165    }
166    /**
167     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
168     * This version of the constructor uses a Map with Coord keys and Coord array values to determine a
169     * branching path for the dungeon to take; each key will connect once to each of the Coords in its value, and you
170     * usually don't want to connect in both directions. You call the different carver-adding methods to affect what the
171     * dungeon will look like, putCaveCarvers(), putBoxRoomCarvers(), and putRoundRoomCarvers(), defaulting to only
172     * caves if none are called. You call generate() after adding carvers, which returns a char[][] for a map.
173     * @param width the width of the final map in cells
174     * @param height the height of the final map in cells
175     * @param rng an RNG object to use for random choices; this make a lot of random choices.
176     * @param connections a Map of Coord keys to arrays of Coord to connect to next; shouldn't connect both ways
177     * @param roomSizeMultiplier a float multiplier that will be applied to each room's width and height
178     * @see SerpentMapGenerator a class that uses this technique
179     */
180    public MixedGenerator(int width, int height, RNG rng, Map<Coord, List<Coord>> connections,
181                          float roomSizeMultiplier) {
182        this.width = width;
183        this.height = height;
184        roomWidth = (width / 64.0f) * roomSizeMultiplier;
185        roomHeight = (height / 64.0f) * roomSizeMultiplier;
186        if(width <= 2 || height <= 2)
187            throw new IllegalStateException("width and height must be greater than 2");
188        this.rng = rng;
189        dungeon = new char[width][height];
190        environment = new int[width][height];
191        marked = new boolean[width][height];
192        walled = new boolean[width][height];
193        fixedRooms = new boolean[width][height];
194        Arrays.fill(dungeon[0], '#');
195        Arrays.fill(environment[0], UNTOUCHED);
196        for (int i = 1; i < width; i++) {
197            System.arraycopy(dungeon[0], 0, dungeon[i], 0, height);
198            System.arraycopy(environment[0], 0, environment[i], 0, height);
199        }
200        totalPoints = 0;
201        for(List<Coord> vals : connections.values())
202        {
203            totalPoints += vals.size();
204        }
205        points = new IntVLA(totalPoints);
206        for (Map.Entry<Coord, List<Coord>> kv : connections.entrySet()) {
207            Coord c1 = kv.getKey();
208            for (Coord c2 : kv.getValue()) {
209                points.add(((c1.x & 0xff) << 24) | ((c1.y & 0xff) << 16) | ((c2.x & 0xff) << 8) | (c2.y & 0xff));
210            }
211        }
212        carvers = new EnumMap<>(CarverType.class);
213    }

215    /**
216     * This prepares a map generator that will generate a map with the given width and height, using the given RNG.
217     * This version of the constructor uses an {@link squidpony.squidgrid.mapping.locks.IRoomLayout} to set up rooms,
218     * almost always produced by {@link squidpony.squidgrid.mapping.locks.generators.LayoutGenerator}. This method does
219     * alter the individual Room objects inside layout, making the center of each room match where that center is placed
220     * in the dungeon this generates. You call the different carver-adding methods to affect what the dungeon will look
221     * like, i.e. {@link #putCaveCarvers(int)}, {@link #putBoxRoomCarvers(int)} , {@link #putRoundRoomCarvers(int)},
222     * {@link #putWalledBoxRoomCarvers(int)}, and {@link #putWalledRoundRoomCarvers(int)}, defaulting to only caves if
223     * none are called (using rooms is recommended for this constructor). You call generate() after adding carvers,
224     * which returns a char[][] for a map and sets the environment to be fetched with {@link #getEnvironment()}, which
225     * is usually needed for {@link SectionDungeonGenerator} to correctly place doors and various other features.
226     * @param width the width of the final map in cells
227     * @param height the height of the final map in cells
228     * @param rng an RNG object to use for random choices; this make a lot of random choices.
229     * @param layout an IRoomLayout that will almost always be produced by LayoutGenerator; the rooms will be altered
230     * @param roomSizeMultiplier a float multiplier that will be applied to each room's width and height
231     * @see SerpentMapGenerator a class that uses this technique
232     */
233    public MixedGenerator(int width, int height, RNG rng, IRoomLayout layout,
234                          float roomSizeMultiplier) {
235        this.width = width;
236        this.height = height;
237        Rect2I bounds = layout.getExtentBounds();
238        int offX = bounds.getBottomLeft().x, offY = bounds.getBottomLeft().y;
239        float rw = (width) / (bounds.width+1f), rh = (height) / (bounds.height+1f);
240        this.roomWidth = roomSizeMultiplier * rw * 0.125f;
241        this.roomHeight = roomSizeMultiplier * rh * 0.125f;
242        if(width <= 2 || height <= 2)
243            throw new IllegalStateException("width and height must be greater than 2");
244        this.rng = rng;
245        dungeon = new char[width][height];
246        environment = new int[width][height];
247        marked = new boolean[width][height];
248        walled = new boolean[width][height];
249        fixedRooms = new boolean[width][height];
250        ArrayTools.fill(dungeon, '#');
251        ArrayTools.fill(environment, UNTOUCHED);
252        totalPoints = layout.roomCount();
253        points = new IntVLA(totalPoints);
254        Coord c2;
255        Set<Room> rooms = layout.getRooms(), removing = new HashSet<>(rooms);
256        Room t;
257        for (Room room : rooms) {
258            Coord c1 = room.getCenter();
259            if (!bounds.contains(c1)) {
260                removing.remove(room);
261            }
262            else {
263                room.setCenter(Coord.get(
264                        (int) ((c1.x - offX + 0.75f) * (rw)) & 0xff,
265                        (int) ((c1.y - offY + 0.75f) * (rh)) & 0xff));
266            }
267        }
268
269        for (Room room : rooms) {
270            Coord c1 = room.getCenter();
271            for (Edge e : room.getEdges()) {
272                if (removing.contains(t = layout.get(e.getTargetRoomId()))) {
273                    c2 = t.getCenter();
274                    points.add((c1.x << 24)
275                            | (c1.y << 16)
276                            | (c2.x << 8)
277                            | c2.y);
278                }
279            }
280            removing.remove(room);
281        }
282        totalPoints = points.size;
283        carvers = new EnumMap<>(CarverType.class);
284    }

286    /**
287     * Changes the number of "carvers" that will create caves from one room to the next. If count is 0 or less, no caves
288     * will be made. If count is at least 1, caves are possible, and higher numbers relative to the other carvers make
289     * caves more likely. Carvers are shuffled when used, then repeat if exhausted during generation. Since typically
290     * about 30-40 rooms are carved, large totals for carver count aren't really needed; aiming for a total of 10
291     * between the count of putCaveCarvers(), putBoxRoomCarvers(), putRoundRoomCarvers(), putWalledBoxRoomCarvers(), and
292     * putWalledRoundRoomCarvers() is reasonable.
293     * @param count the number of carvers making caves between rooms; only matters in relation to other carvers
294     */
295    public void putCaveCarvers(int count)
296    {
297        carvers.put(CarverType.CAVE, count);
298    }

366    /**
367     * Uses the added carvers (or just makes caves if none were added) to carve from point to point in sequence, if it
368     * was provided by the constructor, or evenly-spaced randomized points if it was not. This will never carve out
369     * cells on the very edge of the map. Uses the numbers of the various kinds of carver that were added relative to
370     * each other to determine how frequently to use a given carver type.
371     * @return a char[][] where '#' is a wall and '.' is a floor or corridor; x first y second
372     */
373    public char[][] generate()
374    {
375        CarverType[] carvings = carvers.keySet().toArray(new CarverType[carvers.size()]);
376        int[] carvingsCounters = new int[carvings.length];
377        int totalLength = 0;
378        for (int i = 0; i < carvings.length; i++) {
379            carvingsCounters[i] = carvers.get(carvings[i]);
380            totalLength += carvingsCounters[i];
381        }
382        CarverType[] allCarvings = new CarverType[totalLength];
383
384        for (int i = 0, c = 0; i < carvings.length; i++) {
385            for (int j = 0; j < carvingsCounters[i]; j++) {
386                allCarvings[c++] = carvings[i];
387            }
388        }
389        if(allCarvings.length == 0)
390        {
391            allCarvings = new CarverType[]{CarverType.CAVE};
392            totalLength = 1;
393        }
394        else
395            allCarvings = rng.shuffle(allCarvings, new CarverType[allCarvings.length]);
396
397        for (int p = 0, c = 0; p < totalPoints; p++, c = (c+1) % totalLength) {
398            int pair = points.get(p);
399            Coord start = Coord.get(pair >>> 24 & 0xff, pair >>> 16 & 0xff),
400                  end   = Coord.get(pair >>> 8 & 0xff, pair & 0xff);
401            CarverType ct = allCarvings[c];
402            Direction dir;
403            switch (ct)
404            {
405                case CAVE:
406                    markPiercing(end);
407                    markEnvironmentCave(end.x, end.y);
408                    store();
409                    double weight = 0.75;
410                    do {
411                        Coord cent = markPlusCave(start);
412                        if(cent != null)
413                        {
414                            markPiercingCave(cent);
415                            markPiercingCave(cent.translate(1, 0));
416                            markPiercingCave(cent.translate(-1, 0));
417                            markPiercingCave(cent.translate(0, 1));
418                            markPiercingCave(cent.translate(0, -1));
419                            weight = 0.95;
420                        }
421                        dir = stepWobbly(start, end, weight);
422                        start = start.translate(dir);
423                    }while (dir != Direction.NONE);
424                    break;
425                case BOX:
426                    markRectangle(end, rng.between(1, 5), rng.between(1, 5));
427                    markRectangle(start, rng.between(1, 4), rng.between(1, 4));
428                    store();
429                    dir = Direction.getDirection(end.x - start.x, end.y - start.y);
430                    if(dir.isDiagonal())
431                        dir = rng.nextBoolean() ? Direction.getCardinalDirection(dir.deltaX, 0)
432                                : Direction.getCardinalDirection(0, -dir.deltaY);
433                    while (start.x != end.x && start.y != end.y)
434                    {
435                        markPiercing(start);
436                        markEnvironmentCorridor(start.x, start.y);
437                        start = start.translate(dir);
438                    }
439                    markRectangle(start, 1, 1);
440                    dir = Direction.getCardinalDirection(end.x - start.x, -(end.y - start.y));
441                    while (!(start.x == end.x && start.y == end.y))
442                    {
443                        markPiercing(start);
444                        markEnvironmentCorridor(start.x, start.y);
445                        start = start.translate(dir);
446                    }
447                    break;
448                case BOX_WALLED:
449                    markRectangleWalled(end, rng.between(1, 5), rng.between(1, 5));
450                    markRectangleWalled(start, rng.between(1, 4), rng.between(1, 4));
451                    store();
452                    dir = Direction.getDirection(end.x - start.x, end.y - start.y);
453                    if(dir.isDiagonal())
454                        dir = rng.nextBoolean() ? Direction.getCardinalDirection(dir.deltaX, 0)
455                                : Direction.getCardinalDirection(0, -dir.deltaY);
456                    while (start.x != end.x && start.y != end.y)
457                    {
458                        markPiercing(start);
459                        markEnvironmentCorridor(start.x, start.y);
460                        start = start.translate(dir);
461                    }
462                    markRectangleWalled(start, 1, 1);
463                    dir = Direction.getCardinalDirection(end.x - start.x, -(end.y - start.y));
464                    while (!(start.x == end.x && start.y == end.y))
465                    {
466                        markPiercing(start);
467                        markEnvironmentCorridor(start.x, start.y);
468                        start = start.translate(dir);
469                    }
470                    break;
471                case ROUND:
472                    markCircle(end, rng.between(2, 6));
473                    markCircle(start, rng.between(2, 6));
474                    store();
475                    dir = Direction.getDirection(end.x - start.x, end.y - start.y);
476                    if(dir.isDiagonal())
477                        dir = rng.nextBoolean() ? Direction.getCardinalDirection(dir.deltaX, 0)
478                                : Direction.getCardinalDirection(0, -dir.deltaY);
479                    while (start.x != end.x && start.y != end.y)
480                    {
481                        markPiercing(start);
482                        markEnvironmentCorridor(start.x, start.y);
483                        start = start.translate(dir);
484                    }
485                    markCircle(start, 2);
486                    dir = Direction.getCardinalDirection(end.x - start.x, -(end.y - start.y));
487                    while (!(start.x == end.x && start.y == end.y))
488                    {
489                        markPiercing(start);
490                        markEnvironmentCorridor(start.x, start.y);
491                        start = start.translate(dir);
492                    }
493                    break;
494                case ROUND_WALLED:
495                    markCircleWalled(end, rng.between(2, 6));
496                    markCircleWalled(start, rng.between(2, 6));
497                    store();
498                    dir = Direction.getDirection(end.x - start.x, end.y - start.y);
499                    if(dir.isDiagonal())
500                        dir = rng.nextBoolean() ? Direction.getCardinalDirection(dir.deltaX, 0)
501                                : Direction.getCardinalDirection(0, -dir.deltaY);
502                    while (start.x != end.x && start.y != end.y)
503                    {
504                        markPiercing(start);
505                        markEnvironmentCorridor(start.x, start.y);
506                        start = start.translate(dir);
507                    }
508                    markCircleWalled(start, 2);
509                    dir = Direction.getCardinalDirection(end.x - start.x, -(end.y - start.y));
510                    while (!(start.x == end.x && start.y == end.y))
511                    {
512                        markPiercing(start);
513                        markEnvironmentCorridor(start.x, start.y);
514                        start = start.translate(dir);
515                    }
516                    break;
517            }
518            store();
519        }
520        for (int x = 0; x < width; x++) {
521            for (int y = 0; y < height; y++) {
522                if(fixedRooms[x][y])
523                    markPiercingRoom(x, y);
524            }
525        }
526        store();
527        markEnvironmentWalls();
528        generated = true;
529        return dungeon;
530    }

550    /**
551     * Internal use. Takes cells that have been previously marked and permanently stores them as floors in the dungeon.
552     */
553    protected void store()
554    {
555        for (int i = 0; i < width; i++) {
556            for (int j = 0; j < height; j++) {
557                if(marked[i][j])
558                {
559                    dungeon[i][j] = '.';
560                    marked[i][j] = false;
561                }
562            }
563        }
564    }

566    /**
567     * Internal use. Finds all floor cells by environment and marks untouched adjacent (8-way) cells as walls, using the
568     * appropriate type for the nearby floor.
569     */
570    protected void markEnvironmentWalls()
571    {
572        for (int i = 0; i < width; i++) {
573            for (int j = 0; j < height; j++) {
574                if(environment[i][j] == UNTOUCHED)
575                {
576                    boolean allWalls = true;
577                    //lowest precedence, also checks for any floors
578                    for (int x = Math.max(0, i - 1); x <= Math.min(width - 1, i + 1); x++) {
579
580                        for (int y = Math.max(0, j - 1); y <= Math.min(height - 1, j + 1); y++) {
581                            if (environment[x][y] == CORRIDOR_FLOOR) {
582                                markEnvironment(i, j, CORRIDOR_WALL);
583                            }
584                            if(dungeon[x][y] == '.')
585                                allWalls = false;
586                        }
587                    }
588                    //if there are no floors we don't need to check twice again.
589                    if(allWalls)
590                        continue;
591                    //more precedence
592                    for (int x = Math.max(0, i - 1); x <= Math.min(width - 1, i + 1); x++) {
593
594                        for (int y = Math.max(0, j - 1); y <= Math.min(height - 1, j + 1); y++) {
595                            if (environment[x][y] == CAVE_FLOOR) {
596                                markEnvironment(i, j, CAVE_WALL);
597                            }
598                        }
599                    }
600                    //highest precedence
601                    for (int x = Math.max(0, i - 1); x <= Math.min(width - 1, i + 1); x++) {
602
603                        for (int y = Math.max(0, j - 1); y <= Math.min(height - 1, j + 1); y++) {
604                            if (environment[x][y] == ROOM_FLOOR) {
605                                markEnvironment(i, j, ROOM_WALL);
606                            }
607                        }
608                    }
609
610                }
611            }
612        }
613    }

732     * @param pos center position to mark
733     * @return null if the center of the plus shape wasn't blocked by wall, otherwise the Coord of the center
734     */
735    private Coord markPlus(Coord pos) {
736        Coord block = null;
737        if (mark(pos.x, pos.y))
738            block = pos;
739        mark(pos.x + 1, pos.y);
740        mark(pos.x - 1, pos.y);
741        mark(pos.x, pos.y + 1);
742        mark(pos.x, pos.y - 1);

749     * @param pos center position to mark

768    /**
769     * Internal use. Marks a rectangle of points centered on pos, extending halfWidth in both x directions and
770     * halfHeight in both vertical directions. Marks all cells in the rectangle as room floors.
771     * @param pos center position to mark
772     * @param halfWidth the distance from the center to extend horizontally
773     * @param halfHeight the distance from the center to extend vertically
774     * @return null if no points in the rectangle were blocked by walls, otherwise a Coord blocked by a wall
775     */
776    private Coord markRectangle(Coord pos, int halfWidth, int halfHeight)
777    {
778        halfWidth = Math.max(1, Math.round(halfWidth * roomWidth));
779        halfHeight = Math.max(1, Math.round(halfHeight * roomHeight));
780        Coord block = null;
781        for (int i = pos.x - halfWidth; i <= pos.x + halfWidth; i++) {
782            for (int j = pos.y - halfHeight; j <= pos.y + halfHeight; j++) {
783                if(mark(i, j))
784                    block = Coord.get(i, j);
785                else
786                    markEnvironmentRoom(i, j);
787            }
788        }
789        return block;

794     * a blocking wall that can only be passed by corridors and small cave openings. Marks all cells in the rectangle as
795     * room floors.

823    /**
824     * Internal use. Marks a circle of points centered on pos, extending out to radius in Euclidean measurement. Marks
825     * all cells in the circle as room floors.
826     * @param pos center position to mark
827     * @param radius radius to extend in all directions from center
828     * @return null if no points in the circle were blocked by walls, otherwise a Coord blocked by a wall
829     */
830    private Coord markCircle(Coord pos, int radius)
831    {
832        Coord block = null;
833        int high;
834        radius = Math.max(1, Math.round(radius * Math.min(roomWidth, roomHeight)));
835        for (int dx = -radius; dx <= radius; ++dx)
836        {
837            high = (int)Math.floor(Math.sqrt(radius * radius - dx * dx));
838            for (int dy = -high; dy <= high; ++dy)
839            {
840                if(mark(pos.x + dx, pos.y + dy))
841                    block = pos.translate(dx, dy);
842                else
843                    markEnvironmentRoom(pos.x + dx, pos.y + dy);
844            }
845        }
846        return block;
847    }
848    /**
849     * Internal use. Marks a circle of points centered on pos, extending out to radius in Euclidean measurement.
850     * Also considers the area just beyond each wall, but not corners, to be a blocking wall that can only be passed by
851     * corridors and small cave openings. Marks all cells in the circle as room floors.
852     * @param pos center position to mark
853     * @param radius radius to extend in all directions from center
854     * @return null if no points in the circle were blocked by walls, otherwise a Coord blocked by a wall
855     */
856    private Coord markCircleWalled(Coord pos, int radius)
857    {
858        Coord block = null;
859        int high;
860        radius = Math.max(1, Math.round(radius * Math.min(roomWidth, roomHeight)));
861        for (int dx = -radius; dx <= radius; ++dx)
862        {
863            high = (int)Math.floor(Math.sqrt(radius * radius - dx * dx));
864            for (int dy = -high; dy <= high; ++dy)
865            {
866                if(mark(pos.x + dx, pos.y + dy))
867                    block = pos.translate(dx, dy);
868                else
869                    markEnvironmentRoom(pos.x + dx, pos.y + dy);
870            }
871        }
872        for (int dx = -radius; dx <= radius; ++dx)
873        {
874            high = (int)Math.floor(Math.sqrt(radius * radius - dx * dx));
875            int dx2 = Math.max(1, Math.min(pos.x + dx, width - 2));
876            for (int dy = -high; dy <= high; ++dy)
877            {
878                int dy2 = Math.max(1, Math.min(pos.y + dy, height - 2));
879
880                wallOff(dx2, dy2-1);
881                wallOff(dx2+1, dy2-1);
882                wallOff(dx2-1, dy2-1);
883                wallOff(dx2, dy2);
884                wallOff(dx2+1, dy2);
885                wallOff(dx2-1, dy2);
886                wallOff(dx2, dy2+1);
887                wallOff(dx2+1, dy2+1);
888                wallOff(dx2-1, dy2+1);
889
890            }
891        }
892        return block;
893    }

895    /**
896     * Internal use. Drunkard's walk algorithm, single step. Based on Michael Patraw's C code, used for cave carving.
897     * http://mpatraw.github.io/libdrunkard/
898     * @param current the current point
899     * @param target the point to wobble towards
900     * @param weight between 0.5 and 1.0, usually. 0.6 makes very random caves, 0.9 is almost a straight line.
901     * @return a Direction, either UP, DOWN, LEFT, or RIGHT if we should move, or NONE if we have reached our target
902     */
903    private Direction stepWobbly(Coord current, Coord target, double weight)
904    {
905        int dx = target.x - current.x;
906        int dy = target.y - current.y;
907
908        if (dx >  1) dx = 1;
909        if (dx < -1) dx = -1;
910        if (dy >  1) dy = 1;
911        if (dy < -1) dy = -1;
912
913        double r = rng.nextDouble();
914        Direction dir;
915        if (dx == 0 && dy == 0)
916        {
917            return Direction.NONE;
918        }
919        else if (dx == 0 || dy == 0)
920        {
921            int dx2 = (dx == 0) ? dx : dy, dy2 = (dx == 0) ? dy : dx;
922            if (r >= (weight * 0.5))
923            {
924                r -= weight * 0.5;
925                if (r < weight * (1.0 / 6) + (1 - weight) * (1.0 / 3))
926                {
927                    dx2 = -1;
928                    dy2 = 0;
929                }
930                else if (r < weight * (2.0 / 6) + (1 - weight) * (2.0 / 3))
931                {
932                    dx2 = 1;
933                    dy2 = 0;
934                }
935                else
936                {
937                    dx2 = 0;
938                    dy2 *= -1;
939                }
940            }
941            dir = Direction.getCardinalDirection(dx2, -dy2);
942
943        }
944        else
945        {
946            if (r < weight * 0.5)
947            {
948                dy = 0;
949            }
950            else if (r < weight)
951            {
952                dx = 0;
953            }
954            else if (r < weight + (1 - weight) * 0.5)
955            {
956                dx *= -1;
957                dy = 0;
958            }
959            else
960            {
961                dx = 0;
962                dy *= -1;
963            }
964            dir = Direction.getCardinalDirection(dx, -dy);
965        }
966        if(current.x + dir.deltaX <= 0 || current.x + dir.deltaX >= width - 1) {
967            if (current.y < target.y) dir = Direction.DOWN;
968            else if (current.y > target.y) dir = Direction.UP;
969        }
970        else if(current.y + dir.deltaY <= 0 || current.y + dir.deltaY >= height - 1) {
971            if (current.x < target.x) dir = Direction.RIGHT;
972            else if (current.x > target.x) dir = Direction.LEFT;
973        }
974        return dir;
975    }
