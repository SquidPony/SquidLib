044    /**
045     * Randomly chooses a {@link Room} within the given collection that has at
046     * least one adjacent empty space.
047     * 
048     * @param roomCollection    the collection of rooms to choose from
049     * @return  the room that was chosen, or null if there are no rooms with
050     *          adjacent empty spaces
051     */
052    protected Room chooseRoomWithFreeEdge(Collection<Room> roomCollection,
053                                          int keyLevel) {
054        List<Room> rooms = new ArrayList<Room>(roomCollection);
055        random.shuffle(rooms);
056        Room room;
057        IntVLA near;
058        for (int i = 0; i < rooms.size(); ++i) {
059            room = rooms.get(i);
060            near = constraints.getAdjacentRooms(room.id, keyLevel);
061            for (int j = 0; j < near.size; j++) {
062                if (dungeon.get(near.get(j)) == null) {
063                    return room;
064                }
065            }
066        }
067        return null;
068    }
069    
070    /**
071     * Randomly chooses a {@link Direction} in which the given {@link Room} has
072     * an adjacent empty space.
073     * 
074     * @param room  the room
075     * @return  the Direction of the empty space chosen adjacent to the Room or
076     *          null if there are no adjacent empty spaces
077     */
078    protected int chooseFreeEdge(Room room, int keyLevel) {
079        IntVLA neighbors = new IntVLA(constraints.getAdjacentRooms(room.id, keyLevel));
080        neighbors.shuffle(random);
081        while (neighbors.size > 0) {
082            int choice = neighbors.getRandomElement(random);
083            if (dungeon.get(choice) == null)
084                return choice;
085            neighbors.removeValue(choice);
086        }
087        assert false;
088        throw new GenerationFailureException("Internal error: Room doesn't have a free edge");
089    }

091    /**
092     * Maps 'keyLevel' to the set of rooms within that keyLevel.
093     * <p>
094     * A 'keyLevel' is the count of the number of unique keys are needed for all
095     * the locks we've placed. For example, all the rooms in keyLevel 0 are
096     * accessible without collecting any keys, while to get to rooms in
097     * keyLevel 3, the player must have collected at least 3 keys.
098     */
099    protected class KeyLevelRoomMapping {
100        protected List<List<Room>> map = new ArrayList<List<Room>>(
101                constraints.getMaxKeys());
102        
103        List<Room> getRooms(int keyLevel) {
104            while (keyLevel >= map.size()) map.add(null);
105            if (map.get(keyLevel) == null)
106                map.set(keyLevel, new ArrayList<Room>());
107            return map.get(keyLevel);
108        }
109        
110        void addRoom(int keyLevel, Room room) {
111            getRooms(keyLevel).add(room);
112        }
113        
114        int keyCount() {
115            return map.size();
116        }
117    }

131    /**
132     * Comparator objects for sorting {@link Room}s in a couple of different
133     * ways. These are used to determine in which rooms of a given keyLevel it
134     * is best to place the next key.
135     * 
136     * @see #placeKeys
137     */
138    protected static final Comparator<Room>
139    EDGE_COUNT_COMPARATOR = new Comparator<Room>() {
140        @Override
141        public int compare(Room arg0, Room arg1) {
142            return arg0.linkCount() - arg1.linkCount();
143        }
144    },
145    INTENSITY_COMPARATOR = new Comparator<Room>() {
146        @Override
147        public int compare(Room arg0, Room arg1) {
148            return arg0.getIntensity() > arg1.getIntensity() ? -1
149                    : arg0.getIntensity() < arg1.getIntensity() ? 1
150                            : 0;
151        }
152    };

154    /**
155     * Sets up the dungeon's entrance room.
156     * 
157     * @param levels    the keyLevel -> room-set mapping to update
158     * @see KeyLevelRoomMapping 
159     */
160    protected void initEntranceRoom(KeyLevelRoomMapping levels)
161            throws RetryException {
162        int id;
163        IntVLA possibleEntries = constraints.initialRooms();
164        assert possibleEntries.size > 0;
165        id = possibleEntries.getRandomElement(random);
166        
167        Room entry = new Room(id, constraints.getCoords(id), null, Symbol.START, new Condition());
168        dungeon.add(entry);
169        
170        levels.addRoom(0, entry);
171    }

173    /**
174     * Decides whether to add a new lock (and keyLevel) at this point.
175     *
176     * @param keyLevel the number of distinct locks that have been placed into
177     *      the map so far
178     * @param numRooms the number of rooms at the current keyLevel
179     * @param targetRoomsPerLock the number of rooms the generator has chosen
180     *      as the target number of rooms to place at each keyLevel (which
181     *      subclasses can ignore, if desired).
182     */
183    protected boolean shouldAddNewLock(int keyLevel, int numRooms, int targetRoomsPerLock) {
184        int usableKeys = constraints.getMaxKeys();
185        if (isBossRoomLocked())
186            usableKeys -= 1;
187        return numRooms >= targetRoomsPerLock && keyLevel < usableKeys;
188    }

190    /**
191     * Fill the dungeon's space with rooms and doors (some locked).
192     * Keys are not inserted at this point.
193     * 
194     * @param levels    the keyLevel -> room-set mapping to update
195     * @throws RetryException if it fails
196     * @see KeyLevelRoomMapping
197     */
198    protected void placeRooms(KeyLevelRoomMapping levels, int roomsPerLock)
199            throws RetryException, OutOfRoomsException {
200        
201        // keyLevel: the number of keys required to get to the new room
202        int keyLevel = 0;
203        int latestKey = Symbol.NOTHING;
204        // condition that must hold true for the player to reach the new room
205        // (the set of keys they must have).
206        Condition cond = new Condition();
207        
208        // Loop to place rooms and link them
209        while (dungeon.roomCount() < constraints.getMaxRooms()) {
210            
211            boolean doLock = false;
212            
213            // Decide whether we need to place a new lock
214            // (Don't place the last lock, since that's reserved for the boss)
215            if (shouldAddNewLock(keyLevel, levels.getRooms(keyLevel).size(), roomsPerLock)) {
216                latestKey = keyLevel++;
217                cond = cond.and(latestKey);
218                doLock = true;
219            }
220            
221            // Find an existing room with a free edge:
222            Room parentRoom = null;
223            if (!doLock && random.nextIntHasty(10) > 0)
224                parentRoom = chooseRoomWithFreeEdge(levels.getRooms(keyLevel),
225                        keyLevel);
226            if (parentRoom == null) {
227                parentRoom = chooseRoomWithFreeEdge(dungeon.getRooms(),
228                        keyLevel);
229                doLock = true;
230            }
231            
232            if (parentRoom == null)
233                throw new OutOfRoomsException();
234            
235            // Decide which direction to put the new room in relative to the
236            // parent
237            int nextId = chooseFreeEdge(parentRoom, keyLevel);
238            Set<Coord> coords = constraints.getCoords(nextId);
239            Room room = new Room(nextId, coords, parentRoom, Symbol.NOTHING, cond);
240            
241            // Add the room to the dungeon
242            assert dungeon.get(room.id) == null;
243            //synchronized(dungeon) {
244            dungeon.add(room);
245            parentRoom.addChild(room);
246            dungeon.link(parentRoom, room, doLock ? latestKey : Symbol.NOTHING);
247            // }
248            levels.addRoom(keyLevel, room);
249        }
250    }

252    /**
253     * Places the BOSS and GOAL rooms within the dungeon, in existing rooms.
254     * These rooms are moved into the next keyLevel.
255     * 
256     * @param levels    the keyLevel -> room-set mapping to update
257     * @throws RetryException if it fails
258     * @see KeyLevelRoomMapping
259     */
260    protected void placeBossGoalRooms(KeyLevelRoomMapping levels)
261            throws RetryException {
262        List<Room> possibleGoalRooms = new ArrayList<Room>(dungeon.roomCount());
263        
264        int goalSym = Symbol.GOAL,
265               bossSym = Symbol.BOSS;
266        
267        for (Room room: dungeon.getRooms()) {
268            if (room.getChildren().size() > 0 || room.getItem() != Symbol.NOTHING)
269                continue;
270            Room parent = room.getParent();
271            if (parent == null || parent.getChildren().size() != 1 ||
272                    room.getItem() != Symbol.NOTHING ||
273                    !parent.getPrecond().implies(room.getPrecond()))
274                continue;
275            if (isGenerateGoal()) {
276                if (!constraints.roomCanFitItem(room.id, goalSym) ||
277                        !constraints.roomCanFitItem(parent.id, bossSym))
278                    continue;
279            } else {
280                if (!constraints.roomCanFitItem(room.id, bossSym))
281                    continue;
282            }
283            possibleGoalRooms.add(room);
284        }
285        
286        if (possibleGoalRooms.size() == 0) throw new RetryException();
287        
288        Room goalRoom = random.getRandomElement(possibleGoalRooms),
289             bossRoom = goalRoom.getParent();
290        
291        if (!isGenerateGoal()) {
292            bossRoom = goalRoom;
293            goalRoom = null;
294        }
295        
296        if (goalRoom != null) goalRoom.setItem(goalSym);
297        bossRoom.setItem(bossSym);
298        
299        if (isBossRoomLocked()) {
300            int oldKeyLevel = bossRoom.getPrecond().getKeyLevel(),
301                newKeyLevel = Math.min(levels.keyCount(), constraints.getMaxKeys());
302            
303            List<Room> oklRooms = levels.getRooms(oldKeyLevel);
304            if (goalRoom != null) oklRooms.remove(goalRoom);
305            oklRooms.remove(bossRoom);
306            
307            if (goalRoom != null) levels.addRoom(newKeyLevel, goalRoom);
308            levels.addRoom(newKeyLevel, bossRoom);
309            
310            int bossKey = newKeyLevel-1;
311            Condition precond = bossRoom.getPrecond().and(bossKey);
312            bossRoom.setPrecond(precond);
313            if (goalRoom != null) goalRoom.setPrecond(precond);
314            
315            if (newKeyLevel == 0) {
316                dungeon.link(bossRoom.getParent(), bossRoom);
317            } else {
318                dungeon.link(bossRoom.getParent(), bossRoom, bossKey);
319            }
320            if (goalRoom != null) dungeon.link(bossRoom, goalRoom);
321        }
322    }

324    /**
325     * Removes the given {@link Room} and all its descendants from the given
326     * list.
327     * 
328     * @param rooms the list of Rooms to remove nodes from
329     * @param room  the Room whose descendants to remove from the list
330     */ 
331    protected void removeDescendantsFromList(List<Room> rooms, Room room) {
332        rooms.remove(room);
333        for (Room child: room.getChildren()) {
334            removeDescendantsFromList(rooms, child);
335        }
336    }

342     * @param room  the Room to add extra preconditions to
343     * @param cond  the extra preconditions to add
344     */
345    protected void addPrecond(Room room, Condition cond) {
346        room.setPrecond(room.getPrecond().and(cond));
347        for (Room child: room.getChildren()) {
348            addPrecond(child, cond);

352    /**
353     * Randomly locks descendant rooms of the given {@link Room} with
354     * {@link Edge}s that require the switch to be in the given state.
355     * <p>
356     * If the given state is EITHER, the required states will be random.
357     * 
358     * @param room          the room whose child to lock
359     * @param givenState    the state to require the switch to be in for the
360     *                      child rooms to be accessible
361     * @return              true if any locks were added, false if none were
362     *                      added (which can happen due to the way the random
363     *                      decisions are made)
364     * @see SwitchState
365     */
366    protected boolean switchLockChildRooms(Room room,
367            SwitchState givenState) {
368        boolean anyLocks = false;
369        SwitchState state = givenState != SwitchState.EITHER
370                ? givenState
371                : (random.nextBoolean()
372                    ? SwitchState.ON
373                    : SwitchState.OFF);
374        
375        for (Edge edge: room.getEdges()) {
376            int neighborId = edge.getTargetRoomId();
377            Room nextRoom = dungeon.get(neighborId);
378            if (room.getChildren().contains(nextRoom)) {
379                if (room.getEdge(neighborId).getSymbol() == Symbol.NOTHING &&
380                        random.nextIntHasty(4) != 0) {
381                    dungeon.link(room, nextRoom, state.toSymbol());
382                    addPrecond(nextRoom, new Condition(state.toSymbol()));
383                    anyLocks = true;
384                } else {
385                    anyLocks |= switchLockChildRooms(nextRoom, state);
386                }
387                
388                if (givenState == SwitchState.EITHER) {
389                    state = state.invert();
390                }
391            }
392        }
393        return anyLocks;
394    }

396    /**
397     * Returns a path from the goal to the dungeon entrance, along the 'parent'
398     * relations.
399     * 
400     * @return  a list of linked {@link Room}s starting with the goal room and
401     *          ending with the start room.
402     */
403    protected List<Room> getSolutionPath() {
404        List<Room> solution = new ArrayList<Room>();
405        Room room = dungeon.findGoal();
406        while (room != null) {
407            solution.add(room);
408            room = room.getParent();
409        }
410        return solution;
411    }

413    /**
414     * Makes some {@link Edge}s within the dungeon require the dungeon's switch
415     * to be in a particular state, and places the switch in a room in the
416     * dungeon.
417     * 
418     * @throws RetryException if it fails
419     */
420    protected void placeSwitches() throws RetryException {
421        // Possible TODO: have multiple switches on separate circuits
422        // At the moment, we only have one switch per dungeon.
423        if (constraints.getMaxSwitches() <= 0) return;
424        
425        List<Room> solution = getSolutionPath();
426        
427        for (int attempt = 0; attempt < 10; ++attempt) {
428            
429            List<Room> rooms = new ArrayList<Room>(dungeon.getRooms());
430            random.shuffle(rooms);
431            random.shuffle(solution);
432
433            // Pick a base room from the solution path so that the player
434            // will have to encounter a switch-lock to solve the dungeon.
435            Room baseRoom = null;
436            for (Room room: solution) {
437                if (room.getChildren().size() > 1 && room.getParent() != null) {
438                    baseRoom = room;
439                    break;
440                }
441            }
442            if (baseRoom == null) throw new RetryException();
443            Condition baseRoomCond = baseRoom.getPrecond();
444            
445            removeDescendantsFromList(rooms, baseRoom);
446            
447            int switchSym = Symbol.SWITCH;
448            
449            Room switchRoom = null;
450            for (Room room: rooms) {
451                if (room.getItem() == Symbol.NOTHING &&
452                        baseRoomCond.implies(room.getPrecond()) &&
453                        constraints.roomCanFitItem(room.id, switchSym)) {
454                    switchRoom = room;
455                    break;
456                }
457            }
458            if (switchRoom == null) continue;
459            
460            if (switchLockChildRooms(baseRoom, SwitchState.EITHER)) {
461                switchRoom.setItem(switchSym);
462                return;
463            }
464        }
465        throw new RetryException();
466    }

468    /**
469     * Randomly links up some adjacent rooms to make the dungeon graph less of
470     * a tree.
471     * 
472     * @throws RetryException if it fails
473     */
474    protected void graphify() throws RetryException {
475        IntVLA near;
476        for (Room room: dungeon.getRooms()) {
477            
478            if (room.isGoal() || room.isBoss()) continue;
479            near = constraints.getAdjacentRooms(room.id, Integer.MAX_VALUE);
480
481            for (int i = 0; i < near.size; i++) {
482
483                    // Doesn't matter what the keyLevel is; later checks about
484                    // preconds ensure linkage doesn't trivialize the puzzle.
485                int nextId = near.get(i);
486                if (room.getEdge(nextId) != null) continue;
487                
488                Room nextRoom = dungeon.get(nextId);
489                if (nextRoom == null || nextRoom.isGoal() || nextRoom.isBoss())
490                    continue;
491                
492                boolean forwardImplies = room.getPrecond().implies(nextRoom.getPrecond()),
493                        backwardImplies = nextRoom.getPrecond().implies(room.getPrecond());
494                if (forwardImplies && backwardImplies) {
495                    // both rooms are at the same keyLevel.
496                    if (random.nextDouble() >=
497                            constraints.edgeGraphifyProbability(room.id, nextRoom.id))
498                        continue;
499                    
500                    dungeon.link(room, nextRoom);
501                } else {
502                    int difference = room.getPrecond().singleSymbolDifference(
503                            nextRoom.getPrecond());
504                    if (difference == Symbol.NOTHING || (!Symbol.isSwitchState(difference) &&
505                            random.nextDouble() >=
506                                constraints.edgeGraphifyProbability(room.id, nextRoom.id)))
507                        continue;
508                    dungeon.link(room, nextRoom, difference);
509                }
510            }
511        }
512    }

514    /**
515     * Places keys within the dungeon in such a way that the dungeon is
516     * guaranteed to be solvable.
517     * 
518     * @param levels    the keyLevel -> room-set mapping to use
519     * @throws RetryException if it fails
520     * @see KeyLevelRoomMapping
521     */
522    protected void placeKeys(KeyLevelRoomMapping levels) throws RetryException {
523        // Now place the keys. For every key-level but the last one, place a
524        // key for the next level in it, preferring rooms with fewest links
525        // (dead end rooms).
526        for (int key = 0; key < levels.keyCount()-1; ++key) {
527            List<Room> rooms = levels.getRooms(key);
528            random.shuffle(rooms);
529            // Collections.sort is stable: it doesn't reorder "equal" elements,
530            // which means the shuffling we just did is still useful.
531            Collections.sort(rooms, INTENSITY_COMPARATOR);
532            // Alternatively, use the EDGE_COUNT_COMPARATOR to put keys at
533            // 'dead end' rooms.
534
535            boolean placedKey = false;
536            for (Room room: rooms) {
537                if (room.getItem() == Symbol.NOTHING && constraints.roomCanFitItem(room.id, key)) {
538                    room.setItem(key);
539                    placedKey = true;
540                    break;
541                }
542            }
543            if (!placedKey)
544                // there were no rooms into which the key would fit
545                throw new RetryException();
546        }
547    }

549    protected static final double
550            INTENSITY_GROWTH_JITTER = 0.1,
551            INTENSITY_EASE_OFF = 0.2;
552    
553    /**
554     * Recursively applies the given intensity to the given {@link Room}, and
555     * higher intensities to each of its descendants that are within the same
556     * keyLevel.
557     * <p>
558     * Intensities set by this method may (will) be outside of the normal range
559     * from 0.0 to 1.0. See {@link #normalizeIntensity} to correct this.
560     * 
561     * @param room      the room to set the intensity of
562     * @param intensity the value to set intensity to (some randomn variance is
563     *                  added)
564     * @see Room
565     */
566    protected double applyIntensity(Room room, double intensity) {
567        intensity *= 1.0 - INTENSITY_GROWTH_JITTER/2.0 +
568                INTENSITY_GROWTH_JITTER * random.nextDouble();
569        
570        room.setIntensity(intensity);
571        
572        double maxIntensity = intensity;
573        for (Room child: room.getChildren()) {
574            if (room.getPrecond().implies(child.getPrecond())) {
575                maxIntensity = Math.max(maxIntensity, applyIntensity(child,
576                        intensity + 1.0));
577            }
578        }
579        
580        return maxIntensity;
581    }
582
583    /**
584     * Scales intensities within the dungeon down so that they all fit within
585     * the range 0 <= intensity < 1.0.
586     * 
587     * @see Room
588     */
589    protected void normalizeIntensity() {
590        double maxIntensity = 0.0;
591        for (Room room: dungeon.getRooms()) {
592            maxIntensity = Math.max(maxIntensity, room.getIntensity());
593        }
594        for (Room room: dungeon.getRooms()) {
595            room.setIntensity(room.getIntensity() * 0.99 / maxIntensity);
596        }
597    }

599    /**
600     * Computes the 'intensity' of each {@link Room}. Rooms generally get more
601     * intense the deeper they are into the dungeon.
602     * 
603     * @param levels    the keyLevel -> room-set mapping to update
604     * @throws RetryException if it fails
605     * @see KeyLevelRoomMapping
606     * @see Room
607     */
608    protected void computeIntensity(KeyLevelRoomMapping levels)
609            throws RetryException {
610        
611        double nextLevelBaseIntensity = 0.0;
612        for (int level = 0; level < levels.keyCount(); ++level) {
613            
614            double intensity = nextLevelBaseIntensity *
615                    (1.0 - INTENSITY_EASE_OFF);
616            
617            for (Room room: levels.getRooms(level)) {
618                if (room.getParent() == null ||
619                        !room.getParent().getPrecond().
620                            implies(room.getPrecond())) {
621                    nextLevelBaseIntensity = Math.max(
622                            nextLevelBaseIntensity,
623                            applyIntensity(room, intensity));
624                }
625            }
626        }
627        
628        normalizeIntensity();
629        
630        dungeon.findBoss().setIntensity(1.0);
631        Room goalRoom = dungeon.findGoal();
632        if (goalRoom != null)
633            goalRoom.setIntensity(0.0);
634    }

650    @Override
651    public void generate() {
652        int attempt = 0;
653
654        while (true) {
655            try {
656                KeyLevelRoomMapping levels;
657                int roomsPerLock;
658                if (constraints.getMaxKeys() > 0) {
659                    roomsPerLock = constraints.getMaxRooms() /
660                        constraints.getMaxKeys();
661                } else {
662                    roomsPerLock = constraints.getMaxRooms();
663                }
664                while (true) {
665                    dungeon = new RoomLayout();
666                    
667                    // Maps keyLevel -> Rooms that were created when lockCount had that
668                    // value
669                    levels = new KeyLevelRoomMapping();
670                    
671                    // Create the entrance to the dungeon:
672                    initEntranceRoom(levels);
673                
674                    try {
675                        // Fill the dungeon with rooms:
676                        placeRooms(levels, roomsPerLock);
677                        break;
678                    } catch (OutOfRoomsException e) {
679                        // We can run out of rooms where certain links have
680                        // predetermined locks. Example: if a river bisects the
681                        // map, the keyLevel for rooms in the river > 0 because
682                        // crossing water requires a key. If there are not
683                        // enough rooms before the river to build up to the
684                        // key for the river, we've run out of rooms.
685
686                        //log("Ran out of rooms. roomsPerLock was "+roomsPerLock);
687                        roomsPerLock = roomsPerLock * constraints.getMaxKeys() /
688                                (constraints.getMaxKeys() + 1);
689                        //log("roomsPerLock is now "+roomsPerLock);
690                        
691                        if (roomsPerLock == 0) {
692                            throw new GenerationFailureException(
693                                    "Failed to place rooms. Have you forgotten to disable boss-locking?");
694                            // If the boss room is locked, the final key is used
695                            // only for the boss room. So if the final key is
696                            // also used to cross the river, rooms cannot be
697                            // placed.
698                        }
699                    }
700                }
701                
702                // Place the boss and goal rooms:
703                placeBossGoalRooms(levels);
704                
705                // Place switches and the locks that require it:
706                placeSwitches();
707        
708                // Make the dungeon less tree-like:
709                graphify();
710                
711                computeIntensity(levels);
712                
713                // Place the keys within the dungeon:
714                placeKeys(levels);
715                
716                if (levels.keyCount()-1 != constraints.getMaxKeys())
717                    throw new RetryException();
718
719                checkAcceptable();
720                
721                return;
722            
723            } catch (RetryException e) {
724                if (++ attempt > MAX_RETRIES) {
725                    throw new GenerationFailureException("Dungeon generator failed", e);
726                }
727                //log("Retrying dungeon generation...");
728            }
729        }
730        
731    }
