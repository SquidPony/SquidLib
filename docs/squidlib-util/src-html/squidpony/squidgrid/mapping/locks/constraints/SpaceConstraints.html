001package squidpony.squidgrid.mapping.locks.constraints;
002
003import squidpony.squidmath.Coord;
004import squidpony.squidmath.IntVLA;
005
006/**
007 * Constrains the coordinates where Rooms may be placed to be only those within
008 * the {@link SpaceMap}, as well as placing limitations on the number of keys
009 * and switches.
010 * 
011 * @see CountConstraints
012 * @see SpaceMap
013 */
014public class SpaceConstraints extends CountConstraints {
015
016    public static final int DEFAULT_MAX_KEYS = 4,
017            DEFAULT_MAX_SWITCHES = 1;
018    
019    protected SpaceMap spaceMap;
020    
021    public SpaceConstraints(SpaceMap spaceMap) {
022        super(spaceMap.numberSpaces(), DEFAULT_MAX_KEYS, DEFAULT_MAX_SWITCHES);
023        this.spaceMap = spaceMap;
024    }
025
026    @Override
027    protected boolean validRoomCoords(Coord c) {
028        return spaceMap.get(c);
029    }
030
031    @Override
032    public IntVLA initialRooms() {
033        IntVLA ids = new IntVLA();
034        for (Coord xy: spaceMap.getBottomSpaces()) {
035            ids.add(getRoomId(xy));
036        }
037        return ids;
038    }
039    
040    
041
042}
