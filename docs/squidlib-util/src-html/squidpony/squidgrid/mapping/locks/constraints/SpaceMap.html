001package squidpony.squidgrid.mapping.locks.constraints;
002
003import squidpony.squidgrid.mapping.locks.Room;
004import squidpony.squidgrid.mapping.locks.generators.ILayoutGenerator;
005import squidpony.squidmath.Coord;
006import squidpony.squidmath.OrderedSet;
007
008import java.util.*;
009
010/**
011 * Controls which spaces are valid for an
012 * {@link ILayoutGenerator} to create
013 * {@link Room}s in.
014 * <p>
015 * Essentially just a Set<{@link Coord}> with some convenience methods.
016 * 
017 * @see Coord
018 * @see SpaceConstraints
019 */
020public class SpaceMap {
021    protected OrderedSet<Coord> spaces = new OrderedSet<>();
022    
023    public int numberSpaces() {
024        return spaces.size();
025    }
026    
027    public boolean get(Coord c) {
028        return spaces.contains(c);
029    }
030    
031    public void set(Coord c, boolean val) {
032        if (val)
033            spaces.add(c);
034        else
035            spaces.remove(c);
036    }
037    
038    private Coord getFirst() {
039        return spaces.first();
040    }
041    
042    public ArrayList<Coord> getBottomSpaces() {
043        ArrayList<Coord> bottomRow = new ArrayList<Coord>();
044        bottomRow.add(getFirst());
045        int bottomY = getFirst().y;
046        for (Coord space: spaces) {
047            if (space.y > bottomY) {
048                bottomY = space.y;
049                bottomRow.clear();
050                bottomRow.add(space);
051            } else if (space.y == bottomY) {
052                bottomRow.add(space);
053            }
054        }
055        return bottomRow;
056    }
057}
