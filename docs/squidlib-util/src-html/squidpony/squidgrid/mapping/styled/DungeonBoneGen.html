147    private Tile chooseTile(Tile[] list, int numlist, int[] y_positions, int[] x_positions) {
148        int a = c_color[y_positions[0]][x_positions[0]];
149        int b = c_color[y_positions[1]][x_positions[1]];
150        int c = c_color[y_positions[2]][x_positions[2]];
151        int d = c_color[y_positions[3]][x_positions[3]];
152        int e = c_color[y_positions[4]][x_positions[4]];
153        int f = c_color[y_positions[5]][x_positions[5]];
154        int i, n, match = Integer.MAX_VALUE, pass;
155        for (pass = 0; pass < 2; ++pass) {
156            n = 0;
157            // pass #1:
158            //   count number of variants that match this partial set of constraints
159            // pass #2:
160            //   stop on randomly selected match
161            for (i = 0; i < numlist; ++i) {
162                Tile tile = list[i];
163                if ((a < 0 || a == tile.a_constraint) &&
164                        (b < 0 || b == tile.b_constraint) &&
165                        (c < 0 || c == tile.c_constraint) &&
166                        (d < 0 || d == tile.d_constraint) &&
167                        (e < 0 || e == tile.e_constraint) &&
168                        (f < 0 || f == tile.f_constraint)) {
169                    n += 1;
170                    if (n > match) {
171                        // use list[i]
172                        // update constraints to reflect what we placed
173                        c_color[y_positions[0]][x_positions[0]] = tile.a_constraint;
174                        c_color[y_positions[1]][x_positions[1]] = tile.b_constraint;
175                        c_color[y_positions[2]][x_positions[2]] = tile.c_constraint;
176                        c_color[y_positions[3]][x_positions[3]] = tile.d_constraint;
177                        c_color[y_positions[4]][x_positions[4]] = tile.e_constraint;
178                        c_color[y_positions[5]][x_positions[5]] = tile.f_constraint;
179                        return tile;
180                    }
181                }
182            }
183            if (n == 0) {
184                return null;
185            }
186            match = rng.nextInt(n);
187        }
188        return null;
189    }

191    private Tile chooseTile(Tile[] list, int numlist, boolean upright, int[] y_positions, int[] x_positions) {
192        int a, b, c, d, e, f;
193        if (upright) {
194            a = h_color[y_positions[0]][x_positions[0]];
195            b = v_color[y_positions[1]][x_positions[1]];
196            c = v_color[y_positions[2]][x_positions[2]];
197            d = v_color[y_positions[3]][x_positions[3]];
198            e = v_color[y_positions[4]][x_positions[4]];
199            f = h_color[y_positions[5]][x_positions[5]];
200        } else {
201            a = h_color[y_positions[0]][x_positions[0]];
202            b = h_color[y_positions[1]][x_positions[1]];
203            c = v_color[y_positions[2]][x_positions[2]];
204            d = v_color[y_positions[3]][x_positions[3]];
205            e = h_color[y_positions[4]][x_positions[4]];
206            f = h_color[y_positions[5]][x_positions[5]];
207        }
208        int i, n, match = Integer.MAX_VALUE, pass;
209        for (pass = 0; pass < 2; ++pass) {
210            n = 0;
211            // pass #1:
212            //   count number of variants that match this partial set of constraints
213            // pass #2:
214            //   stop on randomly selected match
215            for (i = 0; i < numlist; ++i) {
216                Tile tile = list[i];
217                if ((a < 0 || a == tile.a_constraint) &&
218                        (b < 0 || b == tile.b_constraint) &&
219                        (c < 0 || c == tile.c_constraint) &&
220                        (d < 0 || d == tile.d_constraint) &&
221                        (e < 0 || e == tile.e_constraint) &&
222                        (f < 0 || f == tile.f_constraint)) {
223                    n += 1;
224                    if (n > match) {
225                        // use list[i]
226                        // update constraints to reflect what we placed
227                        if (upright) {
228                            h_color[y_positions[0]][x_positions[0]] = tile.a_constraint;
229                            v_color[y_positions[1]][x_positions[1]] = tile.b_constraint;
230                            v_color[y_positions[2]][x_positions[2]] = tile.c_constraint;
231                            v_color[y_positions[3]][x_positions[3]] = tile.d_constraint;
232                            v_color[y_positions[4]][x_positions[4]] = tile.e_constraint;
233                            h_color[y_positions[5]][x_positions[5]] = tile.f_constraint;
234                        } else {
235                            h_color[y_positions[0]][x_positions[0]] = tile.a_constraint;
236                            h_color[y_positions[1]][x_positions[1]] = tile.b_constraint;
237                            v_color[y_positions[2]][x_positions[2]] = tile.c_constraint;
238                            v_color[y_positions[3]][x_positions[3]] = tile.d_constraint;
239                            h_color[y_positions[4]][x_positions[4]] = tile.e_constraint;
240                            h_color[y_positions[5]][x_positions[5]] = tile.f_constraint;
241                        }
242                        return tile;
243                    }
244                }
245            }
246            if (n == 0) {
247                return null;
248            }
249            match = rng.nextInt(n);
250        }
251        return null;
252    }

261     * @param w  Width of the dungeon to generate in chars.

263     * @return A row-major char[][] with h rows and w columns; it will be filled with '#' for walls and '.' for floors.

269    /**
270     * Changes the outer edge of a char[][] to the wall char, '#'.
271     *
272     * @param map A char[][] that stores map data.
273     * @return
274     */
275    public static char[][] wallWrap(char[][] map) {
276        int upperY = map[0].length - 1;
277        int upperX = map.length - 1;
278        for (int i = 0; i < map.length; i++) {
279            map[i][0] = '#';
280            map[i][upperY] = '#';
281        }
282        for (int i = 0; i < map[0].length; i++) {
283            map[0][i] = '#';
284            map[upperX][i] = '#';
285        }
286        return map;
287    }

289    /**
290     * Changes the outer edge of this dungeon to the wall char, '#'.
291     *
292     * @return The modified dungeon, a char[][].
293     */
294    public char[][] wallWrap() {
295        int upperY = high - 1;
296        int upperX = wide - 1;
297        for (int i = 0; i < wide; i++) {
298            dungeon[i][0] = '#';
299            dungeon[i][upperY] = '#';
300        }
301        for (int i = 0; i < high; i++) {
302            dungeon[0][i] = '#';
303            dungeon[upperX][i] = '#';
304        }
305        return dungeon;
306    }

308    private boolean matchingAdjacent(int y, int x)
309    {
310        return c_color[y][x] == c_color[y + 1][x + 1];
311    }
312
313    private int changeColor(int old_color, int num_options) {
314
315        int offset = 1 + rng.nextInt(num_options - 1);
316        return (old_color + offset) % num_options;
317    }

319    /**
320     * Generate a dungeon given a Tileset.
321     * If you have your own Tileset gained by parsing your own JSON, use
322     * this to generate a dungeon using it. Consider using DungeonBoneGen.wallWrap
323     * to surround the edges with walls. Assigns the returned result to a member
324     * of this class, 'dungeon'.
325     *
326     * @param ts A Tileset; if you don't have one of these available, use a TilesetType enum instead to select a predefined one.
327     * @param h  Height of the dungeon to generate in chars.
328     * @param w  Width of the dungeon to generate in chars.
329     * @return A row-major char[][] with h rows and w columns; it will be filled with '#' for walls and '.' for floors.
330     */
331    public char[][] generate(Tileset ts, int w, int h) {
332        wide = w;
333        high = h;
334        char[][] trans_output = new char[h][w];
335        GreasedRegion out = new GreasedRegion(wide, high);
336        int sidelen = ts.config.short_side_length;
337        int xmax = (w / sidelen) + 6;
338        int ymax = (h / sidelen) + 6;
339        if (xmax > 1006) {
340            return null;
341        }
342        if (ymax > 1006) {
343            return null;
344        }
345        if (ts.config.is_corner) {
346            c_color = new int[ymax][xmax];
347            int i = 0, j = 0, ypos = -1 * sidelen;
348            int[] cc = ts.config.num_colors;
349
350            for (j = 0; j < ymax; ++j) {
351                for (i = 0; i < xmax; ++i) {
352                    c_color[j][i] = rng.nextInt(cc[(i - j + 1) & 3]); // select from cc based on corner type
353                }
354            }
355
356            // Repetition reduction
357            // now go back through and make sure we don't have adjacent 3x2 vertices that are identical,
358            // to avoid really obvious repetition (which happens easily with extreme weights)
359            for (j = 0; j < ymax - 3; ++j) {
360                for (i = 0; i < xmax - 3; ++i) {
361                    int p = (i - j + 1) & 3; // corner type
362                    if (i + 3 >= 1006) {
363                        return null;
364                    }
365                    if (j + 3 >= 1006) {
366                        return null;
367                    }
368                    if (matchingAdjacent(j, i) && matchingAdjacent(j + 1, i) && matchingAdjacent(j + 2, i)
369                            && matchingAdjacent(j, i + 1) && matchingAdjacent(j + 1, i + 1) && matchingAdjacent(j + 2, i + 1)) {
370                        p = ((i + 1) - (j + 1) + 1) & 3;
371                        if (cc[p] > 1)
372                            c_color[j + 1][i + 1] = changeColor(c_color[j + 1][i + 1], cc[p]);
373                    }
374
375                    if (matchingAdjacent(j, i) && matchingAdjacent(j, i + 1) && matchingAdjacent(j, i + 2)
376                            && matchingAdjacent(j + 1, i) && matchingAdjacent(j + 1, i + 1) && matchingAdjacent(j + 1, i + 2)) {
377                        p = ((i + 2) - (j + 1) + 1) & 3;
378                        if (cc[p] > 1)
379                            c_color[j + 1][i + 2] = changeColor(c_color[j + 1][i + 2], cc[p]);
380                    }
381                }
382            }
383
384
385            for (j = -1; ypos < h; ++j) {
386                // a general herringbone row consists of:
387                //    horizontal left block, the bottom of a previous vertical, the top of a new vertical
388                int phase = j & 3;
389                // displace horizontally according to pattern
390                if (phase == 0) {
391                    i = 0;
392                } else {
393                    i = phase - 4;
394                }
395                for (; ; i += 4) {
396                    int xpos = i * sidelen;
397                    if (xpos >= w)
398                        break;
399                    // horizontal left-block
400                    if (xpos + sidelen * 2 >= 0 && ypos >= 0) {
401                        Tile t = chooseTile(
402                                ts.h_tiles, ts.h_tiles.length,
403                                new int[]{j + 2, j + 2, j + 2, j + 3, j + 3, j + 3},
404                                new int[]{i + 2, i + 3, i + 4, i + 2, i + 3, i + 4});
405
406                        if (t == null)
407                            return null;
408
409                        //trans_output = insert(trans_output, t.data, ypos, xpos);
410                        out.or(new GreasedRegion(t.data, t.width, t.height, wide, high).translate(xpos, ypos));
411                    }
412                    xpos += sidelen * 2;
413                    // now we're at the end of a previous vertical one
414                    xpos += sidelen;
415                    // now we're at the start of a new vertical one
416                    if (xpos < w) {
417                        Tile t = chooseTile(
418                                ts.v_tiles, ts.v_tiles.length,
419                                new int[]{j + 2, j + 3, j + 4, j + 2, j + 3, j + 4},
420                                new int[]{i + 5, i + 5, i + 5, i + 6, i + 6, i + 6});
421
422                        if (t == null)
423                            return null;
424                        //trans_output = insert(trans_output, t.data, ypos, xpos);
425                        out.or(new GreasedRegion(t.data, t.width, t.height, wide, high).translate(xpos, ypos));
426                    }
427                }
428                ypos += sidelen;
429            }
430        } else {
431            int i = 0, j = -1, ypos;
432            v_color = new int[ymax][xmax];
433            h_color = new int[ymax][xmax];
434            for (int yy = 0; yy < ymax; yy++) {
435                for (int xx = 0; xx < xmax; xx++) {
436                    v_color[yy][xx] = -1;
437                    h_color[yy][xx] = -1;
438                }
439            }
440
441            ypos = -1 * sidelen;
442            for (j = -1; ypos < h; ++j) {
443                // a general herringbone row consists of:
444                //    horizontal left block, the bottom of a previous vertical, the top of a new vertical
445                int phase = j & 3;
446                // displace horizontally according to pattern
447                if (phase == 0) {
448                    i = 0;
449                } else {
450                    i = phase - 4;
451                }
452                for (; ; i += 4) {
453                    int xpos = i * sidelen;
454                    if (xpos >= w)
455                        break;
456                    // horizontal left-block
457                    if (xpos + sidelen * 2 >= 0 && ypos >= 0) {
458                        Tile t = chooseTile(
459                                ts.h_tiles, ts.h_tiles.length, false,
460                                new int[]{j + 2, j + 2, j + 2, j + 2, j + 3, j + 3},
461                                new int[]{i + 2, i + 3, i + 2, i + 4, i + 2, i + 3});
462
463                        if (t == null)
464                            return null;
465                        //trans_output = insert(trans_output, t.data, ypos, xpos);
466                        out.or(new GreasedRegion(t.data, t.width, t.height, wide, high).translate(xpos, ypos));
467                    }
468                    xpos += sidelen * 2;
469                    // now we're at the end of a previous vertical one
470                    xpos += sidelen;
471                    // now we're at the start of a new vertical one
472                    if (xpos < w) {
473                        Tile t = chooseTile(
474                                ts.v_tiles, ts.v_tiles.length, true,
475                                new int[]{j + 2, j + 2, j + 2, j + 3, j + 3, j + 4},
476                                new int[]{i + 5, i + 5, i + 6, i + 5, i + 6, i + 5});
477
478                        if (t == null)
479                            return null;
480                        //trans_output = insert(trans_output, t.data, ypos, xpos);
481                        out.or(new GreasedRegion(t.data, t.width, t.height, wide, high).translate(xpos, ypos));
482                    }
483                }
484                ypos += sidelen;
485            }
486        }
487        /*for (int x = 0; x < w; x++) {
488            for (int y = 0; y < h; y++) {
489                output[x][y] = trans_output[y][x];
490            }
491        }*/
492        dungeon = out.toChars();
493        return dungeon;
494    }
