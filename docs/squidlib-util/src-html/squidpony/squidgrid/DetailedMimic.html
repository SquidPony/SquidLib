019/**
020 * Similar to MimicFill, this class can be used to imitate the style of an existing piece of data, but this works on
021 * more than just booleans; it can produce similar styles of texture (its original use in SynTex), of map, of item
022 * placement, and so on by specifying a different technique for differentiating two int values.
023 * <br>
024 * Two options are now available; the process() method allows the slow analyze() step to be computed once,
025 * before the rest of processing is potentially done many times, but the new neoProcess() method produces
026 * comparable or better results and is drastically faster even without needing analysis. This means
027 * neoProcess(), based on P.F. Harrison's algorithm in SynTex, is strongly recommended now.
028 * <br>
029 * Example results of neoProcess(), with a procedural dungeon first and the mimic versions after it:
030 * https://gist.github.com/tommyettinger/405336fb0fc74838d806c021aabe77da (you may need to click Raw and zoom
031 * out somewhat for it to render well).
032 * <br>
033 * Ported from https://github.com/mxgmn/SynTex by Tommy Ettinger on 6/9/2016.
034 */
035public class DetailedMimic {

047    /**
048     * Constructor that uses an unseeded RNG (effectively a random seed) and the given AestheticDifference. An example
049     * piece of code that implements an AestheticDifference is available in the docs for
050     * {@link AestheticDifference#difference(int, int)}; it is also considered a functional interface if you use Java 8
051     * or newer. You can also use the ready-made implementation {@link AestheticDifference#rgba8888} if you have int
052     * data that represents RGBA8888 colors, which can be obtained from libGDX Colors or SColors in the display module.
053     * @param diff an implementation of the AestheticDifference interface, such as {@link AestheticDifference#rgba8888};
054     *             may be null, but that forces all calls to processing methods to treat discrete as true
055     */
056    public DetailedMimic(AestheticDifference diff)
057    {
058        this(diff, new RNG());
059    }

061    /**
062     * Constructor that uses the given RNG and the given AestheticDifference. An example
063     * piece of code that implements an AestheticDifference is available in the docs for
064     * {@link AestheticDifference#difference(int, int)}; it is also considered a functional interface if you use Java 8
065     * or newer. You can also use the ready-made implementation {@link AestheticDifference#rgba8888} if you have int
066     * data that represents RGBA8888 colors, which can be obtained from libGDX Colors or SColors in the display module.
067     * @param diff an implementation of the AestheticDifference interface, such as {@link AestheticDifference#rgba8888};
068     *             may be null, but that forces all calls to processing methods to treat discrete as true
069     * @param rng an RNG to generate random factors; may be seeded to produce reliable output
070     */
071    public DetailedMimic(AestheticDifference diff, RNG rng)
072    {
073        random = rng;
074        difference = diff;
075        analyzed = null;
076    }

078    /**
079     * DISCOURAGED; use {@link #neoProcess(int[], int, int, int, int, int, int, boolean)} instead, which doesn't need a
080     * separate analysis step.
081     * Analyzes a sample as a 1D int array and stores the needed info to call
082     * {@link #process(int[], int, int, int, int, int, int, double, boolean)} any number of times later on without
083     * recalculating some heavy-weight information.
084     * @param sample a 1D array of ints that can be compared by the AestheticDifference this uses (or any ints if
085     *               discrete is true)
086     * @param width the width of the area in sample this should use (sample can be interpreted as different shapes)
087     * @param height the height of the area in sample this should use (sample can be interpreted as different shapes)
088     * @param detailLevel how much detail to try for; if 0 or less this does nothing, 2 works well in general
089     * @param proximity how far away to consider cells as affecting another; 3 works well
090     * @param discrete false if this can produce ints other than those in the input; true if it uses a fixed set
091     */
092    public void analyze(int[] sample, int width, int height, int detailLevel, int proximity, boolean discrete)
093    {
094        discrete |= (difference == null);
095        if(detailLevel > 0)
096            analysis(sample, width, height, detailLevel, proximity, discrete);
097    }
098    /**
099     * DISCOURAGED; use {@link #neoProcess(int[], int, int, int, int, int, int, boolean)} instead, which doesn't need a
100     * separate analysis step.
101     * Processes a sample as a 1D int array and returns a different 1D int array that mimics the input. If the last time
102     * this was called used the same sample, sampleWidth, and sampleHeight parameters, or if
103     * {@link #analyze(int[], int, int, int, int, boolean)} was called with its width equal to sampleWidth and its
104     * height equal to sampleHeight, then this doesn't need to perform as many expensive calculations.
105     * @param sample a 1D array of ints that can be compared by the AestheticDifference this uses (or any ints if
106     *               discrete is true)
107     * @param sampleWidth the width of the area in sample this should use (sample can be interpreted as different shapes)
108     * @param sampleHeight the height of the area in sample this should use (sample can be interpreted as different shapes)
109     * @param targetWidth the desired width of the output
110     * @param targetHeight the desired height of the output
111     * @param detailLevel how much detail to try for; if 0 or less this doesn't perform analysis and has somewhat lower
112     *                    quality, but 2 works well in general
113     * @param proximity how far away to consider cells as affecting another; 3 works well
114     * @param temperature a level of unpredictability in the output relative to the input; must be greater than 0
115     * @param discrete false if this can produce ints other than those in the input; true if it uses a fixed set
116     * @return a new 1D int array that can be interpreted as having targetWidth and targetHeight, and mimics sample
117     */
118    public int[] process(int[] sample, int sampleWidth, int sampleHeight, int targetWidth, int targetHeight,
119                         int detailLevel, int proximity, double temperature, boolean discrete)
120    {
121        discrete |= (difference == null);
122        if(detailLevel > 0)
123        {
124            if(analyzed == null || analyzed.length != sampleWidth || analyzed.length == 0 ||
125                    analyzed[0].length != sampleHeight)
126                analyze(sample, sampleWidth, sampleHeight, detailLevel, proximity, discrete);
127            return coherentSynthesis(sample, sampleWidth, sampleHeight, analyzed, proximity,
128                    targetWidth, targetHeight, temperature, discrete);
129        }
130        else {
131            return fullSynthesis(sample, sampleWidth, sampleHeight, proximity,
132                    targetWidth, targetHeight, temperature, discrete);
133        }
134    }

136    /**
137     * Processes a 1D int array representing 2D storage of values that can be compared by this object's
138     * AestheticDifference (or any values if that is null or discrete is true), and returns a 1D array representing data
139     * with potentially different dimensions but similar appearance to sample.
140     * @param sample a 1D array of ints that can be compared by the AestheticDifference this uses (or any ints if
141     *               discrete is true)
142     * @param sampleWidth the width of the area in sample this should use (sample can be interpreted as different shapes)
143     * @param sampleHeight the height of the area in sample this should use (sample can be interpreted as different shapes)
144     * @param targetWidth the desired width of the output
145     * @param targetHeight the desired height of the output
146     * @param detailLevel how much detail to try for; here this will always be treated as at least 1
147     * @param proximity how far away to consider cells as affecting another; 3 works well
148     * @param discrete false if this can produce ints other than those in the input; true if it uses a fixed set
149     * @return a new 1D int array that can be interpreted as having targetWidth and targetHeight, and mimics sample
150     */
151    public int[] neoProcess(int[] sample, int sampleWidth, int sampleHeight, int targetWidth, int targetHeight,
152                         int detailLevel, int proximity, boolean discrete)
153    {
154            return reSynthesis(sample, sampleWidth, sampleHeight, proximity, 20,
155                    Math.max(1, detailLevel), discrete || (difference == null), targetWidth, targetHeight);
156    }

162    private void analysis(int[] bitmap, int width, int height, int K, int N, boolean indexed)
163    {
164        int area = width * height;
165        analyzed = new int[area][];
166        OrderedSet<Integer> points = new OrderedSet<>(area);
167        for (int i = 0; i < area; i++) points.add(i);
168
169        double[] similarities = new double[area * area];
170        for (int i = 0; i < area; i++) for (int j = 0; j < area; j++)
171            similarities[i * area + j] = similarities[j * area + i] != 0 ? similarities[j * area + i] :
172                    similarity(i, bitmap, width, height, j, bitmap, width, height, N, null, indexed);
173
174        for (int i = 0; i < area; i++)
175        {
176            analyzed[i] = new int[K];
177            OrderedSet<Integer> copy = new OrderedSet<>(points);
178
179            analyzed[i][0] = i;
180            copy.remove(i);
181
182            for (int k = 1; k < K; k++)
183            {
184                double max = -10000;
185                int argmax = -1;
186
187                for(Integer p : copy)
188                {
189                    double s = similarities[i * area + p];
190                    if (s > max)
191                    {
192                        max = s;
193                        argmax = p;
194                    }
195                }
196
197                analyzed[i][k] = argmax;
198                copy.remove(argmax);
199            }
200        }
201    }

203    private int[] coherentSynthesis(int[] sample, int SW, int SH, int[][] sets, int N, int OW, int OH, double t, boolean indexed)
204    {
205        int[] result = new int[OW * OH];
206        Integer[] origins = new Integer[OW * OH];
207        boolean[][] mask = new boolean[SW][SH], cleanMask = new boolean[SW][SH];
208
209        for (int i = 0; i < OW * OH; i++)
210        {
211            int x = i % OW, y = i / OW;
212            IntDoubleOrderedMap candidates = new IntDoubleOrderedMap();
213            ArrayTools.insert(cleanMask, mask, 0, 0);
214
215            for (int dy = -1; dy <= 1; dy++){
216                for (int dx = -1; dx <= 1; dx++)
217                {
218                    int sx = (x + dx + OW) % OW, sy = (y + dy + OH) % OH;
219                    Integer origin = origins[sy * OW + sx];
220                    if ((dx != 0 || dy != 0) && origin != null)
221                    {
222                        for (int pi = 0, p; pi < sets[origin].length; pi++)
223                        {
224                            p = sets[origin][pi];
225                            int ox = (p % SW - dx + SW) % SW, oy = (p / SW - dy + SH) % SH;
226                            double s = similarity(oy * SW + ox, sample, SW, SH, i, result, OW, OH, N, origins, indexed);
227
228                            if (!mask[ox][oy]) candidates.put(ox + oy * SW, Math.pow(100, s / t));
229                            mask[ox][oy] = true;
230                        }
231                    }
232                }
233            }
234
235            int shifted = candidates.isEmpty() ? random.nextInt(SW) + random.nextInt(SH) * SW : weightedRandom(candidates, random.nextDouble());
236            origins[i] = shifted;
237            result[i] = sample[shifted];
238        }
239
240        return result;
241    }

243    private int[] fullSynthesis(int[] sample, int SW, int SH, int N, int OW, int OH, double t, boolean indexed)
244    {
245        int[] result = new int[OW * OH];
246        Integer[] origins = new Integer[OW * OH];
247
248        if (!indexed) for (int y = 0; y < OH; y++) {
249            for (int x = 0; x < OW; x++){
250                if (y + N >= OH)
251                {
252                    result[x + y * OW] = sample[random.nextInt(SW * SH)];
253                    origins[x + y * OW] = -1;
254                }
255            }
256        }
257
258        for (int i = 0; i < OW * OH; i++)
259        {
260            double[] candidates = new double[SW * SH];
261            double max = -10000;
262            int argmax = -1;
263
264            for (int j = 0; j < SW * SH; j++)
265            {
266                double s = similarity(j, sample, SW, SH, i, result, OW, OH, N, origins, indexed);
267                if (s > max)
268                {
269                    max = s;
270                    argmax = j;
271                }
272
273                if (indexed) candidates[j] = Math.pow(100.0, s / t);
274            }
275
276            if (indexed) argmax = weightedRandom(candidates, random.nextDouble());
277            result[i] = sample[argmax];
278            origins[i] = -1;
279        }
280
281        return result;
282    }

285    private int[] reSynthesis(int[] sample, int SW, int SH, int N, int M, int polish, boolean indexed, int OW, int OH)
286    {
287        IntVLA colors = new IntVLA();
288        int[] indexedSample = new int[Math.min(SW * SH, sample.length)];
289        for (int j = 0; j < indexedSample.length; j++)
290        {
291            int color = sample[j];
292
293            int i = 0;
294            for (int cn = 0; cn < colors.size; cn++)
295            {
296                if(colors.get(cn) == color) break;
297                i++;
298            }
299
300            if (i == colors.size) colors.add(color);
301            indexedSample[j] = i;
302        }
303
304        int colorsNumber = colors.size;
305
306        double[][] colorMetric = null;
307        if (!indexed && colorsNumber <= 1024)
308        {
309            colorMetric = new double[colorsNumber][colorsNumber];
310            for (int x = 0; x < colorsNumber; x++)
311            {
312                for (int y = 0; y < colorsNumber; y++)
313                {
314                    int cx = colors.get(x), cy = colors.get(y);
315                    colorMetric[x][y] = difference.difference(cx, cy);
316                }
317            }
318        }
319
320        int[] origins = new int[OW * OH];
321        for (int i = 0; i < origins.length; i++) origins[i] = -1;
322
323        int[] shuffle = new int[OW * OH];
324        for (int i = 0; i < shuffle.length; i++)
325        {
326            int j = random.nextIntHasty(i + 1);
327            if (j != i) shuffle[i] = shuffle[j];
328            shuffle[j] = i;
329        }
330
331        for (int round = 0; round <= polish; round++) for (int counter = 0; counter < shuffle.length; counter++)
332        {
333            int f = shuffle[counter];
334            int fx = f % OW, fy = f / OW;
335            int neighborsNumber = round > 0 ? 8 : Math.min(8, counter);
336            int neighborsFound = 0;
337
338            int[] candidates = new int[neighborsNumber + M];
339
340            if (neighborsNumber > 0)
341            {
342                int[] neighbors = new int[neighborsNumber];
343                int[] x = new int[4], y = new int[4];
344
345                for (int radius = 1; neighborsFound < neighborsNumber; radius++)
346                {
347                    x[0] = fx - radius;
348                    y[0] = fy - radius;
349                    x[1] = fx - radius;
350                    y[1] = fy + radius;
351                    x[2] = fx + radius;
352                    y[2] = fy + radius;
353                    x[3] = fx + radius;
354                    y[3] = fy - radius;
355
356                    for (int k = 0; k < 2 * radius; k++)
357                    {
358                        for (int d = 0; d < 4; d++)
359                        {
360                            x[d] = (x[d] + 10 * OW) % OW;
361                            y[d] = (y[d] + 10 * OH) % OH;
362
363                            if (neighborsFound >= neighborsNumber) continue;
364                            int point = x[d] + y[d] * OW;
365                            if (origins[point] != -1)
366                            {
367                                neighbors[neighborsFound] = point;
368                                neighborsFound++;
369                            }
370                        }
371
372                        y[0]++;
373                        x[1]++;
374                        y[2]--;
375                        x[3]--;
376                    }
377                }
378
379
380                for (int n = 0; n < neighborsNumber; n++)
381                {
382                    int cx = (origins[neighbors[n]] + (f - neighbors[n]) % OW + 100 * SW) % SW;
383                    int cy = (origins[neighbors[n]] / SW + f / OW - neighbors[n] / OW + 100 * SH) % SH;
384                    candidates[n] = cx + cy * SW;
385                }
386            }
387
388            for (int m = 0; m < M; m++) candidates[neighborsNumber + m] = random.nextIntHasty(SW * SH);
389
390            double max = -1E+10;
391            int argmax = -1;
392
393            for (int c = 0; c < candidates.length; c++)
394            {
395                double sum = random.nextDouble(0.000001);
396                int ix = candidates[c] % SW, iy = candidates[c] / SW, jx = f % OW, jy = f / OW;
397                int SX, SY, FX, FY, S, F;
398                int origin;
399
400                for (int dy = -N; dy <= N; dy++)
401                {
402                    for (int dx = -N; dx <= N; dx++)
403                    {
404                        if (dx != 0 || dy != 0)
405                        {
406                            SX = ix + dx;
407                            if (SX < 0) SX += SW;
408                            else if (SX >= SW) SX -= SW;
409
410                            SY = iy + dy;
411                            if (SY < 0) SY += SH;
412                            else if (SY >= SH) SY -= SH;
413
414                            FX = jx + dx;
415                            if (FX < 0) FX += OW;
416                            else if (FX >= OW) FX -= OW;
417
418                            FY = jy + dy;
419                            if (FY < 0) FY += OH;
420                            else if (FY >= OH) FY -= OH;
421
422                            S = SX + SY * SW;
423                            F = FX + FY * OW;
424
425                            origin = origins[F];
426                            if (origin != -1)
427                            {
428                                if (indexed) sum += sample[origin] == sample[S] ? 1 : -1;
429                                else if (colorMetric != null) sum += colorMetric[indexedSample[origin]][indexedSample[S]];
430                                else sum += difference.difference(sample[origin], sample[S]);
431                            }
432                        }
433                    }
434                }
435
436                if (sum >= max)
437                {
438                    max = sum;
439                    argmax = candidates[c];
440                }
441            }
442
443            origins[f] = argmax;
444        }
445
446        int[] result = new int[OW * OH];
447        for (int i = 0; i < result.length; i++) result[i] = sample[origins[i]];
448        return result;
449    }

451    private double similarity(int i1, int[] b1, int w1, int h1, int i2, int[] b2, int w2, int h2, int N, Integer[] origins, boolean indexed)
452    {
453        double sum = 0;
454        int x1 = i1 % w1, y1 = i1 / w1, x2 = i2 % w2, y2 = i2 / w2;
455
456        for (int dy = -N; dy <= 0; dy++) for (int dx = -N; (dy < 0 && dx <= N) || (dy == 0 && dx < 0); dx++)
457        {
458            int sx1 = (x1 + dx + w1) % w1, sy1 = (y1 + dy + h1) % h1;
459            int sx2 = (x2 + dx + w2) % w2, sy2 = (y2 + dy + h2) % h2;
460
461            int c1 = b1[sx1 + sy1 * w1];
462            int c2 = b2[sx2 + sy2 * w2];
463
464            if (origins == null || origins[sy2 * w2 + sx2] != null)
465            {
466                if (indexed)
467                    sum += c1 == c2 ? 1 : -1;
468                else
469                    sum -= difference.difference(c1, c2);
470                    /*
471                    Color C1 = Color.FromArgb(c1), C2 = Color.FromArgb(c2);
472                    sum -= (double)((C1.R - C2.R) * (C1.R - C2.R) + (C1.G - C2.G) * (C1.G - C2.G) + (C1.B - C2.B) * (C1.B - C2.B)) / 65536.0;
473                    */
474            }
475        }
476
477        return sum;
478    }

480    static int weightedRandom(double[] array, double r)
481    {
482        double sum = 0;
483        for (int j = 0; j < array.length; j++)
484            sum += Math.max(1.0, array[j]);
485
486        for (int j = 0; j < array.length; j++)
487            array[j] /= sum;
488
489        int i = 0;
490        double x = 0;
491
492        while (i < array.length)
493        {
494            x += array[i];
495            if (r <= x) return i;
496            i++;
497        }
498
499        return 0;
500    }
501
502    static int weightedRandom(IntDoubleOrderedMap dic, double r) {
503        int[] ints = dic.keySet().toIntArray();
504        double[] doubles =  dic.values().toDoubleArray();
505        return ints[weightedRandom(doubles, r)];
506    }

508    /**
509     * Utility method to produce 1D int arrays this can process when discrete is true or difference is null.
510     * @param map a 2D char array
511     * @return an int array that can be used as a sample
512     */
513    public static int[] convertCharToInt(char[][] map)
514    {
515        int w = map.length, h = map[0].length;
516        int[] result = new int[w * h];
517        for (int x = 0; x < w; x++) {
518            for (int y = 0; y < h; y++) {
519                result[x * h + y] = map[x][y];
520            }
521        }
522        return result;
523    }

525    /**
526     * Utility method that takes a 1D int array that represents chars (such as a sample produced by
527     * {@link #convertCharToInt(char[][])} or, more likely, the result of processing such a sample with this class) and
528     * returns a 2D char array with the requested width and height (which should match the targetWidth and targetHeight
529     * given during processing).
530     * @param arr a 1D int array that represents char values
531     * @param w the width that arr can be interpreted as; should probably match the targetWidth given in processing
532     * @param h the height that arr can be interpreted as; should probably match the targetHeight given in processing
533     * @return a 2D char array with the given width and height, probably filled with the data from arr
534     */
535    public static char[][] convertIntToChar(int[] arr, int w, int h)
536    {
537        char[][] result = new char[w][h];
538        if(arr == null)
539            return result;
540        for (int x = 0; x < w; x++) {
541            for (int y = 0; y < h; y++) {
542                if(x * h + y >= arr.length)
543                    return result;
544                result[x][y] = (char) arr[x * h + y];
545            }
546        }
547        return result;
548    }
