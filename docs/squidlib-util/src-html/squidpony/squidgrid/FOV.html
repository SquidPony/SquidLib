010/**
011 * This class provides methods for calculating Field of View in grids. Field of
012 * View (FOV) algorithms determine how much area surrounding a point can be
013 * seen. They return a two dimensional array of doubles, representing the amount
014 * of view (typically sight, but perhaps sound, smell, etc.) which the origin
015 * has of each cell.
016 * <br>
017 * The input resistanceMap is considered the percent of opacity. This resistance
018 * is on top of the resistance applied from the light spreading out. You can
019 * obtain a resistance map easily with the DungeonUtility.generateResistances()
020 * method, which uses defaults for common chars used in SquidLib, but you may
021 * also want to create a resistance map manually if a given char means something
022 * very different in your game. This is easy enough to do by looping over all the
023 * x,y positions in your char[][] map and running a switch statement on each char,
024 * assigning a double to the same x,y position in a double[][]. The value should
025 * be between 0.0 (unblocked) for things light passes through, 1.0 (blocked) for
026 * things light can't pass at all, and possibly other values if you have
027 * translucent obstacles.
028 * <br>
029 * The returned light map is considered the percent of light in the cells.
030 * <br>
031 * Not all implementations are required to provide percentage levels of light.
032 * In such cases the returned values will be 0 for no light and 1.0 for fully
033 * lit. Implementations that return this way note so in their documentation.
034 * Currently, all implementations do provide percentage levels.
035 * <br>
036 * All solvers perform bounds checking so solid borders in the map are not
037 * required.
038 * <br>
039 * Static methods are provided to add together FOV maps in the simple way
040 * (disregarding visibility of distant FOV from a given cell), or the more
041 * practical way for roguelikes (where a cell needs to be within line-of-sight
042 * in the first place for a distant light to illuminate it). The second method
043 * relies on an LOS map, which is essentially the same as a very-high-radius
044 * FOV map and can be easily obtained with calculateLOSMap().
045 * <br>
046 * If you want to iterate through cells that are visible in a double[][] returned
047 * by FOV, you can pass that double[][] to the constructor for Region, and you
048 * can use the Region as a reliably-ordered List of Coord (among other things).
049 * The order Region iterates in is somewhat strange, and doesn't, for example,
050 * start at the center of an FOV map, but it will be the same every time you
051 * create a Region with the same FOV map (or the same visible Coords).
052 *
053 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
054 */
055public class FOV implements Serializable {

158    /**
159     * Calculates the Field Of View for the provided map from the given x, y
160     * coordinates. Returns a light map where the values represent a percentage
161     * of fully lit.
162     *
163     * The starting point for the calculation is considered to be at the center
164     * of the origin cell. Radius determinations are determined by the provided
165     * RadiusStrategy.
166     *
167     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
168     * @param startX the horizontal component of the starting location
169     * @param startY the vertical component of the starting location
170     * @param radius the distance the light will extend to
171     * @param radiusTechnique provides a means to calculate the radius as desired
172     * @return the computed light grid
173     */
174    public double[][] calculateFOV(double[][] resistanceMap, int startX, int startY, double radius, Radius radiusTechnique) {
175        double rad = Math.max(1, radius);
176        double decay = 1.0 / rad;
177
178        int width = resistanceMap.length;
179        int height = resistanceMap[0].length;
180
181        initializeLightMap(width, height);
182        light[startX][startY] = 1;//make the starting space full power
183
184        switch (type) {
185            case RIPPLE:
186            case RIPPLE_LOOSE:
187            case RIPPLE_TIGHT:
188            case RIPPLE_VERY_LOOSE:
189                initializeNearLight(width, height);
190                doRippleFOV(light, rippleValue(type), startX, startY, startX, startY, decay, rad, resistanceMap, nearLight, radiusTechnique);
191                break;
192            case SHADOW:
193                // hotfix for too large radius -> set to longest possible straight-line Manhattan distance instead
194                // does not cause problems with brightness falloff because shadowcasting is on/off
195
196                // this should be fixed now, sorta. the distance is checked in the method this calls, so it doesn't ever
197                // run through more than (width + height) iterations of the radius-related loop (which seemed to be the
198                // only problem, running through billions of iterations when Integer/MAX_VALUE is given as a radius).
199                for (Direction d : Direction.DIAGONALS) {
200                    shadowCast(1, 1.0, 0.0, 0, d.deltaX, d.deltaY, 0, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
201                    shadowCast(1, 1.0, 0.0, d.deltaX, 0, 0, d.deltaY, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
202                }
203                break;
204        }
205
206        return light;
207    }

209        /**
210     * Calculates the Field Of View for the provided map from the given x, y
211     * coordinates. Returns a light map where the values represent a percentage
212     * of fully lit.
213     *
214     * The starting point for the calculation is considered to be at the center
215     * of the origin cell. Radius determinations are determined by the provided
216     * RadiusStrategy. A conical section of FOV is lit by this method if
217     * span is greater than 0.
218     *
219     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
220     * @param startX the horizontal component of the starting location
221     * @param startY the vertical component of the starting location
222     * @param radius the distance the light will extend to
223     * @param radiusTechnique provides a means to calculate the radius as desired
224     * @param angle the angle in degrees that will be the center of the FOV cone, 0 points right
225     * @param span the angle in degrees that measures the full arc contained in the FOV cone
226     * @return the computed light grid
227     */
228    public double[][] calculateFOV(double[][] resistanceMap, int startX, int startY, double radius,
229                                   Radius radiusTechnique, double angle, double span) {
230
231        double rad = Math.max(1, radius);
232
233        double decay = 1.0 / rad;
234
235                double angle2 = Math.toRadians((angle > 360.0 || angle < 0.0)
236                                ? GwtCompatibility.IEEEremainder(angle + 720.0, 360.0) : angle);
237        double span2 = Math.toRadians(span);
238        int width = resistanceMap.length;
239        int height = resistanceMap[0].length;
240
241        initializeLightMap(width, height);
242        light[startX][startY] = 1;//make the starting space full power
243
244        switch (type) {
245            case RIPPLE:
246            case RIPPLE_LOOSE:
247            case RIPPLE_TIGHT:
248            case RIPPLE_VERY_LOOSE:
249                initializeNearLight(width, height);
250                doRippleFOV(light, rippleValue(type), startX, startY, startX, startY, decay, rad, resistanceMap, nearLight, radiusTechnique, angle2, span2);
251                break;
252            case SHADOW:
253                // this should be fixed now, sorta. the distance is checked in the method this calls, so it doesn't ever
254                // run through more than (width + height) iterations of the radius-related loop (which seemed to be the
255                // only problem, running through billions of iterations when Integer/MAX_VALUE is given as a radius).
256                int ctr = 0;
257                boolean started = false;
258                for (Direction d : ccw) {
259                    ctr %= 4;
260                    ++ctr;
261                    if (angle <= Math.PI / 2.0 * ctr + span / 2.0)
262                        started = true;
263                    if (started) {
264                        if(ctr < 4 && angle < Math.PI / 2.0 * (ctr - 1) - span / 2.0)
265                            break;
266                        light = shadowCastLimited(1, 1.0, 0.0, 0, d.deltaX, d.deltaY, 0, rad, startX, startY, decay, light, resistanceMap, radiusTechnique, angle2, span2);
267                        light = shadowCastLimited(1, 1.0, 0.0, d.deltaX, 0, 0, d.deltaY, rad, startX, startY, decay, light, resistanceMap, radiusTechnique, angle2, span2);
268                    }
269                }
270                break;
271        }
272
273        return light;
274    }

336     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()

345
346        double rad = Math.max(1, radius);
347        double decay = 1.0 / rad;
348        ArrayTools.fill(light, 0);
349        light[startX][startY] = 1;//make the starting space full power
350
351
352        shadowCast(1, 1.0, 0.0, 0, 1, 1, 0, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
353        shadowCast(1, 1.0, 0.0, 1, 0, 0, 1, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
354
355        shadowCast(1, 1.0, 0.0, 0, 1, -1, 0, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
356        shadowCast(1, 1.0, 0.0, 1, 0, 0, -1, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
357
358        shadowCast(1, 1.0, 0.0, 0, -1, -1, 0, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
359        shadowCast(1, 1.0, 0.0, -1, 0, 0, -1, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
360
361        shadowCast(1, 1.0, 0.0, 0, -1, 1, 0, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
362        shadowCast(1, 1.0, 0.0, -1, 0, 0, 1, rad, startX, startY, decay, light, resistanceMap, radiusTechnique);
363        return light;
364    }

373        private void initializeLightMap(int width, int height) {
374                if (light == null)
375                        light = new double[width][height];
376                else {
377                        if (light.length != width || light[0].length != height)
378                                /* Size changed */
379                                light = new double[width][height];
380                        else {
381                                /*
382                                 * Size did not change, we simply need to erase the previous
383                                 * result
384                                 */
385                                ArrayTools.fill(light, 0.0);
386                        }
387                }
388        }

429    private static void doRippleFOV(double[][] lightMap, int ripple, int x, int y, int startx, int starty, double decay, double radius, double[][] map, boolean[][] indirect, Radius radiusStrategy) {
430        /* Not using Deque's interface, it isn't GWT compatible */
431        final LinkedList<Coord> dq = new LinkedList<>();
432        int width = lightMap.length;
433        int height = lightMap[0].length;
434        dq.offer(Coord.get(x, y));
435        while (!dq.isEmpty()) {
436            Coord p = dq.removeFirst();
437            if (lightMap[p.x][p.y] <= 0 || indirect[p.x][p.y]) {
438                continue;//no light to spread
439            }
440
441            for (Direction dir : Direction.OUTWARDS) {
442                int x2 = p.x + dir.deltaX;
443                int y2 = p.y + dir.deltaY;
444                if (x2 < 0 || x2 >= width || y2 < 0 || y2 >= height //out of bounds
445                        || radiusStrategy.radius(startx, starty, x2, y2) >= radius + 1) {//+1 to cover starting tile
446                    continue;
447                }
448
449                double surroundingLight = nearRippleLight(x2, y2, ripple, startx, starty, decay, lightMap, map, indirect, radiusStrategy);
450                if (lightMap[x2][y2] < surroundingLight) {
451                    lightMap[x2][y2] = surroundingLight;
452                    if (map[x2][y2] < 1) {//make sure it's not a wall
453                        dq.offer(Coord.get(x2, y2));//redo neighbors since this one's light changed
454                    }
455                }
456            }
457        }
458    }

462    private static void doRippleFOV(double[][] lightMap, int ripple, int x, int y, int startx, int starty, double decay, double radius, double[][] map, boolean[][] indirect, Radius radiusStrategy, double angle, double span) {
463        /* Not using Deque's interface, it isn't GWT compatible */
464        final LinkedList<Coord> dq = new LinkedList<>();
465        int width = lightMap.length;
466        int height = lightMap[0].length;
467        dq.offer(Coord.get(x, y));
468        while (!dq.isEmpty()) {
469            Coord p = dq.removeFirst();
470            if (lightMap[p.x][p.y] <= 0 || indirect[p.x][p.y]) {
471                continue;//no light to spread
472            }
473
474            for (Direction dir : ccw_full) {
475                int x2 = p.x + dir.deltaX;
476                int y2 = p.y + dir.deltaY;
477                if (x2 < 0 || x2 >= width || y2 < 0 || y2 >= height //out of bounds
478                        || radiusStrategy.radius(startx, starty, x2, y2) >= radius + 1) {//+1 to cover starting tile
479                    continue;
480                }
481                double newAngle = Math.atan2(y2 - starty, x2 - startx) + Math.PI * 2;
482                                if (Math.abs(GwtCompatibility.IEEEremainder(angle - newAngle + Math.PI * 8, Math.PI * 2)) > span / 2.0)
483                                        continue;
484
485                double surroundingLight = nearRippleLight(x2, y2, ripple, startx, starty, decay, lightMap, map, indirect, radiusStrategy );
486                if (lightMap[x2][y2] < surroundingLight) {
487                    lightMap[x2][y2] = surroundingLight;
488                    if (map[x2][y2] < 1) {//make sure it's not a wall
489                        dq.offer(Coord.get(x2, y2));//redo neighbors since this one's light changed
490                    }
491                }
492            }
493        }

496    private static double nearRippleLight(int x, int y, int rippleNeighbors, int startx, int starty, double decay, double[][] lightMap, double[][] map, boolean[][] indirect, Radius radiusStrategy) {
497        if (x == startx && y == starty) {
498            return 1;
499        }
500        int width = lightMap.length;
501        int height = lightMap[0].length;
502        List<Coord> neighbors = new ArrayList<>();
503        double tmpDistance = 0, testDistance;
504        Coord c;
505        for (Direction di : Direction.OUTWARDS) {
506            int x2 = x + di.deltaX;
507            int y2 = y + di.deltaY;
508            if (x2 >= 0 && x2 < width && y2 >= 0 && y2 < height) {
509                tmpDistance = radiusStrategy.radius(startx, starty, x2, y2);
510                int idx = 0;
511                for(int i = 0; i < neighbors.size() && i <= rippleNeighbors; i++)
512                {
513                    c = neighbors.get(i);
514                    testDistance = radiusStrategy.radius(startx, starty, c.x, c.y);
515                    if(tmpDistance < testDistance) {
516                        break;
517                    }
518                    idx++;
519                }
520                neighbors.add(idx, Coord.get(x2, y2));
521            }
522        }
523
524        if (neighbors.isEmpty()) {
525            return 0;
526        }
527        neighbors = neighbors.subList(0, Math.min(neighbors.size(), rippleNeighbors));
528/*
529        while (neighbors.size() > rippleNeighbors) {
530            Coord p = neighbors.remove(0);
531            double dist = radiusStrategy.radius(startx, starty, p.x, p.y);
532            double dist2 = 0;
533            for (Coord p2 : neighbors) {
534                dist2 = Math.max(dist2, radiusStrategy.radius(startx, starty, p2.x, p2.y));
535            }
536            if (dist < dist2) {//not the largest, put it back
537                neighbors.add(p);
538            }
539        }
540*/
541        double light = 0;
542        int lit = 0, indirects = 0;
543        for (Coord p : neighbors) {
544            if (lightMap[p.x][p.y] > 0) {
545                lit++;
546                if (indirect[p.x][p.y]) {
547                    indirects++;
548                }
549                double dist = radiusStrategy.radius(x, y, p.x, p.y);
550                light = Math.max(light, lightMap[p.x][p.y] - dist * decay - map[p.x][p.y]);
551            }
552        }
553
554        if (map[x][y] >= 1 || indirects >= lit) {
555            indirect[x][y] = true;
556        }
557        return light;
558    }

560    private static double[][] shadowCast(int row, double start, double end, int xx, int xy, int yx, int yy,
561                                  double radius, int startx, int starty, double decay, double[][] lightMap,
562                                  double[][] map, Radius radiusStrategy) {
563        double newStart = 0;
564        if (start < end) {
565            return lightMap;
566        }
567        int width = lightMap.length;
568        int height = lightMap[0].length;
569
570        boolean blocked = false;
571        for (int distance = row; distance <= radius && distance < width + height && !blocked; distance++) {
572            int deltaY = -distance;
573            for (int deltaX = -distance; deltaX <= 0; deltaX++) {
574                int currentX = startx + deltaX * xx + deltaY * xy;
575                int currentY = starty + deltaX * yx + deltaY * yy;
576                double leftSlope = (deltaX - 0.5f) / (deltaY + 0.5f);
577                double rightSlope = (deltaX + 0.5f) / (deltaY - 0.5f);
578
579                if (!(currentX >= 0 && currentY >= 0 && currentX < width && currentY < height) || start < rightSlope) {
580                    continue;
581                } else if (end > leftSlope) {
582                    break;
583                }
584                double deltaRadius = radiusStrategy.radius(deltaX, deltaY);
585                //check if it's within the lightable area and light if needed
586                if (deltaRadius <= radius) {
587                    double bright = 1 - decay * deltaRadius;
588                    lightMap[currentX][currentY] = bright;
589                }
590
591                if (blocked) { //previous cell was a blocking one
592                    if (map[currentX][currentY] >= 1) {//hit a wall
593                        newStart = rightSlope;
594                    } else {
595                        blocked = false;
596                        start = newStart;
597                    }
598                } else {
599                    if (map[currentX][currentY] >= 1 && distance < radius) {//hit a wall within sight line
600                        blocked = true;
601                        lightMap = shadowCast(distance + 1, start, leftSlope, xx, xy, yx, yy, radius, startx, starty, decay, lightMap, map, radiusStrategy);
602                        newStart = rightSlope;
603                    }
604                }
605            }
606        }
607        return lightMap;
608    }

663    /**
664     * Adds an FOV map to another in the simplest way possible; does not check line-of-sight between FOV maps.
665     * Clamps the highest value for any single position at 1.0. Modifies the basis parameter in-place and makes no
666     * allocations; this is different from {@link #addFOVs(double[][]...)}, which creates a new 2D array.
667     * @param basis a 2D double array, which can be empty or returned by calculateFOV() or reuseFOV(); modified!
668     * @param addend another 2D double array that will be added into basis; this one will not be modified
669     * @return the sum of the 2D double arrays passed, using the dimensions of basis if they don't match
670     */
671    public static double[][] addFOVsInto(double[][] basis, double[][] addend)
672    {
673        for (int x = 0; x < basis.length && x < addend.length; x++) {
674                for (int y = 0; y < basis[x].length && y < addend[x].length; y++) {
675                    basis[x][y] = Math.min(1.0, basis[x][y] + addend[x][y]);
676                }
677            }
678        return basis;
679    }
680    /**
681     * Adds multiple FOV maps together in the simplest way possible; does not check line-of-sight between FOV maps.
682     * Clamps the highest value for any single position at 1.0.
683     * @param maps an array or vararg of 2D double arrays, each usually returned by calculateFOV()
684     * @return the sum of all the 2D double arrays passed, using the dimensions of the first if they don't all match
685     */
686    public static double[][] addFOVs(double[][]... maps)
687    {
688        if(maps == null || maps.length == 0)
689            return new double[0][0];
690        double[][] map = ArrayTools.copy(maps[0]);
691        for(int i = 1; i < maps.length; i++)
692        {
693            for (int x = 0; x < map.length && x < maps[i].length; x++) {
694                for (int y = 0; y < map[x].length && y < maps[i][x].length; y++) {
695                    map[x][y] += maps[i][x][y];
696                }
697            }
698        }
699        for (int x = 0; x < map.length; x++) {
700            for (int y = 0; y < map[x].length; y++) {
701                if(map[x][y] > 1.0) map[x][y] = 1.0;
702            }
703        }
704        return map;
705    }

707    /**
708     * Adds multiple FOV maps together in the simplest way possible; does not check line-of-sight between FOV maps.
709     * Clamps the highest value for any single position at 1.0.
710     * @param maps an Iterable of 2D double arrays (most collections implement Iterable),
711     *             each usually returned by calculateFOV()
712     * @return the sum of all the 2D double arrays passed, using the dimensions of the first if they don't all match
713     */
714    public static double[][] addFOVs(Iterable<double[][]> maps)
715    {
716        if(maps == null)
717            return new double[0][0];
718        Iterator<double[][]> it = maps.iterator();
719        if(!it.hasNext())
720            return new double[0][0];
721        double[][] map = ArrayTools.copy(it.next()), t;
722        while (it.hasNext())
723        {
724            t = it.next();
725            for (int x = 0; x < map.length && x < t.length; x++) {
726                for (int y = 0; y < map[x].length && y < t[x].length; y++) {
727                    map[x][y] += t[x][y];
728                }
729            }
730        }
731        for (int x = 0; x < map.length; x++) {
732            for (int y = 0; y < map[x].length; y++) {
733                if(map[x][y] > 1.0) map[x][y] = 1.0;
734            }
735        }
736        return map;
737    }

739    /**
740     * Adds together multiple FOV maps, but only adds to a position if it is visible in the given LOS map. Useful if
741     * you want distant lighting to be visible only if the player has line-of-sight to a lit cell. Typically the LOS map
742     * is calculated by calculateLOSMap(), using the same resistance map used to calculate the FOV maps.
743     * Clamps the highest value for any single position at 1.0.
744     * @param losMap an LOS map such as one generated by calculateLOSMap()
745     * @param maps an array or vararg of 2D double arrays, each usually returned by calculateFOV()
746     * @return the sum of all the 2D double arrays in maps where a cell was visible in losMap
747     */
748    public static double[][] mixVisibleFOVs(double[][] losMap, double[][]... maps)
749    {
750        if(losMap == null || losMap.length == 0)
751            return addFOVs(maps);
752        double[][] map = new double[losMap.length][losMap[0].length];
753        if(maps == null || maps.length == 0)
754            return map;
755        for(int i = 0; i < maps.length; i++)
756        {
757            for (int x = 0; x < losMap.length && x < map.length && x < maps[i].length; x++) {
758                for (int y = 0; y < losMap[x].length && y < map[x].length && y < maps[i][x].length; y++) {
759                    if(losMap[x][y] > 0.0001) {
760                        map[x][y] += maps[i][x][y];
761                        if(map[x][y] > 1.0) map[x][y] = 1.0;
762                    }
763                }
764            }
765        }
766        return map;
767    }
