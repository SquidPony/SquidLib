001package squidpony.squidgrid.zone;
002
003import squidpony.squidmath.Coord;
004
005import java.io.Serializable;
006import java.util.Iterator;
007import java.util.List;
008
009/**
010 * Abstraction over a list of {@link Coord}. This allows to use the short arrays
011 * coming from {@link squidpony.squidmath.CoordPacker}, which are compressed for
012 * better memory usage, regular {@link List lists of Coord}, which are often the
013 * simplest option, or {@link squidpony.squidmath.GreasedRegion GreasedRegions},
014 * which are "greasy" in the fatty-food sense (they are heavier objects, and are
015 * uncompressed) but also "greased" like greased lightning (they are very fast at
016 * spatial transformations on their region).
017 * <p>
018 * Zones are {@link Serializable}, but serialization doesn't change the internal
019 * representation (some would want to pack {@link ListZone} into
020 * {@link CoordPackerZone}s when serializing). I find that overzealous for a
021 * simple interface. If you want your zones to be be packed when serialized,
022 * create {@link CoordPackerZone} yourself. In squidlib-extra, GreasedRegions are
023 * given slightly special treatment during that JSON-like serialization so they
024 * avoid repeating certain information, but they are still going to be larger than
025 * compressed short arrays from CoordPacker.
026 * </p>
027 * <p>
028 * While CoordPacker produces short arrays that can be wrapped in CoordPackerZone
029 * objects, and a List of Coord can be similarly wrapped in a ListZone object,
030 * GreasedRegion extends {@link Zone.Skeleton} and so implements Zone itself.
031 * Unlike CoordPackerZone, which is immutable in practice (changing the short
032 * array reference is impossible and changing the elements rarely works as
033 * planned), GreasedRegion is mutable for performance reasons, and may need copies
034 * to be created if you want to keep around older GreasedRegions.
035 * </p>
036 * @author smelC
037 * @see squidpony.squidmath.CoordPacker
038 * @see squidpony.squidmath.GreasedRegion
039 */
040public interface Zone extends Serializable, Iterable<Coord> {
041
042    /**
043     * @return Whether this zone is empty.
044     */
045    boolean isEmpty();
046
047    /**
048     * @return The number of cells that this zone contains (the size
049     * {@link #getAll()}).
050     */
051    int size();
052
053    /**
054     * @param x
055     * @param y
056     * @return Whether this zone contains the coordinate (x,y).
057     */
058    boolean contains(int x, int y);
059
060    /**
061     * @param c
062     * @return Whether this zone contains {@code c}.
063     */
064    boolean contains(Coord c);
065
066        /**
067         * @param other
068         * @return true if all cells of {@code other} are in {@code this}.
069         */
070        boolean contains(Zone other);
071
072        /**
073         * @param other
074         * @return true if {@code this} and {@code other} have a common cell.
075         */
076        boolean intersectsWith(Zone other);
077
078        /**
079         * @return The approximate center of this zone, or null if this zone
080         *         is empty.
081         */
082        /* @Nullable */ Coord getCenter();
083
084        /**
085         * @return The distance between the leftmost cell and the rightmost cell, or
086         *         anything negative if {@code this} zone is empty.
087         */
088        int getWidth();
089
090        /**
091         * @return The distance between the topmost cell and the lowest cell, or
092         *         anything negative if {@code this} zone is empty.
093         */
094        int getHeight();
095
096        /**
097         * @param smallestOrBiggest
098         * @return The x-coordinate of the Coord within {@code this} that has the
099         *         smallest (or biggest) x-coordinate. Or -1 if the zone is empty.
100         */
101        int x(boolean smallestOrBiggest);
102
103        /**
104         * @param smallestOrBiggest
105         * @return The y-coordinate of the Coord within {@code this} that has the
106         *         smallest (or biggest) y-coordinate. Or -1 if the zone is empty.
107         */
108        int y(boolean smallestOrBiggest);
109
110    /**
111     * @return All cells in this zone.
112     */
113    List<Coord> getAll();
114
115    /**
116     * A convenience partial implementation. Please try for all new
117     * implementations of {@link Zone} to be subtypes of this class. It usually
118     * prove handy at some point to have a common superclass.
119     *
120     * @author smelC
121     */
122    abstract class Skeleton implements Zone {
123
124                private transient Coord center = null;
125                private transient int width = -2;
126                private transient int height = -2;
127
128                private static final long serialVersionUID = 4436698111716212256L;
129
130                @Override
131                /* Convenience implementation, feel free to override */
132                public boolean contains(Coord c) {
133                        return contains(c.x, c.y);
134                }
135
136                @Override
137                /* Convenience implementation, feel free to override */
138                public boolean contains(Zone other) {
139                        for (Coord c : other) {
140                                if (!contains(c))
141                                        return false;
142                        }
143                        return true;
144                }
145
146                @Override
147                public boolean intersectsWith(Zone other) {
148                        final int tsz = size();
149                        final int osz = other.size();
150                        final Iterable<Coord> iteratedOver = tsz < osz ? this : other;
151                        final Zone other_ = tsz < osz ? other : this;
152                        for (Coord c : iteratedOver) {
153                                if (other_.contains(c))
154                                        return true;
155                        }
156                        return false;
157                }
158
159                @Override
160                /*
161                 * Convenience implementation, feel free to override, in particular if
162                 * you can avoid allocating the list usually allocated by getAll().
163                 */
164                public Iterator<Coord> iterator() {
165                        return getAll().iterator();
166                }
167
168                @Override
169                /* Convenience implementation, feel free to override. */
170                public int getWidth() {
171                        if (width == -2)
172                                width = isEmpty() ? -1 : x(false) - x(true);
173                        return width;
174                }
175
176                @Override
177                /* Convenience implementation, feel free to override. */
178                public int getHeight() {
179                        if (height == -2)
180                                height = isEmpty() ? -1 : y(false) - y(true);
181                        return height;
182                }
183
184                @Override
185                /* Convenience implementation, feel free to override. */
186                public int x(boolean smallestOrBiggest) {
187                        return smallestOrBiggest ? smallest(true) : biggest(true);
188                }
189
190                @Override
191                /* Convenience implementation, feel free to override. */
192                public int y(boolean smallestOrBiggest) {
193                        return smallestOrBiggest ? smallest(false) : biggest(false);
194                }
195
196                @Override
197                /* Convenience implementation, feel free to override. */
198                /*
199                 * A possible enhancement would be to check that the center is within
200                 * the zone, and if not to return the coord closest to the center, that
201                 * is in the zone .
202                 */
203                public /* @Nullable */ Coord getCenter() {
204                        if (center == null) {
205                                /* Need to compute it */
206                                if (isEmpty())
207                                        return null;
208                                int x = 0, y = 0;
209                                float nb = 0;
210                                for (Coord c : this) {
211                                        x += c.x;
212                                        y += c.y;
213                                        nb++;
214                                }
215                                /* Remember it */
216                                center = Coord.get(Math.round(x / nb), Math.round(y / nb));
217                        }
218                        return center;
219                }
220
221                private int smallest(boolean xOrY) {
222                        if (isEmpty())
223                                return -1;
224                        int min = Integer.MAX_VALUE;
225                        for (Coord c : this) {
226                                final int val = xOrY ? c.x : c.y;
227                                if (val < min)
228                                        min = val;
229                        }
230                        return min;
231                }
232
233                private int biggest(boolean xOrY) {
234                        int max = -1;
235                        for (Coord c : this) {
236                                final int val = xOrY ? c.x : c.y;
237                                if (max < val)
238                                        max = val;
239                        }
240                        return max;
241                }
242        }
243}
