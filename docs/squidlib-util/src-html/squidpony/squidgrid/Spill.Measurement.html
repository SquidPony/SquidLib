007/**
008 * A randomized flood-fill implementation that can be used for level generation (e.g. filling ponds and lakes), for
009 * gas propagation, or for all sorts of fluid-dynamics-on-the-cheap.
010 * Created by Tommy Ettinger on 4/7/2015.
011 * 
012 * @see Splash An alternative implementation with a lighter API
013 */
014public class Spill {
015    /**
016     * The type of heuristic to use.
017     */
018    public enum Measurement {
019
020        /**
021         * The distance it takes when only the four primary directions can be
022         * moved in. The default.
023         */
024        MANHATTAN,
025        /**
026         * The distance it takes when diagonal movement costs the same as
027         * cardinal movement.
028         */
029        CHEBYSHEV,
030        /**
031         * The distance it takes as the crow flies.
032         */
033        EUCLIDEAN
034    }
035
036    /**
037     * This affects how distance is measured on diagonal directions vs. orthogonal directions. MANHATTAN should form a
038     * diamond shape on a featureless map, while CHEBYSHEV and EUCLIDEAN will form a square. If you only call
039     * Spill.start() once, you should strongly prefer MANHATTAN, even if the rest of the game uses another
040     * measurement, because CHEBYSHEV and EUCLIDEAN can produce odd, gap-filled flood-fills.  Any case where you have
041     * too many gaps can be corrected to varying extent by calling start() more than once with slowly increasing values.
042     * Because start() will extend from the existing area of the Spill, holes are likely to be filled after a few calls,
043     * but if the last call to start() tries to fill too many more cells than the previous one, it can cause holes on
044     * the periphery of the Spill area.
045     */
046    public Measurement measurement = Measurement.MANHATTAN;

236    /**
237     * Used to initialize or re-initialize a Spill that needs a new PhysicalMap because it either wasn't given
238     * one when it was constructed, or because the contents of the terrain have changed permanently (not if a
239     * creature moved; for that you pass the positions of creatures that block paths to scan() or findPath() ).
240     * @param level
241     * @return
242     */
243    public Spill initialize(final boolean[][] level) {
244        fresh = new OrderedSet<>();
245        width = level.length;
246        height = level[0].length;
247        spillMap = new boolean[width][height];
248        physicalMap = new boolean[width][height];
249        for (int y = 0; y < height; y++) {
250            for (int x = 0; x < width; x++) {
251                spillMap[x][y] = level[x][y];
252                physicalMap[x][y] = level[x][y];
253            }
254        }
255        initialized = true;
256        return this;
257    }

311        if(!initialized) return;
312        for (int y = 0; y < height; y++) {
313            for (int x = 0; x < width; x++) {
314                spillMap[x][y] = false;
315            }
316        }

357    /**
358     * Recalculate the spillMap and return the spreadPattern. The cell corresponding to entry will be true,
359     * the cells near that will be true if chosen at random from all passable cells adjacent to a
360     * filled (true) cell, and all other cells will be false. This takes a total number of cells to attempt
361     * to fill (the volume parameter), and will fill less if it has completely exhausted all passable cells.
362     * If the measurement this Spill uses is anything other than MANHATTAN, you can expect many gaps in the first
363     * filled area.  Subsequent calls to start() with the same entry and a higher volume will expand the area
364     * of the Spill, and are likely to fill any gaps after a few subsequent calls. Increasing the volume slowly
365     * is the best way to ensure that gaps only exist on the very edge if you use a non-MANHATTAN measurement.
366     *
367     * @param entry The first cell to spread from, which should really be passable.
368     * @param volume The total number of cells to attempt to fill, which must be non-negative.
369     * @param impassable A Set of Position keys representing the locations of moving obstacles to a
370     *                   path that cannot be moved through; this can be null if there are no such obstacles.
371     * @return An ArrayList of Points that this will enter, in order starting with entry at index 0, until it
372     * reaches its volume or fills its boundaries completely.
373     */
374    public ArrayList<Coord> start(Coord entry, int volume, Set<Coord> impassable) {
375        if(!initialized) return null;
376        if(impassable == null)
377            impassable = new OrderedSet<>();
378        if(!physicalMap[entry.x][entry.y] || impassable.contains(entry))
379            return null;
380        spreadPattern = new ArrayList<>(volume);
381        spillMap[entry.x][entry.y] = true;
382        Coord temp;
383        for(int x = 0; x < spillMap.length; x++)
384        {
385            for(int y = 0; y < spillMap[x].length; y++)
386            {
387                temp = Coord.get(x, y);
388                if(spillMap[x][y] && !impassable.contains(temp))
389                    fresh.add(temp);
390            }
391        }
392
393        Direction[] dirs = (measurement == Measurement.MANHATTAN) ? Direction.CARDINALS : Direction.OUTWARDS;
394        while (!fresh.isEmpty() && spreadPattern.size() < volume) {
395            Coord cell = fresh.randomItem(rng);//toArray(new Coord[fresh.size()])[rng.nextInt(fresh.size())];
396            spreadPattern.add(cell);
397            spillMap[cell.x][cell.y] = true;
398            for (int d = 0; d < dirs.length; d++) {
399                Coord adj = cell.translate(dirs[d].deltaX, dirs[d].deltaY);
400                double h = heuristic(dirs[d]);
401                if (physicalMap[adj.x][adj.y] && !spillMap[adj.x][adj.y] && !impassable.contains(adj) && rng.nextDouble() <= 1.0 / h) {
402                    setFresh(adj);
403                }
404            }
405            fresh.remove(cell);
406        }
407        filled = spreadPattern.size();
408        return spreadPattern;
409    }
410
411    private static final double root2 = Math.sqrt(2.0);
412    private double heuristic(Direction target) {
413        switch (measurement) {
414            case MANHATTAN:
415            case CHEBYSHEV:
416                return 1.0;
417            case EUCLIDEAN:
418                switch (target) {
419                    case DOWN_LEFT:
420                    case DOWN_RIGHT:
421                    case UP_LEFT:
422                    case UP_RIGHT:
423                        return root2;
424                    default:
425                        return  1.0;
426                }
427        }
428        return 1.0;
429    }
