001package squidpony.squidgrid;
002
003import squidpony.squidmath.Coord;
004
005/**
006 * Represents the eight grid directions and the deltaX, deltaY values associated
007 * with those directions.
008 *
009 * The grid referenced has x positive to the right and y positive downwards on
010 * screen.
011 *
012 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
013 */
014public enum Direction {
015
016    UP(0, -1), DOWN(0, 1), LEFT(-1, 0), RIGHT(1, 0), UP_LEFT(-1, -1), UP_RIGHT(1, -1), DOWN_LEFT(-1, 1), DOWN_RIGHT(1, 1), NONE(0, 0);
017    /**
018     * An array which holds only the four cardinal directions.
019     */
020    public static final Direction[] CARDINALS = {UP, DOWN, LEFT, RIGHT};
021    /**
022     * An array which holds only the four cardinal directions in clockwise order.
023     */
024    public static final Direction[] CARDINALS_CLOCKWISE = {UP, RIGHT, DOWN, LEFT};
025    /**
026     * An array which holds only the four cardinal directions in counter-clockwise order.
027     */
028    public static final Direction[] CARDINALS_COUNTERCLOCKWISE = {UP, LEFT, DOWN, RIGHT};
029    /**
030     * An array which holds only the four diagonal directions.
031     */
032    public static final Direction[] DIAGONALS = {UP_LEFT, UP_RIGHT, DOWN_LEFT, DOWN_RIGHT};
033    /**
034     * An array which holds all eight OUTWARDS directions.
035     */
036    public static final Direction[] OUTWARDS = {UP, DOWN, LEFT, RIGHT, UP_LEFT, UP_RIGHT, DOWN_LEFT, DOWN_RIGHT};
037    /**
038     * An array which holds all eight OUTWARDS directions in clockwise order.
039     */
040    public static final Direction[] CLOCKWISE = {UP, UP_RIGHT, RIGHT, DOWN_RIGHT, DOWN, DOWN_LEFT, LEFT, UP_LEFT};
041    /**
042     * An array which holds all eight OUTWARDS directions in counter-clockwise order.
043     */
044    public static final Direction[] COUNTERCLOCKWISE = {UP, UP_LEFT, LEFT, DOWN_LEFT, DOWN, DOWN_RIGHT, RIGHT, UP_RIGHT};
045    /**
046     * The x coordinate difference for this direction.
047     */
048    public final int deltaX;
049    /**
050     * The y coordinate difference for this direction.
051     */
052    public final int deltaY;
053
054    /**
055     * Returns the direction that most closely matches the input.
056     *
057     * This can be used to get the primary magnitude intercardinal direction
058     * from an origin point to an event point, such as a mouse click on a grid.
059     *
060     * If the point given is exactly on a boundary between directions then the
061     * direction clockwise is returned.
062     *
063     * @param x
064     * @param y
065     * @return
066     */
067    public static Direction getDirection(int x, int y) {
068        if (x == 0 && y == 0) {
069            return NONE;
070        }
071
072        double angle = Math.atan2(y, x);
073        double degree = Math.toDegrees(angle);
074        degree += 450;//rotate to all positive and 0 is up
075        degree %= 360;//normalize
076        if (degree < 22.5) {
077            return UP;
078        } else if (degree < 67.5) {
079            return UP_RIGHT;
080        } else if (degree < 112.5) {
081            return RIGHT;
082        } else if (degree < 157.5) {
083            return DOWN_RIGHT;
084        } else if (degree < 202.5) {
085            return DOWN;
086        } else if (degree < 247.5) {
087            return DOWN_LEFT;
088        } else if (degree < 292.5) {
089            return LEFT;
090        } else if (degree < 337.5) {
091            return UP_LEFT;
092        } else {
093            return UP;
094        }
095    }
096
097    /**
098     * Returns the direction that most closely matches the input.
099     *
100     * This can be used to get the primary magnitude cardinal direction from an
101     * origin point to an event point, such as a mouse click on a grid.
102     *
103     * If the point given is directly diagonal then the direction clockwise is
104     * returned.
105     *
106     * @param x
107     * @param y
108     * @return
109     */
110    public static Direction getCardinalDirection(int x, int y) {
111        if (x == 0 && y == 0) {
112            return NONE;
113        }
114
115        int absx = Math.abs(x);
116
117        if (y > absx) {
118            return UP;
119        }
120
121        int absy = Math.abs(y);
122
123        if (absy > absx) {
124            return DOWN;
125        }
126
127        if (x > 0) {
128            if (-y == x) {//on diagonal
129                return DOWN;
130            }
131            return RIGHT;
132        }
133
134        if (y == x) {//on diagonal
135            return UP;
136        }
137        return LEFT;
138
139    }
140
141        /**
142         * @param from
143         *            The starting point.
144         * @param to
145         *            The desired point to reach.
146         * @return The direction to follow to go from {@code from} to {@code to}. It
147         *         can be cardinal or diagonal.
148         */
149        public static Direction toGoTo(Coord from, Coord to) {
150                return getDirection(to.x - from.x, to.y - from.y);
151        }
152
153    /**
154     * Returns the Direction one step clockwise including diagonals.
155     *
156     * If considering only Cardinal directions, calling this twice will get the
157     * next clockwise cardinal direction.
158     *
159     * @return
160     */
161    public Direction clockwise() {
162        switch (this) {
163            case UP:
164                return UP_RIGHT;
165            case DOWN:
166                return DOWN_LEFT;
167            case LEFT:
168                return UP_LEFT;
169            case RIGHT:
170                return DOWN_RIGHT;
171            case UP_LEFT:
172                return UP;
173            case UP_RIGHT:
174                return RIGHT;
175            case DOWN_LEFT:
176                return LEFT;
177            case DOWN_RIGHT:
178                return DOWN;
179            case NONE:
180            default:
181                return NONE;
182        }
183    }
184
185    /**
186     * Returns the Direction one step counterclockwise including diagonals.
187     *
188     * If considering only Cardinal directions, calling this twice will get the
189     * next counterclockwise cardinal direction.
190     *
191     * @return
192     */
193    public Direction counterClockwise() {
194        switch (this) {
195            case UP:
196                return UP_LEFT;
197            case DOWN:
198                return DOWN_RIGHT;
199            case LEFT:
200                return DOWN_LEFT;
201            case RIGHT:
202                return UP_RIGHT;
203            case UP_LEFT:
204                return LEFT;
205            case UP_RIGHT:
206                return UP;
207            case DOWN_LEFT:
208                return DOWN;
209            case DOWN_RIGHT:
210                return RIGHT;
211            case NONE:
212            default:
213                return NONE;
214        }
215    }
216
217    /**
218     * Returns the direction directly opposite of this one.
219     *
220     * @return
221     */
222    public Direction opposite() {
223        switch (this) {
224            case UP:
225                return DOWN;
226            case DOWN:
227                return UP;
228            case LEFT:
229                return RIGHT;
230            case RIGHT:
231                return LEFT;
232            case UP_LEFT:
233                return DOWN_RIGHT;
234            case UP_RIGHT:
235                return DOWN_LEFT;
236            case DOWN_LEFT:
237                return UP_RIGHT;
238            case DOWN_RIGHT:
239                return UP_LEFT;
240            case NONE:
241            default:
242                return NONE;
243        }
244    }
245
246    /**
247     * @return Whether this is a diagonal move.
248     */
249    public boolean isDiagonal() {
250        return (deltaX & deltaY) != 0;
251    }
252
253    /**
254     * @return Whether this is a cardinal-direction move.
255     */
256    public boolean isCardinal() {
257        return (deltaX + deltaY & 1) != 0;
258    }
259
260        /**
261         * @return {@code true} if {@code this} has an upward component.
262         */
263        public boolean hasUp() {
264                switch (this) {
265                case UP:
266                case UP_LEFT:
267                case UP_RIGHT:
268                        return true;
269                case DOWN:
270                case DOWN_LEFT:
271                case DOWN_RIGHT:
272                case LEFT:
273                case NONE:
274                case RIGHT:
275                        return false;
276                }
277                throw new IllegalStateException("Unmatched " + getClass().getSimpleName() + ": " + this);
278        }
279
280        /**
281         * @return {@code true} if {@code this} has a downward component.
282         */
283        public boolean hasDown() {
284                switch (this) {
285                case DOWN:
286                case DOWN_LEFT:
287                case DOWN_RIGHT:
288                        return true;
289                case LEFT:
290                case NONE:
291                case RIGHT:
292                case UP:
293                case UP_LEFT:
294                case UP_RIGHT:
295                        return false;
296                }
297                throw new IllegalStateException("Unmatched " + getClass().getSimpleName() + ": " + this);
298        }
299
300        /**
301         * @return {@code true} if {@code this} has a left component.
302         */
303        public boolean hasLeft() {
304                switch (this) {
305                case DOWN_LEFT:
306                case LEFT:
307                case UP_LEFT:
308                        return true;
309                case DOWN:
310                case DOWN_RIGHT:
311                case NONE:
312                case RIGHT:
313                case UP:
314                case UP_RIGHT:
315                        return false;
316                }
317                throw new IllegalStateException("Unmatched " + getClass().getSimpleName() + ": " + this);
318        }
319
320        /**
321         * @return {@code true} if {@code this} has a right component.
322         */
323        public boolean hasRight() {
324                switch (this) {
325                case RIGHT:
326                case DOWN_RIGHT:
327                case UP_RIGHT:
328                        return true;
329                case DOWN:
330                case NONE:
331                case UP:
332                case DOWN_LEFT:
333                case LEFT:
334                case UP_LEFT:
335                        return false;
336                }
337                throw new IllegalStateException("Unmatched " + getClass().getSimpleName() + ": " + this);
338        }
339
340    Direction(int x, int y) {
341        deltaX = x;
342        deltaY = y;
343    }
