008/**
009 * Line of Sight (LOS) algorithms find if there is or is not a path between two
010 * given points.
011 * <br>
012 * The line found between two points will end at either the target, the
013 * obstruction closest to the start, or the edge of the map.
014 * <br>
015 * For normal line of sight usage, you should prefer Bresenham lines, and these
016 * are the default (they can also be specified by passing {@link #BRESENHAM} to
017 * the constructor). For more specialized usage, there are other kinds of LOS in
018 * this class, like lines that make no diagonal moves between cells (using
019 * {@link #ORTHO}, or lines that check a wide path (but these use different
020 * methods, like {@link #thickReachable(Radius)}).
021 * <br>
022 * Performance-wise, all of these methods are rather fast and about the same speed.
023 * {@link #RAY} is a tiny fraction faster than {@link #BRESENHAM} but produces
024 * rather low-quality lines in comparison. Calculating the visibility of 40,000
025 * lines in a 102x102 dungeon takes within 3% of 950ms (on an Intel i7-4700MQ laptop
026 * processor) for every one of BRESENHAM, DDA, ORTHO, and RAY, even with ORTHO
027 * finding a different kind of line by design.
028 *
029 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
030 * @author Tommy Ettinger Added DDA, ORTHO, and the thick lines; some cleanup
031 * @author smelC optimized several methods
032 */
033public class LOS {

120    /**
121     * Returns true if a line can be drawn from the start point to the target
122     * point without intervening obstructions.
123     *
124     * Uses RadiusStrategy.CIRCLE, or whatever RadiusStrategy was set with setRadiusStrategy .
125     *
126     * @param walls '#' is fully opaque, anything else is fully transparent, as always this uses x,y indexing.
127     * @param startx starting x position on the grid
128     * @param starty starting y position on the grid
129     * @param targetx ending x position on the grid
130     * @param targety ending y position on the grid
131     * @return true if a line can be drawn without being obstructed, false otherwise
132     */
133    public boolean isReachable(char[][] walls, int startx, int starty, int targetx, int targety) {
134        if(walls.length < 1) return false;
135        double[][] resMap = new double[walls.length][walls[0].length];
136        for(int x = 0; x < walls.length; x++)
137        {
138            for(int y = 0; y < walls[0].length; y++)
139            {
140                resMap[x][y] = (walls[x][y] == '#') ? 1.0 : 0.0;
141            }
142        }
143        return isReachable(resMap, startx, starty, targetx, targety, radiusStrategy);
144    }

179        this.resistanceMap = resistanceMap;
180        this.startx = startx;
181        this.starty = starty;

183        this.targety = targety;

195            case DDA:
196                return ddaReachable(radiusStrategy);
197            case THICK:
198                return thickReachable(radiusStrategy);
199        }
200        return false;
201    }

315    private boolean bresenhamReachable(Radius radiusStrategy) {
316        Coord[] path = Bresenham.line2D_(startx, starty, targetx, targety);
317        lastPath = new LinkedList<>();
318        double rad = radiusStrategy.radius(startx, starty, targetx, targety);
319        if(rad == 0.0) {
320            lastPath.add(Coord.get(startx, starty));
321            return true; // already at the point; we can see our own feet just fine!
322        }
323        double decay = 1 / rad;
324        double currentForce = 1;
325        Coord p;
326        for (int i = 0; i < path.length; i++) {
327            p = path[i];
328            lastPath.offer(p);
329            if (p.x == targetx && p.y == targety) {
330                return true;//reached the end
331            }
332            if (p.x != startx || p.y != starty) {//don't discount the start location even if on resistant cell
333                currentForce -= resistanceMap[p.x][p.y];
334            }
335            double r = radiusStrategy.radius(startx, starty, p.x, p.y);
336            if (currentForce - (r * decay) <= 0) {
337                return false;//too much resistance
338            }
339        }
340        return false;//never got to the target point
341    }

400    private boolean thickReachable(Radius radiusStrategy) {
401        lastPath = new LinkedList<>();
402        double dist = radiusStrategy.radius(startx, starty, targetx, targety);
403        double decay = 1.0 / dist; // note: decay can be positive infinity if dist is 0; this is actually OK
404        OrderedSet<Coord> visited = new OrderedSet<>((int) dist + 3);
405        List<List<Coord>> paths = new ArrayList<>(4);
406        /* // actual corners
407        paths.add(DDALine.line(startx, starty, targetx, targety, 0, 0));
408        paths.add(DDALine.line(startx, starty, targetx, targety, 0, 0xffff));
409        paths.add(DDALine.line(startx, starty, targetx, targety, 0xffff, 0));
410        paths.add(DDALine.line(startx, starty, targetx, targety, 0xffff, 0xffff));
411        */
412        // halfway between the center and a corner
413        paths.add(DDALine.line(startx, starty, targetx, targety, 0x3fff, 0x3fff));
414        paths.add(DDALine.line(startx, starty, targetx, targety, 0x3fff, 0xbfff));
415        paths.add(DDALine.line(startx, starty, targetx, targety, 0xbfff, 0x3fff));
416        paths.add(DDALine.line(startx, starty, targetx, targety, 0xbfff, 0xbfff));
417
418        int length = Math.max(paths.get(0).size(), Math.max(paths.get(1).size(),
419                Math.max(paths.get(2).size(), paths.get(3).size())));
420        double[] forces = new double[]{1,1,1,1};
421        boolean[] go = new boolean[]{true, true, true, true};
422        Coord p;
423        for (int d = 0; d < length; d++) {
424            for (int pc = 0; pc < 4; pc++) {
425                List<Coord> path = paths.get(pc);
426                if(d < path.size() && go[pc])
427                    p = path.get(d);
428                else continue;
429                if (p.x == targetx && p.y == targety) {
430                    visited.add(p);
431                    lastPath.addAll(visited);
432                    return true;//reached the end
433                }
434                if (p.x != startx || p.y != starty) {//don't discount the start location even if on resistant cell
435                    forces[pc] -= resistanceMap[p.x][p.y];
436                }
437                double r = radiusStrategy.radius(startx, starty, p.x, p.y);
438                if (forces[pc] - (r * decay) <= 0) {
439                    go[pc] = false;
440                    continue;//too much resistance
441                }
442                visited.add(p);
443            }
444        }
445        lastPath.addAll(visited);
446        return false;//never got to the target point
447    }

449    private boolean brushReachable(Radius radiusStrategy, int spread) {
450        lastPath = new LinkedList<>();
451        double dist = radiusStrategy.radius(startx, starty, targetx, targety) + spread * 2, decay = 1 / dist;
452        OrderedSet<Coord> visited = new OrderedSet<>((int) (dist + 3) * spread);
453        List<List<Coord>> paths = new ArrayList<>((int) (radiusStrategy.volume2D(spread) * 1.25));
454        int length = 0;
455        List<Coord> currentPath;
456        for (int i = -spread; i <= spread; i++) {
457            for (int j = -spread; j <= spread; j++) {
458                if(radiusStrategy.inRange(startx, starty, startx + i, starty + j, 0, spread)
459                        && startx + i >= 0 && starty + j >= 0
460                        && startx + i < resistanceMap.length && starty + j < resistanceMap[0].length
461                        && targetx + i >= 0 && targety + j >= 0
462                        && targetx + i < resistanceMap.length && targety + j < resistanceMap[0].length) {
463                    for (int q = 0x3fff; q < 0xffff; q += 0x8000) {
464                        for (int r = 0x3fff; r < 0xffff; r += 0x8000) {
465                            currentPath = DDALine.line(startx+i, starty+j, targetx+i, targety+j, q, r);
466                            paths.add(currentPath);
467                            length = Math.max(length, currentPath.size());
468                        }
469                    }
470                }
471            }
472        }
473        double[] forces = new double[paths.size()];
474        Arrays.fill(forces, 1.0);
475        boolean[] go = new boolean[paths.size()];
476        Arrays.fill(go, true);
477        Coord p;
478        boolean found = false;
479        for (int d = 0; d < length; d++) {
480            for (int pc = 0; pc < paths.size(); pc++) {
481                List<Coord> path = paths.get(pc);
482                if(d < path.size() && go[pc])
483                    p = path.get(d);
484                else continue;
485                if (p.x == targetx && p.y == targety) {
486                    found = true;
487                }
488                if (p.x != startx || p.y != starty) {//don't discount the start location even if on resistant cell
489                    forces[pc] -= resistanceMap[p.x][p.y];
490                }
491                double r = radiusStrategy.radius(startx, starty, p.x, p.y);
492                if (forces[pc] - (r * decay) <= 0) {
493                    go[pc] = false;
494                    continue;//too much resistance
495                }
496                visited.add(p);
497            }
498        }
499        lastPath.addAll(visited);
500        return found;//never got to the target point
501    }

503    private boolean rayReachable(Radius radiusStrategy) {
504        lastPath = new LinkedList<>();//save path for later retrieval
505        if (startx == targetx && starty == targety) {//already there!
506            lastPath.add(Coord.get(startx, starty));
507            return true;
508        }
509
510        int width = resistanceMap.length;
511        int height = resistanceMap[0].length;
512
513        Coord end = Coord.get(targetx, targety);
514        //find out which direction to step, on each axis
515        int stepX = (int) Math.signum(targetx - startx),
516                stepY = (int) Math.signum(targety - starty);
517
518        int deltaY = Math.abs(targetx - startx),
519                deltaX = Math.abs(targety - starty);
520
521        int testX = startx,
522                testY = starty;
523
524        int maxX = deltaX,
525                maxY = deltaY;
526
527        while (testX >= 0 && testX < width && testY >= 0 && testY < height && (testX != targetx || testY != targety)) {
528            lastPath.add(Coord.get(testX, testY));
529            if (maxY - maxX > deltaX) {
530                maxX += deltaX;
531                testX += stepX;
532                if (resistanceMap[testX][testY] >= 1f) {
533                    end = Coord.get(testX, testY);
534                    break;
535                }
536            } else if (maxX - maxY > deltaY) {
537                maxY += deltaY;
538                testY += stepY;
539                if (resistanceMap[testX][testY] >= 1f) {
540                    end = Coord.get(testX, testY);
541                    break;
542                }
543            } else {//directly on diagonal, move both full step
544                maxY += deltaY;
545                testY += stepY;
546                maxX += deltaX;
547                testX += stepX;
548                if (resistanceMap[testX][testY] >= 1f) {
549                    end = Coord.get(testX, testY);
550                    break;
551                }
552            }
553            if (radiusStrategy.radius(testX, testY, startx, starty) > radiusStrategy.radius(startx, starty, end.x, end.y)) {//went too far
554                break;
555            }
556        }
557
558        if (end.x >= 0 && end.x < width && end.y >= 0 && end.y < height) {
559            lastPath.add(Coord.get(end.x, end.y));
560        }
561
562        return end.x == targetx && end.y == targety;
563    }

565    @GwtIncompatible /* Because of Thread */
566    private boolean eliasReachable(Radius radiusStrategy) {
567        if(elias == null)
568            elias = new Elias();
569        List<Coord> ePath = elias.line(startx, starty, targetx, targety);
570        lastPath = new LinkedList<>(ePath);//save path for later retreival
571
572        HashMap<EliasWorker, Thread> pool = new HashMap<>();
573
574        for (Coord p : ePath) {
575            EliasWorker worker = new EliasWorker(p.x, p.y, radiusStrategy);
576            Thread thread = new Thread(worker);
577            thread.start();
578            pool.put(worker, thread);
579        }
580
581        for (EliasWorker w : pool.keySet()) {
582            try {
583                pool.get(w).join();
584            } catch (InterruptedException ex) {
585            }
586            if (w.succeeded) {
587                lastPath = w.path;
588                return true;
589            }
590        }
591
592        return false;//never got to the target point
593    }
594
595    private class EliasWorker implements Runnable {
596
597        private LinkedList<Coord> path;
598        private boolean succeeded = false;
599        private int testx, testy;
600        private Radius eliasRadiusStrategy;
601        EliasWorker(int testx, int testy, Radius radiusStrategy) {
602            this.testx = testx;
603            this.testy = testy;
604            this.eliasRadiusStrategy = radiusStrategy;
605        }
606
607        @Override
608        public void run() {
609            LOS los1 = new LOS(BRESENHAM);
610            LOS los2 = new LOS(BRESENHAM);
611            //if a non-solid midpoint on the path can see both the start and end, consider the two ends to be able to see each other
612            if (resistanceMap[testx][testy] < 1
613                    && eliasRadiusStrategy.radius(startx, starty, testx, testy) <= eliasRadiusStrategy.radius(startx, starty, targetx, targety)
614                    && los1.isReachable(resistanceMap, testx, testy, targetx, targety, eliasRadiusStrategy)
615                    && los2.isReachable(resistanceMap, startx, starty, testx, testy, eliasRadiusStrategy)) {
616
617                //record actual sight path used
618                path = new LinkedList<>(los2.lastPath);
619                path.addAll(los1.lastPath);
620                succeeded = true;
621            }
622        }
623    }
