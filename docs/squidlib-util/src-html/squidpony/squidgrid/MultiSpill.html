008/**
009 * A randomized flood-fill implementation that can be used for level generation (e.g. filling ponds and lakes), for
010 * gas propagation, or for all sorts of fluid-dynamics-on-the-cheap.
011 * Created by Tommy Ettinger on 4/7/2015.

013public class MultiSpill {
014
015    /**
016     * This affects how distance is measured on diagonal directions vs. orthogonal directions. MANHATTAN should form a
017     * diamond shape on a featureless map, while CHEBYSHEV and EUCLIDEAN will form a square. If you only call
018     * Spill.start() once, you should strongly prefer MANHATTAN, even if the rest of the game uses another
019     * measurement, because CHEBYSHEV and EUCLIDEAN can produce odd, gap-filled flood-fills.  Any case where you have
020     * too many gaps can be corrected to varying extent by calling start() more than once with slowly increasing values.
021     * Because start() will extend from the existing area of the Spill, holes are likely to be filled after a few calls,
022     * but if the last call to start() tries to fill too many more cells than the previous one, it can cause holes on
023     * the periphery of the Spill area.
024     */
025    public Measurement measurement = Measurement.MANHATTAN;

199    /**
200     * Used to initialize or re-initialize a Spill that needs a new PhysicalMap because it either wasn't given
201     * one when it was constructed, or because the contents of the terrain have changed permanently.
202     * @param level a short[][] that should have been the spillMap of another MultiSpill
203     * @return this for chaining
204     */
205    public MultiSpill initialize(final short[][] level) {
206        fresh = new ArrayList<>();
207        width = level.length;
208        height = level[0].length;
209        spillMap = new short[width][height];
210        anySpillMap = new GreasedRegion(level, 1, 0x7fff);
211        anyFreshMap = new GreasedRegion(width, height);
212        physicalMap = new boolean[width][height];
213        for (int y = 0; y < height; y++) {
214            for (int x = 0; x < width; x++) {
215                spillMap[x][y] = level[x][y];
216                physicalMap[x][y] = level[x][y] >= 0;
217            }
218        }
219        initialized = true;
220        return this;
221    }

224     * Used to initialize or re-initialize a Spill that needs a new PhysicalMap because it either wasn't given
225     * one when it was constructed, or because the contents of the terrain have changed permanently (not if a
226     * creature moved; for that you pass the positions of creatures that block paths to scan() or findPath() ).
227     * @param level a char[][] that should use '#' for walls and '.' for floors
228     * @return this for chaining

276     * Resets the spillMap to being empty.
277     */
278    public void resetMap() {
279        if(!initialized) return;
280        anySpillMap.clear();
281        for (int y = 0; y < height; y++) {
282            for (int x = 0; x < width; x++) {
283                spillMap[x][y] = -1;
284            }
285        }
286    }

320        if(!initialized) return;
321        fresh.get(idx).add(Coord.get(x, y));
322        anyFreshMap.insert(x, y);
323    }

325    protected void setFresh(int idx, final Coord pt) {
326        if(!initialized) return;
327        if(anyFreshMap.contains(pt.x, pt.y))
328            return;
329        fresh.get(idx).add(pt);
330        anyFreshMap.insert(pt);
331    }

333    /**
334     * Recalculate the spillMap and return the spreadPattern. The cell corresponding to a Coord in entries will be true,
335     * the cells near each of those will be true if chosen at random from all passable cells adjacent to a
336     * filled (true) cell, and all other cells will be false. This takes a total number of cells to attempt
337     * to fill (the volume parameter), which can be negative to simply fill the whole map, and will fill less if it has
338     * completely exhausted all passable cells from all sources in entries.
339     * If the measurement this Spill uses is anything other than MANHATTAN, you can expect many gaps in the first
340     * filled area.  Subsequent calls to start() with the same entry and a higher volume will expand the area
341     * of the Spill, and are likely to fill any gaps after a few subsequent calls. Increasing the volume slowly
342     * is the best way to ensure that gaps only exist on the very edge if you use a non-MANHATTAN measurement.
343     *
344     * @param entries the first cell for each spiller to spread from, which should really be passable.
345     * @param volume the total number of cells to attempt to fill; if negative will fill the whole map.
346     * @param impassable a Collection, ideally a Set or GreasedRegion, holding Coord items representing the locations of
347     *                   moving obstacles to a fill that cannot be moved through; null means no obstacles exist.
348     * @return an ArrayList of Points that this will enter, in order starting with entry at index 0, until it
349     * reaches its volume or fills its boundaries completely.
350     */
351    public ArrayList<ArrayList<Coord>> start(List<Coord> entries, int volume, Collection<Coord> impassable) {
352        if(!initialized) return null;
353        if(impassable == null)
354            impassable = Collections.emptySet();
355        if(volume < 0)
356            volume = Integer.MAX_VALUE;
357        ArrayList<Coord> spillers = new ArrayList<>(entries);
358        spreadPattern = new ArrayList<>(spillers.size());
359        fresh.clear();
360        filled = 0;
361        boolean hasFresh = false;
362        for (short i = 0; i < spillers.size(); i++) {
363            spreadPattern.add(new ArrayList<Coord>(128));
364            OrderedSet<Coord> os = new OrderedSet<>(128);
365            fresh.add(os);
366            Coord c = spillers.get(i);
367            spillMap[c.x][c.y] = i;
368            if(!impassable.contains(c)) {
369                os.add(c);
370                hasFresh = true;
371            }
372        }
373        Direction[] dirs = (measurement == Measurement.MANHATTAN) ? Direction.CARDINALS : Direction.OUTWARDS;
374        OrderedSet<Coord> currentFresh;
375        while (hasFresh && filled < volume) {
376            hasFresh = false;
377            for (short i = 0; i < spillers.size() && filled < volume; i++) {
378                currentFresh = fresh.get(i);
379                if(currentFresh.isEmpty())
380                    continue;
381                else
382                    hasFresh = true;
383                Coord cell = currentFresh.randomItem(rng);//.toArray(new Coord[currentFresh.size()])[rng.nextInt(currentFresh.size())];
384
385                spreadPattern.get(i).add(cell);
386                spillMap[cell.x][cell.y] = i;
387                filled++;
388                anySpillMap.insert(cell.x, cell.y);
389
390                for (int d = 0; d < dirs.length; d++) {
391                    Coord adj = cell.translate(dirs[d].deltaX, dirs[d].deltaY);
392                    if(!adj.isWithin(width, height))
393                        continue;
394                    double h = heuristic(dirs[d]);
395                    if (physicalMap[adj.x][adj.y] && !anySpillMap.contains(adj.x, adj.y) && !impassable.contains(adj) && rng.nextDouble(h) <= 1.0) {
396                        setFresh(i, adj);
397                    }
398                }
399                currentFresh.remove(cell);
400                anyFreshMap.remove(cell.x, cell.y);
401            }
402        }
403        return spreadPattern;
404    }

406    /**
407     * Recalculate the spillMap and return the spreadPattern. The cell corresponding to a key in entries will be true,
408     * the cells near each of those will be true if chosen at random from all passable cells adjacent to a
409     * filled (true) cell, and all other cells will be false. This takes a total number of cells to attempt
410     * to fill (the volume parameter), which can be negative to simply fill the whole map, and will fill less if it has
411     * completely exhausted all passable cells from all sources in entries. It uses the values in entries to determine
412     * whether it should advance from a particular key in that step or not; this choice is pseudo-random. If you have
413     * some values that are at or near 1.0 and some values that are closer to 0.0, you should expect the keys for the
414     * higher values to spread further out than the keys associated with lower values.
415     * <br>
416     * If the measurement this Spill uses is anything other than MANHATTAN, you can expect many gaps in the first
417     * filled area.  Subsequent calls to start() with the same entry and a higher volume will expand the area
418     * of the Spill, and are likely to fill any gaps after a few subsequent calls. Increasing the volume slowly
419     * is the best way to ensure that gaps only exist on the very edge if you use a non-MANHATTAN measurement.
420     * <br>
421     * The intended purpose for this method is filling contiguous areas of dungeon with certain terrain features, but it
422     * has plenty of other uses as well.
423     * @param entries key: the first cell for each spiller to spread from. value: the bias toward advancing this key;
424     *                1.0 will always advance, 0.0 will never advance beyond the key, in between will randomly choose
425     * @param volume the total number of cells to attempt to fill; if negative will fill the whole map.
426     * @param impassable a Collection, ideally a Set or GreasedRegion, holding Coord items representing the locations of
427     *                   moving obstacles to a fill that cannot be moved through; null means no obstacles exist.
428     * @return an ArrayList of Points that this will enter, in order starting with entry at index 0, until it
429     * reaches its volume or fills its boundaries completely.
430     */
431    public ArrayList<ArrayList<Coord>> start(OrderedMap<Coord, Double> entries, int volume, Collection<Coord> impassable) {
432        if(!initialized || entries == null) return null;
433        if(impassable == null)
434            impassable = Collections.emptySet();
435        if(volume < 0)
436            volume = Integer.MAX_VALUE;
437        int sz = entries.size();
438        ArrayList<Coord> spillers = new ArrayList<>(entries.keySet());
439        ArrayList<Double>
440                biases = new ArrayList<>(sz);
441        spreadPattern = new ArrayList<>(sz);
442        fresh.clear();
443        filled = 0;
444        boolean hasFresh = false;
445        for (short i = 0; i < sz; i++) {
446            spreadPattern.add(new ArrayList<Coord>(128));
447            OrderedSet<Coord> os = new OrderedSet<Coord>(128);
448            fresh.add(os);
449            Coord c = spillers.get(i);
450            Double d = entries.getAt(i);
451            biases.add(d);
452            if (d <= 0.0001 || c.x < 0 || c.y < 0)
453                continue;
454            spillMap[c.x][c.y] = i;
455            if (!impassable.contains(c)) {
456                os.add(c);
457                hasFresh = true;
458            }
459        }
460
461        Direction[] dirs = (measurement == Measurement.MANHATTAN) ? Direction.CARDINALS : Direction.OUTWARDS;
462
463        while (hasFresh && filled < volume) {
464            hasFresh = false;
465            for (short i = 0; i < spillers.size() && filled < volume; i++) {
466                OrderedSet<Coord> currentFresh = fresh.get(i);
467                if(currentFresh.isEmpty())
468                    continue;
469                else
470                    hasFresh = true;
471                Coord cell = currentFresh.randomItem(rng);//toArray(new Coord[currentFresh.size()])[rng.nextInt(currentFresh.size())];
472                if(rng.nextDouble() < biases.get(i)) {
473
474                    spreadPattern.get(i).add(cell);
475                    spillMap[cell.x][cell.y] = i;
476                    filled++;
477                    anySpillMap.insert(cell.x, cell.y);
478
479
480                    for (int d = 0; d < dirs.length; d++) {
481                        Coord adj = cell.translate(dirs[d].deltaX, dirs[d].deltaY);
482                        if(!adj.isWithin(width, height))
483                            continue;
484                        double h = heuristic(dirs[d]);
485                        if (physicalMap[adj.x][adj.y] && !anySpillMap.contains(adj.x, adj.y) && !impassable.contains(adj)
486                                && rng.nextDouble(h) <= 1.0) {
487                            setFresh(i, adj);
488                        }
489                    }
490                    currentFresh.remove(cell);
491                    anyFreshMap.remove(cell.x, cell.y);
492                }
493            }
494        }
495        return spreadPattern;
496    }

498    private static final double root2 = Math.sqrt(2.0);
499    private double heuristic(Direction target) {
500        switch (measurement) {
501            case MANHATTAN:
502            case CHEBYSHEV:
503                return 1.0;
504            case EUCLIDEAN:
505                switch (target) {
506                    case DOWN_LEFT:
507                    case DOWN_RIGHT:
508                    case UP_LEFT:
509                    case UP_RIGHT:
510                        return root2;
511                    default:
512                        return  1.0;
513                }
514        }
515        return 1.0;
516    }
