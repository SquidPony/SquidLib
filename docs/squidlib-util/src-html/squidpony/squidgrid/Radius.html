080    public double radius(double dx, double dy, double dz) {
081        dx = Math.abs(dx);
082        dy = Math.abs(dy);
083        dz = Math.abs(dz);
084        double radius = 0;
085        switch (this) {
086            case SQUARE:
087            case CUBE:
088                radius = Math.max(dx, Math.max(dy, dz));//radius is longest axial distance
089                break;
090            case DIAMOND:
091            case OCTAHEDRON:
092                radius = dx + dy + dz;//radius is the manhattan distance
093                break;
094            case CIRCLE:
095            case SPHERE:
096                radius = Math.sqrt(dx * dx + dy * dy + dz * dz);//standard spherical radius
097        }
098        return radius;
099    }

125    public Coord onUnitShape(double distance, RNG rng) {
126        int x = 0, y = 0;
127        switch (this) {
128            case SQUARE:
129            case CUBE:
130                x = rng.between((int) -distance, (int) distance + 1);
131                y = rng.between((int) -distance, (int) distance + 1);
132                break;
133            case DIAMOND:
134            case OCTAHEDRON:
135                x = rng.between((int) -distance, (int) distance + 1);
136                y = rng.between((int) -distance, (int) distance + 1);
137                if (radius(x, y) > distance) {
138                    if (x > 0) {
139                        if (y > 0) {
140                            x = (int) (distance - x);
141                            y = (int) (distance - y);
142                        } else {
143                            x = (int) (distance - x);
144                            y = (int) (-distance - y);
145                        }
146                    } else {
147                        if (y > 0) {
148                            x = (int) (-distance - x);
149                            y = (int) (distance - y);
150                        } else {
151                            x = (int) (-distance - x);
152                            y = (int) (-distance - y);
153                        }
154                    }
155                }
156                break;
157            case CIRCLE:
158            case SPHERE:
159                double radius = distance * Math.sqrt(rng.between(0.0, 1.0));
160                double theta = rng.between(0, PI2);
161                x = (int) Math.round(Math.cos(theta) * radius);
162                y = (int) Math.round(Math.sin(theta) * radius);
163        }
164
165        return Coord.get(x, y);
166    }

168    public Coord3D onUnitShape3D(double distance, RNG rng) {
169        int x = 0, y = 0, z = 0;
170        switch (this) {
171            case SQUARE:
172            case DIAMOND:
173            case CIRCLE:
174                Coord p = onUnitShape(distance, rng);
175                return new Coord3D(p.x, p.y, 0);//2D strategies
176            case CUBE:
177                x = rng.between((int) -distance, (int) distance + 1);
178                y = rng.between((int) -distance, (int) distance + 1);
179                z = rng.between((int) -distance, (int) distance + 1);
180                break;
181            case OCTAHEDRON:
182            case SPHERE:
183                do {
184                    x = rng.between((int) -distance, (int) distance + 1);
185                    y = rng.between((int) -distance, (int) distance + 1);
186                    z = rng.between((int) -distance, (int) distance + 1);
187                } while (radius(x, y, z) > distance);
188        }
189
190        return new Coord3D(x, y, z);
191    }
192    public double volume2D(double radiusLength)
193    {
194        switch (this) {
195            case SQUARE:
196            case CUBE:
197                return (radiusLength * 2 + 1) * (radiusLength * 2 + 1);
198            case DIAMOND:
199            case OCTAHEDRON:
200                return radiusLength * (radiusLength + 1) * 2 + 1;
201            default:
202                return Math.PI * radiusLength * radiusLength + 1;
203        }
204    }
205    public double volume3D(double radiusLength)
206    {
207        switch (this) {
208            case SQUARE:
209            case CUBE:
210                return (radiusLength * 2 + 1) * (radiusLength * 2 + 1) * (radiusLength * 2 + 1);
211            case DIAMOND:
212            case OCTAHEDRON:
213                double total = radiusLength * (radiusLength + 1) * 2 + 1;
214                for(double i = radiusLength - 1; i >= 0; i--)
215                {
216                    total += (i * (i + 1) * 2 + 1) * 2;
217                }
218                return total;
219            default:
220                return Math.PI * radiusLength * radiusLength * radiusLength * 4.0 / 3.0 + 1;
221        }
222    }

232    public OrderedSet<Coord> perimeter(Coord center, int radiusLength, boolean surpassEdges, int width, int height)
233    {
234        OrderedSet<Coord> rim = new OrderedSet<>(4 * radiusLength);
235        if(!surpassEdges && (center.x < 0 || center.x >= width || center.y < 0 || center.y > height))
236            return rim;
237        if(radiusLength < 1) {
238            rim.add(center);
239            return rim;
240        }
241        switch (this) {
242            case SQUARE:
243            case CUBE:
244            {
245                for (int i = center.x - radiusLength; i <= center.x + radiusLength; i++) {
246                    int x = i;
247                    if(!surpassEdges) x = clamp(i, 0, width);
248                    rim.add(Coord.get(x, clamp(center.y - radiusLength, 0, height)));
249                    rim.add(Coord.get(x, clamp(center.y + radiusLength, 0, height)));
250                }
251                for (int j = center.y - radiusLength; j <= center.y + radiusLength; j++) {
252                    int y = j;
253                    if(!surpassEdges) y = clamp(j, 0, height);
254                    rim.add(Coord.get(clamp(center.x - radiusLength, 0, height), y));
255                    rim.add(Coord.get(clamp(center.x + radiusLength, 0, height), y));
256                }
257            }
258            break;
259            case DIAMOND:
260            case OCTAHEDRON: {
261                int xUp = center.x + radiusLength, xDown = center.x - radiusLength,
262                        yUp = center.y + radiusLength, yDown = center.y - radiusLength;
263                if(!surpassEdges) {
264                    xDown = clamp(xDown, 0, width);
265                    xUp = clamp(xUp, 0, width);
266                    yDown = clamp(yDown, 0, height);
267                    yUp = clamp(yUp, 0, height);
268                }
269
270                rim.add(Coord.get(xDown, center.y));
271                rim.add(Coord.get(xUp, center.y));
272                rim.add(Coord.get(center.x, yDown));
273                rim.add(Coord.get(center.x, yUp));
274
275                for (int i = xDown + 1, c = 1; i < center.x; i++, c++) {
276                    int x = i;
277                    if(!surpassEdges) x = clamp(i, 0, width);
278                    rim.add(Coord.get(x, clamp(center.y - c, 0, height)));
279                    rim.add(Coord.get(x, clamp(center.y + c, 0, height)));
280                }
281                for (int i = center.x + 1, c = 1; i < center.x + radiusLength; i++, c++) {
282                    int x = i;
283                    if(!surpassEdges) x = clamp(i, 0, width);
284                    rim.add(Coord.get(x, clamp(center.y + radiusLength - c, 0, height)));
285                    rim.add(Coord.get(x, clamp(center.y - radiusLength + c, 0, height)));
286                }
287            }
288            break;
289            default:
290            {
291                double theta;
292                int x, y, denom = 1;
293                boolean anySuccesses;
294                while(denom <= 256) {
295                    anySuccesses = false;
296                    for (int i = 1; i <= denom; i+=2)
297                    {
298                        theta = i * (PI2 / denom);
299                        x = (int) (Math.cos(theta) * (radiusLength + 0.25)) + center.x;
300                        y = (int) (Math.sin(theta) * (radiusLength + 0.25)) + center.y;
301                        
302                        if (!surpassEdges) {
303                            x = clamp(x, 0, width);
304                            y = clamp(y, 0, height);
305                        }
306                        Coord p = Coord.get(x, y);
307                        boolean test = !rim.contains(p);
308
309                        rim.add(p);
310                        anySuccesses = test || anySuccesses;
311                    }
312                    if(!anySuccesses)
313                        break;
314                    denom *= 2;
315                }
316
317            }
318        }
319        return rim;
320    }
321    public Coord extend(Coord center, Coord middle, int radiusLength, boolean surpassEdges, int width, int height)
322    {
323        if(!surpassEdges && (center.x < 0 || center.x >= width || center.y < 0 || center.y > height ||
324                middle.x < 0 || middle.x >= width || middle.y < 0 || middle.y > height))
325            return Coord.get(0, 0);
326        if(radiusLength < 1) {
327            return center;
328        }
329        double theta = Math.atan2(middle.y - center.y, middle.x - center.x),
330                cosTheta = Math.cos(theta), sinTheta = Math.sin(theta);
331
332        Coord end = Coord.get(middle.x, middle.y);
333        switch (this) {
334            case SQUARE:
335            case CUBE:
336            case DIAMOND:
337            case OCTAHEDRON:
338            {
339                int rad2 = 0;
340                if(surpassEdges)
341                {
342                    while (radius(center.x, center.y, end.x, end.y) < radiusLength) {
343                        rad2++;
344                        end = Coord.get((int) Math.round(cosTheta * rad2) + center.x
345                                , (int) Math.round(sinTheta * rad2) + center.y);
346                    }
347                }
348                else {
349                    while (radius(center.x, center.y, end.x, end.y) < radiusLength) {
350                        rad2++;
351                        end = Coord.get(clamp((int) Math.round(cosTheta * rad2) + center.x, 0, width)
352                                      , clamp((int) Math.round(sinTheta * rad2) + center.y, 0, height));
353                        if (end.x == 0 || end.x == width - 1 || end.y == 0 || end.y == height - 1)
354                            return end;
355                    }
356                }
357
358                return end;
359            }
360            default:
361            {
362                end = Coord.get(clamp( (int) Math.round(cosTheta * radiusLength) + center.x, 0, width)
363                        , clamp( (int) Math.round(sinTheta * radiusLength) + center.y, 0, height));
364                if(!surpassEdges) {
365                    long edgeLength = 0;
366//                    if (end.x == 0 || end.x == width - 1 || end.y == 0 || end.y == height - 1)
367                    if (end.x < 0)
368                    {
369                        // wow, we lucked out here. the only situation where cos(angle) is 0 is if the angle aims
370                        // straight up or down, and then x cannot be < 0 or >= width.
371                        edgeLength = Math.round((0 - center.x) / cosTheta);
372                        end = end.setY(clamp((int) Math.round(sinTheta * edgeLength) + center.y, 0, height));
373                    }
374                    else if(end.x >= width)
375                    {
376                        // wow, we lucked out here. the only situation where cos(angle) is 0 is if the angle aims
377                        // straight up or down, and then x cannot be < 0 or >= width.
378                        edgeLength = Math.round((width - 1 - center.x) / cosTheta);
379                        end = end.setY(clamp((int) Math.round(sinTheta * edgeLength) + center.y, 0, height));
380                    }
381
382                    if (end.y < 0)
383                    {
384                        // wow, we lucked out here. the only situation where sin(angle) is 0 is if the angle aims
385                        // straight left or right, and then y cannot be < 0 or >= height.
386                        edgeLength = Math.round((0 - center.y) / sinTheta);
387                        end = end.setX(clamp((int) Math.round(cosTheta * edgeLength) + center.x, 0, width));
388                    }
389                    else if(end.y >= height)
390                    {
391                        // wow, we lucked out here. the only situation where sin(angle) is 0 is if the angle aims
392                        // straight left or right, and then y cannot be < 0 or >= height.
393                        edgeLength = Math.round((height - 1 - center.y) / sinTheta);
394                        end = end.setX(clamp((int) Math.round(cosTheta * edgeLength) + center.x, 0, width));
395                    }
396                }
397                return end;
398            }
399        }
400    }

403     * Compares two Radius enums as if they are both in a 2D plane; that is, Radius.SPHERE is treated as equal to
404     * Radius.CIRCLE, Radius.CUBE is equal to Radius.SQUARE, and Radius.OCTAHEDRON is equal to Radius.DIAMOND.
405     * @param other the Radius to compare this to
406     * @return true if the 2D versions of both Radius enums are the same shape.

408    public boolean equals2D(Radius other)
409    {
410        switch (this)
411        {
412            case CIRCLE:
413            case SPHERE:
414                return other == CIRCLE || other == SPHERE;
415            case SQUARE:
416            case CUBE:
417                return other == SQUARE || other == CUBE;
418            default:
419                return other == DIAMOND || other == OCTAHEDRON;
420        }
421    }

428    public int roughDistance(int xPos, int yPos) {
429        int x = Math.abs(xPos), y = Math.abs(yPos);
430        switch (this) {
431            case CIRCLE:
432            case SPHERE:
433            {
434                if(x == y)
435                    return 3 * x;
436                else if(x < y)
437                    return 3 * x + 2 * (y - x);
438                else
439                    return 3 * y + 2 * (x - y);
440            }
441            case DIAMOND:
442            case OCTAHEDRON:
443                return 2 * (x + y);
444            default:
445                return 2 * Math.max(x, y);
446        }
447    }

459    public List<Coord> pointsInside(int centerX, int centerY, int radiusLength, boolean surpassEdges, int width, int height, List<Coord> buf)
460    {
461        List<Coord> contents = buf == null ? new ArrayList<Coord>((int)Math.ceil(volume2D(radiusLength))) : buf;
462        if(!surpassEdges && (centerX < 0 || centerX >= width || centerY < 0 || centerY >= height))
463            return contents;
464        if(radiusLength < 1) {
465            contents.add(Coord.get(centerX, centerY));
466            return contents;
467        }
468        switch (this) {
469            case SQUARE:
470            case CUBE:
471            {
472                for (int i = centerX - radiusLength; i <= centerX + radiusLength; i++) {
473                    for (int j = centerY - radiusLength; j <= centerY + radiusLength; j++) {
474                        if(!surpassEdges && (i < 0 || j < 0 || i >= width || j >= height))
475                            continue;
476                        contents.add(Coord.get(i, j));
477                    }
478                }
479            }
480            break;
481            case DIAMOND:
482            case OCTAHEDRON: {
483                for (int i = centerX - radiusLength; i <= centerX + radiusLength; i++) {
484                    for (int j = centerY - radiusLength; j <= centerY + radiusLength; j++) {
485                        if ((Math.abs(centerX - i) + Math.abs(centerY- j) > radiusLength) ||
486                                (!surpassEdges && (i < 0 || j < 0 || i >= width || j >= height)))
487                            continue;
488                        contents.add(Coord.get(i, j));
489                    }
490                }
491            }
492            break;
493            default:
494            {
495                int high;
496                for (int dx = -radiusLength; dx <= radiusLength; ++dx) {
497                    high = (int) Math.sqrt(radiusLength * radiusLength - dx * dx);
498                    for (int dy = -high; dy <= high; ++dy) {
499                        if (!surpassEdges && (centerX + dx < 0 || centerY + dy < 0 ||
500                                centerX + dx >= width || centerY + dy >= height))
501                            continue;
502                        contents.add(Coord.get(centerX + dx, centerY + dy));
503                    }
504                }
505            }
506        }
507        return contents;
508    }

624    /**
625     * Given an Iterable of Coord (such as a List or Set), a distance to expand outward by (using this Radius), and the
626     * bounding height and width of the map, gets a "thickened" group of Coord as a Set where each Coord in points has
627     * been expanded out by an amount no greater than distance. As an example, you could call this on a line generated
628     * by Bresenham, OrthoLine, or an LOS object's getLastPath() method, and expand the line into a thick "brush stroke"
629     * where this Radius affects the shape of the ends. This will never produce a Coord with negative x or y, a Coord
630     * with x greater than or equal to width, or a Coord with y greater than or equal to height.
631     * @param distance the distance, as measured by this Radius, to expand each Coord on points up to
632     * @param width the bounding width of the map (exclusive)
633     * @param height the bounding height of the map (exclusive)
634     * @param points an Iterable (such as a List or Set) of Coord that this will make a "thickened" version of
635     * @return a Set of Coord that covers a wider area than what points covers; each Coord will be unique (it's a Set)
636     */
637    public Set<Coord> expand(int distance, int width, int height, Iterable<Coord> points)
638    {
639        List<Coord> around = pointsInside(Coord.get(distance, distance), distance, false, width, height);
640        OrderedSet<Coord> expanded = new OrderedSet<>(around.size() * 16);
641        int tx, ty;
642        for(Coord pt : points)
643        {
644            for(Coord ar : around)
645            {
646                tx = pt.x + ar.x - distance;
647                ty = pt.y + ar.y - distance;
648                if(tx >= 0 && tx < width && ty >= 0 && ty < height)
649                    expanded.add(Coord.get(tx, ty));
650            }
651        }
652        return expanded;
653    }
