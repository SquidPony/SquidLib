274    private static final char[] vowels = new char[]{'a', 'e', 'i', 'o'};//not using y because it looks strange as a vowel in names
275    private static final int LAST_LETTER_CANDIDATES_MAX = 52;
276
277    private RNG rng;
278    private String[] names;
279    private int consonantLimit;
280    private ArrayList<Integer> sizes;
281
282    private TreeMap<Character, HashMap<Character, ProbabilityTable<Character>>> letters;
283    private ArrayList<Character> firstLetterSamples;
284    private ArrayList<Character> lastLetterSamples;
285    private DamerauLevenshteinAlgorithm dla = new DamerauLevenshteinAlgorithm(1, 1, 1, 1);

287    /**
288     * Creates the generator by seeding the provided list of names.
289     *
290     * @param names an array of Strings that are typical names to be emulated
291     */
292    public WeightedLetterNamegen(String[] names) {
293        this(names, 2);

296    /**
297     * Creates the generator by seeding the provided list of names.
298     *
299     * @param names an array of Strings that are typical names to be emulated
300     * @param consonantLimit the maximum allowed consonants in a row
301     */
302    public WeightedLetterNamegen(String[] names, int consonantLimit) {
303        this(names, consonantLimit, new RNG());
304    }

306    /**
307     * Creates the generator by seeding the provided list of names.
308     *
309     * @param names an array of Strings that are typical names to be emulated
310     * @param consonantLimit the maximum allowed consonants in a row
311     * @param rng the source of randomness to be used
312     */
313    public WeightedLetterNamegen(String[] names, int consonantLimit, RNG rng) {
314        this.names = names;
315        this.consonantLimit = consonantLimit;
316        this.rng = rng;
317        init();
318    }
319
320    /**
321     * Initialization, statistically measures letter likelihood.
322     */
323    private void init() {
324        sizes = new ArrayList<>();
325        letters = new TreeMap<>();
326        firstLetterSamples = new ArrayList<>();
327        lastLetterSamples = new ArrayList<>();
328
329        for (int i = 0; i < names.length - 1; i++) {
330            String name = names[i];
331            if (name == null || name.length() < 1) {
332                continue;
333            }
334
335            // (1) Insert size
336            sizes.add(name.length());
337
338            // (2) Grab first letter
339            firstLetterSamples.add(name.charAt(0));
340
341            // (3) Grab last letter
342            lastLetterSamples.add(name.charAt(name.length() - 1));
343
344            // (4) Process all letters
345            for (int n = 0; n < name.length() - 1; n++) {
346                char letter = name.charAt(n);
347                char nextLetter = name.charAt(n + 1);
348
349                // Create letter if it doesn't exist
350                HashMap<Character, ProbabilityTable<Character>> wl = letters.get(letter);
351                if (wl == null) {
352                    wl = new HashMap<>();
353                    letters.put(letter, wl);
354                }
355                ProbabilityTable<Character> wlg = wl.get(letter);
356                if (wlg == null) {
357                    wlg = new ProbabilityTable<>();
358                    wl.put(letter, wlg);
359                }
360                wlg.add(nextLetter, 1);
361
362                // If letter was uppercase (beginning of name), also add a lowercase entry
363                if (Category.Lu.contains(letter)) {
364                    letter = Character.toLowerCase(letter);
365
366                    wlg = wl.get(letter);
367                    if (wlg == null) {
368                        wlg = new ProbabilityTable<>();
369                        wl.put(letter, wlg);
370                    }
371                    wlg.add(nextLetter, 1);
372                }
373            }
374        }
375    }

377    public String[] generate() {
378        return generate(1);
379    }
380
381    public String[] generate(int amountToGenerate) {
382        ArrayList<String> result = new ArrayList<>();
383
384        int nameCount = 0;
385        while (nameCount < amountToGenerate) {
386            String name = "";
387
388            // Pick size
389            int size = rng.getRandomElement(sizes);
390
391            // Pick first letter
392            char firstLetter = rng.getRandomElement(firstLetterSamples);
393
394            name += firstLetter;
395
396            for (int i = 1; i < size - 2; i++) {
397                name += getRandomNextLetter(name.charAt(name.length() - 1));
398            }
399
400            // Attempt to find a last letter
401            for (int lastLetterFits = 0; lastLetterFits < LAST_LETTER_CANDIDATES_MAX; lastLetterFits++) {
402                char lastLetter = rng.getRandomElement(lastLetterSamples);
403                char intermediateLetterCandidate = getIntermediateLetter(name.charAt(name.length() - 1), lastLetter);
404
405                // Only attach last letter if the candidate is valid (if no candidate, the antepenultimate letter always occurs at the end)
406                if (Category.L.contains(intermediateLetterCandidate)) {
407                    name += intermediateLetterCandidate;
408                    name += lastLetter;
409                    break;
410                }
411            }
412
413            String nameString = name;
414
415            // Check that the word has no triple letter sequences, and that the Levenshtein distance is kosher
416            if (validateGrouping(name) && checkLevenshtein(nameString)) {
417                result.add(nameString);
418
419                // Only increase the counter if we've successfully added a name
420                nameCount++;
421            }
422        }
423
424        return result.toArray(new String[result.size()]);
425    }

427    /**
428     * Searches for the best fit letter between the letter before and the letter
429     * after (non-random). Used to determine penultimate letters in names.
430     *
431     * @param   letterBefore    The letter before the desired letter.
432     * @param   letterAfter     The letter after the desired letter.
433     * @return  The best fit letter between the provided letters.
434     */
435    private char getIntermediateLetter(char letterBefore, char letterAfter) {
436        if (Category.L.contains(letterBefore) && Category.L.contains(letterAfter)) {
437            // First grab all letters that come after the 'letterBefore'
438            HashMap<Character, ProbabilityTable<Character>> wl = letters.get(letterBefore);
439            if (wl == null) {
440                return getRandomNextLetter(letterBefore);
441            }
442            Set<Character> letterCandidates = wl.get(letterBefore).items();
443
444            char bestFitLetter = '\'';
445            int bestFitScore = 0;
446
447            // Step through candidates, and return best scoring letter
448            for (char letter : letterCandidates) {
449                wl = letters.get(letter);
450                if (wl == null) {
451                    continue;
452                }
453                ProbabilityTable<Character> weightedLetterGroup = wl.get(letterBefore);
454                if (weightedLetterGroup != null) {
455                    int letterCounter = weightedLetterGroup.weight(letterAfter);
456                    if (letterCounter > bestFitScore) {
457                        bestFitLetter = letter;
458                        bestFitScore = letterCounter;
459                    }
460                }
461            }
462
463            return bestFitLetter;
464        } else {
465            return '-';
466        }
467    }

469    /**
470     * Checks that no three letters happen in succession.
471     *
472     * @param   name    The name array (easier to iterate)
473     * @return  True if no triple letter sequence is found.
474     */
475    private boolean validateGrouping(String name) {
476        for (int i = 2; i < name.length(); i++) {
477            if (name.charAt(i) == name.charAt(i - 1) && name.charAt(i) == name.charAt(i - 2)) {
478                return false;
479            }
480        }
481        int consonants = 0;
482        for (int i = 0; i < name.length(); i++) {
483            if (isVowel(name.charAt(i))) {
484                consonants = 0;
485            } else {
486                consonants++;
487            }
488
489            if (consonants > consonantLimit) {
490                return false;
491            }
492        }
493        return true;
494    }
495
496    private boolean isVowel(char c) {
497        switch(c)
498        {
499            case 'a':
500            case 'e':
501            case 'i':
502            case 'o':
503                return true;
504            default:
505                return false;
506        }
507    }
508
509    /**
510     * Checks that the Damerau-Levenshtein distance of this name is within a
511     * given bias from a name on the master list.
512     *
513     * @param   name    The name string.
514     * @return  True if a name is found that is within the bias.
515     */
516    private boolean checkLevenshtein(String name) {
517        int levenshteinBias = name.length() / 2;
518
519        for (String name1 : names) {
520            int levenshteinDistance = dla.execute(name, name1);
521            if (levenshteinDistance <= levenshteinBias) {
522                return true;
523            }
524        }
525
526        return false;
527    }
528
529    private char getRandomNextLetter(char letter) {
530        if (letters.containsKey(letter)) {
531            return letters.get(letter).get(letter).random();
532        } else {
533            return vowels[rng.nextIntHasty(4)];
534        }
535    }
