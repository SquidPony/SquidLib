001package squidpony.squidmath;
002
003import java.io.Serializable;
004
005/**
006 * Similar to ThunderRNG (emphasizes speed over quality), but unlike ThunderRNG this is a StatefulRandomness.
007 * Being a StatefulRandomness means you can pass a DashRNG to the {@link StatefulRNG#StatefulRNG(RandomnessSource)}
008 * constructor and have expected results, and also that you can get and set the state on a DashRNG directly. If a
009 * DashRNG is stored in some variable that is then passed to {@link RNG#RNG(RandomnessSource)}, you can call
010 * {@link #setState(long)} on the DashRNG variable to set the state used by the RNG, which can be very useful.
011 * <br>
012 * The {@link #nextLong()} method on this class can produce 64-bit data in somewhere between 3/4 and 2/3 the time needed
013 * by LightRNG, and only slightly more than ThunderRNG. This class fails many more statistical tests than ThunderRNG or
014 * especially LightRNG; this probably doesn't matter for games. If you need something closer to a cryptographic RNG, you
015 * can use IsaacRNG (though it won't be as fast). The speed of DashRNG probably makes breaking it easier, especially in
016 * conjunction with its poor statistical quality. It's likely that the period of DashRNG is a full 2 to the 64 (0 seed
017 * is allowed), but the distribution is also likely to cover much less than the full range of all longs.
018 * <br>
019 * Created by Tommy Ettinger on 4/30/2017.
020 */
021public class DashRNG implements StatefulRandomness, Serializable {
022    private static final long serialVersionUID = 1L;
023
024    /**
025     * Constructs a DashRNG using two calls to {@link Math#random()} to make a 64-bit mostly-random seed.
026     */
027    public DashRNG(){
028        this((long)(Math.random() * 0x100000000L) << 32 | (long)(Math.random() * 0x100000000L));
029    }
030
031    /**
032     * Constructs a DashRNG using the given seed without changes.
033     * @param seed any long
034     */
035    public DashRNG(long seed) {
036        state = seed;
037    }
038
039    /**
040     * Constructs a DashRNG using the 64-bit hash of the given CharSequence (such as a String) as its seed.
041     * Uses {@link CrossHash.Wisp#hash64(CharSequence)} to get a 64-bit long seed from the CharSequence given.
042     * @param seed
043     */
044    public DashRNG(CharSequence seed)
045    {
046        state = CrossHash.Wisp.hash64(seed);
047    }
048
049    public long state;
050    /**
051     * Get the current internal state of the StatefulRandomness as a long.
052     *
053     * @return the current internal state of this object.
054     */
055    @Override
056    public long getState() {
057        return state;
058    }
059
060    /**
061     * Set the current internal state of this StatefulRandomness with a long.
062     *
063     * @param state a 64-bit long. DashRNG can handle any long parameter.
064     */
065    @Override
066    public void setState(long state) {
067        this.state = state;
068    }
069
070    /**
071     * Using this method, any algorithm that might use the built-in Java Random
072     * can interface with this randomness source.
073     *
074     * @param bits the number of bits to be returned
075     * @return the integer containing the appropriate number of bits
076     */
077    @Override
078    public final int next( final int bits ) {
079        return (int)( nextLong() >>> (64 - bits) );
080    }
081
082    /**
083     * Using this method, any algorithm that needs to efficiently generate more
084     * than 32 bits of random data can interface with this randomness source.
085     * <p>
086     * Get a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
087     *
088     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
089     */
090    @Override
091    public final long nextLong() {
092        //return ((state += 0x9E3779B97F4A7C15L) >>> 8) * 0x8329C6EB9E6AD3E3L ^ (state >>> 9) * 0xC6BC279692B5CC83L;
093//        final long z = (state += 0x9E3779B97F4A7C15L);
094//        final int r = (int)(z >>> 58);
095//        return (z >>> r) | (z << (64 - r));
096        final long z = (state += 0x9E3779B97F4A7C15L);
097        return (z >> (z >>> 59)) * 0xC6BC279692B5CC83L;
098//        return (z >>> (z >>> 58)) | (z << (64L - (z >>> 58)));
099
100//        final long z = state + 0x9E3779B97F4A7C15L;
101//        return state = (z ^ (z >> 8)) * 0x8329C6EB9E6AD3E3L;
102
103    }
104
105    /**
106     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
107     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
108     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
109     *
110     * @return a copy of this RandomnessSource
111     */
112    @Override
113    public RandomnessSource copy() {
114        return new DashRNG(state);
115    }
116
117    /**
118     * A simple "output stage" applied to state, this method does not update state on its own. If you expect to call
119     * this method more than once, you should add a large long that is an odd number to state as part of the call, such
120     * as with {@code DashRNG.determine(state += 0x9E3779B97F4A7C15L)}. Here, "large long" means at least two of the
121     * upper 5 bits should be set to be safe. The golden-ratio-derived constant 0x9E3779B97F4A7C15L should be fine.
122     * This method doesn't offer particularly good quality assurances, but should be very fast.
123     * @param state should be changed when you call this by adding some large odd number, e.g. 0x9E3779B97F4A7C15L
124     * @return an altered version of state that should be very fast to compute but doesn't promise great quality
125     */
126    public static long determine(final long state)
127    {
128        return (state >> (state >>> 59)) * 0xC6BC279692B5CC83L;
129
130    }
131}
