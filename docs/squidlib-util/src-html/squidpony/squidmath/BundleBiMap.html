011/**
012 * An ordered bidirectional map-like data structure, with "bundles" of unique E (for Element) keys and unique S (for
013 * Single) keys updated together like a map that can be queried by E key bundles, S keys, or int indices, as well as
014 * like a multimap that can be queried by lone E keys. A bundle can be specified as a Collection of E values, an array
015 * of E values, or in two parts as either of the above given to a BundleBiMap method along with an int array that can
016 * represent variations on E keys (e.g. some quantity for an E value that permits different amounts to be passed with
017 * the unique E values to getS and have it yield a different S for a, say, 1 Foo and 1 Bar vs. 2 Foo and 3 Bar). This
018 * allows an intended purpose for this class as a way to describe crafting recipes that may need various amounts of an
019 * item to be mixed in, and since the quantities are optional, it can still be used to group multiple E keys with S
020 * keys. The multimap property allows you to use an E element key to get all of the S keys that are associated with a
021 * bundle that contains that E key (you can also get the ordering indices for those S keys, which you can use to get the
022 * full bundle if you want).
023 * <br>
024 * Does not implement any interfaces that you would expect for a data structure, because almost every method it has
025 * needs to specify whether it applies to E or S items, or otherwise doesn't fit a normal Collection-like interface's
026 * requirements.
027 * <br>
028 * Closely related to Arrangement, K2, and K2V1 in implementation.
029 * <br>
030 * Created by Tommy Ettinger on 4/26/2017.
031 */
032@Beta
033public class BundleBiMap<E, S>
034{
035    private Arrangement<E> elements;
036    private K2<int[][], S> bm;
037    private ArrayList<IntVLA> mm;

048     * Constructs a BundleBiMap with the expected number of indices to hold (the number of bundles and number of S items

050     * @param expected how many bundle-to-single pairings this is expected to hold
051     */
052    public BundleBiMap(int expected)
053    {
054        this(expected, Arrangement.DEFAULT_LOAD_FACTOR);
055    }

057    /**
058     * Constructs a BundleBiMap with the expected number of indices to hold (the number of bundles and number of S items
059     * is always the same, and this will be more efficient if expected is greater than that number) and the load factor
060     * to use, between 0.1f and 0.8f usually (using load factors higher than 0.8f can cause problems).
061     * @param expected the amount of indices (the count of bundles is the same as the count of S items) this should hold
062     * @param f the load factor, probably between 0.1f and 0.8f
063     */
064    public BundleBiMap(int expected, float f)
065    {
066        elements = new Arrangement<>(expected, f);
067        bm = new K2<>(expected, f, CrossHash.int2DHasher, CrossHash.defaultHasher);
068        mm = new ArrayList<>(expected * 4);
069    }

072    /**
073     * Constructs a BundleBiMap using another BundleBiMap to copy.
074     * @param other the other BundleBiMap to copy
075     */
076    public BundleBiMap(BundleBiMap<? extends E, ? extends S> other)
077    {
078        if(other == null)
079        {
080            elements = new Arrangement<>(64, 0.75f);
081            bm = new K2<>(16, 0.75f, CrossHash.int2DHasher, CrossHash.defaultHasher);
082            mm = new ArrayList<>(64);
083        }
084        else
085        {
086            elements = new Arrangement<>(other.elements);
087            bm = new K2<>(other.bm);
088            mm = new ArrayList<>(other.mm);
089        }
090    }

172    /**
173     * Adds a bundle of E keys and a S key at the same point in the ordering (the end) to this BundleBiMap. Neither
174     * parameter can be present in this collection before this is called.
175     * @param e an array of E keys to add; the array cannot already be present, nor can it be null
176     * @param s e S key to add; cannot already be present
177     * @return true if this collection changed as e result of this call
178     */
179    public boolean put(E[] e, S s)
180    {
181        if(e == null) return false;
182        elements.addAllIfAbsent(e);
183        int[][] bundle = new int[][]{elements.getArray(e)};
184        if(!bm.put(bundle, s))
185            return false;
186        int len = e.length;
187        for (int i = 0; i < len; i++) {
188            mm.get(bundle[0][i]).add(bm.size()-1);
189        }
190        return true;
191    }

194     * (the end) to this BundleBiMap. Neither the S key nor the bundle (effectively, the pair of e and variation) can be
195     * present in this collection before this is called.
196     * @param e an array of E keys to add; the array cannot already have been inserted with an equivalent variation, nor can it be null
197     * @param variation an int array that can be used to make a different version of e, i.e. the same things at different quantities; cannot be null
198     * @param s e S key to add; cannot already be present

208        int len = e.length;
209        for (int i = 0; i < len; i++) {
210            mm.get(bundle[0][i]).add(bm.size()-1);
211        }
212        return true;
213    }

257    /**
258     * Puts all unique E and S keys in {@code aKeys} and {@code bKeys} into this K2 at the end. If an E in aKeys or a S
259     * in bKeys is already present when this would add one, this will not put the E and S keys at that point in the
260     * iteration order, and will place the next unique E and S it finds in the arguments at that position instead.
261     * @param aKeys an Iterable or Collection of E keys to add; should all be unique (like a Set)
262     * @param bKeys an Iterable or Collection of S keys to add; should all be unique (like a Set)
263     * @return true if this collection changed as a result of this call
264     */
265    public boolean putAll(Iterable<? extends Collection<? extends E>> aKeys, Iterable<? extends S> bKeys)
266    {
267        if(aKeys == null || bKeys == null) return false;
268        Iterator<? extends Collection<? extends E>> aIt = aKeys.iterator();
269        Iterator<? extends S> bIt = bKeys.iterator();
270        boolean changed = false;
271        while (aIt.hasNext() && bIt.hasNext())
272        {
273            changed = put(aIt.next(), bIt.next()) || changed;
274        }
275        return changed;
276    }
277    /**
278     * Puts all unique E and S keys in {@code other} into this K2, respecting other's ordering. If an E or a S in other
279     * is already present when this would add one, this will not put the E and S keys at that point in the iteration
280     * order, and will place the next unique E and S it finds in the arguments at that position instead.
281     * @param other another K2 collection with the same E and S types
282     * @return true if this collection changed as a result of this call
283     */
284    public boolean putAll(BundleBiMap<? extends E, ? extends S> other)
285    {
286        if(other == null) return false;
287        boolean changed = false;
288        int sz = other.size();
289        for (int i = 0; i < sz; i++) {
290            int[][] bundle = other.bm.getAAt(i);
291            if(bundle == null || bundle.length == 0) continue;
292            if(bundle.length == 1)
293                changed |= put(elements.keysAt(bundle[0]), other.bm.getBAt(i));
294            else
295                changed |= put(elements.keysAt(bundle[0]), bundle[1], other.bm.getBAt(i));
296        }
297        return changed;
298    }

544     * Reorders this BundleBiMap using {@code ordering}, which has the same length as this object's {@link #size()}
545     * and can be generated with {@link ArrayTools#range(int)} (which, if applied, would produce no
546     * change to the current ordering), {@link RNG#randomOrdering(int)} (which gives a random ordering, and if
547     * applied immediately would be the same as calling {@link #shuffle(RNG)}), or made in some other way. If you
548     * already have an ordering and want to make a different ordering that can undo the change, you can use
549     * {@link ArrayTools#invertOrdering(int[])} called on the original ordering. The effects of this method, if called
550     * with an ordering that has repeat occurrences of an int or contains ints that are larger than its size should
551     * permit, are undefined other than the vague definition of "probably bad, unless you like crashes."
552     * @param ordering an int array or vararg that must contain each int from 0 to {@link #size()}
553     * @return this for chaining
554     */
555    public BundleBiMap<E, S> reorder(int... ordering)
556    {
557        if(ordering == null || ordering.length != bm.size()) return this;
558        bm.reorder(ordering);
559        int len = mm.size();
560        for (int i = 0; i < len; i++) {
561            IntVLA iv = mm.get(i);
562            for (int j = 0; j < iv.size; j++) {
563                iv.set(i, ordering[iv.get(i)]);
564            }
565        }
566        return this;
567    }
568
569    /**
570     * Generates a random ordering with rng and applies the same ordering to all kinds of keys this has; they will
571     * maintain their current association to other keys but their ordering/indices will change.
572     * @param rng an RNG to produce the random ordering this will use
573     * @return this for chaining
574     */
575    public BundleBiMap<E, S> shuffle(RNG rng)
576    {
577        return reorder(rng.randomOrdering(bm.size()));
578    }
