003/**
004 * This is Ken Perlin's third revision of his noise function. It is sometimes
005 * referred to as "Simplex Noise". Results are bound by (-1, 1) inclusive.
006 *
007 *
008 * It is significantly faster than his earlier versions. This particular version
009 * was originally from Stefan Gustavson. This is much preferred to the earlier
010 * versions of Perlin Noise due to the reasons noted in the articles:
011 * <ul>
012 * <li>http://www.itn.liu.se/~stegu/simplexnoise/simplexnoise.pdf</li>
013 * <li>http://webstaff.itn.liu.se/~stegu/TNM022-2005/perlinnoiselinks/ch02.pdf</li>
014 * </ul>
015 * But, Gustavson's paper is not without its own issues, particularly for 2D noise.
016 * More detail is noted here,
017 * http://stackoverflow.com/questions/18885440/why-does-simplex-noise-seem-to-have-more-artifacts-than-classic-perlin-noise#21568753
018 * and some changes have been made to 2D noise generation to reduce angular artifacts.
019 * Specifically for the 2D gradient table, code based on Gustavson's paper used 12
020 * points, with some duplicates, and not all on the unit circle. In this version,
021 * points are used on the unit circle starting at (1,0) and moving along the circle
022 * in increments of 1.61803398875 radians, that is, the golden ratio phi, getting the
023 * sin and cosine of 15 points after the starting point and storing them as constants.
024 * This definitely doesn't have a noticeable 45 degree angle artifact, though it does
025 * have, to a lesser extent, some other minor artifacts.
026 * <br>
027 * You can also consider {@link WhirlingNoise} as an alternative, which can be faster
028 * and also reduces the likelihood of angular artifacts. WhirlingNoise does not scale its
029 * input (it doesn't need to), so it won't produce the same results as PerlinNoise for the
030 * same inputs, but it will produce similar shape, density, and aesthetic quality of noise.
031 * @see WhirlingNoise A subclass that has a faster implementation and some different qualities.
032 */

153    /**
154     * 2D simplex noise.
155     * This doesn't use its parameters verbatim; xin and yin are both effectively divided by
156     * ({@link Math#E} * {@link Math#PI}), because without a step like that, any integer parameters would return 0 and
157     * only doubles with a decimal component would produce actual noise. This step allows integers to be passed in a
158     * arguments, and changes the cycle at which 0 is repeated to multiples of (E*PI).
159     *
160     * @param xin x input; works well if between 0.0 and 1.0, but anything is accepted
161     * @param yin y input; works well if between 0.0 and 1.0, but anything is accepted
162     * @return noise from -1.0 to 1.0, inclusive
163     */
164    public static double noise(double xin, double yin) {
165        xin *= epi;
166        yin *= epi;
167        double noise0, noise1, noise2; // from the three corners
168        // Skew the input space to determine which simplex cell we're in
169        double skew = (xin + yin) * F2; // Hairy factor for 2D
170        int i = (int) Math.floor(xin + skew);
171        int j = (int) Math.floor(yin + skew);
172        double t = (i + j) * G2;
173        double X0 = i - t; // Unskew the cell origin back to (x,y) space
174        double Y0 = j - t;
175        double x0 = xin - X0; // The x,y distances from the cell origin
176        double y0 = yin - Y0;
177        // For the 2D case, the simplex shape is an equilateral triangle.
178        // Determine which simplex we are in.
179        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
180        // coords
181        if (x0 > y0) {
182            i1 = 1;
183            j1 = 0;
184        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
185        else {
186            i1 = 0;
187            j1 = 1;
188        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
189        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
190        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
191        // where
192        // c = (3-sqrt(3))/6
193        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
194        // unskewed coords
195        double y1 = y0 - j1 + G2;
196        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
197        // unskewed coords
198        double y2 = y0 - 1.0 + 2.0 * G2;
199        // Work out the hashed gradient indices of the three simplex corners
200        int ii = i & 255;
201        int jj = j & 255;
202        int gi0 = perm[ii + perm[jj]] & 15;
203        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
204        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
205        // Calculate the contribution from the three corners
206        double t0 = 0.5 - x0 * x0 - y0 * y0;
207        if (t0 < 0) {
208            noise0 = 0.0;
209        } else {
210            t0 *= t0;
211            noise0 = t0 * t0 * dot(phiGrad2[gi0], x0, y0); // (x,y) of grad3 used
212            // for 2D gradient
213        }
214        double t1 = 0.5 - x1 * x1 - y1 * y1;
215        if (t1 < 0) {
216            noise1 = 0.0;
217        } else {
218            t1 *= t1;
219            noise1 = t1 * t1 * dot(phiGrad2[gi1], x1, y1);
220        }
221        double t2 = 0.5 - x2 * x2 - y2 * y2;
222        if (t2 < 0) {
223            noise2 = 0.0;
224        } else {
225            t2 *= t2;
226            noise2 = t2 * t2 * dot(phiGrad2[gi2], x2, y2);
227        }
228        // Add contributions from each corner to get the final noise value.
229        // The result is scaled to return values in the interval [-1,1].
230        return 70.0 * (noise0 + noise1 + noise2);
231    }

233    /**
234     * 3D simplex noise.
235     *
236     * @param xin X input
237     * @param yin Y input
238     * @param zin Z input
239     * @return noise from -1.0 to 1.0, inclusive
240     */
241    public static double noise(double xin, double yin, double zin) {
242        xin *= epi;
243        yin *= epi;
244        zin *= epi;
245        double n0, n1, n2, n3; // Noise contributions from the four corners
246        // Skew the input space to determine which simplex cell we're in
247        double s = (xin + yin + zin) * F3; // Very nice and simple skew
248        // factor for 3D
249        int i = (int) Math.floor(xin + s);
250        int j = (int) Math.floor(yin + s);
251        int k = (int) Math.floor(zin + s);
252        double t = (i + j + k) * G3;
253        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
254        double Y0 = j - t;
255        double Z0 = k - t;
256        double x0 = xin - X0; // The x,y,z distances from the cell origin
257        double y0 = yin - Y0;
258        double z0 = zin - Z0;
259        // For the 3D case, the simplex shape is a slightly irregular
260        // tetrahedron.
261        // Determine which simplex we are in.
262        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
263        // coords
264        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
265        // coords
266        if (x0 >= y0) {
267            if (y0 >= z0) {
268                i1 = 1;
269                j1 = 0;
270                k1 = 0;
271                i2 = 1;
272                j2 = 1;
273                k2 = 0;
274            } // X Y Z order
275            else if (x0 >= z0) {
276                i1 = 1;
277                j1 = 0;
278                k1 = 0;
279                i2 = 1;
280                j2 = 0;
281                k2 = 1;
282            } // X Z Y order
283            else {
284                i1 = 0;
285                j1 = 0;
286                k1 = 1;
287                i2 = 1;
288                j2 = 0;
289                k2 = 1;
290            } // Z X Y order
291        } else { // x0<y0
292            if (y0 < z0) {
293                i1 = 0;
294                j1 = 0;
295                k1 = 1;
296                i2 = 0;
297                j2 = 1;
298                k2 = 1;
299            } // Z Y X order
300            else if (x0 < z0) {
301                i1 = 0;
302                j1 = 1;
303                k1 = 0;
304                i2 = 0;
305                j2 = 1;
306                k2 = 1;
307            } // Y Z X order
308            else {
309                i1 = 0;
310                j1 = 1;
311                k1 = 0;
312                i2 = 1;
313                j2 = 1;
314                k2 = 0;
315            } // Y X Z order
316        }
317        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
318        // (x,y,z),
319        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
320        // (x,y,z), and
321        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
322        // (x,y,z), where
323        // c = 1/6.
324        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
325        // coords
326        double y1 = y0 - j1 + G3;
327        double z1 = z0 - k1 + G3;
328        double x2 = x0 - i2 + 2.0 * G3; // Offsets for third corner in
329        // (x,y,z) coords
330        double y2 = y0 - j2 + 2.0 * G3;
331        double z2 = z0 - k2 + 2.0 * G3;
332        double x3 = x0 - 1.0 + 3.0 * G3; // Offsets for last corner in
333        // (x,y,z) coords
334        double y3 = y0 - 1.0 + 3.0 * G3;
335        double z3 = z0 - 1.0 + 3.0 * G3;
336        // Work out the hashed gradient indices of the four simplex corners
337        int ii = i & 255;
338        int jj = j & 255;
339        int kk = k & 255;
340        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
341        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
342        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
343        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
344        // Calculate the contribution from the four corners
345        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
346        if (t0 < 0) {
347            n0 = 0.0;
348        } else {
349            t0 *= t0;
350            n0 = t0 * t0 * dot(grad3[gi0], x0, y0, z0);
351        }
352        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
353        if (t1 < 0) {
354            n1 = 0.0;
355        } else {
356            t1 *= t1;
357            n1 = t1 * t1 * dot(grad3[gi1], x1, y1, z1);
358        }
359        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
360        if (t2 < 0) {
361            n2 = 0.0;
362        } else {
363            t2 *= t2;
364            n2 = t2 * t2 * dot(grad3[gi2], x2, y2, z2);
365        }
366        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3;
367        if (t3 < 0) {
368            n3 = 0.0;
369        } else {
370            t3 *= t3;
371            n3 = t3 * t3 * dot(grad3[gi3], x3, y3, z3);
372        }
373        // Add contributions from each corner to get the final noise value.
374        // The result is scaled to stay just inside [-1,1]
375        return 32.0 * (n0 + n1 + n2 + n3);
376    }

378    /**
379     * 4D simplex noise.
380     *
381     * @param x X position
382     * @param y Y position
383     * @param z Z position
384     * @param w Fourth-dimensional position. It is I, the Fourth-Dimensional Ziltoid the Omniscient!
385     * @return noise from -1.0 to 1.0, inclusive
386     */
387    public static double noise(double x, double y, double z, double w) {
388        x *= epi;
389        y *= epi;
390        z *= epi;
391        w *= epi;
392        // The skewing and unskewing factors are hairy again for the 4D case
393        double n0, n1, n2, n3, n4; // Noise contributions from the five
394        // corners
395        // Skew the (x,y,z,w) space to determine which cell of 24 simplices
396        // we're in
397        double s = (x + y + z + w) * F4; // Factor for 4D skewing
398        int i = (int) Math.floor(x + s);
399        int j = (int) Math.floor(y + s);
400        int k = (int) Math.floor(z + s);
401        int l = (int) Math.floor(w + s);
402        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
403        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
404        double Y0 = j - t;
405        double Z0 = k - t;
406        double W0 = l - t;
407        double x0 = x - X0; // The x,y,z,w distances from the cell origin
408        double y0 = y - Y0;
409        double z0 = z - Z0;
410        double w0 = w - W0;
411        // For the 4D case, the simplex is a 4D shape I won't even try to
412        // describe.
413        // To find out which of the 24 possible simplices we're in, we need
414        // to
415        // determine the magnitude ordering of x0, y0, z0 and w0.
416        // The method below is a good way of finding the ordering of x,y,z,w
417        // and
418        // then find the correct traversal order for the simplex we’re in.
419        // First, six pair-wise comparisons are performed between each
420        // possible pair
421        // of the four coordinates, and the results are used to add up binary
422        // bits
423        // for an integer index.
424        int c =
425                (x0 > y0 ? 32 : 0)
426              + (x0 > z0 ? 16 : 0)
427              + (y0 > z0 ? 8 : 0 )
428              + (x0 > w0 ? 4 : 0 )
429              + (y0 > w0 ? 2 : 0 )
430              + (z0 > w0 ? 1 : 0 );
431        int i1, j1, k1, l1; // The integer offsets for the second simplex
432        // corner
433        int i2, j2, k2, l2; // The integer offsets for the third simplex
434        // corner
435        int i3, j3, k3, l3; // The integer offsets for the fourth simplex
436        // corner
437        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
438        // order.
439        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
440        // y<w and x<w
441        // impossible. Only the 24 indices which have non-zero entries make
442        // any sense.
443        // We use a thresholding to set the coordinates in turn from the
444        // largest magnitude.
445        // The number 3 in the "simplex" array is at the position of the
446        // largest coordinate.
447        i1 = simplex[c][0] >= 3 ? 1 : 0;
448        j1 = simplex[c][1] >= 3 ? 1 : 0;
449        k1 = simplex[c][2] >= 3 ? 1 : 0;
450        l1 = simplex[c][3] >= 3 ? 1 : 0;
451        // The number 2 in the "simplex" array is at the second largest
452        // coordinate.
453        i2 = simplex[c][0] >= 2 ? 1 : 0;
454        j2 = simplex[c][1] >= 2 ? 1 : 0;
455        k2 = simplex[c][2] >= 2 ? 1 : 0;
456        l2 = simplex[c][3] >= 2 ? 1 : 0;
457        // The number 1 in the "simplex" array is at the second smallest
458        // coordinate.
459        i3 = simplex[c][0] >= 1 ? 1 : 0;
460        j3 = simplex[c][1] >= 1 ? 1 : 0;
461        k3 = simplex[c][2] >= 1 ? 1 : 0;
462        l3 = simplex[c][3] >= 1 ? 1 : 0;
463        // The fifth corner has all coordinate offsets = 1, so no need to
464        // look that up.
465        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w)
466        // coords
467        double y1 = y0 - j1 + G4;
468        double z1 = z0 - k1 + G4;
469        double w1 = w0 - l1 + G4;
470        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in
471        // (x,y,z,w) coords
472        double y2 = y0 - j2 + 2.0 * G4;
473        double z2 = z0 - k2 + 2.0 * G4;
474        double w2 = w0 - l2 + 2.0 * G4;
475        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in
476        // (x,y,z,w) coords
477        double y3 = y0 - j3 + 3.0 * G4;
478        double z3 = z0 - k3 + 3.0 * G4;
479        double w3 = w0 - l3 + 3.0 * G4;
480        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in
481        // (x,y,z,w) coords
482        double y4 = y0 - 1.0 + 4.0 * G4;
483        double z4 = z0 - 1.0 + 4.0 * G4;
484        double w4 = w0 - 1.0 + 4.0 * G4;
485        // Work out the hashed gradient indices of the five simplex corners
486        int ii = i & 255;
487        int jj = j & 255;
488        int kk = k & 255;
489        int ll = l & 255;
490        int gi0 = perm[ii + perm[jj + perm[kk + perm[ll]]]] & 31;
491        int gi1
492                = perm[ii + i1 + perm[jj + j1 + perm[kk + k1 + perm[ll + l1]]]] & 31;
493        int gi2
494                = perm[ii + i2 + perm[jj + j2 + perm[kk + k2 + perm[ll + l2]]]] & 31;
495        int gi3
496                = perm[ii + i3 + perm[jj + j3 + perm[kk + k3 + perm[ll + l3]]]] & 31;
497        int gi4
498                = perm[ii + 1 + perm[jj + 1 + perm[kk + 1 + perm[ll + 1]]]] & 31;
499        // Calculate the contribution from the five corners
500        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0;
501        if (t0 < 0) {
502            n0 = 0.0;
503        } else {
504            t0 *= t0;
505            n0 = t0 * t0 * dot(grad4[gi0], x0, y0, z0, w0);
506        }
507        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1;
508        if (t1 < 0) {
509            n1 = 0.0;
510        } else {
511            t1 *= t1;
512            n1 = t1 * t1 * dot(grad4[gi1], x1, y1, z1, w1);
513        }
514        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2;
515        if (t2 < 0) {
516            n2 = 0.0;
517        } else {
518            t2 *= t2;
519            n2 = t2 * t2 * dot(grad4[gi2], x2, y2, z2, w2);
520        }
521        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3;
522        if (t3 < 0) {
523            n3 = 0.0;
524        } else {
525            t3 *= t3;
526            n3 = t3 * t3 * dot(grad4[gi3], x3, y3, z3, w3);
527        }
528        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4;
529        if (t4 < 0) {
530            n4 = 0.0;
531        } else {
532            t4 *= t4;
533            n4 = t4 * t4 * dot(grad4[gi4], x4, y4, z4, w4);
534        }
535        // Sum up and scale the result to cover the range [-1,1]
536        return 27.0 * (n0 + n1 + n2 + n3 + n4);
537    }
