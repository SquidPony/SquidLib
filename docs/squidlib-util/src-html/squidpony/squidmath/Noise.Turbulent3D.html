034    public static class Layered1D implements Noise1D {
035        protected int octaves;
036        protected Noise1D basis;
037        public double frequency;
038        public Layered1D() {
039            this(ValueNoise.instance);
040        }
041
042        public Layered1D(Noise1D basis) {
043            this(basis, 2);
044        }
045
046        public Layered1D(Noise1D basis, final int octaves) {
047            this(basis, octaves, 1.0);
048        }
049        public Layered1D(Noise1D basis, final int octaves, double frequency) {
050            this.basis = basis;
051            this.frequency = frequency;
052            this.octaves = Math.max(1, Math.min(63, octaves));
053        }
054
055
056        @Override
057        public double getNoise(double x) {
058            x *= frequency;
059            int s = 1 << (octaves - 1);
060            double n = 0.0, i_s = 1.0;
061            for (int o = 0; o < octaves; o++, s >>= 1) {
062                n += basis.getNoise(x * (i_s *= 0.5) + (o << 6)) * s;
063            }
064            return n / ((1 << octaves) - 1.0);
065        }
066
067        @Override
068        public double getNoiseWithSeed(double x, final int seed) {
069            x *= frequency;
070            int s = 1 << (octaves - 1), seed2 = seed;
071            double n = 0.0, i_s = 1.0;
072            for (int o = 0; o < octaves; o++, s >>= 1) {
073                //seed2 = PintRNG.determine(seed2);
074                n += basis.getNoiseWithSeed(x * (i_s *= 0.5), (seed2 += 0x9E3779B9)) * s;
075            }
076            return n / ((1 << octaves) - 1.0);
077        }
