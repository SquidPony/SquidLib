008/**
009 * Another experimental noise class. Extends PerlinNoise and should have similar quality, but can be faster and has less
010 * periodic results. This is still considered experimental because the exact output may change in future versions, along
011 * with the scale (potentially) and the parameters it takes. In general, {@link #noise(double, double)} and
012 * {@link #noise(double, double, double)} should have similar appearance to {@link PerlinNoise#noise(double, double)}
013 * and {@link PerlinNoise#noise(double, double, double)}, but are not forced to a zoomed-in scale like PerlinNoise makes
014 * its results, are less likely to repeat sections of noise, and are also somewhat faster (a 20% speedup can be expected
015 * over PerlinNoise using those two methods). Sound good? This also performs minimal (if any) arithmetic on 64-bit long
016 * numbers, instead using double and int, a style that is ideal for GWT (this is true of PerlinNoise too).
017 * <br>
018 * Created by Tommy Ettinger on 12/14/2016.
019 */
020@Beta
021public class WhirlingNoise extends PerlinNoise implements Noise.Noise2D, Noise.Noise3D, Noise.Noise4D {

025    private static int fastFloor(double t) {
026        return t >= 0 ? (int) t : (int) t - 1;
027    }
028    private static int fastFloor(float t) {
029        return t >= 0 ? (int) t : (int) t - 1;
030    }
031    protected static final float root3 = 1.7320508f, root5 = 2.236068f,
032            F2f = 0.5f * (root3 - 1f),
033            G2f = (3f - root3) * 0.16666667f,
034            F3f = 0.33333334f,
035            G3f = 0.16666667f,
036            F4f = (root5 - 1f) * 0.25f,
037            G4f = (5f - root5) * 0.05f,
038            unit1_4f =  0.70710678118f, unit1_8f = 0.38268343236f, unit3_8f = 0.92387953251f;
039
040    protected static final float[][] grad2f = {
041            {1f, 0f}, {-1f, 0f}, {0f, 1f}, {0f, -1f},
042            {unit3_8f, unit1_8f}, {unit3_8f, -unit1_8f}, {-unit3_8f, unit1_8f}, {-unit3_8f, -unit1_8f},
043            {unit1_4f, unit1_4f}, {unit1_4f, -unit1_4f}, {-unit1_4f, unit1_4f}, {-unit1_4f, -unit1_4f},
044            {unit1_8f, unit3_8f}, {unit1_8f, -unit3_8f}, {-unit1_8f, unit3_8f}, {-unit1_8f, -unit3_8f}};
045
046    protected static float dotf(final float g[], final float x, final float y) {
047        return g[0] * x + g[1] * y;
048    }
049
050    protected static float dotf(final int g[], final float x, final float y, final float z) {
051        return g[0] * x + g[1] * y + g[2] * z;
052    }

155    /**
156     * 2D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
157     * different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method. Roughly
158     * 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks because
159     * it uses a pseudo-random function (curiously, {@link PintRNG#determine(int)}, which is optimized for the
160     * limitations of GWT but is rather fast here) instead of a number chosen from a single 256-element array.
161     *
162     * @param xin X input; works well if between 0.0 and 1.0, but anything is accepted
163     * @param yin Y input; works well if between 0.0 and 1.0, but anything is accepted
164     * @return noise from -1.0 to 1.0, inclusive
165     */
166    public static double noise(final double xin, final double yin) {
167        //xin *= epi;
168        //yin *= epi;
169        double noise0, noise1, noise2; // from the three corners
170        // Skew the input space to determine which simplex cell we're in
171        double skew = (xin + yin) * F2; // Hairy factor for 2D
172        int i = fastFloor(xin + skew);
173        int j = fastFloor(yin + skew);
174        double t = (i + j) * G2;
175        double X0 = i - t; // Unskew the cell origin back to (x,y) space
176        double Y0 = j - t;
177        double x0 = xin - X0; // The x,y distances from the cell origin
178        double y0 = yin - Y0;
179        // For the 2D case, the simplex shape is an equilateral triangle.
180        // Determine which simplex we are in.
181        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
182        // coords
183        if (x0 > y0) {
184            i1 = 1;
185            j1 = 0;
186        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
187        else {
188            i1 = 0;
189            j1 = 1;
190        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
191        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
192        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
193        // where
194        // c = (3-sqrt(3))/6
195        double x1 = x0 - i1 + G2; // Offsets for middle corner in (x,y)
196        // unskewed coords
197        double y1 = y0 - j1 + G2;
198        double x2 = x0 - 1.0 + 2.0 * G2; // Offsets for last corner in (x,y)
199        // unskewed coords
200        double y2 = y0 - 1.0 + 2.0 * G2;
201        // Work out the hashed gradient indices of the three simplex corners
202        /*
203        int ii = i & 255;
204        int jj = j & 255;
205        int gi0 = perm[ii + perm[jj]] & 15;
206        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
207        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
208        */
209        /*
210        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
211                i + i1 + ((j + j1) * 0x9E3779B9),
212                i + 1 + ((j + 1) * 0x9E3779B9),
213                seed);
214        int gi0 = hash & 15;
215        int gi1 = (hash >>>= 4) & 15;
216        int gi2 = (hash >>> 4) & 15;
217        */
218        int gi0 = determine(i + determine(j)) & 15;
219        int gi1 = determine(i + i1 + determine(j + j1)) & 15;
220        int gi2 = determine(i + 1 + determine(j + 1)) & 15;
221
222        // Calculate the contribution from the three corners
223        double t0 = 0.5 - x0 * x0 - y0 * y0;
224        if (t0 < 0) {
225            noise0 = 0.0;
226        } else {
227            t0 *= t0;
228            noise0 = t0 * t0 * dot(grad2[gi0], x0, y0);
229            // for 2D gradient
230        }
231        double t1 = 0.5 - x1 * x1 - y1 * y1;
232        if (t1 < 0) {
233            noise1 = 0.0;
234        } else {
235            t1 *= t1;
236            noise1 = t1 * t1 * dot(grad2[gi1], x1, y1);
237        }
238        double t2 = 0.5 - x2 * x2 - y2 * y2;
239        if (t2 < 0) {
240            noise2 = 0.0;
241        } else {
242            t2 *= t2;
243            noise2 = t2 * t2 * dot(grad2[gi2], x2, y2);
244        }
245        // Add contributions from each corner to get the final noise value.
246        // The result is scaled to return values in the interval [-1,1].
247        return 70.0 * (noise0 + noise1 + noise2);
248    }

252     * 2D simplex noise returning a float; extremely similar to {@link #noise(double, double)}, but this may be slightly
253     * faster or slightly slower. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result
254     * will be different when passing the same arguments to {@link PerlinNoise#noise(double, double)} and this method.
255     *
256     * @param x x input; works well if between 0.0 and 1.0, but anything is accepted
257     * @param y y input; works well if between 0.0 and 1.0, but anything is accepted
258     * @return noise from -1.0 to 1.0, inclusive

260    public static float noiseAlt(double x, double y) {
261        //xin *= epi;
262        //yin *= epi;
263        float noise0, noise1, noise2; // from the three corners
264        float xin = (float)x, yin = (float)y;
265        // Skew the input space to determine which simplex cell we're in
266        float skew = (xin + yin) * F2f; // Hairy factor for 2D
267        int i = fastFloor(xin + skew);
268        int j = fastFloor(yin + skew);
269        float t = (i + j) * G2f;
270        float X0 = i - t; // Unskew the cell origin back to (x,y) space
271        float Y0 = j - t;
272        float x0 = xin - X0; // The x,y distances from the cell origin
273        float y0 = yin - Y0;
274        // For the 2D case, the simplex shape is an equilateral triangle.
275        // Determine which simplex we are in.
276        int i1, j1; // Offsets for second (middle) corner of simplex in (i,j)
277        // coords
278        if (x0 > y0) {
279            i1 = 1;
280            j1 = 0;
281        } // lower triangle, XY order: (0,0)->(1,0)->(1,1)
282        else {
283            i1 = 0;
284            j1 = 1;
285        } // upper triangle, YX order: (0,0)->(0,1)->(1,1)
286        // A step of (1,0) in (i,j) means a step of (1-c,-c) in (x,y), and
287        // a step of (0,1) in (i,j) means a step of (-c,1-c) in (x,y),
288        // where
289        // c = (3-sqrt(3))/6
290        float x1 = x0 - i1 + G2f; // Offsets for middle corner in (x,y)
291        // unskewed coords
292        float y1 = y0 - j1 + G2f;
293        float x2 = x0 - 1f + 2f * G2f; // Offsets for last corner in (x,y)
294        // unskewed coords
295        float y2 = y0 - 1f + 2f * G2f;
296        // Work out the hashed gradient indices of the three simplex corners
297        /*
298        int ii = i & 255;
299        int jj = j & 255;
300        int gi0 = perm[ii + perm[jj]] & 15;
301        int gi1 = perm[ii + i1 + perm[jj + j1]] & 15;
302        int gi2 = perm[ii + 1 + perm[jj + 1]] & 15;
303        */
304        /*
305        int hash = (int) rawNoise(i + (j * 0x9E3779B9),
306                i + i1 + ((j + j1) * 0x9E3779B9),
307                i + 1 + ((j + 1) * 0x9E3779B9),
308                seed);
309        int gi0 = hash & 15;
310        int gi1 = (hash >>>= 4) & 15;
311        int gi2 = (hash >>> 4) & 15;
312        */
313        int gi0 = PintRNG.determine(i + PintRNG.determine(j)) & 15;
314        int gi1 = PintRNG.determine(i + i1 + PintRNG.determine(j + j1)) & 15;
315        int gi2 = PintRNG.determine(i + 1 + PintRNG.determine(j + 1)) & 15;
316
317        // Calculate the contribution from the three corners
318        float t0 = 0.5f - x0 * x0 - y0 * y0;
319        if (t0 < 0) {
320            noise0 = 0f;
321        } else {
322            t0 *= t0;
323            noise0 = t0 * t0 * dotf(grad2f[gi0], x0, y0);
324            // for 2D gradient
325        }
326        float t1 = 0.5f - x1 * x1 - y1 * y1;
327        if (t1 < 0) {
328            noise1 = 0f;
329        } else {
330            t1 *= t1;
331            noise1 = t1 * t1 * dotf(grad2f[gi1], x1, y1);
332        }
333        float t2 = 0.5f - x2 * x2 - y2 * y2;
334        if (t2 < 0) {
335            noise2 = 0f;
336        } else {
337            t2 *= t2;
338            noise2 = t2 * t2 * dotf(grad2f[gi2], x2, y2);
339        }
340        // Add contributions from each corner to get the final noise value.
341        // The result is scaled to return values in the interval [-1,1].
342        return 70f * (noise0 + noise1 + noise2);
343    }

345    /**
346     * 3D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
347     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double)} and this method.
348     * Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in chunks
349     * because it uses a pseudo-random function (curiously, {@link PintRNG#determine(int)}, which is optimized for the
350     * limitations of GWT but is rather fast here) instead of a number chosen from a single 256-element array.
351     * @param xin X input
352     * @param yin Y input
353     * @param zin Z input
354     * @return noise from -1.0 to 1.0, inclusive
355     */
356    public static double noise(final double xin, final double yin, final double zin) {
357        //xin *= epi;
358        //yin *= epi;
359        //zin *= epi;
360        double n0, n1, n2, n3; // Noise contributions from the four corners
361        // Skew the input space to determine which simplex cell we're in
362        double s = (xin + yin + zin) * F3; // Very nice and simple skew
363        // factor for 3D
364        int i = fastFloor(xin + s);
365        int j = fastFloor(yin + s);
366        int k = fastFloor(zin + s);
367        double t = (i + j + k) * G3;
368        double X0 = i - t; // Unskew the cell origin back to (x,y,z) space
369        double Y0 = j - t;
370        double Z0 = k - t;
371        double x0 = xin - X0; // The x,y,z distances from the cell origin
372        double y0 = yin - Y0;
373        double z0 = zin - Z0;
374        // For the 3D case, the simplex shape is a slightly irregular
375        // tetrahedron.
376        // Determine which simplex we are in.
377        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
378        // coords
379        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
380        // coords
381        if (x0 >= y0) {
382            if (y0 >= z0) {
383                i1 = 1;
384                j1 = 0;
385                k1 = 0;
386                i2 = 1;
387                j2 = 1;
388                k2 = 0;
389            } // X Y Z order
390            else if (x0 >= z0) {
391                i1 = 1;
392                j1 = 0;
393                k1 = 0;
394                i2 = 1;
395                j2 = 0;
396                k2 = 1;
397            } // X Z Y order
398            else {
399                i1 = 0;
400                j1 = 0;
401                k1 = 1;
402                i2 = 1;
403                j2 = 0;
404                k2 = 1;
405            } // Z X Y order
406        } else { // x0<y0
407            if (y0 < z0) {
408                i1 = 0;
409                j1 = 0;
410                k1 = 1;
411                i2 = 0;
412                j2 = 1;
413                k2 = 1;
414            } // Z Y X order
415            else if (x0 < z0) {
416                i1 = 0;
417                j1 = 1;
418                k1 = 0;
419                i2 = 0;
420                j2 = 1;
421                k2 = 1;
422            } // Y Z X order
423            else {
424                i1 = 0;
425                j1 = 1;
426                k1 = 0;
427                i2 = 1;
428                j2 = 1;
429                k2 = 0;
430            } // Y X Z order
431        }
432        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
433        // (x,y,z),
434        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
435        // (x,y,z), and
436        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
437        // (x,y,z), where
438        // c = 1/6.
439        double x1 = x0 - i1 + G3; // Offsets for second corner in (x,y,z)
440        // coords
441        double y1 = y0 - j1 + G3;
442        double z1 = z0 - k1 + G3;
443        double x2 = x0 - i2 + F3; // Offsets for third corner in
444        // (x,y,z) coords
445        double y2 = y0 - j2 + F3;
446        double z2 = z0 - k2 + F3;
447        double x3 = x0 - 0.5; // Offsets for last corner in
448        // (x,y,z) coords
449        double y3 = y0 - 0.5;
450        double z3 = z0 - 0.5;
451        // Work out the hashed gradient indices of the four simplex corners
452
453        /*
454        int ii = i & 255;
455        int jj = j & 255;
456        int kk = k & 255;
457
458        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
459        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
460        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
461        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
462        */
463
464        int gi0 = determineBounded(i + determine(j + determine(k)), 12);
465        int gi1 = determineBounded(i + i1 + determine(j + j1 + determine(k + k1)), 12);
466        int gi2 = determineBounded(i + i2 + determine(j + j2 + determine(k + k2)), 12);
467        int gi3 = determineBounded(i + 1 + determine(j + 1 + determine(k + 1)), 12);
468
469        /*
470        int hash = (int) rawNoise(i + ((j + k * 0x632BE5AB) * 0x9E3779B9),
471                i + i1 + ((j + j1 + (k + k1) * 0x632BE5AB) * 0x9E3779B9),
472                i + i2 + ((j + j2 + (k + k2) * 0x632BE5AB) * 0x9E3779B9),
473                i + 1 + ((j + 1 + ((k + 1) * 0x632BE5AB)) * 0x9E3779B9),
474                seed);
475        int gi0 = (hash >>>= 4) % 12;
476        int gi1 = (hash >>>= 4) % 12;
477        int gi2 = (hash >>>= 4) % 12;
478        int gi3 = (hash >>> 4) % 12;
479        */
480
481        //int hash = (int) rawNoise(i, j, k, seed);
482        //int gi0 = (hash >>>= 4) % 12, gi1 = (hash >>>= 4) % 12, gi2 = (hash >>>= 4) % 12, gi3 = (hash >>>= 4) % 12;
483        // Calculate the contribution from the four corners
484        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0;
485        if (t0 < 0) {
486            n0 = 0.0;
487        } else {
488            t0 *= t0;
489            n0 = t0 * t0 * dot(grad3[gi0], x0, y0, z0);
490        }
491        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1;
492        if (t1 < 0) {
493            n1 = 0.0;
494        } else {
495            t1 *= t1;
496            n1 = t1 * t1 * dot(grad3[gi1], x1, y1, z1);
497        }
498        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2;
499        if (t2 < 0) {
500            n2 = 0.0;
501        } else {
502            t2 *= t2;
503            n2 = t2 * t2 * dot(grad3[gi2], x2, y2, z2);
504        }
505        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3;
506        if (t3 < 0) {
507            n3 = 0.0;
508        } else {
509            t3 *= t3;
510            n3 = t3 * t3 * dot(grad3[gi3], x3, y3, z3);
511        }
512        // Add contributions from each corner to get the final noise value.
513        // The result is scaled to stay just inside [-1,1]
514        return 32.0 * (n0 + n1 + n2 + n3);
515    }

517    /**
518     * 3D simplex noise returning a float; extremely similar to {@link #noise(double, double, double)}, but this may
519     * be slightly faster or slightly slower. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of
520     * the result will be different when passing the same arguments to {@link PerlinNoise#noise(double, double, double)}
521     * and this method.
522     *
523     * @param x X input
524     * @param y Y input
525     * @param z Z input
526     * @return noise from -1.0 to 1.0, inclusive
527     */
528    public static float noiseAlt(double x, double y, double z) {
529        //xin *= epi;
530        //yin *= epi;
531        //zin *= epi;
532        float xin = (float)x, yin = (float)y, zin = (float)z;
533        float n0, n1, n2, n3; // Noise contributions from the four corners
534        // Skew the input space to determine which simplex cell we're in
535        float s = (xin + yin + zin) * F3f; // Very nice and simple skew
536        // factor for 3D
537        int i = fastFloor(xin + s);
538        int j = fastFloor(yin + s);
539        int k = fastFloor(zin + s);
540        float t = (i + j + k) * G3f;
541        float X0 = i - t; // Unskew the cell origin back to (x,y,z) space
542        float Y0 = j - t;
543        float Z0 = k - t;
544        float x0 = xin - X0; // The x,y,z distances from the cell origin
545        float y0 = yin - Y0;
546        float z0 = zin - Z0;
547        // For the 3D case, the simplex shape is a slightly irregular
548        // tetrahedron.
549        // Determine which simplex we are in.
550        int i1, j1, k1; // Offsets for second corner of simplex in (i,j,k)
551        // coords
552        int i2, j2, k2; // Offsets for third corner of simplex in (i,j,k)
553        // coords
554        if (x0 >= y0) {
555            if (y0 >= z0) {
556                i1 = 1;
557                j1 = 0;
558                k1 = 0;
559                i2 = 1;
560                j2 = 1;
561                k2 = 0;
562            } // X Y Z order
563            else if (x0 >= z0) {
564                i1 = 1;
565                j1 = 0;
566                k1 = 0;
567                i2 = 1;
568                j2 = 0;
569                k2 = 1;
570            } // X Z Y order
571            else {
572                i1 = 0;
573                j1 = 0;
574                k1 = 1;
575                i2 = 1;
576                j2 = 0;
577                k2 = 1;
578            } // Z X Y order
579        } else { // x0<y0
580            if (y0 < z0) {
581                i1 = 0;
582                j1 = 0;
583                k1 = 1;
584                i2 = 0;
585                j2 = 1;
586                k2 = 1;
587            } // Z Y X order
588            else if (x0 < z0) {
589                i1 = 0;
590                j1 = 1;
591                k1 = 0;
592                i2 = 0;
593                j2 = 1;
594                k2 = 1;
595            } // Y Z X order
596            else {
597                i1 = 0;
598                j1 = 1;
599                k1 = 0;
600                i2 = 1;
601                j2 = 1;
602                k2 = 0;
603            } // Y X Z order
604        }
605        // A step of (1,0,0) in (i,j,k) means a step of (1-c,-c,-c) in
606        // (x,y,z),
607        // a step of (0,1,0) in (i,j,k) means a step of (-c,1-c,-c) in
608        // (x,y,z), and
609        // a step of (0,0,1) in (i,j,k) means a step of (-c,-c,1-c) in
610        // (x,y,z), where
611        // c = 1/6.
612        float x1 = x0 - i1 + G3f; // Offsets for second corner in (x,y,z)
613        // coords
614        float y1 = y0 - j1 + G3f;
615        float z1 = z0 - k1 + G3f;
616        float x2 = x0 - i2 + F3f; // Offsets for third corner in
617        // (x,y,z) coords
618        float y2 = y0 - j2 + F3f;
619        float z2 = z0 - k2 + F3f;
620        float x3 = x0 - 0.5f; // Offsets for last corner in
621        // (x,y,z) coords
622        float y3 = y0 - 0.5f;
623        float z3 = z0 - 0.5f;
624        // Work out the hashed gradient indices of the four simplex corners
625
626        /*
627        int ii = i & 255;
628        int jj = j & 255;
629        int kk = k & 255;
630
631        int gi0 = perm[ii + perm[jj + perm[kk]]] % 12;
632        int gi1 = perm[ii + i1 + perm[jj + j1 + perm[kk + k1]]] % 12;
633        int gi2 = perm[ii + i2 + perm[jj + j2 + perm[kk + k2]]] % 12;
634        int gi3 = perm[ii + 1 + perm[jj + 1 + perm[kk + 1]]] % 12;
635        */
636
637        int gi0 = PintRNG.determineBounded(i + PintRNG.determine(j + PintRNG.determine(k)), 12);
638        int gi1 = PintRNG.determineBounded(i + i1 + PintRNG.determine(j + j1 + PintRNG.determine(k + k1)), 12);
639        int gi2 = PintRNG.determineBounded(i + i2 + PintRNG.determine(j + j2 + PintRNG.determine(k + k2)), 12);
640        int gi3 = PintRNG.determineBounded(i + 1 + PintRNG.determine(j + 1 + PintRNG.determine(k + 1)), 12);
641
642        /*
643        int hash = (int) rawNoise(i + ((j + k * 0x632BE5AB) * 0x9E3779B9),
644                i + i1 + ((j + j1 + (k + k1) * 0x632BE5AB) * 0x9E3779B9),
645                i + i2 + ((j + j2 + (k + k2) * 0x632BE5AB) * 0x9E3779B9),
646                i + 1 + ((j + 1 + ((k + 1) * 0x632BE5AB)) * 0x9E3779B9),
647                seed);
648        int gi0 = (hash >>>= 4) % 12;
649        int gi1 = (hash >>>= 4) % 12;
650        int gi2 = (hash >>>= 4) % 12;
651        int gi3 = (hash >>> 4) % 12;
652        */
653
654        //int hash = (int) rawNoise(i, j, k, seed);
655        //int gi0 = (hash >>>= 4) % 12, gi1 = (hash >>>= 4) % 12, gi2 = (hash >>>= 4) % 12, gi3 = (hash >>>= 4) % 12;
656        // Calculate the contribution from the four corners
657        float t0 = 0.6f - x0 * x0 - y0 * y0 - z0 * z0;
658        if (t0 < 0) {
659            n0 = 0f;
660        } else {
661            t0 *= t0;
662            n0 = t0 * t0 * dotf(grad3[gi0], x0, y0, z0);
663        }
664        float t1 = 0.6f - x1 * x1 - y1 * y1 - z1 * z1;
665        if (t1 < 0) {
666            n1 = 0f;
667        } else {
668            t1 *= t1;
669            n1 = t1 * t1 * dotf(grad3[gi1], x1, y1, z1);
670        }
671        float t2 = 0.6f - x2 * x2 - y2 * y2 - z2 * z2;
672        if (t2 < 0) {
673            n2 = 0f;
674        } else {
675            t2 *= t2;
676            n2 = t2 * t2 * dotf(grad3[gi2], x2, y2, z2);
677        }
678        float t3 = 0.6f - x3 * x3 - y3 * y3 - z3 * z3;
679        if (t3 < 0) {
680            n3 = 0f;
681        } else {
682            t3 *= t3;
683            n3 = t3 * t3 * dotf(grad3[gi3], x3, y3, z3);
684        }
685        // Add contributions from each corner to get the final noise value.
686        // The result is scaled to stay just inside [-1,1]
687        return 32f * (n0 + n1 + n2 + n3);
688    }

690    /**
691     * 4D simplex noise. Unlike {@link PerlinNoise}, uses its parameters verbatim, so the scale of the result will be
692     * different when passing the same arguments to {@link PerlinNoise#noise(double, double, double, double)} and this
693     * method. Roughly 20-25% faster than the equivalent method in PerlinNoise, plus it has less chance of repetition in
694     * chunks because it uses a pseudo-random function (curiously, {@link PintRNG#determine(int)}, which is optimized
695     * for the limitations of GWT but is rather fast here) instead of a number chosen from a single 256-element array.
696     * @param x X input
697     * @param y Y input
698     * @param z Z input
699     * @param w W input (fourth-dimensional)
700     * @return noise from -1.0 to 1.0, inclusive
701     */
702    public static double noise(double x, double y, double z, double w) {
703        // The skewing and unskewing factors are hairy again for the 4D case
704
705        // Skew the (x,y,z,w) space to determine which cell of 24 simplices
706        // we're in
707        double s = (x + y + z + w) * F4; // Factor for 4D skewing
708        int i = fastFloor(x + s);
709        int j = fastFloor(y + s);
710        int k = fastFloor(z + s);
711        int l = fastFloor(w + s);
712        double t = (i + j + k + l) * G4; // Factor for 4D unskewing
713        double X0 = i - t; // Unskew the cell origin back to (x,y,z,w) space
714        double Y0 = j - t;
715        double Z0 = k - t;
716        double W0 = l - t;
717        double x0 = x - X0; // The x,y,z,w distances from the cell origin
718        double y0 = y - Y0;
719        double z0 = z - Z0;
720        double w0 = w - W0;
721        // For the 4D case, the simplex is a 4D shape I won't even try to
722        // describe.
723        // To find out which of the 24 possible simplices we're in, we need
724        // to
725        // determine the magnitude ordering of x0, y0, z0 and w0.
726        // The method below is a good way of finding the ordering of x,y,z,w
727        // and
728        // then find the correct traversal order for the simplex we’re in.
729        // First, six pair-wise comparisons are performed between each
730        // possible pair
731        // of the four coordinates, and the results are used to add up binary
732        // bits
733        // for an integer index.
734        int c = (x0 > y0 ? 32 : 0) | (x0 > z0 ? 16 : 0) | (y0 > z0 ? 8 : 0) |
735                (x0 > w0 ? 4 : 0) | (y0 > w0 ? 2 : 0) | (z0 > w0 ? 1 : 0);
736
737        // simplex[c] is a 4-vector with the numbers 0, 1, 2 and 3 in some
738        // order.
739        // Many values of c will never occur, since e.g. x>y>z>w makes x<z,
740        // y<w and x<w
741        // impossible. Only the 24 indices which have non-zero entries make
742        // any sense.
743        // We use a thresholding to set the coordinates in turn from the
744        // largest magnitude.
745        // The number 3 in the "simplex" array is at the position of the
746        // largest coordinate.
747
748        // The integer offsets for the second simplex corner
749        int i1 = simplex[c][0] >= 3 ? 1 : 0;
750        int j1 = simplex[c][1] >= 3 ? 1 : 0;
751        int k1 = simplex[c][2] >= 3 ? 1 : 0;
752        int l1 = simplex[c][3] >= 3 ? 1 : 0;
753        // The number 2 in the "simplex" array is at the second largest
754        // coordinate.
755
756        // The integer offsets for the third simplex corner
757        int i2 = simplex[c][0] >= 2 ? 1 : 0;
758        int j2 = simplex[c][1] >= 2 ? 1 : 0;
759        int k2 = simplex[c][2] >= 2 ? 1 : 0;
760        int l2 = simplex[c][3] >= 2 ? 1 : 0;
761        // The number 1 in the "simplex" array is at the second smallest
762        // coordinate.
763
764        // The integer offsets for the fourth simplex corner
765        int i3 = simplex[c][0] >= 1 ? 1 : 0;
766        int j3 = simplex[c][1] >= 1 ? 1 : 0;
767        int k3 = simplex[c][2] >= 1 ? 1 : 0;
768        int l3 = simplex[c][3] >= 1 ? 1 : 0;
769        // The fifth corner has all coordinate offsets = 1, so no need to
770        // look that up.
771        double x1 = x0 - i1 + G4; // Offsets for second corner in (x,y,z,w) coords
772        double y1 = y0 - j1 + G4;
773        double z1 = z0 - k1 + G4;
774        double w1 = w0 - l1 + G4;
775        double x2 = x0 - i2 + 2.0 * G4; // Offsets for third corner in (x,y,z,w) coords
776        double y2 = y0 - j2 + 2.0 * G4;
777        double z2 = z0 - k2 + 2.0 * G4;
778        double w2 = w0 - l2 + 2.0 * G4;
779        double x3 = x0 - i3 + 3.0 * G4; // Offsets for fourth corner in (x,y,z,w) coords
780        double y3 = y0 - j3 + 3.0 * G4;
781        double z3 = z0 - k3 + 3.0 * G4;
782        double w3 = w0 - l3 + 3.0 * G4;
783        double x4 = x0 - 1.0 + 4.0 * G4; // Offsets for last corner in (x,y,z,w) coords
784        double y4 = y0 - 1.0 + 4.0 * G4;
785        double z4 = z0 - 1.0 + 4.0 * G4;
786        double w4 = w0 - 1.0 + 4.0 * G4;
787
788        int gi0 = determine(i + determine(j + determine(k + determine(l)))) & 31;
789        int gi1 = determine(i + i1 + determine(j + j1 + determine(k + k1 + determine(l + l1)))) & 31;
790        int gi2 = determine(i + i2 + determine(j + j2 + determine(k + k2 + determine(l + l2)))) & 31;
791        int gi3 = determine(i + i3 + determine(j + j3 + determine(k + k3 + determine(l + l3)))) & 31;
792        int gi4 = determine(i + 1 + determine(j + 1 + determine(k + 1 + determine(l + 1)))) & 31;
793
794        // Noise contributions from the five corners are n0 to n4
795
796        // Calculate the contribution from the five corners
797        double t0 = 0.6 - x0 * x0 - y0 * y0 - z0 * z0 - w0 * w0, n0;
798        if (t0 < 0) {
799            n0 = 0.0;
800        } else {
801            t0 *= t0;
802            n0 = t0 * t0 * dot(grad4[gi0], x0, y0, z0, w0);
803        }
804        double t1 = 0.6 - x1 * x1 - y1 * y1 - z1 * z1 - w1 * w1, n1;
805        if (t1 < 0) {
806            n1 = 0.0;
807        } else {
808            t1 *= t1;
809            n1 = t1 * t1 * dot(grad4[gi1], x1, y1, z1, w1);
810        }
811        double t2 = 0.6 - x2 * x2 - y2 * y2 - z2 * z2 - w2 * w2,  n2;
812        if (t2 < 0) {
813            n2 = 0.0;
814        } else {
815            t2 *= t2;
816            n2 = t2 * t2 * dot(grad4[gi2], x2, y2, z2, w2);
817        }
818        double t3 = 0.6 - x3 * x3 - y3 * y3 - z3 * z3 - w3 * w3, n3;
819        if (t3 < 0) {
820            n3 = 0.0;
821        } else {
822            t3 *= t3;
823            n3 = t3 * t3 * dot(grad4[gi3], x3, y3, z3, w3);
824        }
825        double t4 = 0.6 - x4 * x4 - y4 * y4 - z4 * z4 - w4 * w4, n4;
826        if (t4 < 0) {
827            n4 = 0.0;
828        } else {
829            t4 *= t4;
830            n4 = t4 * t4 * dot(grad4[gi4], x4, y4, z4, w4);
831        }
832        // Sum up and scale the result to cover the range [-1,1]
833        return 27.0 * (n0 + n1 + n2 + n3 + n4);
834    }
