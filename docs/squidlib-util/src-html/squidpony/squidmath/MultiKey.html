001package squidpony.squidmath;
002
003import squidpony.ArrayTools;
004import squidpony.annotation.Beta;
005
006import java.util.Collection;
007import java.util.Iterator;
008import java.util.SortedSet;
009
010/**
011 * An ordered multi-directional "map" that keeps 1 or more keysets organized so you can look up one key given a key in
012 * another keyset, and do the same for the indices of keys. Does not have generic type parameters, which is needed
013 * because we handle arbitrary counts of keysets, and each keyset could have its own type. You can use most of the
014 * normal Map methods here, though they often need an int as their first argument, {@code which}, that specifies which
015 * keyset the method applies to. For example, {@link #contains(int, Object)} checks for the presence of the second
016 * parameter in the keyset specified by the first parameter. Adding items to a MultiKey uses {@link #put(Object...)},
017 * and that does not take a {@code which} parameter because it needs to add an item to every keyset to succeed, and will
018 * do nothing if the array or varargs passed to it has a different length than the {@link #keyCount} of the MultiKey.
019 * Created by Tommy Ettinger on 10/23/2016.
020 */
021@Beta
022@SuppressWarnings("unchecked")
023public class MultiKey {
024    public final int keyCount;
025    Arrangement[] keys;
026
027    /**
028     * Constructs an empty MultiKey.
029     */
030    public MultiKey()
031    {
032        this(3, 16, 0.5f);
033    }
034
035    /**
036     * Constructs a MultiKey with the expected number of indices to hold (the number of items in each keyset is always
037     * the same, and this will be more efficient if expected is greater than that number).
038     * @param expected how many items this should be ready to hold; can resize if needed
039     */
040    public MultiKey(int keyCount, int expected)
041    {
042        this(keyCount, expected, 0.5f);
043    }
044
045    /**
046     * Constructs a MultiKey with the expected number of indices to hold (the number of items in each keyset is always
047     * the same, and this will be more efficient if expected is greater than that number) and the load factor to use,
048     * between 0.1f and 0.8f usually (using load factors higher than 0.8f can cause problems).
049     * @param expected the amount of indices (the number of items in each keyset is always the same) this should hold
050     * @param f the load factor, probably between 0.1f and 0.8f
051     */
052    public MultiKey(int keyCount, int expected, float f)
053    {
054        this.keyCount = keyCount;
055        keys = new Arrangement[keyCount];
056        for (int i = 0; i < keyCount; i++) {
057            keys[i] = new Arrangement(expected, f);
058        }
059    }
060
061    /**
062     * Constructs a MultiKey from a Collection of Iterables that will be processed in tandem, adding a unique item from
063     * each keyset in keysets if and only if it can also add a unique item from all other keysets, otherwise skipping
064     * that iteration in each Iterable.
065     * @param keysets a Collection of Iterable data structures, each containing items that should all be unique
066     */
067    public MultiKey(Collection<Iterable> keysets) {
068        if (keysets == null) {
069            keyCount = 0;
070            keys = new Arrangement[0];
071        } else {
072            keyCount = keysets.size();
073            keys = new Arrangement[keyCount];
074            for (int i = 0; i < keyCount; i++) {
075                keys[i] = new Arrangement();
076            }
077            putAll(keysets);
078        }
079    }
080
081    public MultiKey(MultiKey other)
082    {
083        if(other == null)
084        {
085            keyCount = 0;
086            keys = new Arrangement[0];
087        }
088        else
089        {
090            keyCount = other.keyCount;
091            keys = new Arrangement[keyCount];
092            for (int i = 0; i < keyCount; i++) {
093                keys[i] = new Arrangement(other.keys[i]);
094            }
095        }
096    }
097
098    public MultiKey(Arrangement[] keysets)
099    {
100        if(keysets == null)
101        {
102            keyCount = 0;
103            keys = new Arrangement[0];
104        }
105        else
106        {
107            keyCount = keysets.length;
108            keys = new Arrangement[keyCount];
109            int minLength = Integer.MAX_VALUE;
110            for (int k = 0; k < keyCount; k++) {
111                if(keysets[k] == null)
112                    return;
113                minLength = Math.min(minLength, keysets[k].size);
114            }
115            for (int k = 0; k < keyCount; k++) {
116                keys[k] = keysets[k].take(minLength);
117            }
118        }
119    }
120    /**
121     * Returns true if this contains key in the keyset specified by which.
122     * @param which which keyset to check in
123     * @param key the key to check the presence of
124     * @return true if key is present in the keyset at which; false otherwise
125     */
126    public boolean contains(int which, Object key) {
127        if(which >= 0 && which < keyCount)
128            return keys[which].containsKey(key);
129        return false;
130    }
131
132    /**
133     * Returns true if index is between 0 (inclusive) and {@link #size()} (exclusive), or false otherwise.
134     * @param index the index to check
135     * @return true if index is a valid index in the ordering of this MultiKey
136     */
137    public boolean containsIndex(int index) {
138        if(keyCount > 0) return keys[0].containsValue(index);
139        return false;
140    }
141
142    /**
143     * Given an index of a keyset (which) and an Object key, finds the position in the ordering that key has in the
144     * keyset at which, or -1 if key is not present.
145     * <br>
146     * Unlike {@link java.util.List#indexOf(Object)}, this runs in constant time.
147     * @param which which keyset to check in
148     * @param key the key to find the position of
149     * @return the int index of key in the ordering, or -1 if it is not present
150     */
151    public int indexOf(int which, Object key)
152    {
153        if(which >= 0 && which < keyCount)
154            return keys[which].getInt(key);
155        return -1;
156    }
157
158    /**
159     * Given an index of a keyset (which) and an int index, finds the associated key in the keyset specified by which
160     * (using index as a point in the ordering).
161     * @param which which keyset to get from
162     * @param index an int index into this MultiKey
163     * @return the key (in the keyset found by which) with index for its position in the ordering, or null if index or which was invalid
164     */
165    public Object getAt(int which, int index)
166    {
167        if(which >= 0 && which < keyCount)
168            return keys[which].keyAt(index);
169        return null;
170    }
171
172    /**
173     * Given an int index, finds the associated keys at all keysets (using index as a point in the ordering) and returns
174     * them as a newly-allocated Object array.
175     * @param index an int index into this MultiKey
176     * @return the array of keys with index for their position in the ordering, or an array of null if index was invalid
177     * @see #getAllAt(int, Object[]) getAllAt can avoid allocating a new array each time
178     */
179    public Object[] getAllAt(int index) {
180        Object[] os = new Object[keyCount];
181        for (int i = 0; i < keyCount; i++) {
182            os[i] = keys[i].keyAt(index);
183        }
184        return os;
185    }
186
187    /**
188     * Given an int index and an Object array to reuse, finds the associated keys at all keysets (using index as a point
189     * in the ordering) and fills into with those keys, up to keyCount items.
190     * @param index an int index into this MultiKey
191     * @param into an Object array to reuse and fill with items; will be returned as-is if smaller than keyCount
192     * @return the array of keys with index for their position in the ordering, or an array of null if index was invalid
193     */
194    public Object[] getAllAt(int index, Object[] into) {
195        if (into != null && into.length >= keyCount) {
196            for (int i = 0; i < keyCount; i++) {
197                into[i] = keys[i].keyAt(index);
198            }
199        }
200        return into;
201    }
202
203    /**
204     * Given an index of the keyset to look up a key in (lookingUp), an index of the keyset to get from (getting), and
205     * an Object key to look up (key), finds the Object key in the keyset specified by getting that is associated with
206     * key in the keyset specified by lookingUp. key and the returned value will be at the same point in the ordering.
207     * @param lookingUp which keyset to look up the {@code key} parameter in
208     * @param getting which keyset to get a value from
209     * @param key an object to use as a key, which should be the right type for the keyset at lookingUp
210     * @return the object from getting's keyset that is associated with key, or null if key was not present
211     */
212    public Object getFrom(int lookingUp, int getting, Object key)
213    {
214        if(lookingUp >= 0 && lookingUp < keyCount && getting >= 0 && getting < keyCount)
215            return keys[getting].keyAt(keys[lookingUp].getInt(key));
216        return null;
217    }
218
219    /**
220     * Gets a random key from the keyset specified by which using the given RNG.
221     * @param which which keyset to get a random key from
222     * @param random generates a random index to get a key with
223     * @return a randomly chosen key from the keyset specified, or null if this is empty
224     */
225    public Object randomKey(int which, RNG random)
226    {
227        if(which >= 0 && which < keyCount)
228            return keys[which].randomKey(random);
229        return null;
230    }
231    /**
232     * Gets a random key from a random keyset in this MultiKey using the given RNG.
233     * @param random generates a random keyset index and random item index, to get a random key
234     * @return a randomly chosen Object key from possibly any keyset in this, or null if this is empty
235     */
236    public Object randomKey(RNG random)
237    {
238        if(keyCount > 0)
239            return keys[random.nextIntHasty(keyCount)].randomKey(random);
240        return null;
241    }
242
243    /**
244     * In the keyset specified by {@code which}, changes an existing key, {@code past}, to another key, {@code future},
245     * if past exists in that keyset and future does not yet exist in that keyset. This will retain past's point in the
246     * ordering for future, so the associated other key(s) will still be associated in the same way.
247     * @param which which keyset to alter the items in
248     * @param past a key, that must exist in the keyset specified by which, and will be changed
249     * @param future a key, that cannot currently exist in the keyset specified by which, but will if this succeeds
250     * @return this for chaining
251     */
252    public MultiKey alter(int which, Object past, Object future)
253    {
254        if(which >= 0 && which < keyCount && keys[which].containsKey(past) && !keys[which].containsKey(future))
255            keys[which].alter(past, future);
256        return this;
257    }
258    /**
259     * In the keyset specified by {@code which}, changes the key at {@code index} to another key, {@code future}, if
260     * index is valid and future does not yet exist in that keyset. The position in the ordering for future will be the
261     * same as index, and the same as the key this replaced, if this succeeds, so the other key(s) at that position will
262     * still be associated in the same way.
263     * @param which which keyset to alter the items in
264     * @param index a position in the ordering to change; must be at least 0 and less than {@link #size()}
265     * @param future a key, that cannot currently exist in the keyset specified by which, but will if this succeeds
266     * @return this for chaining
267     */
268    public MultiKey alterAt(int which, int index, Object future)
269    {
270        if(which >= 0 && which < keyCount && !keys[which].containsKey(future) && index >= 0 && index < keys[which].size)
271            keys[which].alter(keys[which].keyAt(index), future);
272        return this;
273    }
274
275    /**
276     * Adds a key to each keyset at the same point in the ordering (the end) of this MultiKey. The length of k must
277     * match the keyCount of this MultiKey, and the nth item in k will go into the nth keyset. No item in k can be
278     * present in the matching keyset in this before this is called. If you want to change or update an existing key,
279     * use {@link #alter(int, Object, Object)} or {@link #alterAt(int, int, Object)}.
280     * @param k an array or varargs of keys to add after the last index of this MultiKey; length must equal keyCount
281     * @return true if this collection changed as a result of this call
282     */
283    public boolean put(Object... k)
284    {
285        if(k != null && keyCount > 0 && k.length == keyCount) {
286            for (int i = 0; i < keyCount; i++) {
287                if(keys[i].containsKey(k[i]))
288                    return false;
289            }
290            for (int i = 0; i < keyCount; i++) {
291                keys[i].add(k[i]);
292            }
293            return true;
294        }
295        return false;
296    }
297
298    /**
299     * Goes through all Iterable items in {@code k} and adds their unique items into their corresponding keyset at the
300     * end. If an item from the nth Iterable is already present in the nth keyset in this when this would add one, this
301     * will not put any keys at that point in the iteration order, and will place the next unique group of items it
302     * finds in the arguments at that position instead.
303     * @param k a Collection of Iterable s of keys to add to their respective keysets; should all be unique (like a Set)
304     * @return true if this collection changed as a result of this call
305     */
306    public boolean putAll(Collection<Iterable> k)
307    {
308        if(k == null || k.size() != keyCount) return false;
309        boolean changed = false;
310        Iterator[] its = new Iterator[keyCount];
311        int idx = 0;
312        for (Iterable it : k) {
313            its[idx++] = it.iterator();
314        }
315        Object[] os = new Object[keyCount];
316        while (true)
317        {
318            for (int i = 0; i < keyCount; i++) {
319                if(!its[i].hasNext())
320                    return changed;
321                os[i] = its[i].next();
322            }
323            changed = put(os) || changed;
324        }
325    }
326    /**
327     * Puts all unique keys in {@code other} into this MultiKey, respecting other's ordering. If a key in other is
328     * already present when this would add one, this will not put the keys at that point in the iteration
329     * order, and will place the next unique items it finds in other at that position instead.
330     * @param other another MultiKey collection with the same keyCount
331     * @return true if this collection changed as a result of this call
332     */
333    public boolean putAll(MultiKey other)
334    {
335        if(other == null || other.keyCount != keyCount) return false;
336        boolean changed = false;
337        int sz = other.size();
338        Object[] os = new Object[keyCount];
339        for (int i = 0; i < sz; i++) {
340            changed = put(other.getAllAt(i, os)) || changed;
341        }
342        return changed;
343    }

345    /**
346     * Adds a key to each keyset at the given index in the ordering of this MultiKey. The length of k must
347     * match the keyCount of this MultiKey, and the nth item in k will go into the nth keyset. No item in k can be
348     * present in the matching keyset in this before this is called. If you want to change or update an existing key,
349     * use {@link #alter(int, Object, Object)} or {@link #alterAt(int, int, Object)}.
350     * @param k an array or varargs of keys to add after the last index of this MultiKey; length must equal keyCount
351     * @return true if this collection changed as a result of this call
352     */
353    public boolean putAt(int index, Object... k)
354    {
355        if(k != null && keyCount > 0 && k.length == keyCount) {
356            for (int i = 0; i < keyCount; i++) {
357                if(keys[i].containsKey(k[i]))
358                    return false;
359            }
360            for (int i = 0; i < keyCount; i++) {
361                keys[i].addAt(index, k[i]);
362            }
363            return true;
364        }
365        return false;
366    }
367
368    /**
369     * Removes a given Object key from the keyset specified by which, if {@code removing} exists in that keyset, and
370     * also removes any keys associated with its point in the ordering.
371     * @param which which keyset to remove the item from; if {@code removing} isn't in that keyset, this does nothing
372     * @param removing the key to remove
373     * @return this for chaining
374     */
375    public MultiKey remove(int which, Object removing)
376    {
377        if(which >= 0 && which < keyCount) {
378            int i = keys[which].removeInt(removing);
379            if(i >= 0) {
380                for (int j = 0; j < keyCount; j++) {
381                    if(j != which)
382                        keys[j].removeAt(i);
383                }
384            }
385        }
386        return this;
387    }
388
389    /**
390     * Removes a given point in the ordering, if {@code index} is at least 0 and less than {@link #size()}.
391     * @param index the position in the ordering to remove
392     * @return this for chaining
393     */
394    public MultiKey removeAt(int index)
395    {
396        if(index >= 0 && keyCount > 0 && index < keys[0].size) {
397            for (int i = 0; i < keyCount; i++) {
398                keys[i].removeAt(index);
399            }
400        }
401        return this;
402    }

404    /**
405     * Reorders this MultiKey using {@code ordering}, which have the same length as this MultiKey's {@link #size()}
406     * and can be generated with {@link ArrayTools#range(int)} (which, if applied, would produce no
407     * change to the current ordering), {@link RNG#randomOrdering(int)} (which gives a random ordering, and if
408     * applied immediately would be the same as calling {@link #shuffle(RNG)}), or made in some other way. If you
409     * already have an ordering and want to make a different ordering that can undo the change, you can use
410     * {@link ArrayTools#invertOrdering(int[])} called on the original ordering.
411     * @param ordering an int array or vararg that should contain each int from 0 to {@link #size()} (or less)
412     * @return this for chaining
413     */
414    public MultiKey reorder(int... ordering)
415    {
416        if(ordering != null) {
417            for (int i = 0; i < keyCount; i++) {
418                keys[i].reorder(ordering);
419            }
420        }
421        return this;
422    }
423
424    /**
425     * Generates a random ordering with rng and applies the same ordering to all kinds of keys this has; they will
426     * maintain their current association to other keys but their ordering/indices will change.
427     * @param rng an RNG to produce the random ordering this will use
428     * @return this for chaining
429     */
430    public MultiKey shuffle(RNG rng)
431    {
432        if(keyCount > 0) {
433            int[] ordering = rng.randomOrdering(keys[0].size);
434            for (int i = 0; i < keyCount; i++) {
435                keys[i].reorder(ordering);
436            }
437        }
438        return this;
439    }

441    /**
442     * Creates a new iterator over the keys this holds in the keyset specified by which. This can be problematic for
443     * garbage collection if called very frequently; it may be better to access items by index (which also lets you
444     * access other keys associated with that index) using {@link #getAt(int, int)} in a for(int i=0...) loop.
445     * @return a newly-created iterator over this MultiKey's keys in the specified keyset
446     */
447    public Iterator iterator(int which)
448    {
449        if(which >= 0 && which < keyCount)
450            return keys[which].iterator();
451        return null;
452    }

454    /**
455     * Gets and caches the keys in the keyset specified by which as a Collection that implements SortedSet (and so also
456     * implements Set).
457     * @param which which keyset to get as a separate value
458     * @return the keys from the keyset specified by which, as a SortedSet
459     */
460    public SortedSet getSet(int which) {
461        if(which >= 0 && which < keyCount)
462            return keys[which].keySet();
463        return null;
464    }
465
466    /**
467     * To be called sparingly, since this allocates a new OrderedSet instead of reusing one.
468     * @param which which keyset to get as a separate value
469     * @return the keys from the keyset specified by which, as an OrderedSet
470     */
471    public OrderedSet getOrderedSet(int which) {
472        if(which >= 0 && which < keyCount)
473            return keys[which].keysAsOrderedSet();
474        return null;
475
476    }
477
478    public int keyCount() {
479        return keyCount;
480    }
481    public int valueCount() {
482        return 0;
483    }
484
485    public int size() {
486        return (keyCount > 0) ? keys[0].size : 0;
487    }
488
489    public boolean isEmpty()
490    {
491        return keyCount > 0 && keys[0].size > 0;
492    }
