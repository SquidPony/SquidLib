025package squidpony.squidmath;
026
027import squidpony.StringKit;
028
029import java.io.Serializable;
030
031/**
032 * This is a RandomnessSource in the PCG-Random family. It performs pseudo-
033 * random modifications to the output based on the techniques from the
034 * Permuted Congruential Generators created by M.E. O'Neill.
035 * Specifically, this variant is:
036 * RXS M XS -- random xorshift, mcg multiply, fixed xorshift
037 *
038 * The most statistically powerful generator, but all those steps
039 * make it slower than some of the others.
040 * <br>
041 * Even though benchmarks on similar programs in C would lead you to
042 * believe this should be somewhat faster than LightRNG, benchmarking
043 * with JMH seems to show LightRNG being roughly 16% faster than
044 * PermutedRNG, and both drastically faster than java.util.Random .
045 * @author Melissa E. O'Neill (Go HMC!)
046 * @author Tommy Ettinger
047 * @see PintRNG PintRNG is similar to this algorithm but uses only 32-bit math, where possible; meant for GWT.
048 */
049public class PermutedRNG implements RandomnessSource, StatefulRandomness, Serializable
050{
051        /** 2 raised to the 53, - 1. */
052    private static final long DOUBLE_MASK = ( 1L << 53 ) - 1;
053    /** 2 raised to the -53. */
054    private static final double NORM_53 = 1. / ( 1L << 53 );
055    /** 2 raised to the 24, -1. */
056    private static final long FLOAT_MASK = ( 1L << 24 ) - 1;
057    /** 2 raised to the -24. */
058    private static final double NORM_24 = 1. / ( 1L << 24 );
059    /**
060     * The state can be seeded with any value.
061     */
062    public long state;
063
064        private static final long serialVersionUID = 3637443966125527620L;
065
066    /** Creates a new generator seeded using Math.random. */
067    public PermutedRNG() {
068        this((long)Math.floor(Math.random() * Long.MAX_VALUE));
069    }
070
071    public PermutedRNG(final long seed) {
072        state = (seed + 1442695040888963407L) * 6364136223846793005L + 1442695040888963407L;
073    }
074
075    @Override
076    public int next( final int bits ) {
077        return (int)( nextLong() & ( 1L << bits ) - 1 );
078    }
079
080    /**
081     * Can return any int, positive or negative, of any size permissible in a 32-bit signed integer.
082     * Calls nextLong() exactly one time.
083     * @return any int, all 32 bits are random
084     */
085    public int nextInt() {
086        return (int)nextLong();
087    }
088    /**
089     * Can return any long, positive or negative, of any size permissible in a 64-bit signed integer.
090     *
091     * @return any long, all 64 bits are random
092     */
093    @Override
094    public long nextLong()
095    {
096        // increment  = 1442695040888963407L;
097        // multiplier = 6364136223846793005L;
098
099        long p = (state += 0x9E3779B97F4A7C15L);
100        p ^= p >>> (5 + (p >>> 59));
101        //state = state * 0x5851F42D4C957F2DL + 0x14057B7EF767814FL;
102        return ((p *= 0xAEF17502108EF2D9L) >>> 43) ^ p;
103    }
104
105    /**
106     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
107     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just needs to
108     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
109     *
110     * @return a copy of this RandomnessSource
111     */
112    @Override
113    public RandomnessSource copy() {
114        PermutedRNG next = new PermutedRNG(state);
115        next.setState(state);
116        return next;
117    }
118
119    /**
120     * Exclusive on the upper bound n.  The lower bound is 0.
121     * Will call nextLong() with no arguments at least 1 time, possibly more.
122     * @param bound the upper bound; should be positive
123     * @return a random int less than n and at least equal to 0
124     */
125    public int nextInt( final int bound ) {
126        if (bound <= 0) return 0;
127        return (int)((bound * (nextLong() & 0x7FFFFFFFL)) >> 31);
128    }
129
130    /**
131     * Inclusive lower, exclusive upper.
132     * Will call nextLong() with no arguments at least 1 time, possibly more.
133     * @param lower the lower bound, inclusive, can be positive or negative
134     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
135     * @return a random int at least equal to lower and less than upper
136     */
137    public int nextInt( final int lower, final int upper ) {
138        if ( upper - lower <= 0 ) throw new IllegalArgumentException();
139        return lower + nextInt(upper - lower);
140    }
141
142    /**
143     * Exclusive on the upper bound n. The lower bound is 0.
144     *
145     * Will call nextLong() with no arguments at least 1 time, possibly more.
146     * @param n the upper bound; should be positive
147     * @return a random long less than n
148     */
149    public long nextLong( final long n ) {
150        if ( n <= 0 ) return 0;
151        long threshold = (0x7fffffffffffffffL - n + 1) % n;
152        for (;;) {
153            long bits = nextLong() & 0x7fffffffffffffffL;
154            if (bits >= threshold)
155                return bits % n;
156        }
157    }
158
159    /**
160     * Exclusive on the upper bound n. The lower bound is 0.
161     *
162     * Will call nextLong() at least 1 time, possibly more.
163     * @param lower the lower bound, inclusive, can be positive or negative
164     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
165     * @return a random long at least equal to lower and less than upper
166     */
167    public long nextLong( final long lower, final long upper ) {
168        if ( upper - lower <= 0 ) return 0;
169        return lower + nextLong(upper - lower);
170    }
171
172    /**
173     * Gets a uniform random double in the range [0.0,1.0)
174     *
175     * Calls nextLong() exactly one time.
176     *
177     * @return a random double at least equal to 0.0 and less than 1.0
178     */
179    public double nextDouble() {
180        return ( nextLong() & DOUBLE_MASK ) * NORM_53;
181    }
182
183    /**
184     * Gets a uniform random double in the range [0.0,outer) given a positive parameter outer. If outer
185     * is negative, it will be the (exclusive) lower bound and 0.0 will be the (inclusive) upper bound.
186     *
187     * Calls nextLong() exactly one time.
188     *
189     *  @param outer the exclusive outer bound, can be negative
190     * @return a random double between 0.0 (inclusive) and outer (exclusive)
191     */
192    public double nextDouble(final double outer) {
193        return nextDouble() * outer;
194    }
195
196    /**
197     * Gets a uniform random float in the range [0.0,1.0)
198     *
199     * Calls nextLong() exactly one time.
200     *
201     * @return a random float at least equal to 0.0f and less than 1.0f
202     */
203    public float nextFloat() {
204        return (float)( ( nextLong() & FLOAT_MASK ) * NORM_24 );
205    }
206
207    /**
208     * Gets a random value, true or false.
209     * Calls nextLong() once.
210     * @return a random true or false value.
211     */
212    public boolean nextBoolean() {
213        return nextLong() < 0L;
214    }
215
216    /**
217     * Given a byte array as a parameter, this will fill the array with random bytes (modifying it
218     * in-place). Calls nextLong() {@code Math.ceil(bytes.length / 8.0)} times.
219     * @param bytes a byte array that will have its contents overwritten with random bytes.
220     */
221    public void nextBytes( final byte[] bytes ) {
222        int i = bytes.length, n = 0;
223        while( i != 0 ) {
224            n = Math.min(i, 8 );
225            for ( long bits = nextLong(); n-- != 0; bits >>>= 8 ) bytes[ --i ] = (byte)bits;
226        }
227    }

230    /**
231     * Sets the seed of this generator (which is also the current state).
232     * @param seed the seed to use for this PermutedRNG, as if it was constructed with this seed.
233     */
234    public void setSeed( final long seed ) {
235        state = seed;
236    }
237    /**
238     * Sets the seed (also the current state) of this generator.
239     * @param seed the seed to use for this PermutedRNG, as if it was constructed with this seed.
240     */
241    @Override
242        public void setState( final long seed ) {
243        state = seed;
244    }
245    /**
246     * Gets the current state of this generator.
247     * @return the current seed of this PermutedRNG, changed once per call to nextLong()
248     */
249    @Override
250        public long getState( ) {
251        return state;
252    }
253
254    /**
255     * Advances or rolls back the PermutedRNG's state without actually generating each number. Skip forward
256     * or backward a number of steps specified by advance, where a step is equal to one call to nextLong(),
257     * and returns the random number produced at that step (you can get the state with {@link #getState()}).
258     * @param advance Number of future generations to skip past. Can be negative to backtrack.
259     * @return the number that would be generated after generating advance random numbers.
260     */
261    public long skip(final long advance)
262    {
263        // The method used here is based on Brown, "Random Number Generation
264        // with Arbitrary Stride,", Transactions of the American Nuclear
265        // Society (Nov. 1994).  The algorithm is very similar to fast
266        // exponentiation.
267        //
268        // Even though advance is a signed long, it is treated as unsigned, effectively, for the purposes
269        // of how many iterations it goes through (at most 63 for forwards, 64 for "backwards").
270//        if(advance == 0)
271//            return state;
272//        long acc_mult = 1, acc_plus = 0, cur_mult = 6364136223846793005L, cur_plus = 1442695040888963407L;
273//
274//        do {
275//            if ((advance & 1L) != 0L) {
276//                acc_mult *= cur_mult;
277//                acc_plus = acc_plus*cur_mult + cur_plus;
278//            }
279//            cur_plus *= (cur_mult+1L);
280//            cur_mult *= cur_mult;
281//            advance >>>= 1;
282//        }while (advance > 0L);
283//        return acc_mult * state + acc_plus;
284        long p = (state += 0x9E3779B97F4A7C15L * advance);
285        p ^= p >>> (5 + (p >>> 59));
286        //state = state * 0x5851F42D4C957F2DL + 0x14057B7EF767814FL;
287        return ((p *= 0xAEF17502108EF2D9L) >>> 43) ^ p;
288    }
289
290    @Override
291    public String toString() {
292        return "PermutedRNG with state 0x" + StringKit.hex(state) + 'L';
293    }
294
295    public static long determine(long state)
296    {
297        state ^= state >>> (5 + (state >>> 59));
298        return ((state *= 0xAEF17502108EF2D9L) >>> 43) ^ state;
299    }
300
301    public static int determineBounded(long state, final int bound)
302    {
303        state ^= state >>> (5 + (state >>> 59));
304        return (int)((bound * ((((state *= 0xAEF17502108EF2D9L) >>> 43) ^ state) & 0x7FFFFFFFL)) >>> 31);
305    }
