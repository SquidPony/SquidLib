054/**
055 * Experiments with noise functions that use an alternate "honeycomb" or tiling pattern of cells in higher dimensions,
056 * but defaults to the simplex honeycomb used by SeededNoise if this can't make an attempt to change it. Currently, only
057 * 4D noise methods, such as {@link #noise(float, float, float, float, int)}, use a diferent honeycomb, which turns out
058 * to produce a very grid-aligned noise when used with {@link Noise#seamless2D(double[][], int, int, Noise.Noise4D)}.
059 * This might seem like a problem, but seamless noise normally has a hard time producing grid patterns intentionally, so
060 * if you want a nice circuit-board pattern or something similarly grid-heavy... this might help? The "mead" in the name
061 * is a reference to the beverage made from honey, and the extensive work my brain had to do to understand 4D honeycomb
062 * patterns. It seems that the grid is a result of using a honeycomb that can be formed from alternating hypercubes,
063 * where the hypercubes contribute square faces when projected in some ways.
064 * <br>
065 * Preview in black and white, being fed to seamless2D: http://i.imgur.com/0UBWDVW.png
066 * Preview in color, also being fed to seamless2D but with one call per color channel: http://i.imgur.com/46DI3uz.png
067 * <br>
068 * All functions can take an int seed that should significantly change the pattern of noise produced. Based on code from
069 * Joise; the full library is available at https://github.com/SudoPlayGames/Joise . Joise is derived from the Accidental
070 * Noise Library, available in C++ at http://accidentalnoise.sourceforge.net/index.html .
071 */

161     * Possibly useful outside SeededNoise. A customized 5-input hash that mixes around all inputs fairly well, and
162     * produces 8 bits.
163     * @param x an int to incorporate into the hash
164     * @param y an int to incorporate into the hash
165     * @param z an int to incorporate into the hash
166     * @param w an int to incorporate into the hash

168     * @return a pseudo-random-like int between 0 and 255, inclusive on both

170    //0x89 0x95 0xA3 0xB3 0xC5 0xD3 0xE3
171    public static int hash(final int x, final int y, final int z, final int w, final int seed) {
172        return (
173                z    + 0xD3 * (seed + y)
174              ^ w    + 0xB5 * (x + z   )
175              ^ seed + 0xC1 * (y + w   )
176              ^ x    + 0x95 * (z + seed)
177              ^ y    + 0xA3 * (w + x   )) & 255;
178    }

181     * @param x an int to incorporate into the hash
182     * @param y an int to incorporate into the hash
183     * @param z an int to incorporate into the hash

188    public static int hashAlt(final int x, final int y, final int z, final int w, final int seed) {
189        int a = 0x632BE5AB, result = (0x9E3779B9
190                + (a ^= 0x85157AF5 * seed + x)
191                + (a ^= 0x85157AF5 * x + y)
192                + (a ^= 0x85157AF5 * y + z)
193                + (a ^= 0x85157AF5 * z + w)
194                + (a ^= 0x85157AF5 * w + seed));
195        return (result * (a | 1) ^ (result >>> 11 | result << 21));
196    }

218              ^ y    + 0xB5 * (v + seed)) & 255;

305    public static double noise(final float x, final float y, final float z, final float w, final int seed) {
306        double n = 0.0;
307        //final float s = (x + y + z + w) * F4;
308        final int seed2 = (seed >>> 21) ^ (seed & 0x7ff) - (seed >>> 11 & 0x3ff),
309                i = fastFloor(x * 2.430f + seed2),
310                j = fastFloor(y * 2.430f + seed2),
311                k = fastFloor(z * 2.430f + seed2),
312                l = fastFloor(w * 2.430f + seed2);
313        final float
314                x0 = x * 2.430f + seed2 - i, x05 = x0 - 0.5f,
315                y0 = y * 2.430f + seed2 - j, y05 = y0 - 0.5f,
316                z0 = z * 2.430f + seed2 - k, z05 = z0 - 0.5f,
317                w0 = w * 2.430f + seed2 - l, w05 = w0 - 0.5f;
318        final float dist = x05 * x05 + y05 * y05 + z05 * z05 + w05 * w05,
319        xc, yc, zc, wc;
320        final boolean isCentral = (dist <= 0.35);
321        final int xx, yy, zz, ww;
322        if(isCentral)
323        {
324            xx = i * 2;
325            yy = j * 2;
326            zz = k * 2;
327            ww = l * 2;
328            xc = x0;
329            yc = y0;
330            zc = z0;
331            wc = w0;
332        }
333        else
334        {
335            xx = i * 2 + (x0 < 0.5f ? -1 : 1);
336            yy = j * 2 + (y0 < 0.5f ? -1 : 1);
337            zz = k * 2 + (z0 < 0.5f ? -1 : 1);
338            ww = l * 2 + (w0 < 0.5f ? -1 : 1);
339            xc = (x0 < 0.5f ? 0.5f + x0 : x0 - 0.5f);
340            yc = (y0 < 0.5f ? 0.5f + y0 : y0 - 0.5f);
341            zc = (z0 < 0.5f ? 0.5f + z0 : z0 - 0.5f);
342            wc = (w0 < 0.5f ? 0.5f + w0 : w0 - 0.5f);
343        }
344        /*
345        final int c = (x0 > y0 ? 128 : 0) | (x0 > z0 ? 64 : 0) | (y0 > z0 ? 32 : 0) | (x0 > w0 ? 16 : 0) | (y0 > w0 ? 8 : 0) | (z0 > w0 ? 4 : 0);
346        final int i1 = SIMPLEX[c] >>> 2,
347                j1 = SIMPLEX[c | 1] >>> 2,
348                k1 = SIMPLEX[c | 2] >>> 2,
349                l1 = SIMPLEX[c | 3] >>> 2,
350                i2 = SIMPLEX[c] >>> 1 & 1,
351                j2 = SIMPLEX[c | 1] >>> 1 & 1,
352                k2 = SIMPLEX[c | 2] >>> 1 & 1,

354                i3 = SIMPLEX[c] & 1,

356                k3 = SIMPLEX[c | 2] & 1,
357                l3 = SIMPLEX[c | 3] & 1;
358
359        final float x1 = x0 - i1 + G4,
360                y1 = y0 - j1 + G4,
361                z1 = z0 - k1 + G4,
362                w1 = w0 - l1 + G4,
363                x2 = x0 - i2 + 2f * G4,
364                y2 = y0 - j2 + 2f * G4,
365                z2 = z0 - k2 + 2f * G4,
366                w2 = w0 - l2 + 2f * G4,
367                x3 = x0 - i3 + 3f * G4,
368                y3 = y0 - j3 + 3f * G4,
369                z3 = z0 - k3 + 3f * G4,
370                w3 = w0 - l3 + 3f * G4,
371                x4 = x0 - 1f + 4f * G4,
372                y4 = y0 - 1f + 4f * G4,
373                z4 = z0 - 1f + 4f * G4,
374                w4 = w0 - 1f + 4f * G4;*/
375        int h;
376        float t;
377        h = hashAlt(xx+1, yy, zz, ww, seed);
378        t = xc - 0.4f;
379        if (t >= 0f)
380        {
381            t *= t;
382            n += t * t * ((x05+1) * randomSignedFloat(h += 0x9E3779B9) + (y05) * randomSignedFloat(h += 0x9E3779B9) + (z05) * randomSignedFloat(h += 0x9E3779B9) + (w05) * randomSignedFloat(h + 0x9E3779B9));
383        }
384        h = hashAlt(xx-1, yy, zz, ww, seed);
385        t = 0.6f - xc;
386        if (t >= 0f)
387        {
388            t *= t;
389            n += t * t * ((x05-1) * randomSignedFloat(h += 0x9E3779B9) + (y05) * randomSignedFloat(h += 0x9E3779B9) + (z05) * randomSignedFloat(h += 0x9E3779B9) + (w05) * randomSignedFloat(h + 0x9E3779B9));

404        }

411        }
412        h = hashAlt(xx, yy, zz-1, ww, seed);
413        t = 0.6f - zc;
414        if (t >= 0f)
415        {
416            t *= t;
417            n += t * t * ((x05) * randomSignedFloat(h += 0x9E3779B9) + (y05) * randomSignedFloat(h += 0x9E3779B9) + (z05-1) * randomSignedFloat(h += 0x9E3779B9) + (w05) * randomSignedFloat(h + 0x9E3779B9));
418        }
419        h = hashAlt(xx, yy, zz, ww+1, seed);
420        t = wc - 0.4f;
421        if (t >= 0f)
422        {
423            t *= t;
424            n += t * t * ((x05) * randomSignedFloat(h += 0x9E3779B9) + (y05) * randomSignedFloat(h += 0x9E3779B9) + (z05) * randomSignedFloat(h += 0x9E3779B9) + (w05+1) * randomSignedFloat(h + 0x9E3779B9));
425        }
426        h = hashAlt(xx, yy, zz, ww-1, seed);
427        t = 0.6f - wc;
428        if (t >= 0f)
429        {
430            t *= t;
431            n += t * t * ((x05) * randomSignedFloat(h += 0x9E3779B9) + (y05) * randomSignedFloat(h += 0x9E3779B9) + (z05) * randomSignedFloat(h += 0x9E3779B9) + (w05-1) * randomSignedFloat(h + 0x9E3779B9));
432        }
433        return 4 * n;
434    }
