001package squidpony.squidmath;
002
003import squidpony.StringKit;
004
005import java.io.Serializable;
006
007/**
008 * Similar to DashRNG (emphasizes speed over quality), but like PintRNG this only uses 32-bit int math (good for GWT).
009 * It's likely that the period of FlapRNG is a full 2 to the 32 (0 seed is allowed), but the distribution for 32-bit
010 * values is also likely to cover much less than the full range of all ints. The quality of the random numbers this
011 * produces is probably rather bad, but the speed is on par with ThunderRNG (our current best generator for longs) when
012 * this generates ints and ThunderRNG generates longs. FlapRNG generates ints at a higher speed than ThunderRNG
013 * generates ints, but because FlapRNG needs to generate and mix two ints to make a long, it takes over twice the time
014 * as ThunderRNG to generate longs. Long story short, FlapRNG is a good choice if you need a StatefulRandomness that
015 * uses int math, you don't need great quality, and you do want great speed. If you want a much-higher-quality int-math
016 * StatefulRandomness generator, use PintRNG.
017 * <br>
018 * Created by Tommy Ettinger on 5/1/2017.
019 */
020public class FlapRNG implements StatefulRandomness, Serializable {
021    private static final long serialVersionUID = 1L;
022    public FlapRNG(){
023        this((int)((Math.random() * 2.0 - 1.0) * 0x80000000));
024    }
025    public FlapRNG(final int seed) {
026        state = seed;
027    }
028    public FlapRNG(final long seed) {
029        state = (int)((seed ^ seed >>> 32) & 0xFFFFFFFFL);
030    }
031    public FlapRNG(final CharSequence seed)
032    {
033        state = CrossHash.Wisp.hash(seed);
034    }
035
036    public int state;
037    /**
038     * Get the current internal state of the StatefulRandomness as a long.
039     *
040     * @return the current internal state of this object.
041     */
042    @Override
043    public long getState() {
044        return state;
045    }
046
047    /**
048     * Set the current internal state of this StatefulRandomness with a long.
049     *
050     * @param state a 64-bit long, but this is always truncated when used; the upper 32 bits are discarded
051     */
052    @Override
053    public void setState(final long state) {
054        this.state = (int)(state & 0xFFFFFFFFL);
055    }
056
057    /**
058     * Using this method, any algorithm that might use the built-in Java Random
059     * can interface with this randomness source.
060     *
061     * @param bits the number of bits to be returned
062     * @return the integer containing the appropriate number of bits
063     */
064    @Override
065    public final int next( final int bits ) {
066        return nextInt() >>> (32 - bits);
067    }
068
069    public final int nextInt()
070    {
071        final int z = (state += 0x9E3779B9);
072        return (z >> (z >>> 28)) * 0xC6BC278D;
073    }
074    /**
075     * Using this method, any algorithm that needs to efficiently generate more
076     * than 32 bits of random data can interface with this randomness source.
077     * This particular implementation is not much more efficient than calling
078     * {@link #nextInt()} twice.
079     * <p>
080     * Get a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
081     *
082     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
083     */
084    @Override
085    public final long nextLong() {
086        int z = state + 0x9E3779B9;
087        final long j = (z >> (z >>> 28)) * 0xC6BC279692B5CC83L;
088        z = (state += 0x3C6EF372);
089        return j ^ ((z >> (z >>> 28)) * 0xC6BC279692B5CC83L) << 32;
090    }
091
092    /**
093     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
094     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
095     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
096     *
097     * @return a copy of this RandomnessSource
098     */
099    @Override
100    public RandomnessSource copy() {
101        return new FlapRNG(state);
102    }
103
104    /**
105     * A simple "output stage" applied to state, this method does not update state on its own. If you expect to call
106     * this method more than once, you should add a large int that is an odd number to state as part of the call, such
107     * as with {@code FlapRNG.determine(state += 0x9E3779B9)}. Here, "large long" means at least two of the
108     * upper 5 bits should be set to be safe. The golden-ratio-derived constant 0x9E3779B9 should be fine.
109     * This method doesn't offer particularly good quality assurances, but should be very fast.
110     * @param state should be changed when you call this by adding some large odd number, e.g. 0x9E3779B9
111     * @return an altered version of state that should be very fast to compute but doesn't promise great quality
112     */
113    public static int determine(final int state)
114    {
115        return (state >> (state >>> 28)) * 0xC6BC278D;
116    }
117    /**
118     * Gets a pseudo-random float between 0f (inclusive) and 1f (exclusive) using the given state. Calling this
119     * repeatedly won't have correlation problems if you give it sequential ints.
120     * @param state any int
121     * @return a pseudo-random float from -0f (inclusive) to 1f (exclusive)
122     */
123    public static float randomFloat(int state)
124    {
125        return NumberTools.intBitsToFloat(((((state *= 0x9E3779B9) >> (state >>> 28)) * 0xC6BC278D) >>> 9) | 0x3f800000) - 1f;
126    }
127
128    /**
129     * Gets a pseudo-random float between -1f (inclusive) and 1f (exclusive) using the given state. Calling this
130     * repeatedly won't have correlation problems if you give it sequential ints.
131     * @param state any int
132     * @return a pseudo-random float from -1f (inclusive) to 1f (exclusive)
133     */
134    public static float randomSignedFloat(int state)
135    {
136        return NumberTools.intBitsToFloat(((((state *= 0x9E3779B9) >> (state >>> 28)) * 0xC6BC278D) >>> 9) | 0x40000000) - 3f;
137    }
138
139    @Override
140    public String toString() {
141        return "FlapRNG with state 0x" + StringKit.hex(state);
142    }
143
144    @Override
145    public int hashCode() {
146        return 0x632BE5AB * state;
147    }
148
149    @Override
150    public boolean equals(Object o) {
151        if (this == o) return true;
152        if (o == null || getClass() != o.getClass()) return false;
153
154        FlapRNG flapRNG = (FlapRNG) o;
155
156        return state == flapRNG.state;
157    }
158
159}
