003import squidpony.StringKit;
004import squidpony.annotation.Beta;

006import java.io.Serializable;

008/**
009 * Similar to XoRoRNG, but like PintRNG this only uses 32-bit int math (good for GWT).
010 * It's likely that the period of FlapRNG is a full 2 to the 64 (0 seed is allowed), since this uses a pair of ints for
011 * its state, and acts like XoRoRNG modified to handle all initial states (hopefully, at least... there may be one seed
012 * pair that corresponds to the original "bad seed" of 0) and, of course, use only int math where possible. This is
013 * slightly faster than XoRoRNG at producing ints, and as you would expect, twice as slow when generating longs.
014 * <br>
015 * Created by Tommy Ettinger on 5/1/2017.
016 */
017@Beta
018public class FlapRNG implements StatefulRandomness, Serializable {
019    private static final long serialVersionUID = 1L;
020    public FlapRNG(){
021        this((int)((Math.random() * 2.0 - 1.0) * 0x80000000));
022    }
023    public FlapRNG(final int seed) {
024        state0 = seed;
025        state1 = seed ^ seed >>> (4 + (seed >>> 28));
026        state1 *= 277803737;
027        state1 ^= (state1 >>> 22);
028    }
029    public FlapRNG(final int seed0, final int seed1) {
030        state0 = seed0;
031        state1 = seed1;
032    }
033    public FlapRNG(final long seed) {
034        state0 = (int)(seed & 0xFFFFFFFFL);
035        state1 = (int)(seed >>> 32);
036    }
037    public FlapRNG(final CharSequence seed)
038    {
039        this(CrossHash.Wisp.hash64(seed));
040    }
041
042    public int state0, state1;
043    /**
044     * Get the current internal state of the StatefulRandomness as a long.
045     *
046     * @return the current internal state of this object.
047     */
048    @Override
049    public long getState() {
050        return (long)(state1) << 32 | (state0 & 0xFFFFFFFFL);
051    }
052
053    /**
054     * Set the current internal state of this StatefulRandomness with a long.
055     *
056     * @param state a 64-bit long, but this is always truncated when used; the upper 32 bits are discarded
057     */
058    @Override
059    public void setState(final long state) {
060        state0 = (int)(state & 0xFFFFFFFFL);
061        state1 = (int)(state >>> 32);
062    }
063
064    /**
065     * Using this method, any algorithm that might use the built-in Java Random
066     * can interface with this randomness source.
067     *
068     * @param bits the number of bits to be returned
069     * @return the integer containing the appropriate number of bits
070     */
071    @Override
072    public final int next( final int bits ) {
073        return nextInt() >>> (32 - bits);
074    }
075
076    public final int nextInt()
077    {
078        final int s0 = state0;
079        int s1 = state1;
080        final int result = s0 + s1;
081        s1 ^= s0 + 0x9E3779B9;
082        state0 = (s0 << 26 | s0 >>> 6) ^ s1 ^ (s1 << 7); // a, b
083        state1 = (s1 << 18 | s1 >>> 14); // c
084        return result;
085
086        //0x632BE5AB
087        //final int z = (state += 0x9E3779B9);
088        //return (z >> (z & 15)) * 0xC6BC278D;
089        //final int z = (state + 0x9E3779B9);
090        //return state += ((state + 0x9E3779B9) >> 5) * 0xC6BC278D;
091        //final int z = (state += 0x9E3779B9), r = (z & 15);
092        //return (z >> r) * 0xC6BC278D;
093        //return state += (state >> 5) * 0xC6BC279692B5CC83L + 0x9E3779B97F4A7C15L;
094
095    }
096    /**
097     * Using this method, any algorithm that needs to efficiently generate more
098     * than 32 bits of random data can interface with this randomness source.
099     * This implementation produces a different result than calling {@link #nextInt()} twice and shifting the bits
100     * to make a long from the two ints. It really just multiplies each int of the state by a different large prime
101     * before adding them together (the adding is just like in nextInt(), but the multiplying is just here).
102     * The end result is a long that should take only slightly longer to produce than an int, from a primarily-int
103     * generator! Hooray.
104     * <p>
105     * Get a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive).
106     *
107     * @return a random long between Long.MIN_VALUE and Long.MAX_VALUE (both inclusive)
108     */
109    @Override
110    public final long nextLong() {
111        final int s0 = state0;
112        int s1 = state1;
113        final long result = s0 * 0xD0E89D2D311E289FL + s1 * 0xC6BC279692B5CC83L;
114        s1 ^= s0 + 0x9E3779B9;
115        state0 = (s0 << 26 | s0 >>> 6) ^ s1 ^ (s1 << 7); // a, b
116        state1 = (s1 << 18 | s1 >>> 14); // c
117        return result;
118
119        /*
120        int z = state + 0x9E3779B9;
121        state += (z >> (z >>> 28)) * 0xC6BC279692B5CC83L;
122        z = (state + 0x9E3779B9);
123        return (state) ^ (long)(state += ((z >> (z >>> 28)) * 0xC6BC279692B5CC83L)) << 32;
124        */
125    }
126
127    /**
128     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
129     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
130     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
131     *
132     * @return a copy of this RandomnessSource
133     */
134    @Override
135    public RandomnessSource copy() {
136        return new FlapRNG(state0, state1);
137    }
138
139    /**
140     * A simple "output stage" applied to state, this method does not update state on its own. If you expect to call
141     * this method more than once, you should add a large int that is an odd number to state as part of the call, such
142     * as with {@code FlapRNG.determine(state += 0x9E3779B9)}. Here, "large long" means at least two of the
143     * upper 5 bits should be set to be safe. The golden-ratio-derived constant 0x9E3779B9 should be fine.
144     * This method doesn't offer particularly good quality assurances, but should be very fast.
145     * @param state should be changed when you call this by adding some large odd number, e.g. 0x9E3779B9
146     * @return an altered version of state that should be very fast to compute but doesn't promise great quality
147     */
148    public static int determine(final int state)
149    {
150        return (state >> (state >>> 28)) * 0xC6BC278D;
151    }
152    /**
153     * Gets a pseudo-random float between 0f (inclusive) and 1f (exclusive) using the given state. Calling this
154     * repeatedly won't have correlation problems if you give it sequential ints.
155     * @param state any int
156     * @return a pseudo-random float from -0f (inclusive) to 1f (exclusive)
157     */
158    public static float randomFloat(int state)
159    {
160        return NumberTools.intBitsToFloat(((((state *= 0x9E3779B9) >> (state >>> 28)) * 0xC6BC278D) >>> 9) | 0x3f800000) - 1f;
161    }
162
163    /**
164     * Gets a pseudo-random float between -1f (inclusive) and 1f (exclusive) using the given state. Calling this
165     * repeatedly won't have correlation problems if you give it sequential ints.
166     * @param state any int
167     * @return a pseudo-random float from -1f (inclusive) to 1f (exclusive)
168     */
169    public static float randomSignedFloat(int state)
170    {
171        return NumberTools.intBitsToFloat(((((state *= 0x9E3779B9) >> (state >>> 28)) * 0xC6BC278D) >>> 9) | 0x40000000) - 3f;
172    }
173
174    @Override
175    public String toString() {
176        return "FlapRNG with state0 0x" + StringKit.hex(state0) + ", state1 0x" + StringKit.hex(state1);
177    }
178
179    @Override
180    public int hashCode() {
181        return 0x632BE5AB * state0 ^ state1;
182    }
183
184    @Override
185    public boolean equals(Object o) {
186        if (this == o) return true;
187        if (o == null || getClass() != o.getClass()) return false;
188
189        FlapRNG flapRNG = (FlapRNG) o;
190
191        if (state0 != flapRNG.state0) return false;
192        return state1 == flapRNG.state1;
193    }
194}
