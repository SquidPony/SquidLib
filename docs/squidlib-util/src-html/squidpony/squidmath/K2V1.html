009/**
010 * An ordered multi-directional map-like data structure with two sets of keys and one list of values.
011 *
012 * This is structured so that the two keys, of types A and B, are always associated with each other and the same value,
013 * of type Q, but may have their index in the ordering change. You can look up a B key with an A key or index, an A key
014 * with a B key or index, and a Q value with an A key, B key, or index.
015 * Created by Tommy Ettinger on 10/27/2016.
016 */
017public class K2V1<A, B, Q> {
018    K2<A, B> keys;
019    ArrayList<Q> values;
020
021    /**
022     * Constructs an empty K2V1 with the default parameters: 32 expected indices and a load factor of 0.5f.
023     */
024    public K2V1()
025    {
026        this(32, 0.5f);
027    }
028
029    /**
030     * Constructs an empty K2V1 that can hold {@code expected} indices before resizing and has a load factor of 0.5f.
031     * @param expected the expected number of items of any single type that this will hold; each put counts as one item
032     */
033    public K2V1(int expected)
034    {
035        this(expected, 0.5f);
036    }
037
038    /**
039     * Constructs an empty K2V1 that can hold {@code expected} indices before resizing and has load factor {@code f}.
040     * @param expected the expected number of items of any single type that this will hold; each put counts as one item
041     * @param f the load factor; must be greater than 0 and less than 1, but should ideally be between 0.2 and 0.8
042     */
043    public K2V1(int expected, float f)
044    {
045        keys = new K2<>(expected, f);
046        values = new ArrayList<>(expected);
047    }
048
049    /**
050     * Constructs a K2V1 from an A iterable, a B iterable, and a Q iterable, where the A and B items should be unique
051     * (if they aren't, each item that would be associated with a duplicate A or B will be skipped). The K2V1 this
052     * constructs will only take enough items from all Iterables to exhaust the shortest Iterable, so the lengths can be
053     * different between arguments. If there are no duplicate A keys or duplicate B keys (e.g. you can have an A key
054     * that is equal to a B key, but not to another A key), then all items will be used in the order the Iterable
055     * parameters provide them; otherwise the keys and value that would be associated with a duplicate are skipped.
056     * @param aKeys an Iterable of A keys; if null will be considered empty and this K2V1 will be empty
057     * @param bKeys an Iterable of B keys; if null will be considered empty and this K2V1 will be empty
058     * @param qValues an Iterable of Q values; if null will be considered empty and this K2V1 will be empty
059     */
060    public K2V1(Iterable<A> aKeys, Iterable<B> bKeys, Iterable<Q> qValues)
061    {
062        this(32, 0.5f);
063        putAll(aKeys, bKeys, qValues);
064    }
065
066    /**
067     * Constructs a K2V1 from another K2V1 with ths same A, B, and Q types. This will have an expected size equal to the
068     * current size of other, use the same load factor (f) as other, and will have the same items put into it in the
069     * same order.
070     * @param other a K2V1 to copy into this; must have the same A, B, and Q types
071     */
072    public K2V1(K2V1<A, B, Q> other)
073    {
074        this(other == null ? 32 : other.size(), other == null ? 0.5f : other.keys.keysA.f);
075        putAll(other);
076    }

314    /**
315     * Adds an A key, a B key, and a Q value at the same point in the ordering (the end) to this K2V1. Neither aKey nor
316     * bKey can be present in this collection before this is called. If you want to change or update an existing key,
317     * use {@link #alterA(Object, Object)} or {@link #alterB(Object, Object)}. The value, qValue, has no restrictions.
318     * @param aKey an A key to add; cannot already be present
319     * @param bKey a B key to add; cannot already be present
320     * @param qValue a Q value to add; can be present any number of times
321     * @return true if this collection changed as a result of this call
322     */
323
324    public boolean put(A aKey, B bKey, Q qValue)
325    {
326        if(keys.put(aKey, bKey))
327        {
328            values.add(qValue);
329            return true;
330        }
331        return false;
332    }

334    /**
335     * Adds an A key, a B key, and a Q value at the given index in the ordering to this K2V1. Neither a nor b can be
336     * present in this collection before this is called. If you want to change or update an existing key, use
337     * {@link #alterA(Object, Object)} or {@link #alterB(Object, Object)}. The value, q, has no restrictions. The index
338     * this is given should be at least 0 and no greater than {@link #size()}.
339     * @param index the point in the ordering to place a and b into; later entries will be shifted forward
340     * @param a an A key to add; cannot already be present
341     * @param b a B key to add; cannot already be present
342     * @param q a Q value to add; can be present any number of times
343     * @return true if this collection changed as a result of this call
344     */
345    public boolean putAt(int index, A a, B b, Q q)
346    {
347        if(keys.putAt(index, a, b))
348        {
349            values.add(index, q);
350            return true;
351        }
352        return false;
353    }
354    public boolean putAll(Iterable<A> aKeys, Iterable<B> bKeys, Iterable<Q> qValues)
355    {
356        if(aKeys == null || bKeys == null || qValues == null) return false;
357        Iterator<A> aIt = aKeys.iterator();
358        Iterator<B> bIt = bKeys.iterator();
359        Iterator<Q> qIt = qValues.iterator();
360        boolean changed = false;
361        while (aIt.hasNext() && bIt.hasNext() && qIt.hasNext())
362        {
363            changed = put(aIt.next(), bIt.next(), qIt.next()) || changed;
364        }
365        return changed;
366    }
367    /**
368     * Puts all unique A and B keys in {@code other} into this K2V1, respecting other's ordering. If an A or a B in other
369     * is already present when this would add one, this will not put the A and B keys at that point in the iteration
370     * order, and will place the next unique A and B it finds in the arguments at that position instead.
371     * @param other another K2V1 collection with the same A and B types
372     * @return true if this collection changed as a result of this call
373     */
374    public boolean putAll(K2V1<A, B, Q> other)
375    {
376        if(other == null) return false;
377        boolean changed = false;
378        int sz = other.size();
379        for (int i = 0; i < sz; i++) {
380            changed = put(other.getAAt(i), other.getBAt(i), other.getQAt(i)) || changed;
381        }
382        return changed;
383    }
384    /**
385     * Removes a given A key, if {@code removing} exists in this K2V1's A keys, and also removes any B key or Q value
386     * associated with its point in the ordering.
387     * @param removing the A key to remove
388     * @return this for chaining
389     */
390    public K2V1<A, B, Q> removeA(A removing)
391    {
392        int i = keys.keysA.removeInt(removing);
393        if(i >= 0) {
394            keys.keysB.removeAt(i);
395            values.remove(i);
396        }
397        return this;
398    }

405     */

429    /**
430     * Reorders this K2V1 using {@code ordering}, which have the same length as this K2V1's {@link #size()}
431     * and can be generated with {@link ArrayTools#range(int)} (which, if applied, would produce no
432     * change to the current ordering), {@link RNG#randomOrdering(int)} (which gives a random ordering, and if
433     * applied immediately would be the same as calling {@link #shuffle(RNG)}), or made in some other way. If you
434     * already have an ordering and want to make a different ordering that can undo the change, you can use
435     * {@link ArrayTools#invertOrdering(int[])} called on the original ordering.
436     * @param ordering an int array or vararg that should contain each int from 0 to {@link #size()} (or less)
437     * @return this for chaining
438     */
439    public K2V1<A, B, Q> reorder(int... ordering)
440    {
441        keys.reorder(ordering);
442        ArrayTools.reorder(values, ordering);
443        return this;
444    }

446    /**
447     * Generates a random ordering with rng and applies the same ordering to all keys and values this has; they will
448     * maintain their current association to other keys and values but their ordering/indices will change.
449     * @param rng an RNG to produce the random ordering this will use
450     * @return this for chaining
451     */
452    public K2V1<A, B, Q> shuffle(RNG rng)
453    {
454        int[] ordering = rng.randomOrdering(keys.keysA.size);
455        keys.reorder(ordering);
456        ArrayTools.reorder(values, ordering);
457        return this;
458    }
