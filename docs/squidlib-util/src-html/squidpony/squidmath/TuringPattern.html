003/**
004 * A technique for producing organic-seeming patterns via iterative processing of random values (reaction-diffusion).
005 * The fundamental basis for why this looks organic should be credited to Alan Turing, but the specific algorithm should
006 * be credited to <a href="http://www.jonathanmccabe.com/Cyclic_Symmetric_Multi-Scale_Turing_Patterns.pdf">Jonathan
007 * McCabe</a>, who has produced significant expansions on the scope the original Turing work covered; you can see
008 * <a href=https://www.flickr.com/photos/jonathanmccabe/sets>various examples of art he produced using this and related
009 * techniques</a>.
010 * <br>
011 * This class tries to provide a fairly raw API so different adjustments can be made on top of it.
012 * <br>
013 * Created by Tommy Ettinger on 4/27/2017.
014 */
015public class TuringPattern {
016    private static final double fraction = 1.0 / 0x80000000L;

024     */

026        return initializeInto(new double[width * height]);

029    /**
030     * Refills a substance array that can be given to other static methods. Uses very little 64-bit math,
031     * making it ideal for GWT.
032     * @param substance a 1D double array that will be modified and filled with random values
033     * @return substance, after modification; should be given to other methods
034     */
035    public static double[] initializeInto(double[] substance){
036        if(substance == null) return null;
037        int seed = 65537;
038        for (int i = 0; i < substance.length; i++) {
039            seed += 0x9E3779B9;
040            substance[i] = PintRNG.determine((seed ^ 0xD0E89D2D) >>> 19 | (seed ^ 0xD0E89D2D) << 13) * fraction;
041        }
042        return substance;
043    }

083    /**
084     * Initializes a substance array that can be given to other static methods. Uses an RNG to produce long values that
085     * this turns into doubles in the desired range (that is, -1.0 to 1.0).
086     * @param substance a 1D double array that will be modified and filled with random values
087     * @param rng the random number generator responsible for producing random double values
088     * @return a 1D double array that represents a 2D array with random contents; should be given to other methods
089     */
090    public static double[] initializeInto(double[] substance, RNG rng){
091        if(substance == null || rng == null) return null;
092        for (int i = 0; i < substance.length; i++) {
093            substance[i] = rng.nextDouble(2.0) - 1.0;
094        }
095        return substance;
096    }

110    /**
111     * Initializes a substance array that can be given to other static methods. Uses a Noise2D instance (it may or may
112     * not use the given seed) to produce values between -1.0 and 1.0.
113     * @param width the width of the substance array; should be consistent throughout calls
114     * @param height the height of the substance array; should be consistent throughout calls
115     * @param noise a Noise.Noise2D instance, such as {@link SeededNoise#instance}
116     * @param seed the seed to use with the noise generator
117     * @return a 1D double array that represents a 2D array with random contents; should be given to other methods
118     */
119    public static double[] initializeInto(double[] substance, int width, int height, Noise.Noise2D noise, int seed){
120        if(substance == null || noise == null) return null;
121        int i = 0;
122        for (int x = 0; x < width; x++) {
123            for (int y = 0; y < height; y++) {
124                substance[i++] = noise.getNoiseWithSeed(x, y, seed);
125            }
126        }
127        return substance;
128    }

130    /**
131     * Modifies the data parameter so no value in it is outside the range -1.0 inclusive to 1.0 exclusive. Makes no
132     * guarantees about the values this puts in data beyond that they will be inside that range. Has undefined results
133     * if any values in data are NaN or are infinite, though it probably will still work. If all values in data are
134     * already in the range, this will still change them, though probably not beyond recognition. Does not return a
135     * value because the changes are applied to data in-place.
136     * @param data a double array that will be modified in-place so all values in it will be between -1.0 and 1.0
137     */
138    public static void refit(final double[] data)
139    {
140        if(data == null) return;
141        for (int i = 0; i < data.length; i++) {
142            data[i] = NumberTools.bounce(data[i] + 5);
143        }
144    }

146    /**
147     * Given an offset information array that has been modified and should be returned to its unmodified state, this
148     * uses the given width, height, and radius (which should be the same as what this was originally constructed with)
149     * to modify offsets in-place as if it was freshly-made, even if the array is final.
150     * @param offsets an offset information array that should have been produced by {@link #offsetsCircle(int, int, double)} and may have been distorted
151     * @param width the width of the substance array; should be consistent throughout calls
152     * @param height the height of the substance array; should be consistent throughout calls
153     * @param radius the radius of the circle to
154     * @return an offset information array, as a jagged 2D int array, that can be passed to {@link #step(double[], int[][], double, int[][], double)}
155     */
156    public static int[][] offsetsCircleInto(final int[][] offsets, final int width, final int height, double radius) {
157        if (radius > width || radius > height) radius = Math.min(width, height);
158        radius = Math.max(1, radius);
159        IntVLA ivx = new IntVLA((int) (Math.PI * radius * (radius + 1))),
160                ivy = new IntVLA((int) (Math.PI * radius * (radius + 1)));
161        for (int x = 1; x <= radius; x++) {
162            for (int y = 1; y <= radius; y++) {
163                if (Math.sqrt(x * x + y * y) + 0.5 <= radius) {
164                    ivx.add(x);
165                    ivy.add(y);
166                }
167            }
168        }
169        int pointSize = (ivx.size + (int) radius - 1 << 2) + 1;
170        int[] bx = new int[pointSize], by = new int[pointSize];
171        for (int i = 0; i < ivx.size; i++) {
172            int x = ivx.get(i), y = ivy.get(i);
173            bx[i << 2] = x;
174            by[i << 2] = y;
175            bx[i << 2 | 1] = -x;
176            by[i << 2 | 1] = y;
177            bx[i << 2 | 2] = -x;
178            by[i << 2 | 2] = -y;
179            bx[i << 2 | 3] = x;
180            by[i << 2 | 3] = -y;
181        }
182        for (int i = (ivx.size + 1 << 2), p = 1; i < pointSize - 3 && p <= radius + 0.5; i += 4, p++) {
183            bx[i] = p;
184            by[i] = 0;
185            bx[i | 1] = 0;
186            by[i | 1] = p;
187            bx[i | 2] = -p;
188            by[i | 2] = 0;
189            bx[i | 3] = 0;
190            by[i | 3] = -p;
191        }
192        bx[pointSize - 1] = 0;
193        by[pointSize - 1] = 0;
194        int[] vp;
195        int bxp, byp;
196        for (int p = 0; p < pointSize; p++) {
197            vp = offsets[p];
198            bxp = bx[p];
199            byp = by[p];
200            for (int x = 0; x < width; x++) {
201                for (int y = 0; y < height; y++) {
202                    vp[x * height + y] = ((x + bxp + width) % width) * height + ((y + byp + height) % height);
203                }
204            }
205        }
206        return offsets;
207    }

209    /**
210     * Pre-calculates the indices into a substance array that will need to be averaged per point in that array for an
211     * activator or inhibitor. The radius should usually (by convention) be smaller for an activator and larger for an
212     * inhibitor, but very large radii cause this to require significantly more time; consider a maximum of about 20, or
213     * less depending on how fast it needs to run vs. quality, and how many generations you expect.
214     * @param width the width of the substance array; should be consistent throughout calls
215     * @param height the height of the substance array; should be consistent throughout calls
216     * @param radius the radius of the circle to
217     * @return an offset information array, as a jagged 2D int array, that can be passed to {@link #step(double[], int[][], double, int[][], double)}
218     */
219    public static int[][] offsetsCircle(int width, int height, double radius)
220    {
221        if(radius > width || radius > height) radius = Math.min(width, height);
222        radius = Math.max(1, radius);
223        IntVLA ivx = new IntVLA((int)(Math.PI * radius * (radius+1))),
224               ivy = new IntVLA((int)(Math.PI * radius * (radius+1)));
225        for (int x = 1; x <= radius; x++) {
226            for (int y = 1; y <= radius; y++) {
227                if(Math.sqrt(x * x + y * y) + 0.5 <= radius)
228                {
229                    ivx.add(x);
230                    ivy.add(y);
231                }
232            }
233        }
234        int pointSize = (ivx.size + (int)radius - 1 << 2) + 1;
235        int[] bx = new int[pointSize], by = new int[pointSize];
236        for (int i = 0; i < ivx.size; i++) {
237            int x = ivx.get(i), y = ivy.get(i);
238            bx[i << 2] = x;
239            by[i << 2] = y;
240            bx[i << 2 | 1] = -x;
241            by[i << 2 | 1] = y;
242            bx[i << 2 | 2] = -x;
243            by[i << 2 | 2] = -y;
244            bx[i << 2 | 3] = x;
245            by[i << 2 | 3] = -y;
246        }
247        for (int i = (ivx.size + 1 << 2), p = 1; i < pointSize - 3 && p <= radius + 0.5; i+=4, p++) {
248            bx[i] = p;
249            by[i] = 0;
250            bx[i|1] = 0;
251            by[i|1] = p;
252            bx[i|2] = -p;
253            by[i|2] = 0;
254            bx[i|3] = 0;
255            by[i|3] = -p;
256        }
257        bx[pointSize-1] = 0;
258        by[pointSize-1] = 0;
259        int[][] val = new int[pointSize][width * height];
260        int[] vp;
261        int bxp, byp;
262        for (int p = 0; p < pointSize; p++) {
263            vp = val[p];
264            bxp = bx[p];
265            byp = by[p];
266            for (int x = 0; x < width; x++) {
267                for (int y = 0; y < height; y++) {
268                    vp[x * height + y] = ((x + bxp + width) % width) * height + ((y + byp + height) % height);
269                }
270            }
271        }
272        return val;
273    }

275    /**
276     * Alters the given offset information (as a jagged 2D int array) with the given Noise2D instance and seed. This can
277     * change the quality of the pattern produced significantly, going from stripes to whorls and fungoid shapes.
278     * Modifies offsets in-place, and repeated calls can be a good way to alter the pattern this produces.
279     * @param offsets a jagged 2D int array as produced by {@link #offsetsCircle(int, int, double)}; will be modified!
280     * @param width the width of the full area that will be used by the TuringPattern
281     * @param height the height of the full area that will be used by the TuringPattern
282     * @param noise a Noise2D instance, such as {@link SeededNoise#instance}, that will be used to alter offsets
283     * @param seed a seed for the Noise2D
284     */
285    public static void distort(int[][] offsets, int width, int height, Noise.Noise2D noise, int seed)
286    {
287        int pointSize = offsets.length, altSeed = PintRNG.determine(seed);
288        int[] vp;
289        for (int x = 0; x < width; x++) {
290            for (int y = 0; y < height; y++) {
291                int pushX = (int)(2.5 * (noise.getNoiseWithSeed(x, y, seed) + 1)) - 2,
292                        pushY = (int)(2.5 * (noise.getNoiseWithSeed(x, y, altSeed) + 1)) - 2;
293                for (int p = 0; p < pointSize; p++) {
294                    vp = offsets[p];
295                    int a = vp[x * height + y], px = a / height, py = a % height;
296                    vp[x * height + y] = ((px + pushX + width) % width) * height + ((py + pushY + height) % height);
297                }
298            }
299        }
300    }
301    /**
302     * Alters the given offset information (as a jagged 2D int array) with the given Noise3D instance and seed, allowing
303     * a z position for the 3D component so this can change over time with changing z. This can method change the
304     * quality of the pattern produced significantly, going from stripes to whorls and fungoid shapes.
305     * Modifies offsets in-place, and repeated calls can be a good way to alter the pattern this produces.
306     * @param offsets a jagged 2D int array as produced by {@link #offsetsCircle(int, int, double)}; will be modified!
307     * @param width the width of the full area that will be used by the TuringPattern
308     * @param height the height of the full area that will be used by the TuringPattern
309     * @param noise a Noise3D instance, such as {@link SeededNoise#instance}, that will be used to alter offsets
310     * @param z a z position to be given to the Noise3D along with a point's x and y
311     * @param seed a seed for the Noise3D
312     */
313    public static void distort(int[][] offsets, int width, int height, Noise.Noise3D noise, double z, int seed)
314    {
315        int pointSize = offsets.length, altSeed = PintRNG.determine(seed);
316        int[] vp;
317        for (int x = 0; x < width; x++) {
318            for (int y = 0; y < height; y++) {
319                int pushX = (int)(2.5 * (noise.getNoiseWithSeed(x, y, z, seed) + 1)) - 2,
320                        pushY = (int)(2.5 * (noise.getNoiseWithSeed(x, y, z, altSeed) + 1)) - 2;
321                for (int p = 0; p < pointSize; p++) {
322                    vp = offsets[p];
323                    int a = vp[x * height + y], px = a / height, py = a % height;
324                    vp[x * height + y] = ((px + pushX + width) % width) * height + ((py + pushY + height) % height);
325                }
326            }
327        }

330    /**
331     * Brings together the other methods to advance the substance simulation by one step, modifying substance in-place.
332     * You should have generated substance with either one of the initialize() methods or with {@link #refit(double[])},
333     * and the activator and inhibitor should have been produced by calls to an offsets method like
334     * {@link #offsetsCircle(int, int, double)}, with the same width and height passed to initialize (or if you used
335     * refit, the length of the substance array should be equal to width times height). The activation and inhibition
336     * parameters should be very small (larger numbers will cause more significant jumps in a simulation, but may be
337     * better for single generations; neither amount should have an absolute value larger than 0.1 in general), and
338     * inhibition should have the opposite sign of activation.
339     * @param substance as produced by initialize; will be modified!
340     * @param activator as produced by an offsets method
341     * @param activation the small double amount to use when the activator is dominant; should usually be positive
342     * @param inhibitor as produced by an offsets method
343     * @param inhibition the small double amount to use when the inhibitor is dominant; should usually be negative
344     */
345    public static void step(final double[] substance, final int[][] activator,
346                            final double activation, final int[][] inhibitor, final double inhibition) {
347        double mn = Double.POSITIVE_INFINITY, mx = Double.NEGATIVE_INFINITY, t;
348
349        for (int s = 0; s < substance.length; s++) {
350            double activate = 0.0, inhibit = 0.0;
351            for (int p = 0; p < activator.length; p++) {
352                activate += substance[activator[p][s]];
353            }
354            for (int p = 0; p < inhibitor.length; p++) {
355                inhibit += substance[inhibitor[p][s]];
356            }
357            t = (substance[s] += (activate > inhibit) ? activation : inhibition);
358            mx = Math.max(mx, t);
359            mn = Math.min(mn, t);
360        }
361        t = 2.0 / (mx - mn);
362        for (int s = 0; s < substance.length; s++) {
363            substance[s] = (substance[s] - mn) * t - 1;
364        }
365    }

367    /**
368     * Computes the first part of a step, allowing other adjustments to be mixed in before finishing by calling
369     * {@link #normalize(double[])}. A sample adjustment would be
370     * {@link #addNoise(double[], int, int, double, Noise.Noise2D, int)}. This is probably not very useful yet.
371     * @param substance as produced by initialize; will be modified!
372     * @param activator as produced by an offsets method
373     * @param activation the small double amount to use when the activator is dominant; should usually be positive
374     * @param inhibitor as produced by an offsets method
375     * @param inhibition the small double amount to use when the inhibitor is dominant; should usually be negative
376     * @see #step(double[], int[][], double, int[][], double) step, which is preferred to this method
377     */
378    public static void stepPartial(final double[] substance, final int[][] activator,
379                            final double activation, final int[][] inhibitor, final double inhibition) {
380        for (int s = 0; s < substance.length; s++) {
381            double activate = 0.0, inhibit = 0.0;
382            for (int p = 0; p < activator.length; p++) {
383                activate += substance[activator[p][s]];
384            }
385            for (int p = 0; p < inhibitor.length; p++) {
386                inhibit += substance[inhibitor[p][s]];
387            }
388            substance[s] += (activate > inhibit) ? activation : inhibition;
389        }
390    }

434    /**
435     * Finds the highest and lowest values in the substance array and modifies the whole array so the lowest and highest
436     * values are contracted or expanded to -1.0 and 1.0, respectively, and other values change commensurately.
437     * @param substance a substance array, as produced by initialize, will be modified!
438     */
439    public static void normalize(final double[] substance)
440    {
441        double mn = Double.POSITIVE_INFINITY, mx = Double.NEGATIVE_INFINITY, t;
442        for (int s = 0; s < substance.length; s++) {
443            mx = Math.max(mx, (t = substance[s]));
444            mn = Math.min(mn, t);
445        }
446        t = 2.0 / (mx - mn);
447        for (int s = 0; s < substance.length; s++) {
448            substance[s] = (substance[s] - mn) * t - 1;
449        }
450    }

452    /**
453     * Makes a new 2D double array with the given width and height, using the given substance array for contents.
454     * @param width the width of the 2D array to produce; must be 1 or greater
455     * @param height the height of the 2D array to produce; must be 1 or greater
456     * @param substance a substance array as produced by initialize and modified by step
457     * @return a new 2D double array with the contents of substance and the requested size
458     */
459    public static double[][] reshape(final int width, final int height, final double[] substance)
460    {
461        double[][] val = new double[width][height];
462        int s = 0;
463        for (int x = 0; x < width; x++) {
464            for (int y = 0; y < height; y++) {
465                val[x][y] = substance[s++];
466            }
467        }
468        return val;
469    }

471    /**
472     * Modifies target in-place so it is filled with as much data as possible from substance.
473     * @param target a non-null, non-empty 2D double array; will be modified!
474     * @param substance a substance array as produced by initialize and modified by step
475     */
476    public static void fill(final double[][] target, final double[] substance)
477    {
478        if(target == null || substance == null || target.length == 0 || target[0].length == 0 || substance.length == 0)
479            return;
480        int s = 0, width = target.length, height = target[0].length, size = substance.length;
481        for (int x = 0; x < width && s < size; x++) {
482            for (int y = 0; y < height && s < size; y++) {
483                target[x][y] = substance[s++];
484            }
485        }
486    }
