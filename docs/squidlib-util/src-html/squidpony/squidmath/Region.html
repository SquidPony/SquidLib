003import squidpony.squidgrid.mapping.DungeonUtility;

005import java.io.Serializable;
006import java.util.AbstractList;
007import java.util.Collection;
008
009import static squidpony.squidmath.CoordPacker.*;
010/**
011 * <b>NOTE</b>: You should consider {@link GreasedRegion} before using this class, unless you know Region does exactly
012 * what you want.
013 * <br>
014 * Represents an area or series of areas as one logical unit, and allows iterating over or altering that unit.
015 * This can be useful for getting an iterable data structure from a FOV map, Dijkstra map, or just a map from a dungeon
016 * generator. It can also work with some shapes (circles and rectangles).
017 * Regions must be no larger than 256x256 (actually, the Coords in them should fit in that), even if Coord has had its
018 * expandPool() method called. This is because CoordPacker relies on that maximum size to significantly improve
019 * efficiency, and this is really just a convenience class wrapping CoordPacker to avoid some of its complexity.
020 * GreasedRegion allows larger maps, particularly when the Coord pool has been expanded, and should be much faster at
021 * tasks that involve changing the size or shape of an area. It also is mutable by default, without needing to track
022 * the "dirty" and "clean" states of this class. GreasedRegion does not implement the List interface, though; it does
023 * implement Iterable of Coord.
024 * <br>
025 * Created by Tommy Ettinger on 5/12/2016.
026 * @see GreasedRegion You should prefer GreasedRegion for most usage.
027 */
028public class Region extends AbstractList<Coord> implements Serializable{
029
030    private static final long serialVersionUID = 4015272367863327093L;
031
032    protected short[] raw;
033    protected Coord[] coords;
034    private boolean dirty;
035
036    /**
037     * A constructor for a Region that takes a 2D double array, usually the kind produced by FOV, and stores only Coord
038     * positions that correspond to values greater than 0.0 (actually, greater than epsilon, which here is 0.0001).
039     * This won't behave as-expected if you give it a double[][] that DijkstraMap produces; there's a different
040     * constructor for that purpose.
041     * @param fovMap a 2D double array as produced by FOV
042     */
043    public Region(double[][] fovMap)
044    {
045        this(pack(fovMap));
046    }
047
048    /**
049     * A constructor for a Region that takes a 2D double array, usually produced by DijkstraMap, and a maximum value,
050     * and stores only Coord positions that correspond to values no greater than maximum.
051     * This won't behave as-expected if you give it a double[][] that FOV produces; there's a different
052     * constructor for that purpose.
053     * @param dijkstraMap a 2D double array as produced by DijkstraMap
054     * @param maximum the highest value that a position can have in dijkstraMap and still be given a Coord in this
055     */
056    public Region(double[][] dijkstraMap, double maximum)
057    {
058        this(pack(dijkstraMap, maximum));
059    }
060
061    /**
062     * A constructor for a Region that takes a 2D char array, the kind produced by most map/dungeon generators in this
063     * library, and a vararg or array of char that will have their Coord positions used where those chars appear in map.
064     * This is optimized for the common case of a single char in using if you only want to, for example, store '.' to
065     * make a Region of floors, or '#' for walls.
066     * @param map a 2D char array that is usually the kind returned by a dungeon or map generator
067     * @param using an array or vararg of char that will have their Coords used where they appear in map
068     */
069    public Region(char[][] map, char... using)
070    {
071        this(pack(map, using));
072    }
073
074    /**
075     * A constructor for a Region that takes an array or vararg of Coord and encodes all of them in the Region.
076     * @param points an array or vararg of Coord that will be stored in this Region, none can be null
077     */
078    public Region(Coord... points)
079    {
080        this(packSeveral(points));
081    }
082    /**
083     * A constructor for a Region that takes a Collection of Coord, such as a List or Set, and encodes all of them in
084     * the Region.
085     * @param points a Collection of Coord that will be stored in this Region, none can be null
086     */
087    public Region(Collection<Coord> points)
088    {
089        this(packSeveral(points));
090    }
091
092    /**
093     * A constructor that copies another Region so this Region will have the same contents. If the other Region is
094     * "dirty", this won't perform "cleanup" on it, but will ensure that this Region is "clean" at creation.
095     * None of the reference types in other will be used directly in this Region, so changes made to this Region won't
096     * be reflected in other.
097     * @param other another Region to copy into this one
098     */
099    public Region(Region other)
100    {
101        raw = new short[other.raw.length];
102        System.arraycopy(other.raw, 0, raw, 0, raw.length);
103        if(other.dirty)
104        {
105            coords = allPacked(raw);
106        }
107        else
108        {
109            coords = new Coord[other.coords.length];
110            System.arraycopy(other.coords, 0, coords, 0, coords.length);
111        }
112        dirty = false;
113
114    }
115
116    /**
117     * A constructor for a circular Region (possibly truncated at the edges) with a Coord center, an int radius, and a
118     * maximum width and height that the Coords in this Region will not exceed.
119     * @param center the center of the circular Region
120     * @param circleRadius the radius as an int
121     * @param mapWidth one more than the maximum x-position of any Coord this will contain
122     * @param mapHeight one more than the maximum y-position of any Coord this will contain
123     */
124    public Region(Coord center, int circleRadius, int mapWidth, int mapHeight)
125    {
126        this(circle(center, circleRadius, mapWidth, mapHeight));
127    }
128
129    /**
130     * A constructor for a rectangular Region that stores Coords for the area from (minX,minY) at the minimum corner to
131     * (width + minX - 1, height + minY - 1) at the maximum corner. All parameters should be non-negative and less than
132     * 256, and height and width will be reduced if a Coord in the rectangle would extend to 256 in either dimension.
133     * This doesn't take two Coords as parameters because that would be confusing with the constructor that takes a
134     * vararg or array of Coord for its parameters.
135     * @param minX lowest x-coordinate in the rectangle; should be between 0 and 255
136     * @param minY lowest y-coordinate in the rectangle; should be between 0 and 255
137     * @param width total width of the rectangle; must be non-negative
138     * @param height total height of the rectangle; must be non-negative
139     */
140    public Region(int minX, int minY, int width, int height)
141    {
142        this(rectangle(minX, minY, width, height));
143    }
144    /**
145     * A constructor for a Region that takes a specifically-formatted short array (packed data), as produced by
146     * CoordPacker or sometimes other classes, like RegionMap or RoomFinder. If you don't have such data, the other
147     * constructors are recommended instead.
148     * <br>
149     * Note: arrays of Hilbert indices are occasionally returned in CoordPacker as a different kind of short array, but
150     * those methods are always noted as such and those short arrays won't make sense if passed to this constructor.
151     * They may result in a technically-valid Region with random-seeming contents. In general, if a method in
152     * CoordPacker returns a short array (most of them do), but the name contains Hilbert, it may return the wrong kind
153     * (an array of Hilbert indices is wrong, packed data is right); check the documentation for that method.
154     * @param packedData a short array as produced by CoordPacker (usually), or sometimes RoomFinder or RegionMap
155     */
156    public Region(short[] packedData)
157    {
158        raw = new short[packedData.length];
159        System.arraycopy(packedData, 0, raw, 0, packedData.length);
160        coords = allPacked(raw);
161        dirty = false;
162    }
163
164    /**
165     * Gets a single random Coord from this using the given RNG (which can be seeded); returns null if this is empty.
166     * @param rng the source of random numbers used to get a random Coord from this Region
167     * @return a random Coord in this Region, or null if this is empty
168     */
169    public Coord getRandomCoord(RNG rng)
170    {
171        if(CoordPacker.isEmpty(raw))
172            return null;
173        return singleRandom(raw, rng);
174    }
175
176    /**
177     * Takes this region and walks through its Coords in chunks with length equal to separation, creating a new Region
178     * where one Coord in each chunk is kept and the others are discarded.
179     * @param separation an int where higher numbers mean there will be more distance between Coords, and fewer total
180     * @return a new Region made of the separated Coords
181     */
182    public Region separated(int separation)
183    {
184        return new Region(fractionPacked(raw, separation));
185    }
186
187    /**
188     * Takes this region and walks through its Coords in chunks with length equal to separation, creating a new Region
189     * where one randomly-chosen Coord in each chunk is kept and the others are discarded.
190     * @param separation an int where higher numbers mean there will be more distance between Coords, and fewer total
191     * @param rng the source of random numbers used to randomize Coords used, removing any noticeable pattern
192     * @return a new Region made of the separated Coords
193     */
194    public Region randomSeparated(int separation, RNG rng)
195    {
196        return new Region(CoordPacker.randomSeparated(raw, separation, rng));
197    }

199    /**
200     * Gets a representation of this Region as a 2D boolean array with the given width and height. If this contains a
201     * Coord with x and y less than width and height, that position will be true in the 2D array this returns, otherwise
202     * it will be false.
203     * @param width the width of the 2D array to return
204     * @param height the height of the 2D array to return
205     * @return a 2D boolean array where present Coords in this Region will be true, and everything else will be false
206     */
207    public boolean[][] toBooleanArray(int width, int height)
208    {
209        return CoordPacker.unpack(raw, width, height);
210    }
211
212    /**
213     * Gets a representation of this Region as a 2D char array with the given width and height, using on to represent
214     * present Coords and off to represent absent ones. If this contains a Coord with x and y less than width and
215     * height, that position will be equal to on in the 2D array this returns, otherwise it will be equal to off.
216     * @param width the width of the 2D array to return
217     * @param height the height of the 2D array to return
218     * @param on the char to use when a Coord is present in this Region
219     * @param off the char to use where no Coord is present in this Region
220     * @return a 2D char array where present Coords in this Region will be on, and everything else will be off
221     */
222    public char[][] toCharArray(int width, int height, char on, char off)
223    {
224        return CoordPacker.unpackChar(raw, width, height, on, off);
225    }
226
227    /**
228     * Gets the Coord stored at the given index in this Region, or null if index is out of bounds.
229     * The ordering this class uses may seem unusual, but it is predictable, using the pattern a Hilbert Curve takes to
230     * wind through a 256x256 space (at its maximum). Any two given Coords will always have the same before-after
231     * relationship, regardless of other Coords in the Region. A Region is a dense data structure, like a List or array,
232     * so valid indices shouldn't ever return null.
233     * @param index the index into this Region to get a Coord at; should be less than size() and non-negative.
234     * @return the Coord this Region holds at index, if index is valid; otherwise null
235     */
236    @Override
237    public Coord get(int index) {
238        if(dirty)
239        {
240            coords = allPacked(raw);
241            dirty = false;
242        }
243        if(index >= 0 && index < coords.length)
244            return coords[index];
245        return null;
246    }
247
248    /**
249     * Gets the size of this Region as measured in Coords stored.
250     * Performs "cleanup" if the Region is "dirty," even though this doesn't specifically request a Coord. As such, it
251     * may take longer than a call to size() might be expected to, but only just after a "dirtying" method was called.
252     * @return the number of Coords stored in this Region.
253     */
254    @Override
255    public int size() {
256        if(dirty)
257        {
258            coords = allPacked(raw);
259            dirty = false;
260        }
261        return coords.length;
262    }
263
264    /**
265     * Returns true if there are no Coords in this Region, or false otherwise. Can be more efficient than checking if
266     * size() is greater than 0 because it doesn't depend on the "dirty or clean" state, and can quickly check.
267     * @return
268     */
269    @Override
270    public boolean isEmpty() {
271        return CoordPacker.isEmpty(raw);
272    }
273
274    /**
275     * Adds a Coord to this Region, returning false if the Coord is null or already included in this, or true otherwise.
276     * Causes the Region to be considered "dirty", which will make anything that gets a Coord from this to perform some
277     * "cleanup" on the Coords this holds when a Coord is requested. You can perform multiple "dirtying" operations in
278     * succession without needing more "cleanups" than the one when a Coord is next requested.
279     * @param coord a Coord to add to this region
280     * @return true if the Coord was added and was not already present; false if the Coord as null or already present
281     */
282    @Override
283    public boolean add(Coord coord) {
284        if(coord == null || queryPacked(raw, coord.x, coord.y))
285            return false;
286        raw = insertPacked(raw, coord.x, coord.y);
287        dirty = true;
288        return true;
289    }

291    /**
292     * Gets a direct reference to this Region's "raw packed data"; don't modify it unless you know what you're doing!
293     * It's fine to pass this to methods in CoordPacker, since essentially all of those methods won't modify packed data
294     * given as arguments.
295     * @return the raw packed data this class uses; should not be modified carelessly
296     */
297    public short[] getRaw() {
298        return raw;
299    }
300
301    /**
302     * Sets the "raw packed data" to the given short array, as generated by CoordPacker or some parts of RegionMap.
303     * This hopefully won't need to be consumed too much externally, but is an important bridge between this and
304     * CoordPacker's API, which deals mostly with these special short arrays.
305     * Causes the Region to be considered "dirty", which will make anything that gets a Coord from this to perform some
306     * "cleanup" on the Coords this holds when a Coord is requested. You can perform multiple "dirtying" operations in
307     * succession without needing more "cleanups" than the one when a Coord is next requested.
308     * @param raw a short array of packed data; has a very specific format and should usually not be made manually
309     */
310    public void setRaw(short[] raw) {
311        this.raw = raw;
312        dirty = true;
313    }
314
315    /**
316     * Gets the Coords contained in this as an array, a direct reference that does permit modifying the Coords in your
317     * own code without altering "dirty/clean" status. This method does "cleanup" in itself if necessary, but once the
318     * Coords are returned you can change them at-will. The Coords may not reflect changes made by this object if it
319     * creates a new Coord array, as it often does.
320     * @return the Coords contained in this Region as an array
321     */
322    public Coord[] getCoords() {
323        if(dirty)
324        {
325            coords = allPacked(raw);
326            dirty = false;
327        }
328        return coords;
329    }
330
331    /**
332     * Changes this Region to include the given Coords and disregard its previous contents. If any elements of coords
333     * are null, this Region will hold no Coords (a sort of escape hatch to avoid throwing an exception, since all other
334     * methods in this class fail on null entries without potentially crashing a program).
335     * @param coords an array or vararg of Coord that will be used as the entirety of this Region
336     */
337    public void setCoords(Coord... coords) {
338        if (coords == null)
339            return;
340        raw = packSeveral(coords);
341        if (raw == ALL_WALL) {
342            this.coords = new Coord[0];
343            dirty = false;
344        } else {
345            this.coords = new Coord[coords.length];
346            System.arraycopy(coords, 0, this.coords, 0, coords.length);
347            dirty = false;
348        }
349    }

351    /**
352     * Prints this Region to System.out as a grid of chars with the given width and height, using '.' for Coords this
353     * contains and '#' for empty space. Consider using toCharArray() instead if you need more customization for what
354     * you want printed.
355     * @param width the width in chars of the grid to print
356     * @param height the height in chars of the grid to print
357     */
358    public void debugPrint(int width, int height)
359    {
360        DungeonUtility.debugPrint(toCharArray(width, height, '.', '#'));
361    }
