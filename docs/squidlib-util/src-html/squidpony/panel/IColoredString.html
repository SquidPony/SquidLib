233                 * @param one
234                 * @param two 
235                 * @return Whether {@code one} represents the same content as
236                 *         {@code two}.
237                 */
238                /*
239                 * Method could be smarter, i.e. return true more often, by doing some
240                 * normalization. It is unnecessary if you only create instances of
241                 * IColoredString.Impl.
242                 */
243                public static <T> boolean equals(IColoredString<T> one, IColoredString<T> two) {
244                        if (one == two)
245                                return true;
246
247                        final Iterator<IColoredString.Bucket<T>> oneIt = one.iterator();
248                        final Iterator<IColoredString.Bucket<T>> twoIt = two.iterator();
249                        while (true) {
250                                if (oneIt.hasNext()) {
251                                        if (twoIt.hasNext()) {
252                                                final Bucket<T> oneb = oneIt.next();
253                                                final Bucket<T> twob = twoIt.next();
254                                                if (!equals(oneb.getText(), twob.getText()))
255                                                        return false;
256                                                if (!equals(oneb.getColor(), twob.getColor()))
257                                                        return false;
258                                                continue;
259                                        } else
260                                                /* 'this' not terminated, but 'other' is. */
261                                                return false;
262                                } else {
263                                        if (twoIt.hasNext())
264                                                /* 'this' terminated, but not 'other'. */
265                                                return false;
266                                        else
267                                                /* Both terminated */
268                                                break;
269                                }
270
271                        }
272                        return true;
273                }

290                @Override
291                public void append(String text, T color) {
292                        if (text == null || text.isEmpty())
293                                return;
294
295                        if (fragments.isEmpty())
296                                fragments.add(new Bucket<>(text, color));
297                        else {
298                                final Bucket<T> last = fragments.getLast();
299                                if (equals(last.color, color)) {
300                                        /* Append to the last bucket, to avoid extending the list */
301                                        final Bucket<T> novel = last.append(text);
302                                        fragments.removeLast();
303                                        fragments.addLast(novel);
304                                } else
305                                        fragments.add(new Bucket<>(text, color));
306                        }
307                }

327                @Override
328                public void replaceColor(/* @Nullable */ T old, /* @Nullable */ T new_) {
329                        if (equals(old, new_))
330                                /* Nothing to do */
331                                return;
332
333                        final ListIterator<Bucket<T>> it = fragments.listIterator();
334                        while (it.hasNext()) {
335                                final Bucket<T> bucket = it.next();
336                                if (equals(bucket.color, old)) {
337                                        /* Replace */
338                                        it.remove();
339                                        it.add(new Bucket<T>(bucket.getText(), new_));
340                                }
341                                /* else leave untouched */
342                        }
343                }

349                @Override
350                public void setLength(int len) {
351                        int l = 0;
352                        final ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
353                        while (it.hasNext()) {
354                                final IColoredString.Bucket<T> next = it.next();
355                                final String ftext = next.text;
356                                final int flen = ftext.length();
357                                final int nextl = l + flen;
358                                if (nextl < len) {
359                                        /* Nothing to do */
360                                        l += flen;
361                                        continue;
362                                } else if (nextl == len) {
363                                        /* Delete all next fragments */
364                                        while (it.hasNext()) {
365                                                it.next();
366                                                it.remove();
367                                        }
368                                        /* We'll exit the outer loop right away */
369                                } else {
370                                        assert len < nextl;
371                                        /* Trim this fragment */
372                                        final IColoredString.Bucket<T> trimmed = next.setLength(len - l);
373                                        /* Replace this fragment */
374                                        it.remove();
375                                        it.add(trimmed);
376                                        /* Delete all next fragments */
377                                        while (it.hasNext()) {
378                                                it.next();
379                                                it.remove();
380                                        }
381                                        /* We'll exit the outer loop right away */
382                                }
383                        }
384                }

386                @Override
387                public List<IColoredString<T>> wrap(int width) {
388                        if (width <= 0) {
389                                /* Really, you should not rely on this behavior */
390                                System.err.println("Cannot wrap string in empty display");
391                                final List<IColoredString<T>> result = new LinkedList<>();
392                                result.add(this);
393                                return result;
394                        }
395
396                        final List<IColoredString<T>> result = new ArrayList<>();
397                        if (isEmpty()) {
398                                /*
399                                 * Catch this case early on, as empty lines are eaten below (see
400                                 * code after the while). Checking emptyness is cheap anyway.
401                                 */
402                                result.add(this);
403                                return result;
404                        }
405
406                        IColoredString<T> current = create();
407                        int curlen = 0;
408                        final Iterator<Bucket<T>> it = iterator();
409                        while (it.hasNext()) {
410                                final Bucket<T> next = it.next();
411                                final String bucket = next.getText();
412                                final String[] split = bucket.split(" ");
413                                final T color = next.color;
414                                for (int i = 0; i < split.length; i++) {
415                                        if (i == split.length - 1 && bucket.endsWith(" "))
416                                                /*
417                                                 * Do not loose trailing space that got eaten by
418                                                 * 'bucket.split'.
419                                                 */
420                                                split[i] = split[i] + " ";
421                                        final String chunk = split[i];
422                                        final int chunklen = chunk.length();
423                                        final boolean addLeadingSpace = 0 < curlen && 0 < i;
424                                        if (curlen + chunklen + (addLeadingSpace ? 1 : 0) <= width) {
425                                                if (addLeadingSpace) {
426                                                        /*
427                                                         * Do not forget space on which chunk got split. If
428                                                         * the space is offscreen, it's harmless, hence not
429                                                         * checking it.
430                                                         */
431                                                        current.append(' ', null);
432                                                        curlen++;
433                                                }
434
435                                                /* Can add it */
436                                                current.append(chunk, color);
437                                                /* Extend size */
438                                                curlen += chunklen;
439                                        } else {
440                                                /* Need to wrap */
441                                                /* Flush content so far */
442                                                if (!current.isEmpty())
443                                                        result.add(current);
444                                                /*
445                                                 * else: line was prepared, but did not contain anything
446                                                 */
447                                                if (chunklen <= width) {
448                                                        curlen = chunklen;
449                                                        current = create();
450                                                        current.append(chunk, color);
451                                                        /* Reinit size */
452                                                        curlen = chunklen;
453                                                } else {
454                                                        /*
455                                                         * This word is too long. Adding it and preparing a
456                                                         * new line immediately.
457                                                         */
458                                                        /* Add */
459                                                        result.add(new Impl<>(chunk, color));
460                                                        /* Prepare for next rolls */
461                                                        current = create();
462                                                        /* Reinit size */
463                                                        curlen = 0;
464                                                }
465                                        }
466                                }
467                        }
468
469                        if (!current.isEmpty()) {
470                                /* Flush rest */
471                                result.add(current);
472                        }
473
474                        return result;
475                }

477                @Override
478                /*
479                 * smelC: not the cutest result (we should add spaces both from the left
480                 * and the right, instead of just from the left), but better than
481                 * nothing.
482                 */
483                public IColoredString<T> justify(int width) {
484                        int length = length();
485
486                        if (width <= length)
487                                /*
488                                 * If width==length, we're good. If width<length, we cannot
489                                 * adjust
490                                 */
491                                return this;
492
493                        int totalDiff = width - length;
494                        assert 0 < totalDiff;
495
496                        if (width <= totalDiff * 3)
497                                /* Too much of a difference, it would look very weird. */
498                                return this;
499
500                        final IColoredString.Impl<T> result = create();
501
502                        ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
503                        final int nbb = fragments.size();
504                        final int[] bucketToNbSpaces = new int[nbb];
505                        /* The number of buckets that can contribute to justifying */
506                        int totalNbSpaces = 0;
507                        /* The index of the last bucket that has spaces */
508                        int lastHopeIndex = -1;
509                        {
510                                int i = 0;
511                                while (it.hasNext()) {
512                                        final Bucket<T> next = it.next();
513                                        final int nbs = nbSpaces(next.getText());
514                                        totalNbSpaces += nbs;
515                                        bucketToNbSpaces[i] = nbs;
516                                        i++;
517                                }
518
519                                if (totalNbSpaces == 0)
520                                        /* Cannot do anything */
521                                        return this;
522
523                                for (int j = bucketToNbSpaces.length - 1; 0 <= j; j--) {
524                                        if (0 < bucketToNbSpaces[j]) {
525                                                lastHopeIndex = j;
526                                                break;
527                                        }
528                                }
529                                /* Holds because we ruled out 'totalNbSpaces == 0' before */
530                                assert 0 <= lastHopeIndex;
531                        }
532
533                        int toAddPerSpace = totalNbSpaces == 0 ? 0 : (totalDiff / totalNbSpaces);
534                        int totalRest = totalDiff - (toAddPerSpace * totalNbSpaces);
535                        assert 0 <= totalRest;
536
537                        int bidx = -1;
538
539                        it = fragments.listIterator();
540
541                        while (it.hasNext() && 0 < totalDiff) {
542                                bidx++;
543                                final Bucket<T> next = it.next();
544                                final String bucket = next.getText();
545                                final int blength = bucket.length();
546                                final int localNbSpaces = bucketToNbSpaces[bidx];
547                                if (localNbSpaces == 0) {
548                                        /* Cannot change it */
549                                        result.append(next);
550                                        continue;
551                                }
552                                int localDiff = localNbSpaces * toAddPerSpace;
553                                assert localDiff <= totalDiff;
554                                int nb = localDiff / localNbSpaces;
555                                int localRest = localDiff - (nb * localNbSpaces);
556                                if (localRest == 0 && 0 < totalRest) {
557                                        /*
558                                         * Take one for the group. This avoids flushing all spaces
559                                         * needed in the 'last hope' cases below.
560                                         */
561                                        localRest = 1;
562                                }
563                                assert 0 <= localRest;
564                                assert localRest <= totalRest;
565                                String novel = "";
566                                int eatenSpaces = 1;
567                                for (int i = 0; i < blength; i++) {
568                                        final char c = bucket.charAt(i);
569                                        novel += c;
570                                        if (c == ' ' && (0 < localDiff || 0 < totalDiff || 0 < localRest || 0 < totalRest)) {
571                                                /* Can (and should) add an extra space */
572                                                for (int j = 0; j < nb && 0 < localDiff; j++) {
573                                                        novel += " ";
574                                                        localDiff--;
575                                                        totalDiff--;
576                                                }
577                                                if (0 < localRest || 0 < totalRest) {
578                                                        if (eatenSpaces == localNbSpaces) {
579                                                                /* I'm the last hope for this bucket */
580                                                                for (int j = 0; j < localRest; j++) {
581                                                                        novel += " ";
582                                                                        localRest--;
583                                                                        totalRest--;
584                                                                }
585                                                                if (bidx == lastHopeIndex) {
586                                                                        /* I'm the last hope globally */
587                                                                        while (0 < totalRest) {
588                                                                                novel += " ";
589                                                                                totalRest--;
590                                                                        }
591                                                                }
592                                                        } else {
593                                                                if (0 < localRest && 0 < totalRest) {
594                                                                        /* Not the last hope: take one only */
595                                                                        novel += " ";
596                                                                        localRest--;
597                                                                        totalRest--;
598                                                                }
599                                                        }
600                                                }
601                                                eatenSpaces++;
602                                        }
603                                }
604                                /* I did my job */
605                                assert localRest == 0;
606                                /* If I was the hope, I did my job */
607                                assert bidx != lastHopeIndex || totalRest == 0;
608                                result.append(novel, next.getColor());
609                        }
610
611                        while (it.hasNext())
612                                result.append(it.next());
613
614                        return result;
615                }

647                @Override
648                public T colorAt(int index) {
649                        final ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
650                        int now = 0;
651                        while (it.hasNext()) {
652                                final IColoredString.Bucket<T> next = it.next();
653                                final String ftext = next.text;
654                                final int flen = ftext.length();
655                                final int nextl = now + flen;
656                                if (index <= nextl)
657                                        return next.color;
658                                now += flen;
659                        }
660                        throw new NoSuchElementException("Character at index " + index + " in " + this);
661                }

670                /**
671                 * Given some way of converting from a T value to an in-line markup tag, returns a string representation of

673                 * @param markup an IMarkup implementation
674                 * @return a String with markup inserted inside.

676                @Override
677                public String presentWithMarkup(IMarkup<T> markup) {
678                        final StringBuilder result = new StringBuilder();
679                        boolean open = false;
680                        for (Bucket<T> fragment : fragments) {
681                                if(fragment.color != null) {
682                                        if (open)
683                                                result.append(markup.closeMarkup());
684                                        result.append(markup.getMarkup(fragment.color));
685                                        open = true;
686                                }
687                                else {
688                                        if (open)
689                                                result.append(markup.closeMarkup());
690                                        open = false;
691                                }
692                                result.append(fragment.text);
693                        }
694                        return result.toString();
695                }

714                private int nbSpaces(String s) {
715                        final int bd = s.length();
716                        int result = 0;
717                        for (int i = 0; i < bd; i++) {
718                                final char c = s.charAt(i);
719                                if (c == ' ')
720                                        result++;
721                        }
722                        return result;
723                }

726        /**
727         * A piece of a {@link IColoredString}: a text and its color.
728         * 
729         * @author smelC

731         * @param <T>
732         *            The type of colors;

734        class Bucket<T> {
735
736                protected final String text;
737                protected final/* @Nullable */T color;
738
739                public Bucket(String text, /* @Nullable */T color) {
740                        this.text = text == null ? "" : text;
741                        this.color = color;
742                }

744                /**
745                 * @param text
746                 * @return An instance whose text is {@code this.text + text}. Color is
747                 *         unchanged.
748                 */
749                public Bucket<T> append(String text) {
750                        if (text == null || text.isEmpty())
751                                /* Let's save an allocation */
752                                return this;
753                        else
754                                return new Bucket<>(this.text + text, color);
755                }

757                public Bucket<T> setLength(int l) {
758                        final int here = text.length();
759                        if (here <= l)
760                                return this;
761                        else
762                                return new Bucket<>(text.substring(0, l), color);
763                }
