030    public static int containsPart(CharSequence text, char[] search, CharSequence prefix, CharSequence suffix)
031    {
032        if(prefix == null) prefix = "";
033        if(suffix == null) suffix = "";
034        int bl = prefix.length(), el = suffix.length();
035        if(text == null || text.length() == 0 || search == null || (search.length + bl + el <= 0))
036            return 0;
037        int sl = bl + search.length + el, tl = text.length() - sl, f = 0, sl2 = sl - el;
038        char s = (bl <= 0) ? (search.length <= 0 ? suffix.charAt(0) : search[0]) : prefix.charAt(0);
039        PRIMARY:
040        for (int i = 0; i <= tl; i++) {
041            if(text.charAt(i) == s)
042            {
043                for (int j = i+1, x = 1; x < sl; j++, x++) {
044                    if(x < bl)
045                    {
046                        if (text.charAt(j) != prefix.charAt(x)) {
047                            f = Math.max(f, x);
048                            continue PRIMARY;
049                        }
050                    }
051                    else if(x < sl2)
052                    {
053                        if (text.charAt(j) != search[x-bl]) {
054                            f = Math.max(f, x);
055                            continue PRIMARY;
056                        }
057                    }
058                    else
059                    {
060                        if (text.charAt(j) != suffix.charAt(x - sl2)) {
061                            f = Math.max(f, x);
062                            continue PRIMARY;
063                        }
064                    }
065                }
066                return sl;
067            }
068        }
069        return f;
070    }

272     * Like {@link String#substring(int, int)} but returns "" instead of throwing any sort of Exception.
273     * @param source the String to get a substring from
274     * @param beginIndex the first index, inclusive; will be treated as 0 if negative
275     * @param endIndex the index after the last character (i.e. length, so exclusive); if negative this returns ""
276     * @return the substring of source between beginIndex and endIndex, or "" if any parameters are null/invalid
277     */
278    public static String safeSubstring(String source, int beginIndex, int endIndex)
279    {
280        if(endIndex < 0 || source == null || source.isEmpty()) return "";
281        if(beginIndex < 0) beginIndex = 0;
282        if(endIndex > source.length()) endIndex = source.length();
283        if(beginIndex > endIndex) return "";
284        return source.substring(beginIndex, endIndex);
285    }

288     * Like {@link String#split(String)} but doesn't use any regex for splitting (delimiter is a literal String).
289     * @param source the String to get split-up substrings from
290     * @param delimiter the literal String to split on (not a regex); will not be included in the returned String array
291     * @return a String array consisting of at least one String (all of Source if nothing was split)
292     */
293    public static String[] split(String source, String delimiter) {
294        int amount = count(source, delimiter);
295        if (amount <= 0) return new String[]{source};
296        String[] splat = new String[amount+1];
297        int dl = delimiter.length(), idx = -dl, idx2;
298        for (int i = 0; i < amount; i++) {
299            splat[i] = safeSubstring(source, idx+dl, idx = source.indexOf(delimiter, idx+dl));
300        }
301        if((idx2 = source.indexOf(delimiter, idx+dl)) < 0)
302        {
303            splat[amount] = safeSubstring(source, idx+dl, source.length());
304        }
305        else
306        {
307            splat[amount] = safeSubstring(source, idx+dl, idx2);
308        }
309        return splat;
310    }

431     * Reads in a CharSequence containing only hex digits (only 0-9, a-f, and A-F) with an optional sign at the start
432     * and returns the long they represent, reading at most 16 characters (17 if there is a sign) and returning the
433     * result if valid, or 0 if nothing could be read. The leading sign can be '+' or '-' if present. This can also
434     * represent negative numbers as they are printed by such methods as String.format given a %x in the formatting
435     * string, or this class' {@link #hex(long)} method; that is, if the first char of a 16-char (or longer)
436     * CharSequence is a hex digit 8 or higher, then the whole number represents a negative number, using two's
437     * complement and so on. This means "FFFFFFFFFFFFFFFF" would return the long -1 when passed to this, though you
438     * could also simply use "-1 ".
439     * <br>
440     * Should be fairly close to Java 8's Long.parseUnsignedLong method, which is an odd omission from earlier JDKs.
441     * This doesn't throw on invalid input, though, instead returning 0 if the first char is not a hex digit, or
442     * stopping the parse process early if a non-hex-digit char is read before the end of cs is reached. If the parse is
443     * stopped early, this behaves as you would expect for a number with less digits, and simply doesn't fill the larger
444     * places.
445     * @param cs a CharSequence, such as a String, containing only hex digits with an optional sign (no 0x at the start)
446     * @return the long that cs represents
447     */

451    /**
452     * Reads in a CharSequence containing only hex digits (only 0-9, a-f, and A-F) with an optional sign at the start
453     * and returns the long they represent, reading at most 16 characters (17 if there is a sign) and returning the
454     * result if valid, or 0 if nothing could be read. The leading sign can be '+' or '-' if present. This can also
455     * represent negative numbers as they are printed by such methods as String.format given a %x in the formatting
456     * string, or this class' {@link #hex(long)} method; that is, if the first char of a 16-char (or longer)
457     * CharSequence is a hex digit 8 or higher, then the whole number represents a negative number, using two's
458     * complement and so on. This means "FFFFFFFFFFFFFFFF" would return the long -1 when passed to this, though you
459     * could also simply use "-1 ". If you use both '-' at the start and have the most significant digit as 8 or higher,
460     * such as with "-FFFFFFFFFFFFFFFF", then both indicate a negative number, but the digits will be processed first
461     * (producing -1) and then the whole thing will be multiplied by -1 to flip the sign again (returning 1).
462     * <br>
463     * Should be fairly close to Java 8's Long.parseUnsignedLong method, which is an odd omission from earlier JDKs.
464     * This doesn't throw on invalid input, though, instead returning 0 if the first char is not a hex digit, or
465     * stopping the parse process early if a non-hex-digit char is read before end is reached. If the parse is stopped
466     * early, this behaves as you would expect for a number with less digits, and simply doesn't fill the larger places.
467     * @param cs a CharSequence, such as a String, containing only hex digits with an optional sign (no 0x at the start)
468     * @param start the (inclusive) first character position in cs to read
469     * @param end the (exclusive) last character position in cs to read (this stops after 16 characters if end is too large)
470     * @return the long that cs represents
471     */
472    public static long longFromHex(final CharSequence cs, final int start, int end) {
473        int len, h, lim = 16;
474        if (cs == null || start < 0 || end <= 0 || end - start <= 0
475                || (len = cs.length()) - start <= 0 || end > len)
476            return 0;
477        char c = cs.charAt(start);
478        if (c == '-') {
479            len = -1;
480            h = 0;
481            ++end;
482            lim = 17;
483        } else if (c == '+') {
484            len = 1;
485            h = 0;
486            ++end;
487            lim = 17;
488        } else if (c > 102 || (h = hexCodes[c]) < 0)
489            return 0;
490        else {
491            len = 1;
492        }
493        long data = h;
494        for (int i = start; i < end && i < start + lim; i++) {
495            if ((c = cs.charAt(i)) > 102 || (h = hexCodes[c]) < 0)
496                return data * len;
497            data <<= 4;
498            data |= h;
499        }
500        return data * len;
501    }

562        {
