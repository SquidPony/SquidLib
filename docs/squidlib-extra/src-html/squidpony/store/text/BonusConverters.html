020    public static final StringConvert<OrderedMap<Character, String>> convertMapCharString =
021            Converters.convertOrderedMap(Converters.convertChar, Converters.convertString);
022    public static final StringConvert<List<FakeLanguageGen>> convertListLanguage = Converters.convertList(Converters.convertFakeLanguageGen);

024            = Converters.convertOrderedMap(Converters.convertChar, convertListLanguage);
025    public static final StringConvert<StatefulRNG> convertStatefulRNG = new StringConvert<StatefulRNG>("StatefulRNG") {
026        @Override
027        public String stringify(StatefulRNG item) {
028            return StringKit.hex(item.getState()) + ':' + (item.getRandomness().getClass().getSimpleName());
029        }
030
031        @Override
032        public StatefulRNG restore(String text) {
033            long state = StringKit.longFromHex(text);
034            try {
035                StatefulRandomness sr = (StatefulRandomness) (ClassReflection.forName(text.substring(text.indexOf(':') + 1)).newInstance());
036                sr.setState(state);
037                return new StatefulRNG(sr);
038            }catch (Exception re)
039            {
040                return new StatefulRNG(state);
041            }
042        }
043    };

045    public static final StringConvert<RNG> convertRNG = new StringConvert<RNG>("RNG") {
046        @Override
047        public String stringify(RNG item) {
048            return "RNG:" + (item.getRandomness().getClass().getSimpleName());
049        }
050
051        @Override
052        public RNG restore(String text) {
053            try {
054                RandomnessSource rs = (RandomnessSource) (ClassReflection.forName(text.substring(text.indexOf(':') + 1)).newInstance());
055                return new RNG(rs);
056            }catch (Exception re)
057            {
058                return new RNG();
059            }
060        }
061    };
062    public static <K> StringConvert<ProbabilityTable<K>> convertProbabilityTable(final StringConvert<K> convert) {
063        CharSequence[] types = StringConvert.asArray("ProbabilityTable", convert.name);
064        StringConvert found = StringConvert.lookup(types);
065        if (found != null)
066            return found; // in this case we've already created a StringConvert for this type combination
067        final StringConvert<Arrangement<K>> convertArrange = Converters.convertArrangement(convert);
068        return new StringConvert<ProbabilityTable<K>>(types) {
069            @Override
070            public String stringify(ProbabilityTable<K> item) {
071                StringBuilder sb = new StringBuilder(256);
072                appendQuoted(sb, convertRNG.stringify(item.getRandom()));
073                sb.append(' ');
074                appendQuoted(sb, convertIntVLA.stringify(item.weights));
075                sb.append(' ');
076                appendQuoted(sb, convertArrange.stringify(item.table));
077                for (int i = 0; i < item.extraTable.size(); i++) {
078                    sb.append(' ');
079                    appendQuoted(sb, stringify(item.extraTable.get(i)));
080                }
081                return sb.toString();
082            }
083
084            @Override
085            public ProbabilityTable<K> restore(String text) {
086                ObText.ContentMatcher m = makeMatcher(text);
087                if(!m.find() || !m.hasMatch())
088                    return null;
089                ProbabilityTable<K> pt = new ProbabilityTable<>(convertRNG.restore(m.getMatch()));
090                if(!m.find() || !m.hasMatch())
091                    return pt;
092                pt.weights.addAll(convertIntVLA.restore(m.getMatch()));
093                if(!m.find() || !m.hasMatch())
094                {
095                    pt.weights.clear();
096                    return pt;
097                }
098                pt.table.putAll(convertArrange.restore(m.getMatch()));
099                while (m.find()) {
100                    if (m.hasMatch()) {
101                        pt.extraTable.add(restore(m.getMatch()));
102                    }
103                }
104                return pt;
105            }
106        };
107    }
108
109    public static <K> StringConvert<ProbabilityTable<K>> convertProbabilityTable(final CharSequence type) {
110        return convertProbabilityTable((StringConvert<K>) StringConvert.get(type));
111    }
112
113    public static <K> StringConvert<ProbabilityTable<K>> convertProbabilityTable(final Class<K> type) {
114        return convertProbabilityTable((StringConvert<K>) StringConvert.get(type.getSimpleName()));
115    }

117    public static final StringConvert<SpillWorldMap> convertSpillWorldMap = new StringConvert<SpillWorldMap>("SpillWorldMap") {
118        @Override
119        public String stringify(SpillWorldMap item) {
120            return item.width
121                    + "\t" + item.height
122                    + "\t" + item.name
123                    + '\t' + Converters.convertArrayInt2D.stringify(item.heightMap)
124                    + '\t' + Converters.convertArrayCoord.stringify(item.mountains)
125                    + '\t' + Converters.convertArrayChar2D.stringify(item.politicalMap)
126                    + '\t' + convertMapCharString.stringify(item.atlas)
127                    + '\t' + convertStatefulRNG.stringify(item.rng);
128        }
129
130        @Override
131        public SpillWorldMap restore(String text) {
132            int pos;
133            SpillWorldMap swm = new SpillWorldMap(
134                    Integer.decode(text.substring(0, (pos = text.indexOf('\t')))),
135                    Integer.decode(text.substring(pos+1, (pos = text.indexOf('\t', pos+1)))),
136                    text.substring(pos+1, (pos = text.indexOf('\t', pos+1)))
137            );
138            swm.heightMap = Converters.convertArrayInt2D.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1))));
139            swm.mountains = Converters.convertArrayCoord.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1))));
140            swm.politicalMap = Converters.convertArrayChar2D.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1))));
141            swm.atlas.clear();
142            swm.atlas.putAll(convertMapCharString.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1)))));
143            swm.rng = convertStatefulRNG.restore(text.substring(pos+1));
144            return swm;
145        }
146    };

148    public static final StringConvert<PoliticalMapper> convertPoliticalMapper = new StringConvert<PoliticalMapper>("PoliticalMapper") {
149        @Override
150        public String stringify(PoliticalMapper item) {
151            return item.width
152                    + "\t" + item.height
153                    + "\t" + item.name
154                    + '\t' + Converters.convertArrayChar2D.stringify(item.politicalMap)
155                    + '\t' + convertMapCharString.stringify(item.atlas)
156                    + '\t' + convertMapCharListLanguage.stringify(item.spokenLanguages)
157                    + '\t' + convertStatefulRNG.stringify(item.rng);
158        }
159
160        @Override
161        public PoliticalMapper restore(String text) {
162            int pos;
163            PoliticalMapper pm = new PoliticalMapper();
164            pm.width = Integer.decode(text.substring(0, (pos = text.indexOf('\t'))));
165            pm.height = Integer.decode(text.substring(pos+1, (pos = text.indexOf('\t', pos+1))));
166            pm.name = text.substring(pos+1, (pos = text.indexOf('\t', pos+1)));
167            pm.politicalMap = Converters.convertArrayChar2D.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1))));
168            pm.atlas.clear();
169            pm.atlas.putAll(convertMapCharString.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1)))));
170            pm.spokenLanguages.clear();
171            pm.spokenLanguages.putAll(convertMapCharListLanguage.restore(text.substring(pos+1, (pos = text.indexOf('\t', pos+1)))));
172            pm.rng = convertStatefulRNG.restore(text.substring(pos+1));
173            return pm;
174        }
175    };
