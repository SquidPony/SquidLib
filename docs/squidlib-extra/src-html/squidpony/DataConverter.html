001package squidpony;
002
003import com.badlogic.gdx.utils.JsonWriter;
004import squidpony.store.json.JsonConverter;
005
006/**
007 * Augmented version of LibGDX's Json class that knows how to handle various data types common in SquidLib.
008 * This includes OrderedMap, which notably allows non-String keys (LibGDX's default Map serializer requires keys to be
009 * Strings), but does not currently allow the IHasher to be set (which only should affect OrderedMaps with array keys).
010 * It also makes significantly shorter serialized output for 2D char arrays, GreasedRegion and FakeLanguageGen objects,
011 * and various collections (IntDoubleOrderedMap, IntVLA, Arrangement, K2, and K2V1 at least).
012 * Created by Tommy Ettinger on 1/9/2017.
013 */
014public class DataConverter extends JsonConverter {
015    /**
016     * Creates a new DataConverter using "minimal" output type, so it omits double quotes whenever possible but gives
017     * up compatibility with most other JSON readers. Give the constructor
018     * {@link com.badlogic.gdx.utils.JsonWriter.OutputType#json} if you need full compatibility.
019     */
020    public DataConverter() {
021        super();
022    }
023
024    /**
025     * Creates a new DataConverter with the given OutputType; typically minimal is fine, but compatibility may require
026     * you to use json; the javascript type is a sort of middle ground.
027     * @param outputType a JsonWriter.OutputType enum value that determines what syntax can be omitted from the output
028     */
029    public DataConverter(JsonWriter.OutputType outputType) {
030        super(outputType);
031    }
032}
