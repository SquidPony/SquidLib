001package squidpony;
002
003import squidpony.annotation.Beta;
004import squidpony.store.json.JsonStorage;
005
006/**
007 * Helps games store information in libGDX's Preferences class as Strings, then get it back out.
008 * Created by Tommy Ettinger on 9/16/2016.
009 */
010@Beta
011public class SquidStorage extends JsonStorage {
012    /**
013     * Please don't use this constructor if possible; it simply calls {@link #SquidStorage(String)} with the constant
014     * String "nameless". This could easily overlap with other files/sections in Preferences, so you should always
015     * prefer giving a String argument to the constructor, typically the name of the game.
016     * @see #SquidStorage(String) the recommended constructor to use
017     */
018    public SquidStorage()
019    {
020        super("nameless");
021    }
022
023    /**
024     * Creates a SquidStorage with the given fileName to save using Preferences from libGDX. The name should generally
025     * be the name of this game or application, and must be a valid name for a file (so no slashes, backslashes, colons,
026     * semicolons, or commas for certain, and other non-alphanumeric characters are also probably invalid). You should
027     * not assume anything is present in the Preferences storage unless you have put it there, and this applies doubly
028     * to games or applications other than your own; you should avoid values for fileName that might overlap with
029     * another game's Preferences values.
030     * <br>
031     * To organize saved data into sub-sections, you specify logical units (like different players' saved games) with a
032     * String outerName when you call {@link #store(String)}, and can further distinguish data under the outerName when
033     * you call {@link #put(String, Object)} to put each individual item into the saved storage with its own innerName.
034     * <br>
035     * Calling this also sets up custom serializers for several important types in SquidLib; char[][], OrderedMap,
036     * IntDoubleOrderedMap, FakeLanguageGen, GreasedRegion, and notably Pattern from RegExodus all have smaller
037     * serialized representations than the default. OrderedMap allows non-String keys, which gets around a limitation in
038     * JSON maps normally, and both FakeLanguageGen and Pattern are amazingly smaller with the custom representation.
039     * The custom char[][] representation is about half the normal size by omitting commas after each char.
040     * @param fileName the valid file name to create or open from Preferences; typically the name of the game/app.
041     */
042    public SquidStorage(final String fileName)
043    {
044        super(fileName);
045    }
046
047    /**
048     * Prepares to store the Object {@code o} to be retrieved with {@code innerName} in the current group of objects.
049     * Does not write to a permanent location until {@link #store(String)} is called. The innerName used to store an
050     * object is required to get it back again, and can also be used to remove it before storing (or storing again).
051     * @param innerName one of the two Strings needed to retrieve this later
052     * @param o the Object to prepare to store
053     * @return this for chaining
054     */
055    public SquidStorage put(String innerName, Object o)
056    {
057        super.put(innerName, o);
058        return this;
059    }
060
061    /**
062     * Actually stores all objects that had previously been prepared with {@link #put(String, Object)}, with
063     * {@code outerName} used as a key to retrieve any object in the current group. Flushes the preferences, making the
064     * changes permanent (until overwritten), but does not change the current group (you may want to call this method
065     * again with additional items in the current group, and that would simply involve calling put() again). If you want
066     * to clear the current group, use {@link #clear()}. If you want to remove just one object from the current group,
067     * use {@link #remove(String)}.
068     * @param outerName one of the two Strings needed to retrieve any of the objects in the current group
069     * @return this for chaining
070     */
071    public SquidStorage store(String outerName)
072    {
073        super.store(outerName);
074        return this;
075    }
076
077    /**
078     * Gets a String representation of the data that would be saved when {@link #store(String)} is called. This can be
079     * useful for finding particularly problematic objects that require unnecessary space when serialized.
080     * @return a String that previews what would be stored permanently when {@link #store(String)} is called
081     */
082    public String show()
083    {
084        return super.show();
085    }
086
087    /**
088     * Clears the current group of objects; recommended if you intend to store under multiple outerName keys.
089     * @return this for chaining
090     */
091    public SquidStorage clear()
092    {
093        super.clear();
094        return this;
095    }
096
097    /**
098     * Removes one object from the current group by the {@code innerName} it was prepared with using
099     * {@link #put(String, Object)}. This does not affect already-stored objects unless {@link #store(String)} is called
100     * after this, in which case the new version of the current group, without the object this removed, is stored.
101     * @param innerName the String key used to put an object in the current group with {@link #put(String, Object)}
102     * @return this for chaining
103     */
104    public SquidStorage remove(String innerName)
105    {
106        super.remove(innerName);
107        return this;
108    }
109
110    /**
111     * Gets an object from the storage by the given {@code outerName} key from {@link #store(String)} and
112     * {@code innerName} key from {@link #put(String, Object)}, and uses the class given by {@code type} for the
113     * returned value, assuming it matches the object that was originally put with those keys. If no such object is
114     * present, returns null. Results are undefined if {@code type} doesn't match the actual class of the stored object.
115     * @param outerName the key used to store the group of objects with {@link #store(String)}
116     * @param innerName the key used to store the specific object with {@link #put(String, Object)}
117     * @param type the class of the value; for a class like RNG, use {@code RNG.class}, but changed to fit
118     * @param <T> the type of the value to retrieve; if type was {@code RNG.class}, this would be {@code RNG}
119     * @return the retrieved value if successful, or null otherwise
120     */
121    @SuppressWarnings("unchecked")
122    public <T> T get(String outerName, String innerName, Class<T> type)
123    {
124        return super.get(outerName, innerName, type);
125    }
126
127    /**
128     * Gets the approximate size of the currently-stored preferences. This assumes UTF-16 storage, which is the case for
129     * GWT's LocalStorage. Since GWT is restricted to the size the browser permits for LocalStorage, and this limit can
130     * be rather small (about 5 MB, sometimes more but not reliably), this method is especially useful there, but it may
131     * yield inaccurate sizes on other platforms that save Preferences data differently.
132     * @return the size, in bytes, of the already-stored preferences
133     */
134    public int preferencesSize()
135    {
136        return super.preferencesSize();
137    }
138
139}
